136 int
137 coap_endpoint_snprint(char *buf, size_t size, const coap_endpoint_t *ep)
138 {
139  int n;
140  if(buf == NULL || size == 0) {
141  return 0;
142  }
143  if(ep == NULL) {
144  n = snprintf(buf, size - 1, "(NULL EP)");
145  } else {
146  if(ep->secure) {
147  n = snprintf(buf, size - 1, "coaps://[");
148  } else {
149  n = snprintf(buf, size - 1, "coap://[");
150  }
151  if(n < size - 1) {
152  n += uiplib_ipaddr_snprint(&buf[n], size - n - 1, &ep->ipaddr);
153  }
154  if(n < size - 1) {
155  n += snprintf(&buf[n], size -n - 1, "]:%u", uip_ntohs(ep->port));
156  }
157  }
158  if(n >= size - 1) {
159  buf[size - 1] = '\0';
160  }
161  return n;
162 }

182 static int
183 index_of(const char *data, int offset, int len, uint8_t c)
184 {
185  if(offset < 0) {
186  return offset;
187  }
188  for(; offset < len; offset++) {
189  if(data[offset] == c) {
190  return offset;
191  }
192  }
193  return -1;

196 static int
197 get_port(const char *inbuf, size_t len, uint32_t *value)
198 {
199  int i;
200  *value = 0;
201  for(i = 0; i < len; i++) {
202  if(inbuf[i] >= '0' && inbuf[i] <= '9') {
203  *value = *value * 10 + (inbuf[i] - '0');
204  } else {
205  break;
206  }
207  }
208  return i;
209 }

211 int
212 coap_endpoint_parse(const char *text, size_t size, coap_endpoint_t *ep)
213 {
214  /* Only IPv6 supported */
215  int start = index_of(text, 0, size, '[');
216  int end = index_of(text, start, size, ']');
217  int secure = strncmp((const char *)text, "coaps:", 6) == 0;
218  uint32_t port;
219  if(start > 0 && end > start &&
220  uiplib_ipaddrconv((const char *)&text[start], &ep->ipaddr)) {
221  if(text[end + 1] == ':' &&
222  get_port(text + end + 2, size - end - 2, &port)) {
223  ep->port = UIP_HTONS(port);
224  } else if(secure) {
225  /**
226  * Secure CoAP should use a different port but for now
227  * the same port is used.
228  */
229  LOG_DBG("Using secure port (coaps)\n");
230  ep->port = SERVER_LISTEN_SECURE_PORT;
231  ep->secure = 1;
232  } else {
233  ep->port = SERVER_LISTEN_PORT;
234  ep->secure = 0;
235  }
236  return 1;
237  } else {
238  if(uiplib_ipaddrconv((const char *)&text, &ep->ipaddr)) {
239  ep->port = SERVER_LISTEN_PORT;
240  ep->secure = 0;
241  return 1;
242  }
243  }
244  return 0;
245 }

383 int
384 coap_sendto(const coap_endpoint_t *ep, const uint8_t *data, uint16_t length)
385 {
386  if(ep == NULL) {
387  LOG_WARN("failed to send - no endpoint\n");
388  return -1;
389  }
390 
391  if(!coap_endpoint_is_connected(ep)) {
392  LOG_WARN("endpoint ");
393  LOG_WARN_COAP_EP(ep);
394  LOG_WARN_(" not connected - dropping packet\n");
395  return -1;
396  }
397 
398 #ifdef WITH_DTLS
399  if(coap_endpoint_is_secure(ep)) {
400  if(dtls_context) {
401  int ret;
402 
403  ret = dtls_write(dtls_context, (session_t *)ep, (uint8_t *)data, length);
404  LOG_INFO("sent DTLS to ");
405  LOG_INFO_COAP_EP(ep);
406  if(ret < 0) {
407  LOG_INFO_(" - error %d\n", ret);
408  } else {
409  LOG_INFO_(" %d/%u bytes\n", ret, length);
410  }
411  return ret;
412  } else {
413  LOG_WARN("no DTLS context\n");
414  return -1;
415  }
416  }
417 #endif /* WITH_DTLS */
418 
419  uip_udp_packet_sendto(udp_conn, data, length, &ep->ipaddr, ep->port);
420  LOG_INFO("sent to ");
421  LOG_INFO_COAP_EP(ep);
422  LOG_INFO_(" %u bytes\n", length);
423  return length;
424 }

426 PROCESS_THREAD(coap_engine, ev, data)
427 {
428  PROCESS_BEGIN();
429 
430  /* new connection with remote host */
431  udp_conn = udp_new(NULL, 0, NULL);
432  udp_bind(udp_conn, SERVER_LISTEN_PORT);
433  LOG_INFO("Listening on port %u\n", uip_ntohs(udp_conn->lport));
434 
435 #ifdef WITH_DTLS
436  /* create new context with app-data */
437  dtls_conn = udp_new(NULL, 0, NULL);
438  if(dtls_conn != NULL) {
439  udp_bind(dtls_conn, SERVER_LISTEN_SECURE_PORT);
440  LOG_INFO("DTLS listening on port %u\n", uip_ntohs(dtls_conn->lport));
441  dtls_context = dtls_new_context(dtls_conn);
442  }
443  if(!dtls_context) {
444  LOG_WARN("DTLS: cannot create context\n");
445  } else {
446  dtls_set_handler(dtls_context, &cb);
447  }
448 #endif /* WITH_DTLS */
449 
450  while(1) {
451  PROCESS_YIELD();
452 
453  if(ev == tcpip_event) {
454  if(uip_newdata()) {
455 #ifdef WITH_DTLS
456  if(uip_udp_conn == dtls_conn) {
457  process_secure_data();
458  continue;
459  }
460 #endif /* WITH_DTLS */
461  process_data();
462  }
463  }
464  } /* while (1) */
465 
466  PROCESS_END();
467 }

523 /* This function is the "key store" for tinyDTLS. It is called to
524  * retrieve a key for the given identity within this particular
525  * session. */
526 static int
527 get_psk_info(struct dtls_context_t *ctx,
528  const session_t *session,
529  dtls_credentials_type_t type,
530  const unsigned char *id, size_t id_len,
531  unsigned char *result, size_t result_length)
532 {
533  coap_keystore_psk_entry_t ks;
534 
535  if(dtls_keystore == NULL) {
536  LOG_DBG("--- No key store available ---\n");
537  return 0;
538  }
539 
540  memset(&ks, 0, sizeof(ks));
541  LOG_DBG("---===>>> Getting the Key or ID <<<===---\n");
542  switch(type) {
543  case DTLS_PSK_IDENTITY:
544  if(id && id_len) {
545  ks.identity_hint = id;
546  ks.identity_hint_len = id_len;
547  LOG_DBG("got psk_identity_hint: '");
548  LOG_DBG_COAP_STRING((const char *)id, id_len);
549  LOG_DBG_("'\n");
550  }
551 
552  if(dtls_keystore->coap_get_psk_info) {
553  /* we know that session is a coap endpoint */
554  dtls_keystore->coap_get_psk_info((coap_endpoint_t *)session, &ks);
555  }
556  if(ks.identity == NULL || ks.identity_len == 0) {
557  LOG_DBG("no psk_identity found\n");
558  return 0;
559  }
560 
561  if(result_length < ks.identity_len) {
562  LOG_DBG("cannot return psk_identity -- buffer too small\n");
563  return dtls_alert_fatal_create(DTLS_ALERT_INTERNAL_ERROR);
564  }
565  memcpy(result, ks.identity, ks.identity_len);
566  LOG_DBG("psk_identity with %u bytes found\n", ks.identity_len);
567  return ks.identity_len;
568 
569  case DTLS_PSK_KEY:
570  if(dtls_keystore->coap_get_psk_info) {
571  ks.identity = id;
572  ks.identity_len = id_len;
573  /* we know that session is a coap endpoint */
574  dtls_keystore->coap_get_psk_info((coap_endpoint_t *)session, &ks);
575  }
576  if(ks.key == NULL || ks.key_len == 0) {
577  LOG_DBG("PSK for unknown id requested, exiting\n");
578  return dtls_alert_fatal_create(DTLS_ALERT_ILLEGAL_PARAMETER);
579  }
580 
581  if(result_length < ks.key_len) {
582  LOG_DBG("cannot return psk -- buffer too small\n");
583  return dtls_alert_fatal_create(DTLS_ALERT_INTERNAL_ERROR);
584  }
585  memcpy(result, ks.key, ks.key_len);
586  LOG_DBG("psk with %u bytes found\n", ks.key_len);
587  return ks.key_len;
588 
589  default:
590  LOG_WARN("unsupported key store request type: %d\n", type);
591  }
592 
593  return dtls_alert_fatal_create(DTLS_ALERT_INTERNAL_ERROR);
594 }
