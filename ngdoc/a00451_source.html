151 static radio_result_t
152 get_value(radio_param_t param, radio_value_t *value)
153 {
154  int i, v;
155 
156  if(!value) {
157  return RADIO_RESULT_INVALID_VALUE;
158  }
159  switch(param) {
160  case RADIO_PARAM_POWER_MODE:
161  *value = receive_on ? RADIO_POWER_MODE_ON : RADIO_POWER_MODE_OFF;
162  return RADIO_RESULT_OK;
163  case RADIO_PARAM_CHANNEL:
164  *value = cc2420_get_channel();
165  return RADIO_RESULT_OK;
166  case RADIO_PARAM_RX_MODE:
167  *value = 0;
168  if(getreg(CC2420_MDMCTRL0) & ADR_DECODE) {
169  *value |= RADIO_RX_MODE_ADDRESS_FILTER;
170  }
171  if(getreg(CC2420_MDMCTRL0) & AUTOACK) {
172  *value |= RADIO_RX_MODE_AUTOACK;
173  }
174  if(poll_mode) {
175  *value |= RADIO_RX_MODE_POLL_MODE;
176  }
177  return RADIO_RESULT_OK;
178  case RADIO_PARAM_TX_MODE:
179  *value = 0;
180  if(send_on_cca) {
181  *value |= RADIO_TX_MODE_SEND_ON_CCA;
182  }
183  return RADIO_RESULT_OK;
184  case RADIO_PARAM_TXPOWER:
185  v = cc2420_get_txpower();
186  *value = OUTPUT_POWER_MIN;
187  /* Find the actual estimated output power in conversion table */
188  for(i = 0; i < OUTPUT_NUM; i++) {
189  if(v >= output_power[i].config) {
190  *value = output_power[i].power;
191  break;
192  }
193  }
194  return RADIO_RESULT_OK;
195  case RADIO_PARAM_CCA_THRESHOLD:
196  *value = get_cca_threshold() + RSSI_OFFSET;
197  return RADIO_RESULT_OK;
198  case RADIO_PARAM_RSSI:
199  /* Return the RSSI value in dBm */
200  *value = cc2420_rssi();
201  return RADIO_RESULT_OK;
202  case RADIO_PARAM_LAST_RSSI:
203  /* RSSI of the last packet received */
204  *value = cc2420_last_rssi;
205  return RADIO_RESULT_OK;
206  case RADIO_PARAM_LAST_LINK_QUALITY:
207  /* LQI of the last packet received */
208  *value = cc2420_last_correlation;
209  return RADIO_RESULT_OK;
210  case RADIO_CONST_CHANNEL_MIN:
211  *value = 11;
212  return RADIO_RESULT_OK;
213  case RADIO_CONST_CHANNEL_MAX:
214  *value = 26;
215  return RADIO_RESULT_OK;
216  case RADIO_CONST_TXPOWER_MIN:
217  *value = OUTPUT_POWER_MIN;
218  return RADIO_RESULT_OK;
219  case RADIO_CONST_TXPOWER_MAX:
220  *value = OUTPUT_POWER_MAX;
221  return RADIO_RESULT_OK;
222  default:
223  return RADIO_RESULT_NOT_SUPPORTED;
224  }
225 }

227 static radio_result_t
228 set_value(radio_param_t param, radio_value_t value)
229 {
230  int i;
231 
232  switch(param) {
233  case RADIO_PARAM_POWER_MODE:
234  if(value == RADIO_POWER_MODE_ON) {
235  cc2420_on();
236  return RADIO_RESULT_OK;
237  }
238  if(value == RADIO_POWER_MODE_OFF) {
239  cc2420_off();
240  return RADIO_RESULT_OK;
241  }
242  return RADIO_RESULT_INVALID_VALUE;
243  case RADIO_PARAM_CHANNEL:
244  if(value < 11 || value > 26) {
245  return RADIO_RESULT_INVALID_VALUE;
246  }
247  cc2420_set_channel(value);
248  return RADIO_RESULT_OK;
249  case RADIO_PARAM_RX_MODE:
250  if(value & ~(RADIO_RX_MODE_ADDRESS_FILTER |
251  RADIO_RX_MODE_AUTOACK | RADIO_RX_MODE_POLL_MODE)) {
252  return RADIO_RESULT_INVALID_VALUE;
253  }
254  set_frame_filtering((value & RADIO_RX_MODE_ADDRESS_FILTER) != 0);
255  set_auto_ack((value & RADIO_RX_MODE_AUTOACK) != 0);
256  set_poll_mode((value & RADIO_RX_MODE_POLL_MODE) != 0);
257  return RADIO_RESULT_OK;
258  case RADIO_PARAM_TX_MODE:
259  if(value & ~(RADIO_TX_MODE_SEND_ON_CCA)) {
260  return RADIO_RESULT_INVALID_VALUE;
261  }
262  set_send_on_cca((value & RADIO_TX_MODE_SEND_ON_CCA) != 0);
263  return RADIO_RESULT_OK;
264  case RADIO_PARAM_TXPOWER:
265  if(value < OUTPUT_POWER_MIN || value > OUTPUT_POWER_MAX) {
266  return RADIO_RESULT_INVALID_VALUE;
267  }
268  /* Find the closest higher PA_LEVEL for the desired output power */
269  for(i = 1; i < OUTPUT_NUM; i++) {
270  if(value > output_power[i].power) {
271  break;
272  }
273  }
274  cc2420_set_txpower(output_power[i - 1].config);
275  return RADIO_RESULT_OK;
276  case RADIO_PARAM_CCA_THRESHOLD:
277  cc2420_set_cca_threshold(value - RSSI_OFFSET);
278  return RADIO_RESULT_OK;
279  default:
280  return RADIO_RESULT_NOT_SUPPORTED;
281  }
282 }

334 /*---------------------------------------------------------------------------*/
335 /* Reads a register */
336 static uint16_t
337 getreg(enum cc2420_register regname)
338 {
339  uint16_t value;
340 
341  CC2420_SPI_ENABLE();
342  SPI_WRITE(regname | 0x40);
343  value = (uint8_t)SPI_RXBUF;
344  SPI_TXBUF = 0;
345  SPI_WAITFOREORx();
346  value = SPI_RXBUF << 8;
347  SPI_TXBUF = 0;
348  SPI_WAITFOREORx();
349  value |= SPI_RXBUF;
350  CC2420_SPI_DISABLE();
351 
352  return value;
353 }
354 /*---------------------------------------------------------------------------*/
355 /**
356  * Writes to a register.
357  * Note: the SPI_WRITE(0) seems to be needed for getting the
358  * write reg working on the Z1 / MSP430X platform
359  */
360 static void
361 setreg(enum cc2420_register regname, uint16_t value)
362 {
363  CC2420_SPI_ENABLE();
364  SPI_WRITE_FAST(regname);
365  SPI_WRITE_FAST((uint8_t) (value >> 8));
366  SPI_WRITE_FAST((uint8_t) (value & 0xff));
367  SPI_WAITFORTx_ENDED();
368  SPI_WRITE(0);
369  CC2420_SPI_DISABLE();
370 }
371 /*---------------------------------------------------------------------------*/
372 static void
373 read_ram(uint8_t *buffer, uint16_t adr, uint16_t count)
374 {
375  uint8_t i;
376 
377  CC2420_SPI_ENABLE();
378  SPI_WRITE(0x80 | ((adr) & 0x7f));
379  SPI_WRITE((((adr) >> 1) & 0xc0) | 0x20);
380  SPI_RXBUF;
381  for(i = 0; i < count; i++) {
382  SPI_READ(((uint8_t*) buffer)[i]);
383  }
384  CC2420_SPI_DISABLE();
385 }
386 /*---------------------------------------------------------------------------*/
387 /* Write to RAM in the CC2420 */
388 static void
389 write_ram(const uint8_t *buffer,
390  uint16_t adr,
391  uint16_t count,
392  enum write_ram_order order)
393 {
394  uint8_t i;
395 
396  CC2420_SPI_ENABLE();
397  SPI_WRITE_FAST(0x80 | (adr & 0x7f));
398  SPI_WRITE_FAST((adr >> 1) & 0xc0);
399  if(order == WRITE_RAM_IN_ORDER) {
400  for(i = 0; i < count; i++) {
401  SPI_WRITE_FAST((buffer)[i]);
402  }
403  } else {
404  for(i = count; i > 0; i--) {
405  SPI_WRITE_FAST((buffer)[i - 1]);
406  }
407  }
408  SPI_WAITFORTx_ENDED();
409  CC2420_SPI_DISABLE();
410 }

478 wait_for_transmission(void)
479 {
480  rtimer_clock_t t0;
481  t0 = RTIMER_NOW();
482  while((get_status() & BV(CC2420_TX_ACTIVE))
483  && RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + (RTIMER_SECOND / 10)));

487 on(void)
488 {
489  if(!poll_mode) {
490  CC2420_ENABLE_FIFOP_INT();

498 /*---------------------------------------------------------------------------*/
499 static void
500 off(void)
501 {
502  /* PRINTF("off\n");*/
503  receive_on = 0;
504 
505  /* Wait for transmission to end before turning radio off. */
506  wait_for_transmission();
507 
508  ENERGEST_OFF(ENERGEST_TYPE_LISTEN);
509  strobe(CC2420_SRFOFF);
510  if(!poll_mode) {
511  CC2420_DISABLE_FIFOP_INT();
512  }
513 
514  if(!CC2420_FIFOP_IS_1) {
515  flushrx();
516  }
517 }
518 /*---------------------------------------------------------------------------*/
519 static uint8_t locked, lock_on, lock_off;
520 #define GET_LOCK() locked++
521 static void RELEASE_LOCK(void) {
522  if(locked == 1) {
523  if(lock_on) {
524  on();
525  lock_on = 0;
526  }
527  if(lock_off) {
528  off();
529  lock_off = 0;
530  }
531  }
532  locked--;
533 }

585 /*---------------------------------------------------------------------------*/
586 int
587 cc2420_init(void)
588 {
589  uint16_t reg;
590  {
591  int s = splhigh();
592  cc2420_arch_init(); /* Initalize ports and SPI. */
593  CC2420_DISABLE_FIFOP_INT();
594  CC2420_FIFOP_INT_INIT();
595  splx(s);
596  }
597 
598  /* Turn on voltage regulator and reset. */
599  SET_VREG_ACTIVE();
600  clock_delay(250);
601  SET_RESET_ACTIVE();
602  clock_delay(127);
603  SET_RESET_INACTIVE();
604  clock_delay(125);
605 
606 
607  /* Turn on the crystal oscillator. */
608  strobe(CC2420_SXOSCON);
609  /* And wait until it stabilizes */
610  wait_for_status(BV(CC2420_XOSC16M_STABLE));
611 
612  /* Set auto-ack and frame filtering */
613  set_auto_ack(CC2420_CONF_AUTOACK);
614  set_frame_filtering(CC2420_CONF_AUTOACK);
615 
616  /* Enabling CRC in hardware; this is required by AUTOACK anyway
617  and provides us with RSSI and link quality indication (LQI)
618  information. */
619  reg = getreg(CC2420_MDMCTRL0);
620  reg |= AUTOCRC;
621  setreg(CC2420_MDMCTRL0, reg);
622 
623  /* Set transmission turnaround time to the lower setting (8 symbols
624  = 0.128 ms) instead of the default (12 symbols = 0.192 ms). */
625  /* reg = getreg(CC2420_TXCTRL);
626  reg &= ~(1 << 13);
627  setreg(CC2420_TXCTRL, reg);*/
628 
629 
630  /* Change default values as recomended in the data sheet, */
631  /* correlation threshold = 20, RX bandpass filter = 1.3uA. */
632  setreg(CC2420_MDMCTRL1, CORR_THR(20));
633  reg = getreg(CC2420_RXCTRL1);
634  reg |= RXBPF_LOCUR;
635  setreg(CC2420_RXCTRL1, reg);
636 
637  /* Set the FIFOP threshold to maximum. */
638  setreg(CC2420_IOCFG0, FIFOP_THR(127));
639 
640  init_security();
641 
642  cc2420_set_pan_addr(0xffff, 0x0000, NULL);
643  cc2420_set_channel(CC2420_CONF_CHANNEL);
644  cc2420_set_cca_threshold(CC2420_CONF_CCA_THRESH);
645 
646  flushrx();
647 
648  set_poll_mode(0);
649 
650  process_start(&cc2420_process, NULL);
651  return 1;
652 }

654 static int
655 cc2420_transmit(unsigned short payload_len)
656 {
657  int i;
658 
659  GET_LOCK();
660 
661  /* The TX FIFO can only hold one packet. Make sure to not overrun
662  * FIFO by waiting for transmission to start here and synchronizing
663  * with the CC2420_TX_ACTIVE check in cc2420_send.
664  *
665  * Note that we may have to wait up to 320 us (20 symbols) before
666  * transmission starts.
667  */
668 #ifndef CC2420_CONF_SYMBOL_LOOP_COUNT
669 #error CC2420_CONF_SYMBOL_LOOP_COUNT needs to be set!!!
670 #else
671 #define LOOP_20_SYMBOLS CC2420_CONF_SYMBOL_LOOP_COUNT
672 #endif
673 
674  if(send_on_cca) {
675  strobe(CC2420_SRXON);
676  wait_for_status(BV(CC2420_RSSI_VALID));
677  strobe(CC2420_STXONCCA);
678  } else {
679  strobe(CC2420_STXON);
680  }
681  for(i = LOOP_20_SYMBOLS; i > 0; i--) {
682  if(CC2420_SFD_IS_1) {
683  if(!(get_status() & BV(CC2420_TX_ACTIVE))) {
684  /* SFD went high but we are not transmitting. This means that
685  we just started receiving a packet, so we drop the
686  transmission. */
687  RELEASE_LOCK();
688  return RADIO_TX_COLLISION;
689  }
690  if(receive_on) {
691  ENERGEST_OFF(ENERGEST_TYPE_LISTEN);
692  }
693  ENERGEST_ON(ENERGEST_TYPE_TRANSMIT);
694  /* We wait until transmission has ended so that we get an
695  accurate measurement of the transmission time.*/
696  wait_for_transmission();
697 
698 #ifdef ENERGEST_CONF_LEVELDEVICE_LEVELS
699  ENERGEST_OFF_LEVEL(ENERGEST_TYPE_TRANSMIT,cc2420_get_txpower());
700 #endif
701  ENERGEST_OFF(ENERGEST_TYPE_TRANSMIT);
702  if(receive_on) {
703  ENERGEST_ON(ENERGEST_TYPE_LISTEN);
704  } else {
705  /* We need to explicitly turn off the radio,
706  * since STXON[CCA] -> TX_ACTIVE -> RX_ACTIVE */
707  off();
708  }
709 
710  RELEASE_LOCK();
711  return RADIO_TX_OK;
712  }
713  }
714 
715  /* If we send with cca (cca_on_send), we get here if the packet wasn't
716  transmitted because of other channel activity. */
717  PRINTF("cc2420: do_send() transmission never started\n");
718 
719  RELEASE_LOCK();
720  return RADIO_TX_COLLISION;
721 }

757  if(receive_on == 0) {
758  return 1;
759  }

888 static int
889 cc2420_read(void *buf, unsigned short bufsize)
890 {
891  uint8_t footer[FOOTER_LEN];
892  uint8_t len;
893 
894  if(!CC2420_FIFOP_IS_1) {
895  return 0;
896  }
897 
898  GET_LOCK();
899 
900  getrxdata(&len, 1);
901 
902  if(len > CC2420_MAX_PACKET_LEN) {
903  /* Oops, we must be out of sync. */
904  } else if(len <= FOOTER_LEN) {
905  /* Packet too short */
906  } else if(len - FOOTER_LEN > bufsize) {
907  /* Packet too long */
908  } else {
909  getrxdata((uint8_t *) buf, len - FOOTER_LEN);
910  getrxdata(footer, FOOTER_LEN);
911 
912  if(footer[1] & FOOTER1_CRC_OK) {
913  cc2420_last_rssi = footer[0] + RSSI_OFFSET;
914  cc2420_last_correlation = footer[1] & FOOTER1_CORRELATION;
915  if(!poll_mode) {
916  /* Not in poll mode: packetbuf should not be accessed in interrupt context.
917  * In poll mode, the last packet RSSI and link quality can be obtained through
918  * RADIO_PARAM_LAST_RSSI and RADIO_PARAM_LAST_LINK_QUALITY */
919  packetbuf_set_attr(PACKETBUF_ATTR_RSSI, cc2420_last_rssi);
920  packetbuf_set_attr(PACKETBUF_ATTR_LINK_QUALITY, cc2420_last_correlation);
921  }
922  } else {
923  len = FOOTER_LEN;
924  }
925 
926  if(!poll_mode) {
927  if(CC2420_FIFOP_IS_1) {
928  if(!CC2420_FIFO_IS_1) {
929  /* Clean up in case of FIFO overflow! This happens for every
930  * full length frame and is signaled by FIFOP = 1 and FIFO =
931  * 0. */
932  flushrx();
933  } else {
934  /* Another packet has been received and needs attention. */
935  process_poll(&cc2420_process);
936  }
937  }
938  }
939 
940  RELEASE_LOCK();
941  return len - FOOTER_LEN;
942  }
943 
944  flushrx();
945  RELEASE_LOCK();
946  return 0;
947 }

996 cc2420_cca(void)
997 {
998  int cca;
999  int radio_was_off = 0;
1000 
1001  /* If the radio is locked by an underlying thread (because we are
1002  being invoked through an interrupt), we preted that the coast is
1003  clear (i.e., no packet is currently being transmitted by a
1004  neighbor). */
1005  if(locked) {
1006  return 1;
1007  }
1008 
1009  GET_LOCK();
1010  if(!receive_on) {
1011  radio_was_off = 1;
1012  cc2420_on();
1013  }
1014 
1015  /* Make sure that the radio really got turned on. */
1016  if(!receive_on) {
1017  RELEASE_LOCK();
1018  if(radio_was_off) {
1019  cc2420_off();
1020  }
1021  return 1;
1022  }
1023 
1024  wait_for_status(BV(CC2420_RSSI_VALID));
1025 
1026  cca = CC2420_CCA_IS_1;
1027 
1028  if(radio_was_off) {
1029  cc2420_off();
1030  }
1031  RELEASE_LOCK();
1032  return cca;
