112 /*------------------------------- DIO -------------------------------------- */
113 /*---------------------------------------------------------------------------*/
114 static void
115 new_dio_interval(void)
116 {
117  uint32_t time;
118  clock_time_t ticks;
119 
120  time = 1UL << curr_instance.dag.dio_intcurrent;
121 
122  /* Convert from milliseconds to CLOCK_TICKS. */
123  ticks = (time * CLOCK_SECOND) / 1000;
124  curr_instance.dag.dio_next_delay = ticks;
125 
126  /* random number between I/2 and I */
127  ticks = ticks / 2 + (ticks / 2 * (uint32_t)random_rand()) / RANDOM_RAND_MAX;
128 
129  /*
130  * The intervals must be equally long among the nodes for Trickle to
131  * operate efficiently. Therefore we need to calculate the delay between
132  * the randomized time and the start time of the next interval.
133  */
134  curr_instance.dag.dio_next_delay -= ticks;
135  curr_instance.dag.dio_send = 1;
136  /* reset the redundancy counter */
137  curr_instance.dag.dio_counter = 0;
138 
139  /* schedule the timer */
140  ctimer_set(&curr_instance.dag.dio_timer, ticks, &handle_dio_timer, NULL);
141 
142 #ifdef RPL_CALLBACK_NEW_DIO_INTERVAL
143  RPL_CALLBACK_NEW_DIO_INTERVAL((CLOCK_SECOND * 1UL << curr_instance.dag.dio_intcurrent) / 1000);
144 #endif /* RPL_CALLBACK_NEW_DIO_INTERVAL */
145 }

148 rpl_timers_dio_reset(const char *str)
149 {
150  if(rpl_dag_ready_to_advertise()) {
151  LOG_INFO("reset DIO timer (%s)\n", str);
152 #if !RPL_LEAF_ONLY
153  curr_instance.dag.dio_counter = 0;

155  new_dio_interval();
156 #endif /* RPL_LEAF_ONLY */
157  }

159 /*---------------------------------------------------------------------------*/
160 static void
161 handle_dio_timer(void *ptr)
162 {
163  if(!rpl_dag_ready_to_advertise()) {
164  return; /* We will be scheduled again later */
165  }
166 
167  if(curr_instance.dag.dio_send) {
168  /* send DIO if counter is less than desired redundancy, or if dio_redundancy
169  is set to 0, or if we are the root */
170  if(rpl_dag_root_is_root() || curr_instance.dio_redundancy == 0 ||
171  curr_instance.dag.dio_counter < curr_instance.dio_redundancy) {
172 #if RPL_TRICKLE_REFRESH_DAO_ROUTES
173  if(rpl_dag_root_is_root()) {
174  static int count = 0;
175  if((count++ % RPL_TRICKLE_REFRESH_DAO_ROUTES) == 0) {
176  /* Request new DAO to refresh route. */
177  RPL_LOLLIPOP_INCREMENT(curr_instance.dtsn_out);
178  LOG_INFO("trigger DAO updates with a DTSN increment (%u)\n", curr_instance.dtsn_out);
179  }
180  }
181 #endif /* RPL_TRICKLE_REFRESH_DAO_ROUTES */
182  curr_instance.dag.last_advertised_rank = curr_instance.dag.rank;
183  rpl_icmp6_dio_output(NULL);
184  }
185  curr_instance.dag.dio_send = 0;
186  ctimer_set(&curr_instance.dag.dio_timer, curr_instance.dag.dio_next_delay, handle_dio_timer, NULL);
187  } else {
188  /* check if we need to double interval */
189  if(curr_instance.dag.dio_intcurrent < curr_instance.dio_intmin + curr_instance.dio_intdoubl) {
190  curr_instance.dag.dio_intcurrent++;
191  }
192  new_dio_interval();
193  }
194 }

229 /*---------------------------------------------------------------------------*/
230 static void
231 schedule_dao_refresh(void)
232 {
233  if(curr_instance.used && curr_instance.default_lifetime != RPL_INFINITE_LIFETIME) {
234 #if RPL_WITH_DAO_ACK
235  /* DAO-ACK enabled: the last DAO was ACKed, wait until expiration before refresh */
236  clock_time_t target_refresh = CLOCK_SECOND * RPL_LIFETIME(curr_instance.default_lifetime);
237 #else /* RPL_WITH_DAO_ACK */
238  /* DAO-ACK disabled: use half the expiration time to get two chances to refresh per lifetime */
239  clock_time_t target_refresh = (CLOCK_SECOND * RPL_LIFETIME(curr_instance.default_lifetime) / 2);
240 #endif /* RPL_WITH_DAO_ACK */
241 
242  /* Send between 60 and 120 seconds before target refresh */
243  clock_time_t safety_margin = (60 * CLOCK_SECOND) + (random_rand() % (60 * CLOCK_SECOND));
244 
245  if(target_refresh > safety_margin) {
246  target_refresh -= safety_margin;
247  }
248 
249  /* Increment next sequno */
250  RPL_LOLLIPOP_INCREMENT(curr_instance.dag.dao_curr_seqno);
251  ctimer_set(&curr_instance.dag.dao_timer, target_refresh, handle_dao_timer, NULL);
252  }
253 }
254 /*---------------------------------------------------------------------------*/
255 void
256 rpl_timers_schedule_dao(void)
257 {
258  if(curr_instance.used && curr_instance.mop != RPL_MOP_NO_DOWNWARD_ROUTES) {
259  /* No need for aggregation delay as per RFC 6550 section 9.5, as this only
260  * serves storing mode. Use simply delay instead, with the only PURPOSE
261  * to reduce congestion. */
262  clock_time_t expiration_time = RPL_DAO_DELAY / 2 + (random_rand() % (RPL_DAO_DELAY));
263  /* Increment next seqno */
264  RPL_LOLLIPOP_INCREMENT(curr_instance.dag.dao_curr_seqno);
265  ctimer_set(&curr_instance.dag.dao_timer, expiration_time, handle_dao_timer, NULL);
266  }
267 }
268 /*---------------------------------------------------------------------------*/
269 static void
270 handle_dao_timer(void *ptr)
271 {
272 #if RPL_WITH_DAO_ACK
273  if(rpl_lollipop_greater_than(curr_instance.dag.dao_curr_seqno,
274  curr_instance.dag.dao_last_seqno)) {
275  /* We are sending a new DAO here. Prepare retransmissions */
276  curr_instance.dag.dao_transmissions = 0;
277  } else {
278  /* We are called for the same DAO again */
279  if(curr_instance.dag.dao_last_acked_seqno == curr_instance.dag.dao_last_seqno) {
280  /* The last seqno sent is ACKed! Schedule refresh to avoid route expiration */
281  schedule_dao_refresh();
282  return;
283  }
284  /* We need to re-send the last DAO */
285  if(curr_instance.dag.dao_transmissions >= RPL_DAO_MAX_RETRANSMISSIONS) {
286  /* No more retransmissions. Perform local repair and hope to find another . */
287  rpl_local_repair("DAO max rtx");
288  return;
289  }
290  }
291  /* Increment transmission counter before sending */
292  curr_instance.dag.dao_transmissions++;
293  /* Schedule next retransmission */
294  schedule_dao_retransmission();
295 #else /* RPL_WITH_DAO_ACK */
296  /* No DAO-ACK: assume we are reachable as soon as we send a DAO */
297  if(curr_instance.dag.state == DAG_JOINED) {
298  curr_instance.dag.state = DAG_REACHABLE;
299  }
300  rpl_timers_dio_reset("Reachable");
301 #endif /* !RPL_WITH_DAO_ACK */
302 
303  curr_instance.dag.dao_last_seqno = curr_instance.dag.dao_curr_seqno;
304  /* Send a DAO with own prefix as target and default lifetime */
305  rpl_icmp6_dao_output(curr_instance.default_lifetime);
306 
307 #if !RPL_WITH_DAO_ACK
308  /* There is DAO-ACK, schedule a refresh. Must be done after rpl_icmp6_dao_output,
309  because we increment curr_instance.dag.dao_curr_seqno for the next DAO (refresh) */
310  schedule_dao_refresh();
311 #endif /* !RPL_WITH_DAO_ACK */
312 }

339 clock_time_t
340 get_probing_delay(void)
341 {
342  if(curr_instance.used && curr_instance.dag.urgent_probing_target != NULL) {
343  /* Urgent probing needed (to find out if a neighbor may become preferred parent) */
344  return random_rand() % (CLOCK_SECOND * 4);
345  } else {
346  /* Else, use normal probing interval */
347  return ((RPL_PROBING_INTERVAL) / 2) + random_rand() % (RPL_PROBING_INTERVAL);
348  }
349 }
350 /*---------------------------------------------------------------------------*/
351 rpl_nbr_t *
352 get_probing_target(void)
353 {
354  /* Returns the next probing target. The current implementation probes the urgent
355  * probing target if any, or the preferred parent if its link statistics need refresh.
356  * Otherwise, it picks at random between:
357  * (1) selecting the best neighbor with non-fresh link statistics
358  * (2) selecting the least recently updated neighbor
359  */
360 
361  rpl_nbr_t *nbr;
362  rpl_nbr_t *probing_target = NULL;
363  rpl_rank_t probing_target_rank = RPL_INFINITE_RANK;
364  clock_time_t probing_target_age = 0;
365  clock_time_t clock_now = clock_time();
366 
367  if(curr_instance.used == 0) {
368  return NULL;
369  }
370 
371  /* There is an urgent probing target */
372  if(curr_instance.dag.urgent_probing_target != NULL) {
373  return curr_instance.dag.urgent_probing_target;
374  }
375 
376  /* The preferred parent needs probing */
377  if(curr_instance.dag.preferred_parent != NULL && !rpl_neighbor_is_fresh(curr_instance.dag.preferred_parent)) {
378  return curr_instance.dag.preferred_parent;
379  }
380 
381  /* Now consider probing other non-fresh neighbors. With 2/3 proabability,
382  pick the best non-fresh. Otherwise, pick the lest recently updated non-fresh. */
383 
384  if(random_rand() % 3 != 0) {
385  /* Look for best non-fresh */
386  nbr = nbr_table_head(rpl_neighbors);
387  while(nbr != NULL) {
388  if(!rpl_neighbor_is_fresh(nbr)) {
389  /* nbr needs probing */
390  rpl_rank_t nbr_rank = rpl_neighbor_rank_via_nbr(nbr);
391  if(probing_target == NULL
392  || nbr_rank < probing_target_rank) {
393  probing_target = nbr;
394  probing_target_rank = nbr_rank;
395  }
396  }
397  nbr = nbr_table_next(rpl_neighbors, nbr);
398  }
399  } else {
400  /* Look for least recently updated non-fresh */
401  nbr = nbr_table_head(rpl_neighbors);
402  while(nbr != NULL) {
403  if(!rpl_neighbor_is_fresh(nbr)) {
404  /* nbr needs probing */
405  const struct link_stats *stats = rpl_neighbor_get_link_stats(nbr);
406  if(stats != NULL) {
407  if(probing_target == NULL
408  || clock_now - stats->last_tx_time > probing_target_age) {
409  probing_target = nbr;
410  probing_target_age = clock_now - stats->last_tx_time;
411  }
412  }
413  }
414  nbr = nbr_table_next(rpl_neighbors, nbr);
415  }
416  }
417 
418  return probing_target;
419 }

421 static void
422 handle_probing_timer(void *ptr)
423 {
424  rpl_nbr_t *probing_target = RPL_PROBING_SELECT_FUNC();
425  uip_ipaddr_t *target_ipaddr = rpl_neighbor_get_ipaddr(probing_target);
426 
427  /* Perform probing */
428  if(target_ipaddr != NULL) {
429  const struct link_stats *stats = rpl_neighbor_get_link_stats(probing_target);
430  (void)stats;
431  LOG_INFO("probing ");
432  LOG_INFO_6ADDR(target_ipaddr);
433  LOG_INFO_(" %s last tx %u min ago\n",
434  curr_instance.dag.urgent_probing_target != NULL ? "(urgent)" : "",
435  probing_target != NULL ?
436  (unsigned)((clock_time() - stats->last_tx_time) / (60 * CLOCK_SECOND)) : 0
437  );
438  /* Send probe, e.g. unicast DIO or DIS */
439  RPL_PROBING_SEND_FUNC(target_ipaddr);
440  curr_instance.dag.urgent_probing_target = NULL;
441  } else {
442  LOG_INFO("no neighbor needs probing\n");
443  }
444 
445  /* Schedule next probing */
446  rpl_schedule_probing();
447 }

518 #endif /* LOG_INFO_ENABLED */

521 }
