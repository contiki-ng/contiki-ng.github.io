136 /**
137  * @brief The IEEE 802.15.4 frame has a number of constant/fixed fields that
138  * can be counted to make frame construction and max payload
139  * calculations easier.
140  *
141  * These include:
142  * 1. FCF - 2 bytes - Fixed
143  * 2. Sequence number - 1 byte - Fixed
144  * 3. Addressing fields - 4 - 20 bytes - Variable
145  * 4. Aux security header - 0 - 14 bytes - Variable
146  * 5. CRC - 2 bytes - Fixed
147  */
148 
149 /**
150  * \brief Defines the bitfields of the frame control field (FCF).
151  */
152 typedef struct {
153  uint8_t frame_type; /**< 3 bit. Frame type field, see 802.15.4 */
154  uint8_t security_enabled; /**< 1 bit. True if security is used in this frame */
155  uint8_t frame_pending; /**< 1 bit. True if sender has more data to send */
156  uint8_t ack_required; /**< 1 bit. Is an ack frame required? */
157  uint8_t panid_compression; /**< 1 bit. Is this a compressed header? */
158  /* uint8_t reserved; */ /**< 1 bit. Unused bit */
159  uint8_t sequence_number_suppression; /**< 1 bit. Does the header omit sequence number?, see 802.15.4e */
160  uint8_t ie_list_present; /**< 1 bit. Does the header contain Information Elements?, see 802.15.4e */
161  uint8_t dest_addr_mode; /**< 2 bit. Destination address mode, see 802.15.4 */
162  uint8_t frame_version; /**< 2 bit. 802.15.4 frame version */
163  uint8_t src_addr_mode; /**< 2 bit. Source address mode, see 802.15.4 */
164 } frame802154_fcf_t;

166 /** \brief 802.15.4 security control bitfield. See section 7.6.2.2.1 in 802.15.4 specification */
167 typedef struct {

194 /** \brief Parameters used by the frame802154_create() function. These
195  * parameters are used in the 802.15.4 frame header. See the 802.15.4
196  * specification for details.

199  /* The fields dest_addr and src_addr must come first to ensure they are aligned to the
200  * CPU word size. Needed as they are accessed directly as linkaddr_t*. Note we cannot use
201  * the type linkaddr_t directly here, as we always need 8 bytes, not LINKADDR_SIZE bytes. */
