131 /* UIP_CONF_ND6_SEND_NS enables standard IPv6 Neighbor Discovery Protocol
132  (RFC 4861). We enable it by default when IPv6 is used without RPL.
133  With RPL, the neighbor cache (link-local IPv6 <-> MAC address mapping)
134  is fed whenever receiving DIO. This is often sufficient
135  for RPL routing, i.e. to send to the preferred parent or any child.
136  Link-local unicast to other neighbors may, however, not be possible if
137  we never receive any DIO from them. This may happen if the link from the
138  neighbor to us is weak, if DIO transmissions are suppressed (Trickle
139  timer) or if the neighbor chooses not to transmit DIOs because it is
140  a leaf node or for any reason. */
141 #ifndef UIP_CONF_ND6_SEND_NS
142 #define UIP_CONF_ND6_SEND_NS (NETSTACK_CONF_WITH_IPV6 && !UIP_CONF_IPV6_RPL)
143 #endif /* UIP_CONF_ND6_SEND_NS */
144 /* To speed up the neighbor cache construction,
145  enable UIP_CONF_ND6_AUTOFILL_NBR_CACHE. When a node does not the link-layer
146  address of a neighbor, it will infer it from the link-local IPv6, assuming
147  the node used autoconfiguration. Note that RPL uses its own freshness
148  mechanism to select whether neighbors are still usable as a parent
149  or not, regardless of the neighbor cache. Note that this is not
150  standard-compliant (RFC 4861), as neighbors will be added regardless of
151  their reachability and liveness. */
152 #ifndef UIP_CONF_ND6_AUTOFILL_NBR_CACHE
153 #define UIP_CONF_ND6_AUTOFILL_NBR_CACHE (!UIP_CONF_ND6_SEND_NS)
154 #endif /* UIP_CONF_ND6_AUTOFILL_NBR_CACHE */
155 /* UIP_CONF_ND6_SEND_NA allows to still comply with NDP even if the host does
156  not perform NUD or DAD processes. By default it is activated so the host
157  can still communicate with a full NDP peer. */
158 #ifndef UIP_CONF_ND6_SEND_NA
159 #define UIP_CONF_ND6_SEND_NA (NETSTACK_CONF_WITH_IPV6)
160 #endif /* UIP_CONF_ND6_SEND_NS */
