66 int
67 rpl_ext_header_srh_get_next_hop(uip_ipaddr_t *ipaddr)
68 {
69  uint8_t *uip_next_hdr;
70  int last_uip_ext_len = uip_ext_len;
71  rpl_ns_node_t *dest_node;
72  rpl_ns_node_t *root_node;
73 
74  uip_ext_len = 0;
75  uip_next_hdr = &UIP_IP_BUF->proto;
76 
77  /* Look for routing header */
78  while(uip_next_hdr != NULL && *uip_next_hdr != UIP_PROTO_ROUTING) {
79  switch(*uip_next_hdr) {
80  case UIP_PROTO_HBHO:
81  case UIP_PROTO_DESTO:
82  /*
83  * As per RFC 2460, only the Hop-by-Hop Options header and
84  * Destination Options header can appear before the Routing header.
85  */
86  uip_next_hdr = &UIP_EXT_BUF->next;
87  uip_ext_len += (UIP_EXT_BUF->len << 3) + 8;
88  break;
89  default:
90  uip_next_hdr = NULL;
91  break;
92  }
93  }
94 
95  if(!rpl_is_addr_in_our_dag(&UIP_IP_BUF->destipaddr)) {
96  return 0;
97  }
98 
99  root_node = rpl_ns_get_node(&curr_instance.dag.dag_id);
100  dest_node = rpl_ns_get_node(&UIP_IP_BUF->destipaddr);
101 
102  if((uip_next_hdr != NULL && *uip_next_hdr == UIP_PROTO_ROUTING
103  && UIP_RH_BUF->routing_type == RPL_RH_TYPE_SRH) ||
104  (dest_node != NULL && root_node != NULL &&
105  dest_node->parent == root_node)) {
106  /* Routing header found or the packet destined for a direct child of the root.
107  * The next hop should be already copied as the IPv6 destination
108  * address, via rpl_ext_header_srh_update. We turn this address into a link-local to enable
109  * forwarding to next hop */
110  uip_ipaddr_copy(ipaddr, &UIP_IP_BUF->destipaddr);
111  uip_create_linklocal_prefix(ipaddr);
112  uip_ext_len = last_uip_ext_len;
113  return 1;
114  }
115 
116  LOG_DBG("no SRH found\n");
117  uip_ext_len = last_uip_ext_len;
118  return 0;
119 }

121 int
122 rpl_ext_header_srh_update(void)
123 {
124  uint8_t *uip_next_hdr;
125  int last_uip_ext_len = uip_ext_len;
126  uint8_t cmpri, cmpre;
127  uint8_t ext_len;
128  uint8_t padding;
129  uint8_t path_len;
130  uint8_t segments_left;
131  uip_ipaddr_t current_dest_addr;
132 
133  uip_ext_len = 0;
134  uip_next_hdr = &UIP_IP_BUF->proto;
135 
136  /* Look for routing header */
137  while(uip_next_hdr != NULL && *uip_next_hdr != UIP_PROTO_ROUTING) {
138  switch(*uip_next_hdr) {
139  case UIP_PROTO_HBHO:
140  case UIP_PROTO_DESTO:
141  /*
142  * As per RFC 2460, only the Hop-by-Hop Options header and
143  * Destination Options header can appear before the Routing header.
144  */
145  uip_next_hdr = &UIP_EXT_BUF->next;
146  uip_ext_len += (UIP_EXT_BUF->len << 3) + 8;
147  break;
148  default:
149  uip_next_hdr = NULL;
150  break;
151  }
152  }
153 
154  if(uip_next_hdr == NULL || *uip_next_hdr != UIP_PROTO_ROUTING
155  || UIP_RH_BUF->routing_type != RPL_RH_TYPE_SRH) {
156  LOG_INFO("SRH not found\n");
157  uip_ext_len = last_uip_ext_len;
158  return 0;
159  }
160 
161  /* Parse SRH */
162  segments_left = UIP_RH_BUF->seg_left;
163  ext_len = (UIP_RH_BUF->len * 8) + 8;
164  cmpri = UIP_RPL_SRH_BUF->cmpr >> 4;
165  cmpre = UIP_RPL_SRH_BUF->cmpr & 0x0f;
166  padding = UIP_RPL_SRH_BUF->pad >> 4;
167  path_len = ((ext_len - padding - RPL_RH_LEN - RPL_SRH_LEN - (16 - cmpre)) / (16 - cmpri)) + 1;
168  (void)path_len;
169 
170  LOG_INFO("read SRH, path len %u, segments left %u, Cmpri %u, Cmpre %u, ext len %u (padding %u)\n",
171  path_len, segments_left, cmpri, cmpre, ext_len, padding);
172 
173  /* Update SRH in-place */
174  if(segments_left == 0) {
175  /* We are the final destination, do nothing */
176  } else {
177  uint8_t i = path_len - segments_left; /* The index of the next address to be visited */
178  uint8_t *addr_ptr = ((uint8_t *)UIP_RH_BUF) + RPL_RH_LEN + RPL_SRH_LEN + (i * (16 - cmpri));
179  uint8_t cmpr = segments_left == 1 ? cmpre : cmpri;
180 
181  /* As per RFC6554: swap the IPv6 destination address with address[i] */
182 
183  /* First, copy the current IPv6 destination address */
184  uip_ipaddr_copy(&current_dest_addr, &UIP_IP_BUF->destipaddr);
185  /* Second, update the IPv6 destination address with addresses[i] */
186  memcpy(((uint8_t *)&UIP_IP_BUF->destipaddr) + cmpr, addr_ptr, 16 - cmpr);
187  /* Third, write current_dest_addr to addresses[i] */
188  memcpy(addr_ptr, ((uint8_t *)&current_dest_addr) + cmpr, 16 - cmpr);
189 
190  /* Update segments left field */
191  UIP_RH_BUF->seg_left--;
192 
193  LOG_INFO("SRH next hop ");
194  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
195  LOG_INFO_("\n");
196  }
197 
198  uip_ext_len = last_uip_ext_len;
199  return 1;
200 }

202 /* Utility function for SRH. Counts the number of bytes in common between
203  * two addresses at p1 and p2. */
204 static int
205 count_matching_bytes(const void *p1, const void *p2, size_t n)
206 {
207  int i = 0;
208  for(i = 0; i < n; i++) {
209  if(((uint8_t *)p1)[i] != ((uint8_t *)p2)[i]) {
210  return i;
211  }
212  }
213  return n;
214 }
215 /*---------------------------------------------------------------------------*/
216 /* Used by rpl_ext_header_update to insert a RPL SRH extension header. This
217  * is used at the root, to initiate downward routing. Returns 1 on success,
218  * 0 on failure.
219 */
220 static int
221 insert_srh_header(void)
222 {
223  /* Implementation of RFC6554 */
224  uint8_t temp_len;
225  uint8_t path_len;
226  uint8_t ext_len;
227  uint8_t cmpri, cmpre; /* ComprI and ComprE fields of the RPL Source Routing Header */
228  uint8_t *hop_ptr;
229  uint8_t padding;
230  rpl_ns_node_t *dest_node;
231  rpl_ns_node_t *root_node;
232  rpl_ns_node_t *node;
233  uip_ipaddr_t node_addr;
234 
235  LOG_INFO("SRH creating source routing header with destination ");
236  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
237  LOG_INFO_(" \n");
238 
239  /* Construct source route. We do not do this recursively to keep the runtime stack usage constant. */
240 
241  /* Get link of the destination and root */
242 
243  if(!rpl_is_addr_in_our_dag(&UIP_IP_BUF->destipaddr)) {
244  LOG_ERR("SRH destination not in our DAG\n");
245  return 0;
246  }
247 
248  dest_node = rpl_ns_get_node(&UIP_IP_BUF->destipaddr);
249  if(dest_node == NULL) {
250  LOG_ERR("SRH node not found\n");
251  return 0;
252  }
253 
254  root_node = rpl_ns_get_node(&curr_instance.dag.dag_id);
255  if(root_node == NULL) {
256  LOG_ERR("SRH root node not found\n");
257  return 0;
258  }
259 
260  if(!rpl_ns_is_addr_reachable(&UIP_IP_BUF->destipaddr)) {
261  LOG_ERR("SRH no path found to destination\n");
262  return 0;
263  }
264 
265  /* Compute path length and compression factors (we use cmpri == cmpre) */
266  path_len = 0;
267  node = dest_node->parent;
268  /* For simplicity, we use cmpri = cmpre */
269  cmpri = 15;
270  cmpre = 15;
271 
272  /* Note that in case of a direct child (node == root_node), we insert
273  SRH anyway, as RFC 6553 mandates that routed datagrams must include
274  SRH or the RPL option (or both) */
275 
276  while(node != NULL && node != root_node) {
277 
278  rpl_ns_get_node_global_addr(&node_addr, node);
279 
280  /* How many bytes in common between all nodes in the path? */
281  cmpri = MIN(cmpri, count_matching_bytes(&node_addr, &UIP_IP_BUF->destipaddr, 16));
282  cmpre = cmpri;
283 
284  LOG_INFO("SRH Hop ");
285  LOG_INFO_6ADDR(&node_addr);
286  LOG_INFO_("\n");
287  node = node->parent;
288  path_len++;
289  }
290 
291  /* Extension header length: fixed headers + (n-1) * (16-ComprI) + (16-ComprE)*/
292  ext_len = RPL_RH_LEN + RPL_SRH_LEN
293  + (path_len - 1) * (16 - cmpre)
294  + (16 - cmpri);
295 
296  padding = ext_len % 8 == 0 ? 0 : (8 - (ext_len % 8));
297  ext_len += padding;
298 
299  LOG_INFO("SRH path len: %u, ComprI %u, ComprE %u, ext len %u (padding %u)\n",
300  path_len, cmpri, cmpre, ext_len, padding);
301 
302  /* Check if there is enough space to store the extension header */
303  if(uip_len + ext_len > UIP_BUFSIZE) {
304  LOG_ERR("packet too long: impossible to add source routing header (%u bytes)\n", ext_len);
305  return 1;
306  }
307 
308  /* Move existing ext headers and payload uip_ext_len further */
309  memmove(uip_buf + uip_l2_l3_hdr_len + ext_len,
310  uip_buf + uip_l2_l3_hdr_len, uip_len - UIP_IPH_LEN);
311  memset(uip_buf + uip_l2_l3_hdr_len, 0, ext_len);
312 
313  /* Insert source routing header */
314  UIP_RH_BUF->next = UIP_IP_BUF->proto;
315  UIP_IP_BUF->proto = UIP_PROTO_ROUTING;
316 
317  /* Initialize IPv6 Routing Header */
318  UIP_RH_BUF->len = (ext_len - 8) / 8;
319  UIP_RH_BUF->routing_type = RPL_RH_TYPE_SRH;
320  UIP_RH_BUF->seg_left = path_len;
321 
322  /* Initialize RPL Source Routing Header */
323  UIP_RPL_SRH_BUF->cmpr = (cmpri << 4) + cmpre;
324  UIP_RPL_SRH_BUF->pad = padding << 4;
325 
326  /* Initialize addresses field (the actual source route).
327  * From last to first. */
328  node = dest_node;
329  hop_ptr = ((uint8_t *)UIP_RH_BUF) + ext_len - padding; /* Pointer where to write the next hop compressed address */
330 
331  while(node != NULL && node->parent != root_node) {
332  rpl_ns_get_node_global_addr(&node_addr, node);
333 
334  hop_ptr -= (16 - cmpri);
335  memcpy(hop_ptr, ((uint8_t*)&node_addr) + cmpri, 16 - cmpri);
336 
337  node = node->parent;
338  }
339 
340  /* The next hop (i.e. node whose parent is the root) is placed as the current IPv6 destination */
341  rpl_ns_get_node_global_addr(&node_addr, node);
342  uip_ipaddr_copy(&UIP_IP_BUF->destipaddr, &node_addr);
343 
344  /* In-place update of IPv6 length field */
345  temp_len = UIP_IP_BUF->len[1];
346  UIP_IP_BUF->len[1] += ext_len;
347  if(UIP_IP_BUF->len[1] < temp_len) {
348  UIP_IP_BUF->len[0]++;
349  }
350 
351  uip_ext_len += ext_len;
352  uip_len += ext_len;
353 
354  return 1;
355 }

357 int
358 rpl_ext_header_hbh_update(int uip_ext_opt_offset)
359 {
360  int down;
361  int rank_error_signaled;
362  int loop_detected;
363  uint16_t sender_rank;
364  uint8_t sender_closer;
365  rpl_nbr_t *sender;
366  uint8_t opt_type = UIP_EXT_HDR_OPT_RPL_BUF->opt_type;
367  uint8_t opt_len = UIP_EXT_HDR_OPT_RPL_BUF->opt_len;
368 
369  if(UIP_HBHO_BUF->len != ((RPL_HOP_BY_HOP_LEN - 8) / 8)
370  || opt_type != UIP_EXT_HDR_OPT_RPL
371  || opt_len != RPL_HDR_OPT_LEN) {
372  LOG_ERR("hop-by-hop extension header has wrong size or type (%u %u %u)\n",
373  UIP_HBHO_BUF->len, opt_type, opt_len);
374  return 0; /* Drop */
375  }
376 
377  if(!curr_instance.used || curr_instance.instance_id != UIP_EXT_HDR_OPT_RPL_BUF->instance) {
378  LOG_ERR("unknown instance: %u\n",
379  UIP_EXT_HDR_OPT_RPL_BUF->instance);
380  return 0; /* Drop */
381  }
382 
383  if(UIP_EXT_HDR_OPT_RPL_BUF->flags & RPL_HDR_OPT_FWD_ERR) {
384  LOG_ERR("forward error!\n");
385  return 0; /* Drop */
386  }
387 
388  down = (UIP_EXT_HDR_OPT_RPL_BUF->flags & RPL_HDR_OPT_DOWN) ? 1 : 0;
389  sender_rank = UIP_HTONS(UIP_EXT_HDR_OPT_RPL_BUF->senderrank);
390  sender = nbr_table_get_from_lladdr(rpl_neighbors, packetbuf_addr(PACKETBUF_ADDR_SENDER));
391  rank_error_signaled = (UIP_EXT_HDR_OPT_RPL_BUF->flags & RPL_HDR_OPT_RANK_ERR) ? 1 : 0;
392  sender_closer = sender_rank < curr_instance.dag.rank;
393  loop_detected = (down && !sender_closer) || (!down && sender_closer);
394 
395  LOG_INFO("ext hdr: packet from ");
396  LOG_INFO_6ADDR(&UIP_IP_BUF->srcipaddr);
397  LOG_INFO_(" to ");
398  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
399  LOG_INFO_(" going %s, sender closer %d (%d < %d), rank error %u, loop detected %u\n",
400  down == 1 ? "down" : "up", sender_closer, sender_rank,
401  curr_instance.dag.rank, rank_error_signaled, loop_detected);
402 
403  if(loop_detected) {
404  /* Set forward error flag */
405  UIP_EXT_HDR_OPT_RPL_BUF->flags |= RPL_HDR_OPT_RANK_ERR;
406  }
407 
408  return rpl_process_hbh(sender, sender_rank, loop_detected, rank_error_signaled);
409 }

411 /* In-place update of the RPL HBH extension header, when already present
412  * in the uIP packet. Used by insert_hbh_header and rpl_ext_header_update.
413  * Returns 1 on success, 0 on failure. */
414 static int
415 update_hbh_header(void)
416 {
417  int uip_ext_opt_offset;
418  int last_uip_ext_len;
419 
420  last_uip_ext_len = uip_ext_len;
421  uip_ext_len = 0;
422  uip_ext_opt_offset = 2;
423 
424  if(UIP_IP_BUF->proto == UIP_PROTO_HBHO && UIP_EXT_HDR_OPT_RPL_BUF->opt_type == UIP_EXT_HDR_OPT_RPL) {
425  if(UIP_HBHO_BUF->len != ((RPL_HOP_BY_HOP_LEN - 8) / 8)
426  || UIP_EXT_HDR_OPT_RPL_BUF->opt_len != RPL_HDR_OPT_LEN) {
427 
428  LOG_ERR("hop-by-hop extension header has wrong size (%u %u)\n",
429  UIP_EXT_HDR_OPT_RPL_BUF->opt_len, uip_ext_len);
430  return 0; /* Drop */
431  }
432 
433  if(!curr_instance.used || curr_instance.instance_id != UIP_EXT_HDR_OPT_RPL_BUF->instance) {
434  LOG_ERR("unable to add/update hop-by-hop extension header: incorrect instance\n");
435  uip_ext_len = last_uip_ext_len;
436  return 0; /* Drop */
437  }
438 
439  /* Update sender rank and instance, will update flags next */
440  UIP_EXT_HDR_OPT_RPL_BUF->senderrank = UIP_HTONS(curr_instance.dag.rank);
441  UIP_EXT_HDR_OPT_RPL_BUF->instance = curr_instance.instance_id;
442  }
443 
444  uip_ext_len = last_uip_ext_len;
445  return 1;
446 }
447 /*---------------------------------------------------------------------------*/
448 /* Used by rpl_ext_header_update on packets without an HBH extension header,
449  * for packets initated by non-root nodes.
450  * Inserts and initalizes (via update_hbh_header) a RPL HBH ext header.
451  * Returns 1 on success, 0 on failure. */
452 static int
453 insert_hbh_header(void)
454 {
455  int uip_ext_opt_offset;
456  int last_uip_ext_len;
457  uint8_t temp_len;
458 
459  last_uip_ext_len = uip_ext_len;
460  uip_ext_len = 0;
461  uip_ext_opt_offset = 2;
462 
463  /* Insert hop-by-hop header */
464  LOG_INFO("creating hop-by-hop option\n");
465  if(uip_len + RPL_HOP_BY_HOP_LEN > UIP_BUFSIZE) {
466  LOG_ERR("packet too long: impossible to add hop-by-hop option\n");
467  uip_ext_len = last_uip_ext_len;
468  return 0;
469  }
470 
471  /* Move existing ext headers and payload UIP_EXT_BUF further */
472  memmove(UIP_HBHO_NEXT_BUF, UIP_EXT_BUF, uip_len - UIP_IPH_LEN);
473  memset(UIP_HBHO_BUF, 0, RPL_HOP_BY_HOP_LEN);
474 
475  /* Update IP and HBH protocol and fields */
476  UIP_HBHO_BUF->next = UIP_IP_BUF->proto;
477  UIP_IP_BUF->proto = UIP_PROTO_HBHO;
478 
479  /* Initialize HBH option */
480  UIP_HBHO_BUF->len = (RPL_HOP_BY_HOP_LEN - 8) / 8;
481  UIP_EXT_HDR_OPT_RPL_BUF->opt_type = UIP_EXT_HDR_OPT_RPL;
482  UIP_EXT_HDR_OPT_RPL_BUF->opt_len = RPL_HDR_OPT_LEN;
483  UIP_EXT_HDR_OPT_RPL_BUF->flags = 0;
484  UIP_EXT_HDR_OPT_RPL_BUF->senderrank = UIP_HTONS(curr_instance.dag.rank);
485  UIP_EXT_HDR_OPT_RPL_BUF->instance = curr_instance.instance_id;
486  uip_len += RPL_HOP_BY_HOP_LEN;
487  temp_len = UIP_IP_BUF->len[1];
488  UIP_IP_BUF->len[1] += RPL_HOP_BY_HOP_LEN;
489  if(UIP_IP_BUF->len[1] < temp_len) {
490  UIP_IP_BUF->len[0]++;
491  }
492 
493  uip_ext_len = last_uip_ext_len + RPL_HOP_BY_HOP_LEN;
494 
495  /* Update header before returning */
496  return update_hbh_header();
497 }
498 /*---------------------------------------------------------------------------*/
499 int
500 rpl_ext_header_update(void)
501 {
502  if(!curr_instance.used
503  || uip_is_addr_linklocal(&UIP_IP_BUF->destipaddr)
504  || uip_is_addr_mcast(&UIP_IP_BUF->destipaddr)) {
505  return 1;
506  }
507 
508  if(rpl_dag_root_is_root()) {
509  /* At the root, remove headers if any, and insert SRH or HBH
510  * (SRH is inserted only if the destination is in the DODAG) */
511  rpl_ext_header_remove();
512  if(rpl_is_addr_in_our_dag(&UIP_IP_BUF->destipaddr)) {
513  /* dest is in a DODAG; the packet is going down. */
514  if(curr_instance.mop != RPL_MOP_NO_DOWNWARD_ROUTES) {
515  return insert_srh_header();
516  } else {
517  LOG_ERR("packet going down at root, but no support for downward routing\n");
518  return 0; /* No support for downward routes */
519  }
520  } else {
521  /* dest is outside of DODAGs; no ext header is needed. */
522  return 1;
523  }
524  } else {
525  if(uip_ds6_is_my_addr(&UIP_IP_BUF->srcipaddr)
526  && UIP_IP_BUF->ttl == uip_ds6_if.cur_hop_limit) {
527  /* Insert HBH option at source. Checking the address is not sufficient because
528  * in non-storing mode, a packet may go up and then down the same path again */
529  return insert_hbh_header();
530  } else {
531  /* Update HBH option at forwarders */
532  return update_hbh_header();
533  }
534  }
535 }

537 void
538 rpl_ext_header_remove(void)
539 {
540  uint8_t temp_len;
541  uint8_t rpl_ext_hdr_len;
542  uint8_t *uip_next_hdr;
543 
544  uip_ext_len = 0;
545  uip_next_hdr = &UIP_IP_BUF->proto;
546 
547  /* Look for hop-by-hop and routing headers */
548  while(uip_next_hdr != NULL) {
549  switch(*uip_next_hdr) {
550  case UIP_PROTO_HBHO:
551  case UIP_PROTO_ROUTING:
552  /* Remove hop-by-hop and routing headers */
553  *uip_next_hdr = UIP_EXT_BUF->next;
554  rpl_ext_hdr_len = (UIP_EXT_BUF->len * 8) + 8;
555  temp_len = UIP_IP_BUF->len[1];
556  uip_len -= rpl_ext_hdr_len;
557  UIP_IP_BUF->len[1] -= rpl_ext_hdr_len;
558  if(UIP_IP_BUF->len[1] > temp_len) {
559  UIP_IP_BUF->len[0]--;
560  }
561  LOG_INFO("removing RPL extension header (type %u, len %u)\n", *uip_next_hdr, rpl_ext_hdr_len);
562  memmove(UIP_EXT_BUF, ((uint8_t *)UIP_EXT_BUF) + rpl_ext_hdr_len, uip_len - UIP_IPH_LEN);
563  break;
564  case UIP_PROTO_DESTO:
565  /*
566  * As per RFC 2460, any header other than the Destination
567  * Options header does not appear between the Hop-by-Hop
568  * Options header and the Routing header.
569  *
570  * We're moving to the next header only if uip_next_hdr has
571  * UIP_PROTO_DESTO. Otherwise, we'll return.
572  */
573  /* Move to next header */
574  uip_next_hdr = &UIP_EXT_BUF->next;
575  uip_ext_len += (UIP_EXT_BUF->len << 3) + 8;
576  default:
577  return;
578  }
579  }
