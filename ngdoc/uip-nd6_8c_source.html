156  /**
157  * Neighbor Solicitation Processing
158  *
159  * The NS can be received in 3 cases (procedures):
160  * - sender is performing DAD (ip src = unspecified, no SLLAO option)
161  * - sender is performing NUD (ip dst = unicast)
162  * - sender is performing address resolution (ip dest = solicited node mcast
163  * address)
164  *
165  * We do:
166  * - if the tgt belongs to me, reply, otherwise ignore
167  * - if i was performing DAD for the same address, two cases:
168  * -- I already sent a NS, hence I win
169  * -- I did not send a NS yet, hence I lose
170  *
171  * If we need to send a NA in response (i.e. the NS was done for NUD, or
172  * address resolution, or DAD and there is a conflict), we do it in this
173  * function: set src, dst, tgt address in the three cases, then for all cases
174  * set the rest, including SLLAO
175  *
176  */
177 #if UIP_ND6_SEND_NA
178 static void
179 ns_input(void)
180 {
181  uint8_t flags;
182  LOG_INFO("Received NS from ");
183  LOG_INFO_6ADDR(&UIP_IP_BUF->srcipaddr);
184  LOG_INFO_(" to ");
185  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
186  LOG_INFO_(" with target address ");
187  LOG_INFO_6ADDR((uip_ipaddr_t *) (&UIP_ND6_NS_BUF->tgtipaddr));
188  LOG_INFO_("\n");
189  UIP_STAT(++uip_stat.nd6.recv);
190 
191 #if UIP_CONF_IPV6_CHECKS
192  if((UIP_IP_BUF->ttl != UIP_ND6_HOP_LIMIT) ||
193  (uip_is_addr_mcast(&UIP_ND6_NS_BUF->tgtipaddr)) ||
194  (UIP_ICMP_BUF->icode != 0)) {
195  LOG_ERR("NS received is bad\n");
196  goto discard;
197  }
198 #endif /* UIP_CONF_IPV6_CHECKS */
199 
200  /* Options processing */
201  nd6_opt_llao = NULL;
202  nd6_opt_offset = UIP_ND6_NS_LEN;
203  while(uip_l3_icmp_hdr_len + nd6_opt_offset < uip_len) {
204 #if UIP_CONF_IPV6_CHECKS
205  if(UIP_ND6_OPT_HDR_BUF->len == 0) {
206  LOG_ERR("NS received is bad\n");
207  goto discard;
208  }
209 #endif /* UIP_CONF_IPV6_CHECKS */
210  switch (UIP_ND6_OPT_HDR_BUF->type) {
211  case UIP_ND6_OPT_SLLAO:
212  nd6_opt_llao = &uip_buf[uip_l2_l3_icmp_hdr_len + nd6_opt_offset];
213 #if UIP_CONF_IPV6_CHECKS
214  /* There must be NO option in a DAD NS */
215  if(uip_is_addr_unspecified(&UIP_IP_BUF->srcipaddr)) {
216  LOG_ERR("NS received is bad\n");
217  goto discard;
218  } else {
219 #endif /*UIP_CONF_IPV6_CHECKS */
220  uip_lladdr_t lladdr_aligned;
221  extract_lladdr_from_llao_aligned(&lladdr_aligned);
222  nbr = uip_ds6_nbr_lookup(&UIP_IP_BUF->srcipaddr);
223  if(nbr == NULL) {
224  uip_ds6_nbr_add(&UIP_IP_BUF->srcipaddr, &lladdr_aligned,
225  0, NBR_STALE, NBR_TABLE_REASON_IPV6_ND, NULL);
226  } else {
227  const uip_lladdr_t *lladdr = uip_ds6_nbr_get_ll(nbr);
228  if(lladdr == NULL) {
229  goto discard;
230  }
231  if(memcmp(&nd6_opt_llao[UIP_ND6_OPT_DATA_OFFSET],
232  lladdr, UIP_LLADDR_LEN) != 0) {
233  if(nbr_table_update_lladdr((const linkaddr_t *)lladdr, (const linkaddr_t *)&lladdr_aligned, 1) == 0) {
234  /* failed to update the lladdr */
235  goto discard;
236  }
237  nbr->state = NBR_STALE;
238  } else {
239  if(nbr->state == NBR_INCOMPLETE) {
240  nbr->state = NBR_STALE;
241  }
242  }
243  }
244 #if UIP_CONF_IPV6_CHECKS
245  }
246 #endif /*UIP_CONF_IPV6_CHECKS */
247  break;
248  default:
249  LOG_WARN("ND option not supported in NS");
250  break;
251  }
252  nd6_opt_offset += (UIP_ND6_OPT_HDR_BUF->len << 3);
253  }
254 
255  addr = uip_ds6_addr_lookup(&UIP_ND6_NS_BUF->tgtipaddr);
256  if(addr != NULL) {
257  if(uip_is_addr_unspecified(&UIP_IP_BUF->srcipaddr)) {
258  /* DAD CASE */
259 #if UIP_ND6_DEF_MAXDADNS > 0
260 #if UIP_CONF_IPV6_CHECKS
261  if(!uip_is_addr_solicited_node(&UIP_IP_BUF->destipaddr)) {
262  LOG_ERR("NS received is bad\n");
263  goto discard;
264  }
265 #endif /* UIP_CONF_IPV6_CHECKS */
266  if(addr->state != ADDR_TENTATIVE) {
267  uip_create_linklocal_allnodes_mcast(&UIP_IP_BUF->destipaddr);
268  uip_ds6_select_src(&UIP_IP_BUF->srcipaddr, &UIP_IP_BUF->destipaddr);
269  flags = UIP_ND6_NA_FLAG_OVERRIDE;
270  goto create_na;
271  } else {
272  /** \todo if I sent a NS before him, I win */
273  uip_ds6_dad_failed(addr);
274  goto discard;
275  }
276 #else /* UIP_ND6_DEF_MAXDADNS > 0 */
277  goto discard; /* DAD CASE */
278 #endif /* UIP_ND6_DEF_MAXDADNS > 0 */
279  }
280 #if UIP_CONF_IPV6_CHECKS
281  if(uip_ds6_is_my_addr(&UIP_IP_BUF->srcipaddr)) {
282  /**
283  * \NOTE do we do something here? we both are using the same address.
284  * If we are doing dad, we could cancel it, though we should receive a
285  * NA in response of DAD NS we sent, hence DAD will fail anyway. If we
286  * were not doing DAD, it means there is a duplicate in the network!
287  */
288  LOG_ERR("NS received is bad\n");
289  goto discard;
290  }
291 #endif /*UIP_CONF_IPV6_CHECKS */
292 
293  /* Address resolution case */
294  if(uip_is_addr_solicited_node(&UIP_IP_BUF->destipaddr)) {
295  uip_ipaddr_copy(&UIP_IP_BUF->destipaddr, &UIP_IP_BUF->srcipaddr);
296  uip_ipaddr_copy(&UIP_IP_BUF->srcipaddr, &UIP_ND6_NS_BUF->tgtipaddr);
297  flags = UIP_ND6_NA_FLAG_SOLICITED | UIP_ND6_NA_FLAG_OVERRIDE;
298  goto create_na;
299  }
300 
301  /* NUD CASE */
302  if(uip_ds6_addr_lookup(&UIP_IP_BUF->destipaddr) == addr) {
303  uip_ipaddr_copy(&UIP_IP_BUF->destipaddr, &UIP_IP_BUF->srcipaddr);
304  uip_ipaddr_copy(&UIP_IP_BUF->srcipaddr, &UIP_ND6_NS_BUF->tgtipaddr);
305  flags = UIP_ND6_NA_FLAG_SOLICITED | UIP_ND6_NA_FLAG_OVERRIDE;
306  goto create_na;
307  } else {
308 #if UIP_CONF_IPV6_CHECKS
309  LOG_ERR("NS received is bad\n");
310  goto discard;
311 #endif /* UIP_CONF_IPV6_CHECKS */
312  }
313  } else {
314  goto discard;
315  }
316 
317 
318 create_na:
319  /* If the node is a router it should set R flag in NAs */
320 #if UIP_CONF_ROUTER
321  flags = flags | UIP_ND6_NA_FLAG_ROUTER;
322 #endif
323  uip_ext_len = 0;
324  UIP_IP_BUF->vtc = 0x60;
325  UIP_IP_BUF->tcflow = 0;
326  UIP_IP_BUF->flow = 0;
327  UIP_IP_BUF->len[0] = 0; /* length will not be more than 255 */
328  UIP_IP_BUF->len[1] = UIP_ICMPH_LEN + UIP_ND6_NA_LEN + UIP_ND6_OPT_LLAO_LEN;
329  UIP_IP_BUF->proto = UIP_PROTO_ICMP6;
330  UIP_IP_BUF->ttl = UIP_ND6_HOP_LIMIT;
331 
332  UIP_ICMP_BUF->type = ICMP6_NA;
333  UIP_ICMP_BUF->icode = 0;
334 
335  UIP_ND6_NA_BUF->flagsreserved = flags;
336  memcpy(&UIP_ND6_NA_BUF->tgtipaddr, &addr->ipaddr, sizeof(uip_ipaddr_t));
337 
338  create_llao(&uip_buf[uip_l2_l3_icmp_hdr_len + UIP_ND6_NA_LEN],
339  UIP_ND6_OPT_TLLAO);
340 
341  UIP_ICMP_BUF->icmpchksum = 0;
342  UIP_ICMP_BUF->icmpchksum = ~uip_icmp6chksum();
343 
344  uip_len =
345  UIP_IPH_LEN + UIP_ICMPH_LEN + UIP_ND6_NA_LEN + UIP_ND6_OPT_LLAO_LEN;
346 
347  UIP_STAT(++uip_stat.nd6.sent);
348  LOG_INFO("Sending NA to ");
349  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
350  LOG_INFO_(" from ");
351  LOG_INFO_6ADDR(&UIP_IP_BUF->srcipaddr);
352  LOG_INFO_(" with target address ");
353  LOG_INFO_6ADDR(&UIP_ND6_NA_BUF->tgtipaddr);
354  LOG_INFO_("\n");
355  return;
356 
357 discard:
358  uip_clear_buf();
359  return;
360 }
361 #endif /* UIP_ND6_SEND_NA */

366 void
367 uip_nd6_ns_output(uip_ipaddr_t * src, uip_ipaddr_t * dest, uip_ipaddr_t * tgt)
368 {
369  uip_ext_len = 0;
370  UIP_IP_BUF->vtc = 0x60;
371  UIP_IP_BUF->tcflow = 0;
372  UIP_IP_BUF->flow = 0;
373  UIP_IP_BUF->proto = UIP_PROTO_ICMP6;
374  UIP_IP_BUF->ttl = UIP_ND6_HOP_LIMIT;
375 
376  if(dest == NULL) {
377  uip_create_solicited_node(tgt, &UIP_IP_BUF->destipaddr);
378  } else {
379  uip_ipaddr_copy(&UIP_IP_BUF->destipaddr, dest);
380  }
381  UIP_ICMP_BUF->type = ICMP6_NS;
382  UIP_ICMP_BUF->icode = 0;
383  UIP_ND6_NS_BUF->reserved = 0;
384  uip_ipaddr_copy((uip_ipaddr_t *) &UIP_ND6_NS_BUF->tgtipaddr, tgt);
385  UIP_IP_BUF->len[0] = 0; /* length will not be more than 255 */
386  /*
387  * check if we add a SLLAO option: for DAD, MUST NOT, for NUD, MAY
388  * (here yes), for Address resolution , MUST
389  */
390  if(!(uip_ds6_is_my_addr(tgt))) {
391  if(src != NULL) {
392  uip_ipaddr_copy(&UIP_IP_BUF->srcipaddr, src);
393  } else {
394  uip_ds6_select_src(&UIP_IP_BUF->srcipaddr, &UIP_IP_BUF->destipaddr);
395  }
396  if (uip_is_addr_unspecified(&UIP_IP_BUF->srcipaddr)) {
397  LOG_ERR("Dropping NS due to no suitable source address\n");
398  uip_clear_buf();
399  return;
400  }
401  UIP_IP_BUF->len[1] =
402  UIP_ICMPH_LEN + UIP_ND6_NS_LEN + UIP_ND6_OPT_LLAO_LEN;
403 
404  create_llao(&uip_buf[uip_l2_l3_icmp_hdr_len + UIP_ND6_NS_LEN],
405  UIP_ND6_OPT_SLLAO);
406 
407  uip_len =
408  UIP_IPH_LEN + UIP_ICMPH_LEN + UIP_ND6_NS_LEN + UIP_ND6_OPT_LLAO_LEN;
409  } else {
410  uip_create_unspecified(&UIP_IP_BUF->srcipaddr);
411  UIP_IP_BUF->len[1] = UIP_ICMPH_LEN + UIP_ND6_NS_LEN;
412  uip_len = UIP_IPH_LEN + UIP_ICMPH_LEN + UIP_ND6_NS_LEN;
413  }
414 
415  UIP_ICMP_BUF->icmpchksum = 0;
416  UIP_ICMP_BUF->icmpchksum = ~uip_icmp6chksum();
417 
418  UIP_STAT(++uip_stat.nd6.sent);
419  LOG_INFO("Sending NS to ");
420  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
421  LOG_INFO_(" from ");
422  LOG_INFO_6ADDR(&UIP_IP_BUF->srcipaddr);
423  LOG_INFO_(" with target address ");
424  LOG_INFO_6ADDR(tgt);
425  LOG_INFO_("\n");
426  return;
427 }
428 #endif /* UIP_ND6_SEND_NS */

430 #if UIP_ND6_SEND_NS
431 /*------------------------------------------------------------------*/
432 /**
433  * Neighbor Advertisement Processing
434  *
435  * we might have to send a pkt that had been buffered while address
436  * resolution was performed (if we support buffering, see UIP_CONF_QUEUE_PKT)
437  *
438  * As per RFC 4861, on link layer that have addresses, TLLAO options MUST be
439  * included when responding to multicast solicitations, SHOULD be included in
440  * response to unicast (here we assume it is for now)
441  *
442  * NA can be received after sending NS for DAD, Address resolution or NUD. Can
443  * be unsolicited as well.
444  * It can trigger update of the state of the neighbor in the neighbor cache,
445  * router in the router list.
446  * If the NS was for DAD, it means DAD failed
447  *
448  */
449 static void
450 na_input(void)
451 {
452  uint8_t is_llchange;
453  uint8_t is_router;
454  uint8_t is_solicited;
455  uint8_t is_override;
456  uip_lladdr_t lladdr_aligned;
457 
458  LOG_INFO("Received NA from ");
459  LOG_INFO_6ADDR(&UIP_IP_BUF->srcipaddr);
460  LOG_INFO_(" to ");
461  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
462  LOG_INFO_(" with target address ");
463  LOG_INFO_6ADDR((uip_ipaddr_t *) (&UIP_ND6_NA_BUF->tgtipaddr));
464  LOG_INFO_("\n");
465  UIP_STAT(++uip_stat.nd6.recv);
466 
467  /*
468  * booleans. the three last one are not 0 or 1 but 0 or 0x80, 0x40, 0x20
469  * but it works. Be careful though, do not use tests such as is_router == 1
470  */
471  is_llchange = 0;
472  is_router = ((UIP_ND6_NA_BUF->flagsreserved & UIP_ND6_NA_FLAG_ROUTER));
473  is_solicited =
474  ((UIP_ND6_NA_BUF->flagsreserved & UIP_ND6_NA_FLAG_SOLICITED));
475  is_override =
476  ((UIP_ND6_NA_BUF->flagsreserved & UIP_ND6_NA_FLAG_OVERRIDE));
477 
478 #if UIP_CONF_IPV6_CHECKS
479  if((UIP_IP_BUF->ttl != UIP_ND6_HOP_LIMIT) ||
480  (UIP_ICMP_BUF->icode != 0) ||
481  (uip_is_addr_mcast(&UIP_ND6_NA_BUF->tgtipaddr)) ||
482  (is_solicited && uip_is_addr_mcast(&UIP_IP_BUF->destipaddr))) {
483  LOG_ERR("NA received is bad\n");
484  goto discard;
485  }
486 #endif /*UIP_CONF_IPV6_CHECKS */
487 
488  /* Options processing: we handle TLLAO, and must ignore others */
489  nd6_opt_offset = UIP_ND6_NA_LEN;
490  nd6_opt_llao = NULL;
491  while(uip_l3_icmp_hdr_len + nd6_opt_offset < uip_len) {

852 /**
853  * Process a Router Advertisement
854  *
855  * - Possible actions when receiving a RA: add router to router list,
856  * recalculate reachable time, update link hop limit, update retrans timer.
857  * - If MTU option: update MTU.
858  * - If SLLAO option: update entry in neighbor cache
859  * - If prefix option: start autoconf, add prefix to prefix list
860  */
861 void
862 ra_input(void)
863 {
864  uip_lladdr_t lladdr_aligned;
865 
866  LOG_INFO("Received RA from ");
867  LOG_INFO_6ADDR(&UIP_IP_BUF->srcipaddr);
868  LOG_INFO_(" to ");
869  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
870  LOG_INFO_("\n");
871  UIP_STAT(++uip_stat.nd6.recv);
872 
873 #if UIP_CONF_IPV6_CHECKS
874  if((UIP_IP_BUF->ttl != UIP_ND6_HOP_LIMIT) ||
875  (!uip_is_addr_linklocal(&UIP_IP_BUF->srcipaddr)) ||
876  (UIP_ICMP_BUF->icode != 0)) {
877  LOG_ERR("RA received is bad");
878  goto discard;
879  }
880 #endif /*UIP_CONF_IPV6_CHECKS */
881 
882  if(UIP_ND6_RA_BUF->cur_ttl != 0) {
883  uip_ds6_if.cur_hop_limit = UIP_ND6_RA_BUF->cur_ttl;
884  LOG_INFO("uip_ds6_if.cur_hop_limit %u\n", uip_ds6_if.cur_hop_limit);
885  }
886 
887  if(UIP_ND6_RA_BUF->reachable_time != 0) {
888  if(uip_ds6_if.base_reachable_time !=
889  uip_ntohl(UIP_ND6_RA_BUF->reachable_time)) {
890  uip_ds6_if.base_reachable_time = uip_ntohl(UIP_ND6_RA_BUF->reachable_time);
891  uip_ds6_if.reachable_time = uip_ds6_compute_reachable_time();
892  }
893  }
894  if(UIP_ND6_RA_BUF->retrans_timer != 0) {
895  uip_ds6_if.retrans_timer = uip_ntohl(UIP_ND6_RA_BUF->retrans_timer);
896  }
897 
898  /* Options processing */
899  nd6_opt_offset = UIP_ND6_RA_LEN;
900  while(uip_l3_icmp_hdr_len + nd6_opt_offset < uip_len) {
901  if(UIP_ND6_OPT_HDR_BUF->len == 0) {
902  LOG_ERR("RA received is bad");
903  goto discard;
904  }
905  switch (UIP_ND6_OPT_HDR_BUF->type) {
906  case UIP_ND6_OPT_SLLAO:
907  LOG_DBG("Processing SLLAO option in RA\n");
908  nd6_opt_llao = (uint8_t *) UIP_ND6_OPT_HDR_BUF;
909  nbr = uip_ds6_nbr_lookup(&UIP_IP_BUF->srcipaddr);
910  if(!extract_lladdr_from_llao_aligned(&lladdr_aligned)) {
911  /* failed to extract llao - discard packet */
912  goto discard;
913  }
914  if(nbr == NULL) {
915  nbr = uip_ds6_nbr_add(&UIP_IP_BUF->srcipaddr, &lladdr_aligned,
916  1, NBR_STALE, NBR_TABLE_REASON_IPV6_ND, NULL);
917  } else {
918  const uip_lladdr_t *lladdr = uip_ds6_nbr_get_ll(nbr);
919  if(lladdr == NULL) {
920  goto discard;
921  }
922  if(nbr->state == NBR_INCOMPLETE) {
923  nbr->state = NBR_STALE;
924  }
925  if(memcmp(&nd6_opt_llao[UIP_ND6_OPT_DATA_OFFSET],
926  lladdr, UIP_LLADDR_LEN) != 0) {
927  /* change of link layer address */
928  if(nbr_table_update_lladdr((const linkaddr_t *)lladdr, (const linkaddr_t *)&lladdr_aligned, 1) == 0) {
929  /* failed to update the lladdr */
930  goto discard;
931  }
932  nbr->state = NBR_STALE;
933  }
934  nbr->isrouter = 1;
935  }
936  break;
937  case UIP_ND6_OPT_MTU:
938  LOG_DBG("Processing MTU option in RA\n");
939  uip_ds6_if.link_mtu =
940  uip_ntohl(((uip_nd6_opt_mtu *) UIP_ND6_OPT_HDR_BUF)->mtu);
941  break;
942  case UIP_ND6_OPT_PREFIX_INFO:
943  LOG_DBG("Processing PREFIX option in RA\n");
944  nd6_opt_prefix_info = (uip_nd6_opt_prefix_info *) UIP_ND6_OPT_HDR_BUF;
945  if((uip_ntohl(nd6_opt_prefix_info->validlt) >=
946  uip_ntohl(nd6_opt_prefix_info->preferredlt))
947  && (!uip_is_addr_linklocal(&nd6_opt_prefix_info->prefix))) {
948  /* on-link flag related processing */
949  if(nd6_opt_prefix_info->flagsreserved1 & UIP_ND6_RA_FLAG_ONLINK) {
950  prefix =
951  uip_ds6_prefix_lookup(&nd6_opt_prefix_info->prefix,
952  nd6_opt_prefix_info->preflen);
953  if(prefix == NULL) {
954  if(nd6_opt_prefix_info->validlt != 0) {
955  if(nd6_opt_prefix_info->validlt != UIP_ND6_INFINITE_LIFETIME) {
956  prefix = uip_ds6_prefix_add(&nd6_opt_prefix_info->prefix,
957  nd6_opt_prefix_info->preflen,
958  uip_ntohl(nd6_opt_prefix_info->
959  validlt));
960  } else {
961  prefix = uip_ds6_prefix_add(&nd6_opt_prefix_info->prefix,
962  nd6_opt_prefix_info->preflen, 0);
963  }
964  }
965  } else {
966  switch (nd6_opt_prefix_info->validlt) {
967  case 0:
968  uip_ds6_prefix_rm(prefix);
969  break;
970  case UIP_ND6_INFINITE_LIFETIME:
971  prefix->isinfinite = 1;
972  break;
973  default:
974  LOG_DBG("Updating timer of prefix ");
975  LOG_DBG_6ADDR(&prefix->ipaddr);
976  LOG_DBG_(" new value %lu\n", uip_ntohl(nd6_opt_prefix_info->validlt));
977  stimer_set(&prefix->vlifetime,
978  uip_ntohl(nd6_opt_prefix_info->validlt));
979  prefix->isinfinite = 0;
980  break;
981  }
982  }
983  }
984  /* End of on-link flag related processing */
985  /* autonomous flag related processing */
986  if((nd6_opt_prefix_info->flagsreserved1 & UIP_ND6_RA_FLAG_AUTONOMOUS)
987  && (nd6_opt_prefix_info->validlt != 0)
988  && (nd6_opt_prefix_info->preflen == UIP_DEFAULT_PREFIX_LEN)) {
989
