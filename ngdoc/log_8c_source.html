80 void
81 log_6addr(const uip_ipaddr_t *ipaddr)
82 {
83  uint16_t a;
84  unsigned int i;
85  int f;
86 
87  if(ipaddr == NULL) {
88  LOG_OUTPUT("(NULL IP addr)");
89  return;
90  }
91 
92  if(ip64_addr_is_ipv4_mapped_addr(ipaddr)) {
93  /* Printing IPv4-mapped addresses is done according to RFC 4291 */
94  LOG_OUTPUT("::FFFF:%u.%u.%u.%u", ipaddr->u8[12], ipaddr->u8[13], ipaddr->u8[14], ipaddr->u8[15]);
95  } else {
96  for(i = 0, f = 0; i < sizeof(uip_ipaddr_t); i += 2) {
97  a = (ipaddr->u8[i] << 8) + ipaddr->u8[i + 1];
98  if(a == 0 && f >= 0) {
99  if(f++ == 0) {
100  LOG_OUTPUT("::");
101  }
102  } else {
103  if(f > 0) {
104  f = -1;
105  } else if(i > 0) {
106  LOG_OUTPUT(":");
107  }
108  LOG_OUTPUT("%x", a);
109  }
110  }
111  }
112 }

115 log_6addr_compact(const uip_ipaddr_t *ipaddr)
116 {
117  if(ipaddr == NULL) {
118  LOG_OUTPUT("6A-NULL");
119  } else if(uip_is_addr_mcast(ipaddr)) {
120  LOG_OUTPUT("6M-%04x", UIP_HTONS(ipaddr->u16[sizeof(uip_ipaddr_t)/2-1]));
121  } else if(uip_is_addr_linklocal(ipaddr)) {
122  LOG_OUTPUT("6L-%04x", UIP_HTONS(ipaddr->u16[sizeof(uip_ipaddr_t)/2-1]));
123  } else {
124  LOG_OUTPUT("6G-%04x", UIP_HTONS(ipaddr->u16[sizeof(uip_ipaddr_t)/2-1]));
125  }

158 void
159 log_set_level(const char *module, int level)
160 {
161  if(level >= LOG_LEVEL_NONE && level <= LOG_LEVEL_DBG) {
162  int i = 0;
163  int module_all = !strcmp("all", module);
164  while(all_modules[i].name != NULL) {
165  if(module_all || !strcmp(module, all_modules[i].name)) {
166  *all_modules[i].curr_log_level = MIN(level, all_modules[i].max_log_level);
167  }
168  i++;
169  }
170  }
171 }

184  i++;
185  }
186  return -1;
