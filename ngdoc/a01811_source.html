78 static uip_icmp6_input_handler_t *
79 input_handler_lookup(uint8_t type, uint8_t icode)
80 {
81  uip_icmp6_input_handler_t *handler = NULL;
82 
83  for(handler = list_head(input_handler_list);
84  handler != NULL;
85  handler = list_item_next(handler)) {
86  if(handler->type == type &&
87  (handler->icode == icode ||
88  handler->icode == UIP_ICMP6_HANDLER_CODE_ANY)) {
89  return handler;
90  }
91  }
92 
93  return NULL;
94 }

97 uip_icmp6_input(uint8_t type, uint8_t icode)
98 {
99  uip_icmp6_input_handler_t *handler = input_handler_lookup(type, icode);
100 
101  if(handler == NULL) {
102  return UIP_ICMP6_INPUT_ERROR;
103  }
104 
105  if(handler->handler == NULL) {
106  return UIP_ICMP6_INPUT_ERROR;
107  }
108 
109  handler->handler();
110  return UIP_ICMP6_INPUT_SUCCESS;
111 }

119 static void
120 echo_request_input(void)
121 {
122  /*
123  * we send an echo reply. It is trivial if there was no extension
124  * headers in the request otherwise we need to remove the extension
125  * headers and change a few fields
126  */
127  LOG_INFO("Received Echo Request from ");
128  LOG_INFO_6ADDR(&UIP_IP_BUF->srcipaddr);
129  LOG_INFO_(" to ");
130  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
131  LOG_INFO_("\n");
132 
133  /* IP header */
134  UIP_IP_BUF->ttl = uip_ds6_if.cur_hop_limit;
135 
136  if(uip_is_addr_mcast(&UIP_IP_BUF->destipaddr)){
137  uip_ipaddr_copy(&UIP_IP_BUF->destipaddr, &UIP_IP_BUF->srcipaddr);
138  uip_ds6_select_src(&UIP_IP_BUF->srcipaddr, &UIP_IP_BUF->destipaddr);
139  } else {
140  uip_ipaddr_copy(&tmp_ipaddr, &UIP_IP_BUF->srcipaddr);
141  uip_ipaddr_copy(&UIP_IP_BUF->srcipaddr, &UIP_IP_BUF->destipaddr);
142  uip_ipaddr_copy(&UIP_IP_BUF->destipaddr, &tmp_ipaddr);
143  }
144 
145  if(uip_ext_len > 0) {
146  /* Remove extension headers if any */
147  UIP_IP_BUF->proto = UIP_PROTO_ICMP6;
148  uip_len -= uip_ext_len;
149  UIP_IP_BUF->len[0] = ((uip_len - UIP_IPH_LEN) >> 8);
150  UIP_IP_BUF->len[1] = ((uip_len - UIP_IPH_LEN) & 0xff);
151  /* move the echo request payload (starting after the icmp header)
152  * to the new location in the reply.
153  * The shift is equal to the length of the extension headers present
154  * Note: UIP_ICMP_BUF still points to the echo request at this stage
155  */
156  memmove((uint8_t *)UIP_ICMP_BUF + UIP_ICMPH_LEN - uip_ext_len,
157  (uint8_t *)UIP_ICMP_BUF + UIP_ICMPH_LEN,
158  (uip_len - UIP_IPH_LEN - UIP_ICMPH_LEN));
159  uip_ext_len = 0;
160  }
161 
162  /* Below is important for the correctness of UIP_ICMP_BUF and the
163  * checksum
164  */
165 
166  /* Note: now UIP_ICMP_BUF points to the beginning of the echo reply */
167  UIP_ICMP_BUF->type = ICMP6_ECHO_REPLY;
168  UIP_ICMP_BUF->icode = 0;
169  UIP_ICMP_BUF->icmpchksum = 0;
170  UIP_ICMP_BUF->icmpchksum = ~uip_icmp6chksum();
171 
172  LOG_INFO("Sending Echo Reply to ");
173  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
174  LOG_INFO_(" from ");
175  LOG_INFO_6ADDR(&UIP_IP_BUF->srcipaddr);
176  LOG_INFO_("\n");
177  UIP_STAT(++uip_stat.icmp.sent);
178  return;
179 }

181 void
182 uip_icmp6_error_output(uint8_t type, uint8_t code, uint32_t param) {
183  /* check if originating packet is not an ICMP error */
184  if(uip_ext_len) {
185  if(UIP_EXT_BUF->next == UIP_PROTO_ICMP6 && UIP_ICMP_BUF->type < 128) {
186  uip_clear_buf();
187  return;
188  }
189  } else {
190  if(UIP_IP_BUF->proto == UIP_PROTO_ICMP6 && UIP_ICMP_BUF->type < 128) {
191  uip_clear_buf();
192  return;
193  }
194  }
195 
196 #if UIP_CONF_IPV6_RPL
197  rpl_ext_header_remove();
198 #else
199  uip_ext_len = 0;
200 #endif /* UIP_CONF_IPV6_RPL */
201 
202  /* remember data of original packet before shifting */
203  uip_ipaddr_copy(&tmp_ipaddr, &UIP_IP_BUF->destipaddr);
204 
205  uip_len += UIP_IPICMPH_LEN + UIP_ICMP6_ERROR_LEN;
206 
207  if(uip_len > UIP_LINK_MTU) {
208  uip_len = UIP_LINK_MTU;
209  }
210 
211  memmove((uint8_t *)UIP_ICMP6_ERROR_BUF + uip_ext_len + UIP_ICMP6_ERROR_LEN,
212  (void *)UIP_IP_BUF, uip_len - UIP_IPICMPH_LEN - uip_ext_len - UIP_ICMP6_ERROR_LEN);
213 
214  UIP_IP_BUF->vtc = 0x60;
215  UIP_IP_BUF->tcflow = 0;
216  UIP_IP_BUF->flow = 0;
217  if (uip_ext_len) {
218  UIP_FIRST_EXT_BUF->next = UIP_PROTO_ICMP6;
219  } else {
220  UIP_IP_BUF->proto = UIP_PROTO_ICMP6;
221  }
222  UIP_IP_BUF->ttl = uip_ds6_if.cur_hop_limit;
223 
224  /* the source should not be unspecified nor multicast, the check for
225  multicast is done in uip_process */
226  if(uip_is_addr_unspecified(&UIP_IP_BUF->srcipaddr)){
227  uip_clear_buf();
228  return;
229  }
230 
231  uip_ipaddr_copy(&UIP_IP_BUF->destipaddr, &UIP_IP_BUF->srcipaddr);
232 
233  if(uip_is_addr_mcast(&tmp_ipaddr)){
234  if(type == ICMP6_PARAM_PROB && code == ICMP6_PARAMPROB_OPTION){
235  uip_ds6_select_src(&UIP_IP_BUF->srcipaddr, &tmp_ipaddr);
236  } else {
237  uip_clear_buf();
238  return;
239  }
240  } else {
241 #if UIP_CONF_ROUTER
242  /* need to pick a source that corresponds to this node */
243  uip_ds6_select_src(&UIP_IP_BUF->srcipaddr, &tmp_ipaddr);
244 #else
245  uip_ipaddr_copy(&UIP_IP_BUF->srcipaddr, &tmp_ipaddr);
246 #endif
247  }
248 
249  UIP_ICMP_BUF->type = type;
250  UIP_ICMP_BUF->icode = code;
251  UIP_ICMP6_ERROR_BUF->param = uip_htonl(param);
252  UIP_IP_BUF->len[0] = ((uip_len - UIP_IPH_LEN) >> 8);
253  UIP_IP_BUF->len[1] = ((uip_len - UIP_IPH_LEN) & 0xff);
254  UIP_ICMP_BUF->icmpchksum = 0;
255  UIP_ICMP_BUF->icmpchksum = ~uip_icmp6chksum();
256 
257  UIP_STAT(++uip_stat.icmp.sent);
258 
259  LOG_WARN("Sending ICMPv6 ERROR message type %d code %d to ", type, code);
260  LOG_WARN_6ADDR(&UIP_IP_BUF->destipaddr);
261  LOG_WARN_(" from ");
262  LOG_WARN_6ADDR(&UIP_IP_BUF->srcipaddr);
263  LOG_WARN_("\n");
264  return;
265 }

268 void
269 uip_icmp6_send(const uip_ipaddr_t *dest, int type, int code, int payload_len)
270 {
271  LOG_INFO("Sending ICMPv6 packet to ");
272  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
273  LOG_INFO_(", type %u, code %u, len %u\n", type, code, payload_len);
274 
275  UIP_IP_BUF->vtc = 0x60;
276  UIP_IP_BUF->tcflow = 0;
277  UIP_IP_BUF->flow = 0;
278  UIP_IP_BUF->proto = UIP_PROTO_ICMP6;
279  UIP_IP_BUF->ttl = uip_ds6_if.cur_hop_limit;
280  UIP_IP_BUF->len[0] = (UIP_ICMPH_LEN + payload_len) >> 8;
281  UIP_IP_BUF->len[1] = (UIP_ICMPH_LEN + payload_len) & 0xff;
282 
283  memcpy(&UIP_IP_BUF->destipaddr, dest, sizeof(*dest));
284  uip_ds6_select_src(&UIP_IP_BUF->srcipaddr, &UIP_IP_BUF->destipaddr);
285 
286  UIP_ICMP_BUF->type = type;
287  UIP_ICMP_BUF->icode = code;
288 
289  UIP_ICMP_BUF->icmpchksum = 0;
290  UIP_ICMP_BUF->icmpchksum = ~uip_icmp6chksum();
291 
292  uip_len = UIP_IPH_LEN + UIP_ICMPH_LEN + payload_len;
293 
294  UIP_STAT(++uip_stat.icmp.sent);
295  UIP_STAT(++uip_stat.ip.sent);
296 
297  tcpip_ipv6_output();
298 }

300 static void
301 echo_reply_input(void)
302 {
303  int ttl;
304  uip_ipaddr_t sender;
305 
306  LOG_INFO("Received Echo Reply from ");
307  LOG_INFO_6ADDR(&UIP_IP_BUF->srcipaddr);
308  LOG_INFO_(" to ");
309  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
310  LOG_INFO_("\n");
311 
312  uip_ipaddr_copy(&sender, &UIP_IP_BUF->srcipaddr);
313  ttl = UIP_IP_BUF->ttl;
314 
315  if(uip_ext_len > 0) {
316  /* Remove extension headers if any */
317  UIP_IP_BUF->proto = UIP_PROTO_ICMP6;
318  uip_len -= uip_ext_len;
319  UIP_IP_BUF->len[0] = ((uip_len - UIP_IPH_LEN) >> 8);
320  UIP_IP_BUF->len[1] = ((uip_len - UIP_IPH_LEN) & 0xff);
321  /* move the echo reply payload (starting after the icmp header)
322  * to the new location in the reply. The shift is equal to the
323  * length of the extension headers present Note: UIP_ICMP_BUF
324  * still points to the echo request at this stage
325  */
326  memmove((uint8_t *)UIP_ICMP_BUF + UIP_ICMPH_LEN - uip_ext_len,
327  (uint8_t *)UIP_ICMP_BUF + UIP_ICMPH_LEN,
328  (uip_len - UIP_IPH_LEN - UIP_ICMPH_LEN));
329  uip_ext_len = 0;
330  }
331 
332  /* Call all registered applications to let them know an echo reply
333  has been received. */
334  {
335  struct uip_icmp6_echo_reply_notification *n;
336  for(n = list_head(echo_reply_callback_list);
337  n != NULL;
338  n = list_item_next(n)) {
339  if(n->callback != NULL) {
340  n->callback(&sender, ttl,
341  (uint8_t *)&UIP_ICMP_BUF[sizeof(struct uip_icmp_hdr)],
342  uip_len - sizeof(struct uip_icmp_hdr) - UIP_IPH_LEN);
343  }
344  }
345  }
346 
347  uip_clear_buf();
348  return;
349 }
