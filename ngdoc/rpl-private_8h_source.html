157 #ifndef RPL_CONF_MIN_HOPRANKINC
158 /* RFC6550 defines the default MIN_HOPRANKINC as 256.
159  * However, we use MRHOF as a default Objective Function (RFC6719),
160  * which recommends setting MIN_HOPRANKINC with care, in particular
161  * when used with ETX as a metric. ETX is computed as a fixed point
162  * real with a divisor of 128 (RFC6719, RFC6551). We choose to also
163  * use 128 for RPL_MIN_HOPRANKINC, resulting in a rank equal to the
164  * ETX path cost. Larger values may also be desirable, as discussed
165  * in section 6.1 of RFC6719. */
166 #if RPL_OF_OCP == RPL_OCP_MRHOF
167 #define RPL_MIN_HOPRANKINC 128
168 #else /* RPL_OF_OCP == RPL_OCP_MRHOF */
169 #define RPL_MIN_HOPRANKINC 256
170 #endif /* RPL_OF_OCP == RPL_OCP_MRHOF */
171 #else /* RPL_CONF_MIN_HOPRANKINC */
172 #define RPL_MIN_HOPRANKINC RPL_CONF_MIN_HOPRANKINC
173 #endif /* RPL_CONF_MIN_HOPRANKINC */
174 
175 #ifndef RPL_CONF_MAX_RANKINC
176 #define RPL_MAX_RANKINC (7 * RPL_MIN_HOPRANKINC)
177 #else /* RPL_CONF_MAX_RANKINC */
178 #define RPL_MAX_RANKINC RPL_CONF_MAX_RANKINC
179 #endif /* RPL_CONF_MAX_RANKINC */
