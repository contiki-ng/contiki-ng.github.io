162 static
163 PT_THREAD(cmd_ping(struct pt *pt, shell_output_func output, char *args))
164 {
165  static uip_ipaddr_t remote_addr;
166  static struct etimer timeout_timer;
167  char *next_args;
168 
169  PT_BEGIN(pt);
170 
171  SHELL_ARGS_INIT(args, next_args);
172 
173  /* Get argument (remote IPv6) */
174  SHELL_ARGS_NEXT(args, next_args);
175  if(uiplib_ipaddrconv(args, &remote_addr) == 0) {
176  SHELL_OUTPUT(output, "Invalid IPv6: %s\n", args);
177  PT_EXIT(pt);
178  }
179 
180  SHELL_OUTPUT(output, "Pinging ");
181  shell_output_6addr(output, &remote_addr);
182  SHELL_OUTPUT(output, "\n");
183 
184  /* Send ping request */
185  curr_ping_process = PROCESS_CURRENT();
186  curr_ping_output_func = output;
187  etimer_set(&timeout_timer, PING_TIMEOUT);
188  uip_icmp6_send(&remote_addr, ICMP6_ECHO_REQUEST, 0, 4);
189  PT_WAIT_UNTIL(pt, curr_ping_output_func == NULL || etimer_expired(&timeout_timer));
190 
191  if(curr_ping_output_func != NULL) {
192  SHELL_OUTPUT(output, "Timeout\n");
193  curr_ping_output_func = NULL;
194  } else {
195  SHELL_OUTPUT(output, "Received ping reply from ");
196  shell_output_6addr(output, &remote_addr);
197  SHELL_OUTPUT(output, ", len %u, ttl %u, delay %lu ms\n",
198  curr_ping_datalen, curr_ping_ttl, (1000*(clock_time() - timeout_timer.timer.start))/CLOCK_SECOND);
199  }
200 
201  PT_END(pt);
202 }

218 static
219 PT_THREAD(cmd_log(struct pt *pt, shell_output_func output, char *args))
220 {
221  static int prev_level;
222  static int level;
223  char *next_args;
224  char *ptr;
225  char *module;
226 
227  PT_BEGIN(pt);
228 
229  SHELL_ARGS_INIT(args, next_args);
230 
231  /* Get and parse argument: module name */
232  SHELL_ARGS_NEXT(args, next_args);
233  module = args;
234  prev_level = log_get_level(module);
235  if(module == NULL || (strcmp("all", module) && prev_level == -1)) {
236  SHELL_OUTPUT(output, "Invalid first argument: %s\n", module)
237  shell_output_log_levels(output);
238  PT_EXIT(pt);
239  }
240 
241  /* Get and parse argument: log level */
242  SHELL_ARGS_NEXT(args, next_args);
243  if(args == NULL) {
244  level = -1;
245  } else {
246  level = (int)strtol(args, &ptr, 10);
247  }
248  if((level == 0 && args == ptr)
249  || level < LOG_LEVEL_NONE || level > LOG_LEVEL_DBG) {
250  SHELL_OUTPUT(output, "Invalid second argument: %s\n", args);
251  PT_EXIT(pt);
252  }
253 
254  /* Set log level */
255  if(level != prev_level) {
256  log_set_level(module, level);
257 #if MAC_CONF_WITH_TSCH && TSCH_LOG_PER_SLOT
258  if(!strcmp(module, "mac") || !strcmp(module, "all")) {
259  if(level >= LOG_LEVEL_DBG) {
260  tsch_log_init();
261  SHELL_OUTPUT(output, "TSCH logging started\n");
262  } else {
263  tsch_log_stop();
264  SHELL_OUTPUT(output, "TSCH logging stopped\n");
265  }
266  }
267 #endif /* MAC_CONF_WITH_TSCH && TSCH_LOG_PER_SLOT */
268  }
269 
270  shell_output_log_levels(output);
271 
272  PT_END(pt);
273 }

294 PT_THREAD(cmd_rpl_set_root(struct pt *pt, shell_output_func output, char *args))
295 {
296  static int is_on;
297  static uip_ipaddr_t prefix;
298  char *next_args;
299 
300  PT_BEGIN(pt);
301 
302  SHELL_ARGS_INIT(args, next_args);
303 
304  /* Get first arg (0/1) */
305  SHELL_ARGS_NEXT(args, next_args);
306 
307  if(!strcmp(args, "1")) {
308  is_on = 1;
309  } else if(!strcmp(args, "0")) {
310  is_on = 0;
311  } else {
312  SHELL_OUTPUT(output, "Invalid argument: %s\n", args);
313  PT_EXIT(pt);
314  }
315 
316  /* Get first second arg (prefix) */
317  SHELL_ARGS_NEXT(args, next_args);
318  if(args != NULL) {
319  if(uiplib_ipaddrconv(args, &prefix) == 0) {
320  SHELL_OUTPUT(output, "Invalid Prefix: %s\n", args);
321  PT_EXIT(pt);
322  }
323  } else {
324  uip_ip6addr(&prefix, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0, 0, 0, 0);
325  }
326 
327  if(is_on) {
328  if(!rpl_dag_root_is_root()) {
329  SHELL_OUTPUT(output, "Setting as DAG root with prefix ");
330  shell_output_6addr(output, &prefix);
331  SHELL_OUTPUT(output, "/64\n");
332  rpl_dag_root_init(&prefix, NULL);
333  rpl_dag_root_init_dag_immediately();
334  } else {
335  SHELL_OUTPUT(output, "Node is already a DAG root\n");
336  }
337  } else {
338  if(rpl_dag_root_is_root()) {
339  SHELL_OUTPUT(output, "Setting as non-root node: leaving DAG\n");
340  rpl_dag_poison_and_leave();
341  } else {
342  SHELL_OUTPUT(output, "Node is not a DAG root\n");
343  }
344  }
345 
346  PT_END(pt);
347 }

550 static
551 PT_THREAD(cmd_routes(struct pt *pt, shell_output_func output, char *args))
552 {
553  uip_ds6_defrt_t *default_route;
554 #if RPL_WITH_NON_STORING
555  rpl_ns_node_t *link;
556 #endif /* RPL_WITH_NON_STORING */
557 #if RPL_WITH_STORING
558  uip_ds6_route_t *route;
559 #endif /* RPL_WITH_STORING */
560 
561  PT_BEGIN(pt);
562 
563  /* Our default route */
564  SHELL_OUTPUT(output, "Default route:\n");
565  default_route = uip_ds6_defrt_lookup(uip_ds6_defrt_choose());
566  if(default_route != NULL) {
567  SHELL_OUTPUT(output, "-- ");
568  shell_output_6addr(output, &default_route->ipaddr);
569  if(default_route->lifetime.interval != 0) {
570  SHELL_OUTPUT(output, " (lifetime: %lu seconds)\n", (unsigned long)default_route->lifetime.interval);
571  } else {
572  SHELL_OUTPUT(output, " (lifetime: infinite)\n");
573  }
574  } else {
575  SHELL_OUTPUT(output, "-- None\n");
576  }
577 
578 #if RPL_WITH_NON_STORING
579  if(rpl_ns_num_nodes() > 0) {
580  /* Our routing links */
581  SHELL_OUTPUT(output, "Routing links (%u in total):\n", rpl_ns_num_nodes());
582  link = rpl_ns_node_head();
583  while(link != NULL) {
584  uip_ipaddr_t child_ipaddr;
585  uip_ipaddr_t parent_ipaddr;
586  rpl_ns_get_node_global_addr(&child_ipaddr, link);
587  rpl_ns_get_node_global_addr(&parent_ipaddr, link->parent);
588  SHELL_OUTPUT(output, "-- ");
589  shell_output_6addr(output, &child_ipaddr);
590  if(link->parent == NULL) {
591  memset(&parent_ipaddr, 0, sizeof(parent_ipaddr));
592  SHELL_OUTPUT(output, " (DODAG root)");
593  } else {
594  SHELL_OUTPUT(output, " to ");
595  shell_output_6addr(output, &parent_ipaddr);
596  }
597  if(link->lifetime != RPL_ROUTE_INFINITE_LIFETIME) {
598  SHELL_OUTPUT(output, " (lifetime: %lu seconds)\n", (unsigned long)link->lifetime);
599  } else {
600  SHELL_OUTPUT(output, " (lifetime: infinite)\n");
601  }
602  link = rpl_ns_node_next(link);
603  }
604  } else {
605  SHELL_OUTPUT(output, "No routing links\n");
606  }
607 #endif /* RPL_WITH_NON_STORING */
