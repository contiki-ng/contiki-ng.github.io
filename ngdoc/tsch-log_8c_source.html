73 /*---------------------------------------------------------------------------*/
74 /* Process pending log messages */
75 void
76 tsch_log_process_pending(void)
77 {
78  static int last_log_dropped = 0;
79  int16_t log_index;
80  /* Loop on accessing (without removing) a pending input packet */
81  if(log_dropped != last_log_dropped) {
82  printf("[WARN: TSCH-LOG ] logs dropped %u\n", log_dropped);
83  last_log_dropped = log_dropped;
84  }
85  while((log_index = ringbufindex_peek_get(&log_ringbuf)) != -1) {
86  struct tsch_log_t *log = &log_array[log_index];
87  if(log->link == NULL) {
88  printf("[INFO: TSCH-LOG ] {asn-%x.%lx link-NULL} ", log->asn.ms1b, log->asn.ls4b);
89  } else {
90  struct tsch_slotframe *sf = tsch_schedule_get_slotframe_by_handle(log->link->slotframe_handle);
91  printf("[INFO: TSCH-LOG ] {asn-%x.%lx link-%u-%u-%u-%u ch-%u} ",
92  log->asn.ms1b, log->asn.ls4b,
93  log->link->slotframe_handle, sf ? sf->size.val : 0, log->link->timeslot, log->link->channel_offset,
94  tsch_calculate_channel(&log->asn, log->link->channel_offset));
95  }
96  switch(log->type) {
97  case tsch_log_tx:
98  printf("%s-%u-%u tx ",
99  linkaddr_cmp(&log->tx.dest, &linkaddr_null) ? "bc" : "uc", log->tx.is_data, log->tx.sec_level);
100  log_lladdr_compact(&linkaddr_node_addr);
101  printf("->");
102  log_lladdr_compact(&log->tx.dest);
103  printf(", len %u, seq %u, st %d %d",
104  log->tx.datalen, log->tx.seqno, log->tx.mac_tx_status, log->tx.num_tx);
105  if(log->tx.drift_used) {
106  printf(", dr %d", log->tx.drift);
107  }
108  printf("\n");
109  break;
110  case tsch_log_rx:
111  printf("%s-%u-%u rx ",
112  log->rx.is_unicast == 0 ? "bc" : "uc", log->rx.is_data, log->rx.sec_level);
113  log_lladdr_compact(&log->rx.src);
114  printf("->");
115  log_lladdr_compact(log->rx.is_unicast ? &linkaddr_node_addr : NULL);
116  printf(", len %u, seq %u",
117  log->rx.datalen, log->rx.seqno);
118  if(log->rx.drift_used) {
119  printf(", dr %d", log->rx.drift);
120  }
121  printf(", edr %d\n", (int)log->rx.estimated_drift);
122  break;
123  case tsch_log_message:
124  printf("%s\n", log->message);
125  break;
126  }
127  /* Remove input from ringbuf */
128  ringbufindex_get(&log_ringbuf);
129  }
130 }
131 /*---------------------------------------------------------------------------*/
132 /* Prepare addition of a new log.
133  * Returns pointer to log structure if success, NULL otherwise */
134 struct tsch_log_t *
135 tsch_log_prepare_add(void)
136 {
137  int log_index = ringbufindex_peek_put(&log_ringbuf);
138  if(log_index != -1) {
139  struct tsch_log_t *log = &log_array[log_index];
140  log->asn = tsch_current_asn;
141  log->link = current_link;
142  return log;
143  } else {
144  log_dropped++;
145  return NULL;
146  }
147 }
148 /*---------------------------------------------------------------------------*/
149 /* Actually add the previously prepared log */
150 void
151 tsch_log_commit(void)
152 {
153  if(log_active == 1) {
154  ringbufindex_put(&log_ringbuf);
155  process_poll(&tsch_pending_events_process);
156  }
157 }
158 /*---------------------------------------------------------------------------*/
159 /* Initialize log module */
160 void
161 tsch_log_init(void)
162 {
163  if(log_active == 0) {
164  ringbufindex_init(&log_ringbuf, TSCH_LOG_QUEUE_LEN);
165  log_active = 1;
166  }
167 }
168 /*---------------------------------------------------------------------------*/
169 /* Stop log module */
170 void
171 tsch_log_stop(void)
172 {
173  if(log_active == 1) {
174  tsch_log_process_pending();
175  log_active = 0;
176  }
177 }
