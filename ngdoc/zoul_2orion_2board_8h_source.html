106 /** \name UART configuration
107  *
108  * On the eth-gw, the UARTs are connected to the following ports/pins:
109  *
110  * - UART0:
111  * - RX: PA0, connected to CP2104 serial-to-usb converter TX pin
112  * - TX: PA1, connected to CP2104 serial-to-usb converter RX pin
113  * - UART1:
114  * - RX: PC1
115  * - TX: PC0
116  * - CTS: not used
117  * - RTS: not used
118  *
119  * We configure the port to use UART0 and UART1, CTS/RTS only for UART1,
120  * both without a HW pull-up resistor.
121  * UART0 is not exposed anywhere, UART1 pins are exposed over the JP9 connector.
122  * @{
123  */
124 #define UART0_RX_PORT GPIO_A_NUM
125 #define UART0_RX_PIN 0
126 #define UART0_TX_PORT GPIO_A_NUM
127 #define UART0_TX_PIN 1
128 
129 #define UART1_RX_PORT GPIO_C_NUM
130 #define UART1_RX_PIN 1
131 #define UART1_TX_PORT GPIO_C_NUM
132 #define UART1_TX_PIN 0
133 #define UART1_CTS_PORT (-1)
134 #define UART1_CTS_PIN (-1)
