57 /* Per-neighbor RPL information. According to RFC 6550, there exist three
58  * types of neighbors:
59  * - Candidate neighbor set: any neighbor, selected in an implementation
60  * and OF-specific way. The nodes in rpl_neighbors constitute the candidate neighbor set.
61  * - Parent set: the subset of the candidate neighbor set with rank below our rank
62  * - Preferred parent: one node of the parent set
63  */
64 NBR_TABLE_DECLARE(rpl_neighbors);
