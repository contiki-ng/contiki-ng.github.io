33 /**
34  * \file
35  * A quick program that blinks the LEDs
36  * \author
37  * Adam Dunkels <adam@sics.se>
38  */
39 
40 #include "contiki.h"
41 #include "dev/leds.h"
42 /*---------------------------------------------------------------------------*/
43 PROCESS(blink_process, "Blink");
44 AUTOSTART_PROCESSES(&blink_process);
45 /*---------------------------------------------------------------------------*/
46 PROCESS_THREAD(blink_process, ev, data)
47 {
48  PROCESS_EXITHANDLER(goto exit;)
49  PROCESS_BEGIN();
50 
51  while(1) {
52  static struct etimer et;
53  etimer_set(&et, CLOCK_SECOND);
54  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
55  leds_on(LEDS_ALL);
56  etimer_set(&et, CLOCK_SECOND);
57  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
58  leds_off(LEDS_ALL);
59  }
60 
61  exit:
62  leds_off(LEDS_ALL);
63  PROCESS_END();
64 }
