40 /* Register two unit tests that will be executed by using
41  the UNIT_TEST_RUN macro. */
42 UNIT_TEST_REGISTER(arithmetic, "Arith ops");
43 UNIT_TEST_REGISTER(string, "String ops");
44 
45 /* arithmetic: Demonstrates a test that succeeds. The exit point will be
46  the line where UNIT_TEST_END is called. */
47 UNIT_TEST(arithmetic)
48 {
49  int a, b;
50 
51  UNIT_TEST_BEGIN();
52 
53  a = 1;
54  b = 2;
55 
56  UNIT_TEST_ASSERT(a + b == 3);
57 
58  UNIT_TEST_END();
59 }
60 
61 /* string: Demonstrates a test that fails. The exit point will be
62  the line where the call to UNIT_TEST_ASSERT fails. */
63 UNIT_TEST(string)
64 {
65  char str1[] = "A";
66 
67  UNIT_TEST_BEGIN();
68 
69  UNIT_TEST_ASSERT(str1[0] == 'B');
70 
71  UNIT_TEST_END();
72 }
