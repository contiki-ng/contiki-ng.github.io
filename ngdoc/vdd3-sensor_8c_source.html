49 static int
50 value(int type)
51 {
52  int raw = adc_get(SOC_ADC_ADCCON_CH_VDD_3, SOC_ADC_ADCCON_REF_INT,
53  SOC_ADC_ADCCON_DIV_512);
54 
55  if(type == CC2538_SENSORS_VALUE_TYPE_RAW) {
56  return raw;
57  } else if(type == CC2538_SENSORS_VALUE_TYPE_CONVERTED) {
58  return raw * (3 * 1190) / (2047 << 4);
59  }
60 
61  return CC2538_SENSORS_ERROR;
62 }
