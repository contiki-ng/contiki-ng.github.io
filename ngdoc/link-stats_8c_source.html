47 /* Maximum value for the Tx count counter */
48 #define TX_COUNT_MAX 32

50 /* Statistics with no update in FRESHNESS_EXPIRATION_TIMEOUT is not fresh */
51 #define FRESHNESS_EXPIRATION_TIME (10 * 60 * (clock_time_t)CLOCK_SECOND)
52 /* Half time for the freshness counter */
53 #define FRESHNESS_HALF_LIFE (15 * 60 * (clock_time_t)CLOCK_SECOND)
54 /* Statistics are fresh if the freshness counter is FRESHNESS_TARGET or more */
55 #define FRESHNESS_TARGET 4
56 /* Maximum value for the freshness counter */
57 #define FRESHNESS_MAX 16

59 /* EWMA (exponential moving average) used to maintain statistics over time */
60 #define EWMA_SCALE 100

62 #define EWMA_BOOTSTRAP_ALPHA 25

87 link_stats_is_fresh(const struct link_stats *stats)
88 {
89  return (stats != NULL)
90  && clock_time() - stats->last_tx_time < FRESHNESS_EXPIRATION_TIME
91  && stats->freshness >= FRESHNESS_TARGET;

94 #if LINK_STATS_INIT_ETX_FROM_RSSI
95 uint16_t
96 guess_etx_from_rssi(const struct link_stats *stats)
97 {
98  if(stats != NULL) {
99  if(stats->rssi == 0) {
100  return ETX_DEFAULT * ETX_DIVISOR;
101  } else {
102  /* A rough estimate of PRR from RSSI, as a linear function where:
103  * RSSI >= -60 results in PRR of 1
104  * RSSI <= -90 results in PRR of 0
105  * prr = (bounded_rssi - RSSI_LOW) / (RSSI_DIFF)
106  * etx = ETX_DIVOSOR / ((bounded_rssi - RSSI_LOW) / RSSI_DIFF)
107  * etx = (RSSI_DIFF * ETX_DIVOSOR) / (bounded_rssi - RSSI_LOW)
108  * */
109 #define ETX_INIT_MAX 3
110 #define RSSI_HIGH -60
111 #define RSSI_LOW -90
112 #define RSSI_DIFF (RSSI_HIGH - RSSI_LOW)
113  uint16_t etx;
114  int16_t bounded_rssi = stats->rssi;
115  bounded_rssi = MIN(bounded_rssi, RSSI_HIGH);
116  bounded_rssi = MAX(bounded_rssi, RSSI_LOW + 1);
117  etx = RSSI_DIFF * ETX_DIVISOR / (bounded_rssi - RSSI_LOW);
118  return MIN(etx, ETX_INIT_MAX * ETX_DIVISOR);
119  }
120  }
121  return 0xffff;
122 }
123 #endif /* LINK_STATS_INIT_ETX_FROM_RSSI */

125 /* Packet sent callback. Updates stats for transmissions to lladdr */
126 void
127 link_stats_packet_sent(const linkaddr_t *lladdr, int status, int numtx)
128 {
129  struct link_stats *stats;
130 #if !LINK_STATS_ETX_FROM_PACKET_COUNT
131  uint16_t packet_etx;
132  uint8_t ewma_alpha;
133 #endif /* !LINK_STATS_ETX_FROM_PACKET_COUNT */
134 
135  if(status != MAC_TX_OK && status != MAC_TX_NOACK) {
136  /* Do not penalize the ETX when collisions or transmission errors occur. */
137  return;
138  }
139 
140  stats = nbr_table_get_from_lladdr(link_stats, lladdr);
141  if(stats == NULL) {
142  /* Add the neighbor */
143  stats = nbr_table_add_lladdr(link_stats, lladdr, NBR_TABLE_REASON_LINK_STATS, NULL);
144  if(stats != NULL) {
145 #if LINK_STATS_INIT_ETX_FROM_RSSI
146  stats->etx = guess_etx_from_rssi(stats);
147 #else /* LINK_STATS_INIT_ETX_FROM_RSSI */
148  stats->etx = ETX_DEFAULT * ETX_DIVISOR;
149 #endif /* LINK_STATS_INIT_ETX_FROM_RSSI */
150  } else {
151  return; /* No space left, return */
152  }
153  }
154 
155  /* Update last timestamp and freshness */
156  stats->last_tx_time = clock_time();
157  stats->freshness = MIN(stats->freshness + numtx, FRESHNESS_MAX);
158 
159 #if LINK_STATS_ETX_FROM_PACKET_COUNT
160  /* Compute ETX from packet and ACK count */
161  numtx = (status == MAC_TX_NOACK) ? ETX_NOACK_PENALTY : numtx;
162  /* Halve both counter after TX_COUNT_MAX */
163  if(stats->tx_count + numtx > TX_COUNT_MAX) {
164  stats->tx_count /= 2;
165  stats->ack_count /= 2;
166  }
167  /* Update tx_count and ack_count */
168  stats->tx_count += numtx;
169  if(status == MAC_TX_OK) {
170  stats->ack_count++;
171  }
172  /* Compute ETX */
173  if(stats->ack_count > 0) {
174  stats->etx = ((uint16_t)stats->tx_count * ETX_DIVISOR) / stats->ack_count;
175  } else {
176  stats->etx = (uint16_t)MAX(ETX_NOACK_PENALTY, stats->tx_count) * ETX_DIVISOR;
177  }
178 #else /* LINK_STATS_ETX_FROM_PACKET_COUNT */
179  /* Compute ETX using an EWMA */
180 
181  /* ETX used for this update */
182  packet_etx = ((status == MAC_TX_NOACK) ? ETX_NOACK_PENALTY : numtx) * ETX_DIVISOR;
183  /* ETX alpha used for this update */
184  ewma_alpha = link_stats_is_fresh(stats) ? EWMA_ALPHA : EWMA_BOOTSTRAP_ALPHA;
185 
186  /* Compute EWMA and update ETX */
187  stats->etx = ((uint32_t)stats->etx * (EWMA_SCALE - ewma_alpha) +
188  (uint32_t)packet_etx * ewma_alpha) / EWMA_SCALE;
189 #endif /* LINK_STATS_ETX_FROM_PACKET_COUNT */
190 }

192 /* Packet input callback. Updates statistics for receptions on a given link */
193 void
194 link_stats_input_callback(const linkaddr_t *lladdr)
195 {
196  struct link_stats *stats;
197  int16_t packet_rssi = packetbuf_attr(PACKETBUF_ATTR_RSSI);
198 
199  stats = nbr_table_get_from_lladdr(link_stats, lladdr);
200  if(stats == NULL) {
201  /* Add the neighbor */
202  stats = nbr_table_add_lladdr(link_stats, lladdr, NBR_TABLE_REASON_LINK_STATS, NULL);
203  if(stats != NULL) {
204  /* Initialize */
205  stats->rssi = packet_rssi;
206 #if LINK_STATS_INIT_ETX_FROM_RSSI
207  stats->etx = guess_etx_from_rssi(stats);
208 #else /* LINK_STATS_INIT_ETX_FROM_RSSI */
209  stats->etx = ETX_DEFAULT * ETX_DIVISOR;
210 #endif /* LINK_STATS_INIT_ETX_FROM_RSSI */
211  }
212  return;
213  }
214 
215  /* Update RSSI EWMA */
216  stats->rssi = ((int32_t)stats->rssi * (EWMA_SCALE - EWMA_ALPHA) +
217  (int32_t)packet_rssi * EWMA_ALPHA) / EWMA_SCALE;
218 }

220 /* Periodic timer called at a period of FRESHNESS_HALF_LIFE */
221 static void
222 periodic(void *ptr)
223 {
224  /* Age (by halving) freshness counter of all neighbors */
225  struct link_stats *stats;
226  ctimer_reset(&periodic_timer);
227  for(stats = nbr_table_head(link_stats); stats != NULL; stats = nbr_table_next(link_stats, stats)) {
228  stats->freshness >>= 1;
229  }
230 }

232 /* Resets link-stats module */
233 void
234 link_stats_reset(void)
235 {
236  struct link_stats *stats;
237  stats = nbr_table_head(link_stats);
238  while(stats != NULL) {
239  nbr_table_remove(link_stats, stats);
240  stats = nbr_table_next(link_stats, stats);
241  }
242 }
