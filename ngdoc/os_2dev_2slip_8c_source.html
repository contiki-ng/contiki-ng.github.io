64 enum {

71 /*
72  * Variables begin and end manage the buffer space in a cyclic
73  * fashion. The first used byte is at begin and end is one byte past
74  * the last. I.e. [begin, end) is the actively used space.
75  *
76  * If begin != pkt_end we have a packet at [begin, pkt_end),
77  * furthermore, if state == STATE_TWOPACKETS we have one more packet at
78  * [pkt_end, end). If more bytes arrive in state STATE_TWOPACKETS
79  * they are discarded.
80  */
81

95 /* slip_send: forward (IPv4) packets with {UIP_FW_NETIF(..., slip_send)}
96  * was used in slip-bridge.c
97  */
98 uint8_t
99 slip_send(void)
100 {
101  uint16_t i;
102  uint8_t *ptr;
103  uint8_t c;
104 
105  slip_arch_writeb(SLIP_END);
106 
107  ptr = &uip_buf[UIP_LLH_LEN];
108  for(i = 0; i < uip_len; ++i) {
109  c = *ptr++;
110  if(c == SLIP_END) {
111  slip_arch_writeb(SLIP_ESC);
112  c = SLIP_ESC_END;
113  } else if(c == SLIP_ESC) {
114  slip_arch_writeb(SLIP_ESC);
115  c = SLIP_ESC_ESC;
116  }
117  slip_arch_writeb(c);
118  }
119  slip_arch_writeb(SLIP_END);
120 
121  return UIP_FW_OK;
122 }

156 /* Upper half does the polling. */
157 static uint16_t
158 slip_poll_handler(uint8_t *outbuf, uint16_t blen)
159 {
160 #ifdef SLIP_CONF_MICROSOFT_CHAT
161  /* This is a hack and won't work across buffer edge! */
162  if(rxbuf[begin] == 'C') {
163  int i;
164  if(begin < next_free && (next_free - begin) >= 6
165  && memcmp(&rxbuf[begin], "CLIENT", 6) == 0) {
166  state = STATE_TWOPACKETS; /* Interrupts do nothing. */
167  memset(&rxbuf[begin], 0x0, 6);
168 
169  rxbuf_init();
170 
171  for(i = 0; i < 13; i++) {
172  slip_arch_writeb("CLIENTSERVER\300"[i]);
173  }
174  return 0;
175  }
176  }
177 #endif /* SLIP_CONF_MICROSOFT_CHAT */
178 
179 #ifdef SLIP_CONF_ANSWER_MAC_REQUEST
180  else if(rxbuf[begin] == '?') {
181  /* Used by tapslip6 to request mac for auto configure */
182  int i, j;
183  char* hexchar = "0123456789abcdef";
184  if(begin < next_free && (next_free - begin) >= 2
185  && rxbuf[begin + 1] == 'M') {
186  state = STATE_TWOPACKETS; /* Interrupts do nothing. */
187  rxbuf[begin] = 0;
188  rxbuf[begin + 1] = 0;
189 
190  rxbuf_init();
191 
192  linkaddr_t addr = get_mac_addr();
193  /* this is just a test so far... just to see if it works */
194  slip_arch_writeb('!');
195  slip_arch_writeb('M');
196  for(j = 0; j < 8; j++) {
197  slip_arch_writeb(hexchar[addr.u8[j] >> 4]);
198  slip_arch_writeb(hexchar[addr.u8[j] & 15]);
199  }
200  slip_arch_writeb(SLIP_END);
201  return 0;
202  }
203  }
204 #endif /* SLIP_CONF_ANSWER_MAC_REQUEST */
205 
206  /*
207  * Interrupt can not change begin but may change pkt_end.
208  * If pkt_end != begin it will not change again.
209  */
210  if(begin != pkt_end) {
211  uint16_t len;
212  uint16_t cur_next_free;
213  uint16_t cur_ptr;
214  int esc = 0;
215 
216  if(begin < pkt_end) {
217  uint16_t i;
218  len = 0;
219  for(i = begin; i < pkt_end; ++i) {
220  if(len > blen) {
221  len = 0;
222  break;
223  }
224  if (esc) {
225  if(rxbuf[i] == SLIP_ESC_ESC) {
226  outbuf[len] = SLIP_ESC;
227  len++;
228  } else if(rxbuf[i] == SLIP_ESC_END) {
229  outbuf[len] = SLIP_END;
230  len++;
231  }
232  esc = 0;
233  } else if(rxbuf[i] == SLIP_ESC) {
234  esc = 1;
235  } else {
236  outbuf[len] = rxbuf[i];
237  len++;
238  }
239  }
240  } else {
241  uint16_t i;
242  len = 0;
243  for(i = begin; i < RX_BUFSIZE; ++i) {
244  if(len > blen) {
245  len = 0;
246  break;
247  }
248  if (esc) {
249  if(rxbuf[i] == SLIP_ESC_ESC) {
250  outbuf[len] = SLIP_ESC;
251  len++;
252  } else if(rxbuf[i] == SLIP_ESC_END) {
253  outbuf[len] = SLIP_END;
254  len++;
255  }
256  esc = 0;
257  } else if(rxbuf[i] == SLIP_ESC) {
258  esc = 1;
259  } else {
260  outbuf[len] = rxbuf[i];
261  len++;
262  }
263  }
264  for(i = 0; i < pkt_end; ++i) {
265  if(len > blen) {
266  len = 0;
267  break;
268  }
269  if (esc) {
270  if(rxbuf[i] == SLIP_ESC_ESC) {
271  outbuf[len] = SLIP_ESC;
272  len++;
273  } else if(rxbuf[i] == SLIP_ESC_END) {
274  outbuf[len] = SLIP_END;
275  len++;
276  }
277  esc = 0;
278  } else if(rxbuf[i] == SLIP_ESC) {
279  esc = 1;
280  } else {
281  outbuf[len] = rxbuf[i];
282  len++;
283  }
284  }
285  }
286 
287  /* Remove data from buffer together with the copied packet. */
288  pkt_end = pkt_end + 1;
289  if(pkt_end == RX_BUFSIZE) {
290  pkt_end = 0;
291  }
292  if(pkt_end != next_free) {
293  cur_next_free = next_free;
294  cur_ptr = pkt_end;
295  while(cur_ptr != cur_next_free) {
296  if(rxbuf[cur_ptr] == SLIP_END) {
297  uint16_t tmp_begin = pkt_end;
298  pkt_end = cur_ptr;
299  begin = tmp_begin;
300  /* One more packet is buffered, need to be polled again! */
301  process_poll(&slip_process);
302  break;
303  }
304  cur_ptr++;
305  if(cur_ptr == RX_BUFSIZE) {
306  cur_ptr = 0;
307  }
308  }
309  if(cur_ptr == cur_next_free) {
310  /* no more pending full packet found */
311  begin = pkt_end;
312  }
313  } else {
314  begin = pkt_end;
315  }
316  return len;
317  }
318 
319  return 0;
320 }

322 PROCESS_THREAD(slip_process, ev, data)
323 {
324  PROCESS_BEGIN();
325 
326  rxbuf_init();
327 
328  while(1) {
329  PROCESS_YIELD_UNTIL(ev == PROCESS_EVENT_POLL);
330 
331  slip_active = 1;
332 
333  /* Move packet from rxbuf to buffer provided by uIP. */
334  uip_len = slip_poll_handler(&uip_buf[UIP_LLH_LEN],
335  UIP_BUFSIZE - UIP_LLH_LEN);
336 
337  if(uip_len > 0) {
338  if(input_callback) {
339  input_callback();
340  }
341 #ifdef SLIP_CONF_TCPIP_INPUT
342  SLIP_CONF_TCPIP_INPUT();
343 #else
344  tcpip_input();
345 #endif
346  }
347  }
348 
349  PROCESS_END();
350 }

352 int
353 slip_input_byte(unsigned char c)
354 {
355  uint16_t cur_end;
356  switch(state) {
357  case STATE_RUBBISH:
358  if(c == SLIP_END) {
359  state = STATE_OK;
360  }
361  return 0;
362 
363  case STATE_ESC:
364  if(c != SLIP_ESC_END && c != SLIP_ESC_ESC) {
365  state = STATE_RUBBISH;
366  SLIP_STATISTICS(slip_rubbish++);
367  next_free = pkt_end; /* remove rubbish */
368  return 0;
369  }
370  state = STATE_OK;
371  break;
372  }
373 
374  if(c == SLIP_ESC) {
375  state = STATE_ESC;
376  }
377 
378  /* add_char: */
379  cur_end = next_free;
380  next_free = next_free + 1;
381  if(next_free == RX_BUFSIZE) {
382  next_free = 0;
383  }
384  if(next_free == begin) { /* rxbuf is full */
385  state = STATE_RUBBISH;
386  SLIP_STATISTICS(slip_overflow++);
387  next_free = pkt_end; /* remove rubbish */
388  return 0;
389  }
390  rxbuf[cur_end] = c;
391 
392 #ifdef SLIP_CONF_MICROSOFT_CHAT
393  /* There could be a separate poll routine for this. */
394  if(c == 'T' && rxbuf[begin] == 'C') {
395  process_poll(&slip_process);
396  return 1;
397  }
398 #endif /* SLIP_CONF_MICROSOFT_CHAT */
399 
400  if(c == SLIP_END) {
401  /*
402  * We have a new packet, possibly of zero length.
403  *
404  * There may already be one packet buffered.
405  */
406  if(cur_end != pkt_end) { /* Non zero length. */
407  if(begin == pkt_end) { /* None buffered. */
408  pkt_end = cur_end;
409  } else {
410  SLIP_STATISTICS(slip_twopackets++);
411  }
412  process_poll(&slip_process);
413  return 1;
414  } else {
415  /* Empty packet, reset the pointer */
416  next_free = cur_end;
417  }
418  return 0;
419  }
420 
421  return 0;
422 }
