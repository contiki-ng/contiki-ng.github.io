51 /* The ASN is an absolute slot number over 5 bytes. */
52 struct tsch_asn_t {
53  uint32_t ls4b; /* least significant 4 bytes */
54  uint8_t ms1b; /* most significant 1 byte */
55 };
56 
57 /* For quick modulo operation on ASN */
58 struct tsch_asn_divisor_t {
59  uint16_t val; /* Divisor value */
60  uint16_t asn_ms1b_remainder; /* Remainder of the operation 0x100000000 / val */
61 };

65 /* Initialize ASN */
66 #define TSCH_ASN_INIT(asn, ms1b_, ls4b_) do { \
67  (asn).ms1b = (ms1b_); \
68  (asn).ls4b = (ls4b_); \
69 } while(0);
70 
71 /* Increment an ASN by inc (32 bits) */
72 #define TSCH_ASN_INC(asn, inc) do { \
73  uint32_t new_ls4b = (asn).ls4b + (inc); \
74  if(new_ls4b < (asn).ls4b) { (asn).ms1b++; } \
75  (asn).ls4b = new_ls4b; \
76 } while(0);
77 
78 /* Decrement an ASN by inc (32 bits) */
79 #define TSCH_ASN_DEC(asn, dec) do { \
80  uint32_t new_ls4b = (asn).ls4b - (dec); \
81  if(new_ls4b > (asn).ls4b) { (asn).ms1b--; } \
82  (asn).ls4b = new_ls4b; \
83 } while(0);
84 
85 /* Returns the 32-bit diff between asn1 and asn2 */
86 #define TSCH_ASN_DIFF(asn1, asn2) \
87  ((asn1).ls4b - (asn2).ls4b)
88 
89 /* Initialize a struct asn_divisor_t */
90 #define TSCH_ASN_DIVISOR_INIT(div, val_) do { \
91  (div).val = (val_); \
92  (div).asn_ms1b_remainder = ((0xffffffff % (val_)) + 1) % (val_); \
93 } while(0);
94 
95 /* Returns the result (16 bits) of a modulo operation on ASN,
96  * with divisor being a struct asn_divisor_t */
97 #define TSCH_ASN_MOD(asn, div) \
98  ((uint16_t)((asn).ls4b % (div).val) \
99  + (uint16_t)((asn).ms1b * (div).asn_ms1b_remainder % (div).val)) \
100  % (div).val
