50 void
51 uip_udp_packet_send(struct uip_udp_conn *c, const void *data, int len)
52 {
53 #if UIP_UDP
54  if(data != NULL && len <= (UIP_BUFSIZE - (UIP_LLH_LEN + UIP_IPUDPH_LEN))) {
55  uip_udp_conn = c;
56  uip_slen = len;
57  memmove(&uip_buf[UIP_LLH_LEN + UIP_IPUDPH_LEN], data, len);
58  uip_process(UIP_UDP_SEND_CONN);
59 
60 #if UIP_IPV6_MULTICAST
61  /* Let the multicast engine process the datagram before we send it */
62  if(uip_is_addr_mcast_routable(&uip_udp_conn->ripaddr)) {
63  UIP_MCAST6.out();
64  }
65 #endif /* UIP_IPV6_MULTICAST */
66 
67 #if NETSTACK_CONF_WITH_IPV6
68  tcpip_ipv6_output();
69 #else
70  if(uip_len > 0) {
71  tcpip_output();
72  }
73 #endif
74  }
75  uip_slen = 0;
76 #endif /* UIP_UDP */
77 }

79 void
80 uip_udp_packet_sendto(struct uip_udp_conn *c, const void *data, int len,
81  const uip_ipaddr_t *toaddr, uint16_t toport)
82 {
83  uip_ipaddr_t curaddr;
84  uint16_t curport;
85 
86  if(toaddr != NULL) {
87  /* Save current IP addr/port. */
88  uip_ipaddr_copy(&curaddr, &c->ripaddr);
89  curport = c->rport;
90 
91  /* Load new IP addr/port */
92  uip_ipaddr_copy(&c->ripaddr, toaddr);
93  c->rport = toport;
94 
95  uip_udp_packet_send(c, data, len);
96 
97  /* Restore old IP addr/port */
98  uip_ipaddr_copy(&c->ripaddr, &curaddr);
99  c->rport = curport;
100  }
101 }
