30 /**
31  * \file
32  * Orchestra: a slotframe dedicated to unicast data transmission. Designed for
33  * RPL storing mode only, as this is based on the knowledge of the children (and parent).
34  * If receiver-based:
35  * Nodes listen at a timeslot defined as hash(MAC) % ORCHESTRA_SB_UNICAST_PERIOD
36  * Nodes transmit at: for each nbr in RPL children and RPL preferred parent,
37  * hash(nbr.MAC) % ORCHESTRA_SB_UNICAST_PERIOD
38  * If sender-based: the opposite
39  *
40  * \author Simon Duquennoy <simonduq@sics.se>
41  */

69 /*---------------------------------------------------------------------------*/
70 static uint16_t
71 get_node_timeslot(const linkaddr_t *addr)
72 {
73  if(addr != NULL && ORCHESTRA_UNICAST_PERIOD > 0) {
74  return ORCHESTRA_LINKADDR_HASH(addr) % ORCHESTRA_UNICAST_PERIOD;
75  } else {
76  return 0xffff;
77  }
78 }

80 static int
81 neighbor_has_uc_link(const linkaddr_t *linkaddr)
82 {
83  if(linkaddr != NULL && !linkaddr_cmp(linkaddr, &linkaddr_null)) {
84  if((orchestra_parent_knows_us || !ORCHESTRA_UNICAST_SENDER_BASED)
85  && linkaddr_cmp(&orchestra_parent_linkaddr, linkaddr)) {
86  return 1;
87  }
88  if(nbr_table_get_from_lladdr(nbr_routes, (linkaddr_t *)linkaddr) != NULL) {
89  return 1;
90  }
91  }
92  return 0;
93 }
94 /*---------------------------------------------------------------------------*/
95 static void
96 add_uc_link(const linkaddr_t *linkaddr)
97 {
98  if(linkaddr != NULL) {
99  uint16_t timeslot = get_node_timeslot(linkaddr);
100  uint8_t link_options = ORCHESTRA_UNICAST_SENDER_BASED ? LINK_OPTION_RX : LINK_OPTION_TX | UNICAST_SLOT_SHARED_FLAG;
101 
102  if(timeslot == get_node_timeslot(&linkaddr_node_addr)) {
103  /* This is also our timeslot, add necessary flags */
104  link_options |= ORCHESTRA_UNICAST_SENDER_BASED ? LINK_OPTION_TX | UNICAST_SLOT_SHARED_FLAG: LINK_OPTION_RX;
105  }
106 
107  /* Add/update link */
108  tsch_schedule_add_link(sf_unicast, link_options, LINK_TYPE_NORMAL, &tsch_broadcast_address,
109  timeslot, channel_offset);
110  }
111 }
112 /*---------------------------------------------------------------------------*/
113 static void
114 remove_uc_link(const linkaddr_t *linkaddr)
115 {
116  uint16_t timeslot;
117  struct tsch_link *l;
118 
119  if(linkaddr == NULL) {
120  return;
121  }
122 
123  timeslot = get_node_timeslot(linkaddr);
124  l = tsch_schedule_get_link_by_timeslot(sf_unicast, timeslot);
125  if(l == NULL) {
126  return;
127  }
128  /* Does our current parent need this timeslot? */
129  if(timeslot == get_node_timeslot(&orchestra_parent_linkaddr)) {
130  /* Yes, this timeslot is being used, return */
131  return;
132  }
133  /* Does any other child need this timeslot?
134  * (lookup all route next hops) */
135  nbr_table_item_t *item = nbr_table_head(nbr_routes);
136  while(item != NULL) {
137  linkaddr_t *addr = nbr_table_get_lladdr(nbr_routes, item);
138  if(timeslot == get_node_timeslot(addr)) {
139  /* Yes, this timeslot is being used, return */
140  return;
141  }
142  item = nbr_table_next(nbr_routes, item);
143  }
144 
145  /* Do we need this timeslot? */
146  if(timeslot == get_node_timeslot(&linkaddr_node_addr)) {
147  /* This is our link, keep it but update the link options */
148  uint8_t link_options = ORCHESTRA_UNICAST_SENDER_BASED ? LINK_OPTION_TX | UNICAST_SLOT_SHARED_FLAG: LINK_OPTION_RX;
149  tsch_schedule_add_link(sf_unicast, link_options, LINK_TYPE_NORMAL, &tsch_broadcast_address,
150  timeslot, channel_offset);
151  } else {
152  /* Remove link */
153  tsch_schedule_remove_link(sf_unicast, l);
154  }
155 }

169 static int
170 select_packet(uint16_t *slotframe, uint16_t *timeslot)
171 {
172  /* Select data packets we have a unicast link to */
173  const linkaddr_t *dest = packetbuf_addr(PACKETBUF_ADDR_RECEIVER);
174  if(packetbuf_attr(PACKETBUF_ATTR_FRAME_TYPE) == FRAME802154_DATAFRAME
175  && neighbor_has_uc_link(dest)) {
176  if(slotframe != NULL) {
177  *slotframe = slotframe_handle;
178  }
179  if(timeslot != NULL) {
180  *timeslot = ORCHESTRA_UNICAST_SENDER_BASED ? get_node_timeslot(&linkaddr_node_addr) : get_node_timeslot(dest);
181  }
182  return 1;
183  }
184  return 0;
185 }

187 static void
188 new_time_source(const struct tsch_neighbor *old, const struct tsch_neighbor *new)
189 {
190  if(new != old) {
191  const linkaddr_t *old_addr = old != NULL ? &old->addr : NULL;
192  const linkaddr_t *new_addr = new != NULL ? &new->addr : NULL;
193  if(new_addr != NULL) {
194  linkaddr_copy(&orchestra_parent_linkaddr, new_addr);
195  } else {
196  linkaddr_copy(&orchestra_parent_linkaddr, &linkaddr_null);
197  }
198  remove_uc_link(old_addr);
199  add_uc_link(new_addr);
200  }
201 }

203 static void
204 init(uint16_t sf_handle)
205 {
206  slotframe_handle = sf_handle;
207  channel_offset = sf_handle;
208  /* Slotframe for unicast transmissions */
209  sf_unicast = tsch_schedule_add_slotframe(slotframe_handle, ORCHESTRA_UNICAST_PERIOD);
210  uint16_t timeslot = get_node_timeslot(&linkaddr_node_addr);
211  tsch_schedule_add_link(sf_unicast,
212  ORCHESTRA_UNICAST_SENDER_BASED ? LINK_OPTION_TX | UNICAST_SLOT_SHARED_FLAG: LINK_OPTION_RX,
213  LINK_TYPE_NORMAL, &tsch_broadcast_address,
214  timeslot, channel_offset);
215 }
