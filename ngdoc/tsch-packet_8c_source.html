64 /*
65  * We use a local packetbuf_attr array to collect necessary frame settings to
66  * create an EACK because EACK is generated in the interrupt context where
67  * packetbuf and packetbuf_attrs[] may be in use for another purpose.
68  *
69  * We have accessors of eackbuf_attrs: tsch_packet_eackbuf_set_attr() and
70  * tsch_packet_eackbuf_attr(). For some platform, they might need to be
71  * implemented as inline functions. However, for now, we don't provide the
72  * inline option. Such an optimization is left to the compiler for a target
73  * platform.
74  */
75 static struct packetbuf_attr eackbuf_attrs[PACKETBUF_NUM_ATTRS];
76 
77 /*---------------------------------------------------------------------------*/
78 static int
79 tsch_packet_eackbuf_set_attr(uint8_t type, const packetbuf_attr_t val)
80 {
81  eackbuf_attrs[type].val = val;
82  return 1;
83 }

92 /* Construct enhanced ACK packet and return ACK length */
93 int
94 tsch_packet_create_eack(uint8_t *buf, uint16_t buf_len,
95  const linkaddr_t *dest_addr, uint8_t seqno,
96  int16_t drift, int nack)
97 {
98  frame802154_t params;
99  struct ieee802154_ies ies;
100  int hdr_len;
101  int ack_len;
102 
103  if(buf == NULL) {
104  return -1;
105  }
106 
107  memset(eackbuf_attrs, 0, sizeof(eackbuf_attrs));
108 
109  tsch_packet_eackbuf_set_attr(PACKETBUF_ATTR_FRAME_TYPE, FRAME802154_ACKFRAME);
110  tsch_packet_eackbuf_set_attr(PACKETBUF_ATTR_MAC_METADATA, 1);
111  tsch_packet_eackbuf_set_attr(PACKETBUF_ATTR_MAC_SEQNO, seqno);
112 
113  tsch_packet_eackbuf_set_attr(PACKETBUF_ATTR_MAC_NO_DEST_ADDR, 1);
114 #if TSCH_PACKET_EACK_WITH_DEST_ADDR
115  if(dest_addr != NULL) {
116  tsch_packet_eackbuf_set_attr(PACKETBUF_ATTR_MAC_NO_DEST_ADDR, 0);
117  linkaddr_copy((linkaddr_t *)&params.dest_addr, dest_addr);
118  }
119 #endif
120 
121  tsch_packet_eackbuf_set_attr(PACKETBUF_ATTR_MAC_NO_SRC_ADDR, 1);
122 #if TSCH_PACKET_EACK_WITH_SRC_ADDR
123  tsch_packet_eackbuf_set_attr(PACKETBUF_ATTR_MAC_NO_SRC_ADDR, 0);
124  linkaddr_copy((linkaddr_t *)&params.src_addr, &linkaddr_node_addr);
125 #endif
126 
127 #if LLSEC802154_ENABLED
128  if(tsch_is_pan_secured) {
129  tsch_packet_eackbuf_set_attr(PACKETBUF_ATTR_SECURITY_LEVEL,
130  TSCH_SECURITY_KEY_SEC_LEVEL_ACK);
131  tsch_packet_eackbuf_set_attr(PACKETBUF_ATTR_KEY_ID_MODE,
132  FRAME802154_1_BYTE_KEY_ID_MODE);
133  tsch_packet_eackbuf_set_attr(PACKETBUF_ATTR_KEY_INDEX,
134  TSCH_SECURITY_KEY_INDEX_ACK);
135  }
136 #endif /* LLSEC802154_ENABLED */
137 
138  framer_802154_setup_params(tsch_packet_eackbuf_attr, 0, &params);
139  hdr_len = frame802154_hdrlen(&params);
140 
141  memset(buf, 0, buf_len);
142 
143  /* Setup IE timesync */
144  memset(&ies, 0, sizeof(ies));
145  ies.ie_time_correction = drift;
146  ies.ie_is_nack = nack;
147 
148  ack_len =
149  frame80215e_create_ie_header_ack_nack_time_correction(buf + hdr_len,
150  buf_len - hdr_len, &ies);
151  if(ack_len < 0) {
152  return -1;
153  }
154  ack_len += hdr_len;
155 
156  frame802154_create(&params, buf);
157 
158  return ack_len;
159 }

161 /* Parse enhanced ACK packet, extract drift and nack */
162 int
163 tsch_packet_parse_eack(const uint8_t *buf, int buf_size,
164  uint8_t seqno, frame802154_t *frame, struct ieee802154_ies *ies, uint8_t *hdr_len)
165 {
166  uint8_t curr_len = 0;
167  int ret;
168  linkaddr_t dest;
169 
170  if(frame == NULL || buf_size < 0) {
171  return 0;
172  }
173  /* Parse 802.15.4-2006 frame, i.e. all fields before Information Elements */
174  if((ret = frame802154_parse((uint8_t *)buf, buf_size, frame)) < 3) {
175  return 0;
176  }
177  if(hdr_len != NULL) {
178  *hdr_len = ret;
179  }
180  curr_len += ret;
181 
182  /* Check seqno */
183  if(seqno != frame->seq) {
184  return 0;
185  }
186 
187  /* Check destination PAN ID */
188  if(frame802154_check_dest_panid(frame) == 0) {
189  return 0;
190  }
191 
192  /* Check destination address (if any) */
193  if(frame802154_extract_linkaddr(frame, NULL, &dest) == 0 ||
194  (!linkaddr_cmp(&dest, &linkaddr_node_addr)
195  && !linkaddr_cmp(&dest, &linkaddr_null))) {
196  return 0;
197  }
198 
199  if(ies != NULL) {
200  memset(ies, 0, sizeof(struct ieee802154_ies));
201  }
202 
203  if(frame->fcf.ie_list_present) {
204  int mic_len = 0;
205 #if LLSEC802154_ENABLED
206  /* Check if there is space for the security MIC (if any) */
207  mic_len = tsch_security_mic_len(frame);
208  if(buf_size < curr_len + mic_len) {
209  return 0;
210  }
211 #endif /* LLSEC802154_ENABLED */
212  /* Parse information elements. We need to substract the MIC length, as the exact payload len is needed while parsing */
213  if((ret = frame802154e_parse_information_elements(buf + curr_len, buf_size - curr_len - mic_len, ies)) == -1) {
214  return 0;
215  }
216  curr_len += ret;
217  }
218 
219  if(hdr_len != NULL) {
220  *hdr_len += ies->ie_payload_ie_offset;
221  }
222 
223  return curr_len;
224 }

226 /* Create an EB packet */
227 int
228 tsch_packet_create_eb(uint8_t *hdr_len, uint8_t *tsch_sync_ie_offset)
229 {
230  struct ieee802154_ies ies;
231  uint8_t *p;
232  int ie_len;
233  const uint16_t payload_ie_hdr_len = 2;
234 
235  packetbuf_clear();
236 
237  /* Prepare Information Elements for inclusion in the EB */
238  memset(&ies, 0, sizeof(ies));
239 
240  /* Add TSCH timeslot timing IE. */
241 #if TSCH_PACKET_EB_WITH_TIMESLOT_TIMING
242  {
243  int i;
244  ies.ie_tsch_timeslot_id = 1;
245  for(i = 0; i < tsch_ts_elements_count; i++) {
246  ies.ie_tsch_timeslot[i] = RTIMERTICKS_TO_US(tsch_timing[i]);
247  }
248  }
249 #endif /* TSCH_PACKET_EB_WITH_TIMESLOT_TIMING */
250 
251  /* Add TSCH hopping sequence IE */
252 #if TSCH_PACKET_EB_WITH_HOPPING_SEQUENCE
253  if(tsch_hopping_sequence_length.val <= sizeof(ies.ie_hopping_sequence_list)) {
254  ies.ie_channel_hopping_sequence_id = 1;
255  ies.ie_hopping_sequence_len = tsch_hopping_sequence_length.val;
256  memcpy(ies.ie_hopping_sequence_list, tsch_hopping_sequence,
257  ies.ie_hopping_sequence_len);
258  }
259 #endif /* TSCH_PACKET_EB_WITH_HOPPING_SEQUENCE */
260 
261  /* Add Slotframe and Link IE */
262 #if TSCH_PACKET_EB_WITH_SLOTFRAME_AND_LINK
263  {
264  /* Send slotframe 0 with link at timeslot 0 */
265  struct tsch_slotframe *sf0 = tsch_schedule_get_slotframe_by_handle(0);
266  struct tsch_link *link0 = tsch_schedule_get_link_by_timeslot(sf0, 0);
267  if(sf0 && link0) {
268  ies.ie_tsch_slotframe_and_link.num_slotframes = 1;
269  ies.ie_tsch_slotframe_and_link.slotframe_handle = sf0->handle;
270  ies.ie_tsch_slotframe_and_link.slotframe_size = sf0->size.val;
271  ies.ie_tsch_slotframe_and_link.num_links = 1;
272  ies.ie_tsch_slotframe_and_link.links[0].timeslot = link0->timeslot;
273  ies.ie_tsch_slotframe_and_link.links[0].channel_offset =
274  link0->channel_offset;
275  ies.ie_tsch_slotframe_and_link.links[0].link_options =
276  link0->link_options;
277  }
278  }
279 #endif /* TSCH_PACKET_EB_WITH_SLOTFRAME_AND_LINK */
280 
281  p = packetbuf_dataptr();
282 
283  ie_len = frame80215e_create_ie_tsch_synchronization(p,
284  packetbuf_remaininglen(),
285  &ies);
286  if(ie_len < 0) {
287  return -1;
288  }
289  p += ie_len;
290  packetbuf_set_datalen(packetbuf_datalen() + ie_len);
291 
292  ie_len = frame80215e_create_ie_tsch_timeslot(p,
293  packetbuf_remaininglen(),
294  &ies);
295  if(ie_len < 0) {
296  return -1;
297  }
298  p += ie_len;
299  packetbuf_set_datalen(packetbuf_datalen() + ie_len);
300 
301  ie_len = frame80215e_create_ie_tsch_channel_hopping_sequence(p,
302  packetbuf_remaininglen(),
303  &ies);
304  if(ie_len < 0) {
305  return -1;
306  }
307  p += ie_len;
308  packetbuf_set_datalen(packetbuf_datalen() + ie_len);
309 
310  ie_len = frame80215e_create_ie_tsch_slotframe_and_link(p,
311  packetbuf_remaininglen(),
312  &ies);
313  if(ie_len < 0) {
314  return -1;
315  }
316  p += ie_len;
317  packetbuf_set_datalen(packetbuf_datalen() + ie_len);
318 
319 #if 0
320  /* Payload IE list termination: optional */
321  ie_len = frame80215e_create_ie_payload_list_termination(p,
322  packetbuf_remaininglen(),
323  &ies);
324  if(ie_len < 0) {
325  return -1;
326  }
327  p += ie_len;
328  packetbuf_set_datalen(packetbuf_datalen() + ie_len);
329 #endif
330 
331  ies.ie_mlme_len = packetbuf_datalen();
332 
333  /* make room for Payload IE header */
334  memmove((uint8_t *)packetbuf_dataptr() + payload_ie_hdr_len,
335  packetbuf_dataptr(), packetbuf_datalen());
336  packetbuf_set_datalen(packetbuf_datalen() + payload_ie_hdr_len);
337  ie_len = frame80215e_create_ie_mlme(packetbuf_dataptr(),
338  packetbuf_remaininglen(),
339  &ies);
340  if(ie_len < 0) {
341  return -1;
342  }
343 
344  /* allocate space for Header Termination IE, the size of which is 2 octets */
345  packetbuf_hdralloc(2);
346  ie_len = frame80215e_create_ie_header_list_termination_1(packetbuf_hdrptr(),
347  packetbuf_remaininglen(),
348  &ies);
349  if(ie_len < 0) {
350  return -1;
351  }
352 
353  packetbuf_set_attr(PACKETBUF_ATTR_FRAME_TYPE, FRAME802154_BEACONFRAME);
354  packetbuf_set_attr(PACKETBUF_ATTR_MAC_METADATA, 1);
355 
356  packetbuf_set_addr(PACKETBUF_ADDR_SENDER, &linkaddr_node_addr);
357  packetbuf_set_addr(PACKETBUF_ADDR_RECEIVER, &tsch_eb_address);
358 
359 #if LLSEC802154_ENABLED
360  if(tsch_is_pan_secured) {
361  packetbuf_set_attr(PACKETBUF_ATTR_SECURITY_LEVEL,
362  TSCH_SECURITY_KEY_SEC_LEVEL_EB);
363  packetbuf_set_attr(PACKETBUF_ATTR_KEY_ID_MODE,
364  FRAME802154_1_BYTE_KEY_ID_MODE);
365  packetbuf_set_attr(PACKETBUF_ATTR_KEY_INDEX,
366  TSCH_SECURITY_KEY_INDEX_EB);
367  }
368 #endif /* LLSEC802154_ENABLED */
369 
370  if(NETSTACK_FRAMER.create() < 0) {
371  return -1;
372  }
373 
374  if(hdr_len != NULL) {
375  *hdr_len = packetbuf_hdrlen();
376  }
377 
378  /*
379  * Save the offset of the TSCH Synchronization IE, which is expected to be
380  * located just after the Payload IE header, needed to update ASN and join
381  * priority before sending.
382  */
383  if(tsch_sync_ie_offset != NULL) {
384  *tsch_sync_ie_offset = packetbuf_hdrlen() + payload_ie_hdr_len;
385  }
386 
387  return packetbuf_totlen();
388 }

401 /* Parse a IEEE 802.15.4e TSCH Enhanced Beacon (EB) */
402 int
403 tsch_packet_parse_eb(const uint8_t *buf, int buf_size,
404  frame802154_t *frame, struct ieee802154_ies *ies, uint8_t *hdr_len, int frame_without_mic)
405 {
406  uint8_t curr_len = 0;
407  int ret;
408 
409  if(frame == NULL || buf_size < 0) {
410  return 0;
411  }
412 
413  /* Parse 802.15.4-2006 frame, i.e. all fields before Information Elements */
414  if((ret = frame802154_parse((uint8_t *)buf, buf_size, frame)) == 0) {
415  LOG_ERR("! parse_eb: failed to parse frame\n");
416  return 0;
417  }
418 
419  if(frame->fcf.frame_version < FRAME802154_IEEE802154_2015
420  || frame->fcf.frame_type != FRAME802154_BEACONFRAME) {
421  LOG_INFO("! parse_eb: frame is not a valid TSCH beacon. Frame version %u, type %u, FCF %02x %02x\n",
422  frame->fcf.frame_version, frame->fcf.frame_type, buf[0], buf[1]);
423  LOG_INFO("! parse_eb: frame was from 0x%x/", frame->src_pid);
424  LOG_INFO_LLADDR((const linkaddr_t *)&frame->src_addr);
425  LOG_INFO_(" to 0x%x/", frame->dest_pid);
426  LOG_INFO_LLADDR((const linkaddr_t *)&frame->dest_addr);
427  LOG_INFO_("\n");
428  return 0;
429  }
430 
431  if(hdr_len != NULL) {
432  *hdr_len = ret;
433  }
434  curr_len += ret;
435 
436  if(ies != NULL) {
437  memset(ies, 0, sizeof(struct ieee802154_ies));
438  ies->ie_join_priority = 0xff; /* Use max value in case the Beacon does not include a join priority */
439  }
440  if(frame->fcf.ie_list_present) {
441  /* Calculate space needed for the security MIC, if any, before attempting to parse IEs */
442  int mic_len = 0;
443 #if LLSEC802154_ENABLED
444  if(!frame_without_mic) {
445  mic_len = tsch_security_mic_len(frame);
446  if(buf_size < curr_len + mic_len) {
447  return 0;
448  }
449  }
450 #endif /* LLSEC802154_ENABLED */
451 
452  /* Parse information elements. We need to substract the MIC length, as the exact payload len is needed while parsing */
453  if((ret = frame802154e_parse_information_elements(buf + curr_len, buf_size - curr_len - mic_len, ies)) == -1) {
454  LOG_ERR("! parse_eb: failed to parse IEs\n");
455  return 0;
456  }
457  curr_len += ret;
458  }
459 
460  if(hdr_len != NULL) {
461  *hdr_len += ies->ie_payload_ie_offset;
462  }
463 
464  return curr_len;
465 }
