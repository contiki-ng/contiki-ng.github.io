65 mt_start(struct mt_thread *thread, void (* function)(void *), void *data)
66 {
67  /* Call the architecture dependant function to set up the processor
68  stack with the correct parameters. */
69  mtarch_start(&thread->thread, function, data);
70 
71  thread->state = MT_STATE_STARTED;
72 }
73 /*--------------------------------------------------------------------------*/
74 void
75 mt_exec(struct mt_thread *thread)
76 {
77  if(thread->state == MT_STATE_STARTED) {
78  current = thread;
79  /* Switch context to the thread. The function call will not return
80  until the the thread has yielded, or is preempted. */
81  mtarch_exec(&thread->thread);
82  }
83 }
84 /*--------------------------------------------------------------------------*/
85 void
86 mt_yield(void)
87 {
88  mtarch_pstop();
89  /* This function is called from the running thread, and we call the
90  switch function in order to switch the thread to the main Contiki
91  program instead. For us, the switch function will not return
92  until the next time we are scheduled to run. */
93  mtarch_yield();
94 }
95 /*--------------------------------------------------------------------------*/
96 void
97 mt_exit(void)
98 {
99  mtarch_pstop();
100  current->state = MT_STATE_EXITED;
101  mtarch_yield();
102 }
