66 static const uip_ipaddr_t *
67 get_global_address(void)
68 {
69  int i;
70  uint8_t state;
71  uip_ipaddr_t *ipaddr = NULL;
72 
73  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
74  state = uip_ds6_if.addr_list[i].state;
75  if(uip_ds6_if.addr_list[i].isused &&
76  state == ADDR_PREFERRED &&
77  !uip_is_addr_linklocal(&uip_ds6_if.addr_list[i].ipaddr)) {
78  ipaddr = &uip_ds6_if.addr_list[i].ipaddr;
79  }
80  }
81  return ipaddr;
82 }

84 static void
85 create_dag_callback(void *ptr)
86 {
87  const uip_ipaddr_t *root, *ipaddr;
88 
89  root = dag_root();
90  ipaddr = get_global_address();
91 
92  if(root == NULL || uip_ipaddr_cmp(root, ipaddr)) {
93  /* The RPL network we are joining is one that we created, so we
94  become root. */
95  if(to_become_root) {
96  rpl_dag_root_init_dag_immediately();
97  to_become_root = 0;
98  }
99  } else {
100  rpl_dag_t *dag;
101 
102  dag = rpl_get_any_dag();
103 #if DEBUG
104  printf("Found a network we did not create\n");
105  printf("version %d grounded %d preference %d used %d joined %d rank %d\n",
106  dag->version, dag->grounded,
107  dag->preference, dag->used,
108  dag->joined, dag->rank);
109 #endif /* DEBUG */
110 
111  /* We found a RPL network that we did not create so we just join
112  it without becoming root. But if the network has an infinite
113  rank, we assume the network has broken, and we become the new
114  root of the network. */
115 
116  if(dag->rank == RPL_INFINITE_RANK) {
117  if(to_become_root) {
118  rpl_dag_root_init_dag_immediately();
119  to_become_root = 0;
120  }
121  }
122 
123  /* Try again after the grace period */
124  ctimer_set(&c, RPL_DAG_GRACE_PERIOD, create_dag_callback, NULL);
125  }
126 }

145 static void
146 set_global_address(uip_ipaddr_t *prefix, uip_ipaddr_t *iid)
147 {
148  static uip_ipaddr_t root_ipaddr;
149  int i;
150  uint8_t state;
151 
152  /* Assign a unique local address (RFC4193,
153  http://tools.ietf.org/html/rfc4193). */
154  if(prefix == NULL) {
155  uip_ip6addr(&root_ipaddr, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0, 0, 0, 0);
156  } else {
157  memcpy(&root_ipaddr, prefix, 8);
158  }
159  if(iid == NULL) {
160  uip_ds6_set_addr_iid(&root_ipaddr, &uip_lladdr);
161  } else {
162  memcpy(((uint8_t*)&root_ipaddr) + 8, ((uint8_t*)iid) + 8, 8);
163  }
164 
165  uip_ds6_addr_add(&root_ipaddr, 0, ADDR_AUTOCONF);
166 
167  printf("IPv6 addresses: ");
168  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
169  state = uip_ds6_if.addr_list[i].state;
170  if(uip_ds6_if.addr_list[i].isused &&
171  (state == ADDR_TENTATIVE || state == ADDR_PREFERRED)) {
172  uip_debug_ipaddr_print(&uip_ds6_if.addr_list[i].ipaddr);
173  printf("\n");
174  }
175  }
176 }

179 rpl_dag_root_init(uip_ipaddr_t *prefix, uip_ipaddr_t *iid)
180 {
181  static uint8_t initialized = 0;
182 
183  if(!initialized) {
184  to_become_root = 0;
185  set_global_address(prefix, iid);

189  initialized = 1;

193 int
194 rpl_dag_root_init_dag_immediately(void)
195 {
196  struct uip_ds6_addr *root_if;
197  int i;
198  uint8_t state;
199  uip_ipaddr_t *ipaddr = NULL;
200 
201  rpl_dag_root_init(NULL, NULL);
202 
203  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
204  state = uip_ds6_if.addr_list[i].state;
205  if(uip_ds6_if.addr_list[i].isused &&
206  state == ADDR_PREFERRED &&
207  !uip_is_addr_linklocal(&uip_ds6_if.addr_list[i].ipaddr)) {
208  ipaddr = &uip_ds6_if.addr_list[i].ipaddr;
209  }
210  }
211 
212  if(ipaddr != NULL) {
213  root_if = uip_ds6_addr_lookup(ipaddr);
214  if(root_if != NULL) {
215  rpl_dag_t *dag;
216  uip_ipaddr_t prefix;
217 
218  rpl_set_root(RPL_DEFAULT_INSTANCE, ipaddr);
219  dag = rpl_get_any_dag();
220 
221  /* If there are routes in this dag, we remove them all as we are
222  from now on the new dag root and the old routes are wrong */
223  if(RPL_IS_STORING(dag->instance)) {
224  rpl_remove_routes(dag);
225  }
226  if(dag->instance != NULL &&
227  dag->instance->def_route != NULL) {
228  uip_ds6_defrt_rm(dag->instance->def_route);
229  dag->instance->def_route = NULL;
230  }
231 
232  uip_ip6addr(&prefix, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0, 0, 0, 0);
233  rpl_set_prefix(dag, &prefix, 64);
234  PRINTF("rpl_dag_root_init_dag: created a new RPL dag\n");
235  return 0;
236  } else {
237  PRINTF("rpl_dag_root_init_dag: failed to create a new RPL DAG\n");
238  return -1;
239  }
240  } else {
241  PRINTF("rpl_dag_root_init_dag: failed to create a new RPL DAG, no preferred IP address found\n");
242  return -2;
243  }
244 }
