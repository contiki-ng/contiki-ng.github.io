76 static void
77 fade(unsigned char l)
78 {
79  volatile int i;
80  int k, j;
81  for(k = 0; k < 800; ++k) {
82  j = k > 400 ? 800 - k : k;
83 
84  leds_on(l);
85  for(i = 0; i < j; ++i) {
86  asm("nop");
87  }
88  leds_off(l);
89  for(i = 0; i < 400 - j; ++i) {
90  asm("nop");
91  }
92  }
93 }

95 static void
96 set_rf_params(void)
97 {
98  uint16_t short_addr;
99  uint8_t ext_addr[8];
100 
101  ieee_addr_cpy_to(ext_addr, 8);
102 
103  short_addr = ext_addr[7];
104  short_addr |= ext_addr[6] << 8;
105 
106  /* Populate linkaddr_node_addr. Maintain endianness */
107  memcpy(&linkaddr_node_addr, &ext_addr[8 - LINKADDR_SIZE], LINKADDR_SIZE);
108 
109  NETSTACK_RADIO.set_value(RADIO_PARAM_PAN_ID, IEEE802154_PANID);
110  NETSTACK_RADIO.set_value(RADIO_PARAM_16BIT_ADDR, short_addr);
111  NETSTACK_RADIO.set_value(RADIO_PARAM_CHANNEL, CC2538_RF_CHANNEL);
112  NETSTACK_RADIO.set_object(RADIO_PARAM_64BIT_ADDR, ext_addr, 8);
113 }

124 void
125 platform_init_stage_two()
126 {
127  button_sensor_init();
128 
129  /*
130  * Character I/O Initialisation.
131  * When the UART receives a character it will call serial_line_input_byte to
132  * notify the core. The same applies for the USB driver.
133  *
134  * If slip-arch is also linked in afterwards (e.g. if we are a border router)
135  * it will overwrite one of the two peripheral input callbacks. Characters
136  * received over the relevant peripheral will be handled by
137  * slip_input_byte instead
138  */
139 #if UART_CONF_ENABLE
140  uart_init(0);
141  uart_init(1);
142  uart_set_input(SERIAL_LINE_CONF_UART, serial_line_input_byte);
143 #endif
144 
145 #if USB_SERIAL_CONF_ENABLE
146  usb_serial_init();
147  usb_serial_set_input(serial_line_input_byte);
148 #endif
149 
150  serial_line_init();
151 
152  /* Initialise the H/W RNG engine. */
153  random_init(0);
154 
155  udma_init();
156 
157 #if CRYPTO_CONF_INIT
158  crypto_init();
159  crypto_disable();
160 #endif
161 
162  set_rf_params();
163 
164  INTERRUPTS_ENABLE();
165 
166  fade(LEDS_GREEN);
167 }
