48 static struct etimer etimer;
49 
50 PROCESS(burn_process, "Burn node id");
51 AUTOSTART_PROCESSES(&burn_process);
52 /*---------------------------------------------------------------------------*/
53 PROCESS_THREAD(burn_process, ev, data)
54 {
55  PROCESS_BEGIN();
56 
57  etimer_set(&etimer, 5*CLOCK_SECOND);
58  PROCESS_WAIT_UNTIL(etimer_expired(&etimer));
59 
60  watchdog_stop();
61  leds_on(LEDS_RED);
62 #if NODEID
63  printf("Burning node id %d\n", NODEID);
64  node_id_burn(NODEID);
65  leds_on(LEDS_BLUE);
66  node_id_restore();
67  printf("Restored node id %d\n", node_id);
68 #else
69 #error "burn-nodeid must be compiled with nodeid=<the ID of the node>"
70  node_id_restore();
71  printf("Restored node id %d\n", node_id);
72 #endif
73  leds_off(LEDS_RED + LEDS_BLUE);
74  watchdog_start();
75  while(1) {
76  PROCESS_WAIT_EVENT();
77  }
78  PROCESS_END();
79 }
