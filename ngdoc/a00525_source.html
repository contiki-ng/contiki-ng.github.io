147 set_lladdr(void)
148 {
149  linkaddr_t addr;
150  int i;
151 
152  memset(&addr, 0, sizeof(linkaddr_t));
153 #if NETSTACK_CONF_WITH_IPV6
154  for(i = 0; i < sizeof(uip_lladdr.addr); i += 2) {
155  addr.u8[i + 1] = node_id & 0xff;
156  addr.u8[i + 0] = node_id >> 8;
157  }
158 #else /* NETSTACK_CONF_WITH_IPV6 */
159  addr.u8[0] = node_id & 0xff;
160  addr.u8[1] = node_id >> 8;
161 #endif /* NETSTACK_CONF_WITH_IPV6 */
162  linkaddr_set_node_addr(&addr);
163  printf("Contiki started with address ");
164  for(i = 0; i < sizeof(addr.u8) - 1; i++) {
165  printf("%d.", addr.u8[i]);
166  }
167  printf("%d\n", addr.u8[i]);

266 static void
267 process_run_thread_loop(void *data)
268 {
269  /* Yield once during bootup */
270  simProcessRunValue = 1;
271  cooja_mt_yield();
272 
273  contiki_init();
274 
275  while(1)
276  {
277  simProcessRunValue = process_run();
278  while(simProcessRunValue-- > 0) {
279  process_run();
280  }
281  simProcessRunValue = process_nevents();
282 
283  /* Check if we must stay awake */
284  if(simDontFallAsleep) {
285  simDontFallAsleep=0;
286  simProcessRunValue = 1;
287  }
288 
289  /* Return to COOJA */
290  cooja_mt_yield();
291  }
292 }

367 /**
368  * \brief Let mote execute one "block" of code (tick mote).
369  * \param env JNI Environment interface pointer
370  * \param obj unused
371  *
372  * Let mote defined by the active contiki processes and current
373  * process memory execute some program code. This code must not block
374  * or else this function will never return. A typical contiki
375  * process will return when it executes PROCESS_WAIT..() statements.
376  *
377  * Before the control is left to contiki processes, any messages
378  * from the Java part are handled. These may for example be
379  * incoming network data. After the contiki processes return control,
380  * messages to the Java part are also handled (those which may need
381  * special attention).
382  *
383  * This is a JNI function and should only be called via the
384  * responsible Java part (MoteType.java).
385  */
386 JNIEXPORT void JNICALL
387 Java_org_contikios_cooja_corecomm_CLASSNAME_tick(JNIEnv *env, jobject obj)
388 {
389  simProcessRunValue = 0;
390 
391  /* Let all simulation interfaces act first */
392  doActionsBeforeTick();
393 
394  /* Poll etimer process */
395  if(etimer_pending()) {
396  etimer_request_poll();
397  }
398 
399  /* Let rtimers run.
400  * Sets simProcessRunValue */
401  cooja_mt_exec(&rtimer_thread);
402 
403  if(simProcessRunValue == 0) {
404  /* Rtimers done: Let Contiki handle a few events.
405  * Sets simProcessRunValue */
406  cooja_mt_exec(&process_run_thread);
407  }
408 
409  /* Let all simulation interfaces act before returning to java */
410  doActionsAfterTick();
411 
412  /* Do we have any pending timers */
413  simEtimerPending = etimer_pending();
414 
415  /* Save nearest expiration time */
416  simEtimerNextExpirationTime = etimer_next_expiration_time();
417 
418 }
