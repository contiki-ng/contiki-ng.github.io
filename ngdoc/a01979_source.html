46 /* The maximum number of outgoing packets towards each neighbor
47  * Must be power of two to enable atomic ringbuf operations.
48  * Note: the total number of outgoing packets in the system (for
49  * all neighbors) is defined via QUEUEBUF_CONF_NUM */

54  * (in the range [4;256]) */
55 #if QUEUEBUF_CONF_NUM <= 4

57 #elif QUEUEBUF_CONF_NUM <= 8
58 #define TSCH_QUEUE_NUM_PER_NEIGHBOR 8
59 #elif QUEUEBUF_CONF_NUM <= 16
60 #define TSCH_QUEUE_NUM_PER_NEIGHBOR 16
61 #elif QUEUEBUF_CONF_NUM <= 32
62 #define TSCH_QUEUE_NUM_PER_NEIGHBOR 32
63 #elif QUEUEBUF_CONF_NUM <= 64
64 #define TSCH_QUEUE_NUM_PER_NEIGHBOR 64
65 #elif QUEUEBUF_CONF_NUM <= 128

115 /* TSCH packet information */
116 struct tsch_packet {
117  struct queuebuf *qb; /* pointer to the queuebuf to be sent */
118  mac_callback_t sent; /* callback for this packet */
119  void *ptr; /* MAC callback parameter */
120  uint8_t transmissions; /* #transmissions performed for this packet */
121  uint8_t ret; /* status -- MAC return code */
122  uint8_t header_len; /* length of header and header IEs (needed for link-layer security) */
123  uint8_t tsch_sync_ie_offset; /* Offset within the frame used for quick update of EB ASN and join priority */
124 };

126 /* TSCH neighbor information */
127 struct tsch_neighbor {
128  /* Neighbors are stored as a list: "next" must be the first field */
129  struct tsch_neighbor *next;
130  linkaddr_t addr; /* MAC address of the neighbor */
131  uint8_t is_broadcast; /* is this neighbor a virtual neighbor used for broadcast (of data packets or EBs) */
132  uint8_t is_time_source; /* is this neighbor a time source? */
133  uint8_t backoff_exponent; /* CSMA backoff exponent */
134  uint8_t backoff_window; /* CSMA backoff window (number of slots to skip) */
135  uint8_t last_backoff_window; /* Last CSMA backoff window */
136  uint8_t tx_links_count; /* How many links do we have to this neighbor? */
137  uint8_t dedicated_tx_links_count; /* How many dedicated links do we have to this neighbor? */
138  /* Array for the ringbuf. Contains pointers to packets.
139  * Its size must be a power of two to allow for atomic put */
140  struct tsch_packet *tx_array[TSCH_QUEUE_NUM_PER_NEIGHBOR];
141  /* Circular buffer of pointers to packet. */
142  struct ringbufindex tx_ringbuf;
143 };
