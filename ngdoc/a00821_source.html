84  return ((addr1.s6_addr32[0]==addr2.s6_addr32[0]) &&

99 uint8_t
100 sprint_ip6addr(struct in6_addr addr, char * result)
101 {
102  unsigned char i = 0;
103  unsigned char zerocnt = 0;
104  unsigned char numprinted = 0;
105  char * starting = result;
106 
107  *result++='[';
108  while (numprinted < 8) {
109  if ((addr.s6_addr16[i] == 0) && (zerocnt == 0)) {
110  while(addr.s6_addr16[zerocnt + i] == 0) zerocnt++;
111  if (zerocnt == 1) {
112  *result++ = '0';
113  numprinted++;
114  break;
115  }
116  i += zerocnt;
117  numprinted += zerocnt;
118  } else {
119  result += sprintf(result, "%x", (unsigned int) uip_ntohs(addr.s6_addr16[i]));
120  i++;
121  numprinted++;
122  }
123  if (numprinted != 8) *result++ = ':';
124  }
125  *result++=']';
126  *result=0;
127  return (result - starting);
128 }

229 static void
230 set_ethaddr(struct in_addr addr)
231 {
232  PIP_ADAPTER_ADDRESSES adapters;
233  ULONG size = 0;
234 
235  if(GetAdaptersAddresses(AF_INET, GAA_FLAG_SKIP_ANYCAST |
236  GAA_FLAG_SKIP_MULTICAST |
237  GAA_FLAG_SKIP_DNS_SERVER,
238  NULL, NULL, &size) != ERROR_BUFFER_OVERFLOW) {
239  error_exit("error on access to adapter list size\n");
240  }
241  adapters = alloca(size);
242  if(GetAdaptersAddresses(AF_INET, GAA_FLAG_SKIP_ANYCAST |
243  GAA_FLAG_SKIP_MULTICAST |
244  GAA_FLAG_SKIP_DNS_SERVER,
245  NULL, adapters, &size) != ERROR_SUCCESS) {
246  error_exit("error on access to adapter list\n");
247  }
248 
249  while(adapters != NULL) {
250 
251  char buffer[256];
252  WideCharToMultiByte(CP_ACP, 0, adapters->Description, -1,
253  buffer, sizeof(buffer), NULL, NULL);
254  log_message("set_ethaddr: found adapter: ", buffer);
255 
256  if(adapters->FirstUnicastAddress != NULL &&
257  adapters->FirstUnicastAddress->Address.lpSockaddr != NULL &&
258  adapters->FirstUnicastAddress->Address.lpSockaddr->sa_family == AF_INET) {
259 
260  struct in_addr adapter_addr;
261  adapter_addr = ((struct sockaddr_in *)adapters->FirstUnicastAddress->Address.lpSockaddr)->sin_addr;
262  log_message("set_ethaddr: with address: ", inet_ntoa(adapter_addr));
263 
264  if(adapter_addr.s_addr == addr.s_addr) {
265  if(adapters->PhysicalAddressLength != 6) {
266  error_exit("ip addr specified on cmdline does not belong to an ethernet card\n");
267  }
268  wsprintf(buffer, "%02X-%02X-%02X-%02X-%02X-%02X",
269  adapters->PhysicalAddress[0], adapters->PhysicalAddress[1],
270  adapters->PhysicalAddress[2], adapters->PhysicalAddress[3],
271  adapters->PhysicalAddress[4], adapters->PhysicalAddress[5]);
272  log_message("set_ethaddr: ethernetaddr: ", buffer);
273 #if NETSTACK_CONF_WITH_IPV6
274 // int i;for (i=0;i<6;i++) uip_lladdr.addr[i] = adapters->PhysicalAddress[i];
275 #else
276  uip_setethaddr((*(struct uip_eth_addr *)adapters->PhysicalAddress));
277 #endif
278  break;
279  }
280  }
281  adapters = adapters->Next;
282  }
283 
284  if(adapters == NULL) {
285  error_exit("no adapter found with ip addr specified on cmdline\n");
286  }
287 }

352 static void
353 init_pcap(struct in_addr addr)
354 {
355  struct pcap_if *interfaces;
356  struct pcap_addr *paddr;
357  char error[256];
358 
359  if(pcap_findalldevs(&interfaces, error) == -1) {
360  error_exit(error);
361  }
362 
363  while(interfaces != NULL) {
364  log_message("init_pcap: found interface: ", interfaces->description);
365 
366  if(interfaces->addresses != NULL) {
367  for(paddr = interfaces->addresses;
368  paddr != NULL;
369  paddr = paddr->next) {
370  if(paddr->addr != NULL && paddr->addr->sa_family == AF_INET) {
371  struct in_addr interface_addr;
372  interface_addr = ((struct sockaddr_in *)paddr->addr)->sin_addr;
373  log_message("init_pcap: with address: ", inet_ntoa(interface_addr));
374 
375  if(interface_addr.s_addr == addr.s_addr) {
376  pcap = pcap_open_live(interfaces->name, UIP_BUFSIZE, 0, -1, error);
377  if(pcap == NULL) {
378  error_exit(error);
379  }
380 #ifdef UIP_FALLBACK_INTERFACE
381  log_message("init_pcap: Opened as primary interface","");
382 #else
383  log_message("init_pcap: Opened as interface","");
384 #endif
385 // pcap_setdirection(PCAP_D_IN); //Not implemented in windows yet?
386  set_ethaddr(addr);
387 
388 #ifdef UIP_FALLBACK_INTERFACE
389  }
390  if (pfall && pcap) return;
391  if(interface_addr.s_addr == addrfall.s_addr) {
392  pfall = pcap_open_live(interfaces->name, UIP_BUFSIZE, 0, -1, error);
393  if(pfall == NULL) {
394  error_exit(error);
395  }
396  log_message("init_pcap: Opened as fallback interface","");
397  if (pcap) return;
398  }
399 #else
400  return;
401  }
402 #endif
403 
404 #if NETSTACK_CONF_WITH_IPV6
405 
406  } else if(paddr->addr != NULL && paddr->addr->sa_family == AF_INET6) {
407  struct in6_addr interface_addr;
408  interface_addr = ((struct sockaddr_in6 *)paddr->addr)->sin6_addr;
409 
410  sprint_ip6addr(interface_addr, addr6str);
411  log_message("init_pcap: with ipv6 address: ", addr6str);
412 
413  if (issame_ip6addr(interface_addr, addr6)) {
414  pcap = pcap_open_live(interfaces->name, UIP_BUFSIZE, 0, -1, error);
415  if(pcap == NULL) {
416  error_exit(error);
417  }
418 #ifdef UIP_FALLBACK_INTERFACE
419  log_message("init_pcap: Opened as primary interface","");
420 #endif
421  set_ethaddr6(addr6);
422 // pcap_setdirection(PCAP_D_IN); //Not implemented in windows yet?
423 #ifdef UIP_FALLBACK_INTERFACE
424  }
425  if (pfall && pcap) return; //exit when we have both interfaces
426 
427  if (issame_ip6addr(interface_addr, addrfall6)) {
428  pfall = pcap_open_live(interfaces->name, UIP_BUFSIZE, 0, -1, error);
429  if(pfall == NULL) {
430  error_exit(error);
431  }
432  log_message("init_pcap: Opened as fallback interface","");
433  if (pcap) return;
434  }
435 #else
436  return;
437  }
438 #endif
439 #endif /* NETSTACK_CONF_WITH_IPV6 */
440  }
441  }
442  }
443  interfaces = interfaces->next;
444 
445  }
446 
447  if(interfaces == NULL) {
448  error_exit("no interface found with specified ip address\n");
449  }
450 }

452 void
453 wpcap_init(void)
454 {
455  struct in_addr addr;
456  addr.s_addr = INADDR_NONE; //255.255.255.255
457 #ifdef UIP_FALLBACK_INTERFACE
458  addrfall.s_addr = INADDR_NONE;
459 #endif
460 
461  /* Pick up possible ip addresses from command line */
462 #ifdef __CYGWIN__
463  if ((*__argv)[1]) {
464  addr.s_addr = inet_addr((*__argv)[1]);
465 #if NETSTACK_CONF_WITH_IPV6
466  uiplib_ipaddrconv((*__argv)[1],(uip_ipaddr_t*) &addr6.s6_addr);
467 #endif
468 #ifdef UIP_FALLBACK_INTERFACE
469  if ((*__argv)[2]) {
470  addrfall.s_addr = inet_addr((*__argv)[2]);
471 #if NETSTACK_CONF_WITH_IPV6
472  uiplib_ipaddrconv((*__argv)[2],(uip_ipaddr_t*) &addrfall6.s6_addr);
473 #endif
474  }
475 #endif
476  }
477 
478 #else /* __CYGWIN__ */
479 /* VC++ build on win32 platform. Currently the platform has no ipv6 support */
480  addr.s_addr = inet_addr(__argv[1]);
481 #if NETSTACK_CONF_WITH_IPV6
482  if((__argv)[1])
483  uiplib_ipaddrconv((__argv)[1],(uip_ipaddr_t*) &addr6.s6_addr);
484 #endif
485 #ifdef UIP_FALLBACK_INTERFACE
486  addrfall.s_addr = inet_addr(__argv[2]);
487 #if NETSTACK_CONF_WITH_IPV6
488  if((__argv)[2])
489  uiplib_ipaddrconv((__argv)[2],(uip_ipaddr_t*) &addrfall6.s6_addr);
490 #endif
491 #endif
492 #endif /* __CYGWIN__ */
493 
494 #if DEBUG
495  log_message("wpcap_init:Passed ipv4 ", inet_ntoa(addr));
496  sprint_ip6addr(addr6, addr6str);
497  log_message("wpcap_init:Passed ipv6 ", addr6str);
498 #ifdef UIP_FALLBACK_INTERFACE
499  log_message("wpcap_init:Passed fallback ipv4 ", inet_ntoa(addrfall));
500  sprint_ip6addr(addrfall6, addr6str);
501  log_message("wpcap_init:Passed fallback ipv6 ", addr6str);
502 #endif
503 #endif
504 
505  /* Use build defaults if not enough addresses passed */
506 #if NETSTACK_CONF_WITH_IPV6
507 
508 #ifdef UIP_FALLBACK_INTERFACE
509  if(addrfall.s_addr == INADDR_NONE) {
510  if(iszero_ip6addr(addrfall6)) {
511 #ifdef WPCAP_FALLBACK_ADDRESS
512  addrfall.s_addr = inet_addr(WPCAP_FALLBACK_ADDRESS);
513 // if(addrfall.s_addr == INADDR_NONE) { //use ipv6 if contiki-conf.h override
514  uiplib_ipaddrconv(WPCAP_FALLBACK_ADDRESS,(uip_ipaddr_t*) &addrfall6.s6_addr);
515 // }
516 #else
517 // addrfall.s_addr = inet_addr("10.2.10.10");
518  uiplib_ipaddrconv("bbbb::1",(uip_ipaddr_t*) &addrfall6.s6_addr);
519 #endif
520  }
521  }
522 #endif
523 
524  if(addr.s_addr == INADDR_NONE) {
525  if(iszero_ip6addr(addr6)) {
526 #ifdef WPCAP_INTERFACE_ADDRESS
527  addr.s_addr = inet_addr(WPCAP_INTERFACE_ADDRESS);
528 // if(addr.s_addr == INADDR_NONE) {
529  uiplib_ipaddrconv(WPCAP_INTERFACE_ADDRESS,(uip_ipaddr_t*) &addr6.s6_addr);
530 // }
531 #else
532  addr.s_addr = inet_addr("10.10.10.10"); //prefer ipv4 default for legacy compatibility
533 // uiplib_ipaddrconv("fd00::1",(uip_ipaddr_t*) &addr6.s6_addr);
534 #endif
535 
536 #ifdef UIP_FALLBACK_INTERFACE
537  log_message("usage: <program> <ip addr of interface> <ip addr of fallback>\n","");
538  if(addr.s_addr != INADDR_NONE) {
539  log_message("-->I'll try interface address ", inet_ntoa(addr));
540  } else {
541  sprint_ip6addr(addr6, addr6str);
542  log_message("-->I'll try interface address ", addr6str);
543  }
544  if(addrfall.s_addr != INADDR_NONE) {
545  log_message("--> and fallback address ", inet_ntoa(addrfall));
546  } else {
547  sprint_ip6addr(addrfall6, addr6str);
548  log_message("--> and fallback address ", addr6str);
549  }
550 #else
551  if(addr.s_addr != INADDR_NONE) {
552  log_message("usage: <program> <ip addr of ethernet card to share>\n-->I'll try guessing ", inet_ntoa(addr));
553  } else {
554  sprint_ip6addr(addr6, addr6str);
555  log_message("usage: <program> <ip addr of ethernet card to share>\n-->I'll try guessing ", addr6str);
556  }
557 #endif
558  }
559  }
560 #else /* ip4 build */
561  if(addr.s_addr == INADDR_NONE) {
562 #ifdef WPCAP_INTERFACE_ADDRESS
563  addr.s_addr = inet_addr(WPCAP_INTERFACE_ADDRESS);
564 #else
565  addr.s_addr = inet_addr("10.10.10.10");
566 #endif
567  log_message("usage: <program> <ip addr of ethernet card to share>\n-->I'll try guessing ", inet_ntoa(addr));
568  }
569 #endif /* NETSTACK_CONF_WITH_IPV6 */
570 
571 #if DEBUG
572  log_message("wpcap_init:Using ipv4 ", inet_ntoa(addr));
573  sprint_ip6addr(addr6, addr6str);
574  log_message("wpcap_init:Using ipv6 ", addr6str);
575 #ifdef UIP_FALLBACK_INTERFACE
576  log_message("wpcap_init:Using fallback ipv4 ", inet_ntoa(addrfall));
577  sprint_ip6addr(addrfall6, addr6str);
578  log_message("wpcap_init:Using fallback ipv6 ", addr6str);
579 #endif
580 #endif
581 
582  // log_message("wpcap_init:cmdline address: ", inet_ntoa(addr));
583 
584 
585  wpcap = LoadLibrary("wpcap.dll");
586  pcap_findalldevs = (int (*)(struct pcap_if **, char *))
587  GetProcAddress(wpcap, "pcap_findalldevs");
588  pcap_open_live = (struct pcap *(*)(char *, int, int, int, char *))
589  GetProcAddress(wpcap, "pcap_open_live");
590  pcap_next_ex = (int (*)(struct pcap *, struct pcap_pkthdr **, unsigned char **))
591  GetProcAddress(wpcap, "pcap_next_ex");
592  pcap_sendpacket = (int (*)(struct pcap *, unsigned char *, int))
593  GetProcAddress(wpcap, "pcap_sendpacket");
594 
595  if(pcap_findalldevs == NULL || pcap_open_live == NULL ||
596  pcap_next_ex == NULL || pcap_sendpacket == NULL) {
597  error_exit("error on access to winpcap library\n");
598  }
599 
600  init_pcap(addr);
601 
602 }

605 uint16_t
606 wpcap_poll(void)
607 {
