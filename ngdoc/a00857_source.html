152 static void
153 check_for_tcp_syn(void)
154 {
155 #if UIP_TCP
156  /* This is a hack that is needed to start the periodic TCP timer if
157  an incoming packet contains a SYN: since uIP does not inform the
158  application if a SYN arrives, we have no other way of starting
159  this timer. This function is called for every incoming IP packet
160  to check for such SYNs. */
161 #define TCP_SYN 0x02
162  if(UIP_IP_BUF->proto == UIP_PROTO_TCP &&
163  (UIP_TCP_BUF->flags & TCP_SYN) == TCP_SYN) {
164  start_periodic_tcp_timer();
165  }
166 #endif /* UIP_TCP */
167 }

303 static void
304 eventhandler(process_event_t ev, process_data_t data)
305 {
306 #if UIP_TCP
307  unsigned char i;
308  register struct listenport *l;
309 #endif /*UIP_TCP*/
310  struct process *p;
311 
312  switch(ev) {
313  case PROCESS_EVENT_EXITED:
314  /* This is the event we get if a process has exited. We go through
315  the TCP/IP tables to see if this process had any open
316  connections or listening TCP ports. If so, we'll close those
317  connections. */
318 
319  p = (struct process *)data;
320 #if UIP_TCP
321  l = s.listenports;
322  for(i = 0; i < UIP_LISTENPORTS; ++i) {
323  if(l->p == p) {
324  uip_unlisten(l->port);
325  l->port = 0;
326  l->p = PROCESS_NONE;
327  }
328  ++l;
329  }
330 
331  {
332  struct uip_conn *cptr;
333 
334  for(cptr = &uip_conns[0]; cptr < &uip_conns[UIP_TCP_CONNS]; ++cptr) {
335  if(cptr->appstate.p == p) {
336  cptr->appstate.p = PROCESS_NONE;
337  cptr->tcpstateflags = UIP_CLOSED;
338  }
339  }
340  }
341 #endif /* UIP_TCP */
342 #if UIP_UDP
343  {
344  struct uip_udp_conn *cptr;
345 
346  for(cptr = &uip_udp_conns[0];
347  cptr < &uip_udp_conns[UIP_UDP_CONNS]; ++cptr) {
348  if(cptr->appstate.p == p) {
349  cptr->lport = 0;
350  }
351  }
352  }
353 #endif /* UIP_UDP */
354  break;
355 
356  case PROCESS_EVENT_TIMER:
357  /* We get this event if one of our timers have expired. */
358  {
359  /* Check the clock so see if we should call the periodic uIP
360  processing. */
361  if(data == &periodic &&
362  etimer_expired(&periodic)) {
363 #if UIP_TCP
364  for(i = 0; i < UIP_TCP_CONNS; ++i) {
365  if(uip_conn_active(i)) {
366  /* Only restart the timer if there are active
367  connections. */
368  etimer_restart(&periodic);
369  uip_periodic(i);
370  tcpip_ipv6_output();
371  }
372  }
373 #endif /* UIP_TCP */
374  }
375 
376 #if UIP_CONF_IPV6_REASSEMBLY
377  /*
378  * check the timer for reassembly
379  */
380  if(data == &uip_reass_timer &&
381  etimer_expired(&uip_reass_timer)) {
382  uip_reass_over();
383  tcpip_ipv6_output();
384  }
385 #endif /* UIP_CONF_IPV6_REASSEMBLY */
386  /*
387  * check the different timers for neighbor discovery and
388  * stateless autoconfiguration
389  */
390  /*if(data == &uip_ds6_timer_periodic &&
391  etimer_expired(&uip_ds6_timer_periodic)) {
392  uip_ds6_periodic();
393  tcpip_ipv6_output();
394  }*/
395 #if !UIP_CONF_ROUTER
396  if(data == &uip_ds6_timer_rs &&
397  etimer_expired(&uip_ds6_timer_rs)) {
398  uip_ds6_send_rs();
399  tcpip_ipv6_output();
400  }
401 #endif /* !UIP_CONF_ROUTER */
402  if(data == &uip_ds6_timer_periodic &&
403  etimer_expired(&uip_ds6_timer_periodic)) {
404  uip_ds6_periodic();
405  tcpip_ipv6_output();
406  }
407  }
408  break;
409 
410 #if UIP_TCP
411  case TCP_POLL:
412  if(data != NULL) {
413  uip_poll_conn(data);
414  tcpip_ipv6_output();
415  /* Start the periodic polling, if it isn't already active. */
416  start_periodic_tcp_timer();
417  }
418  break;
419 #endif /* UIP_TCP */
420 #if UIP_UDP
421  case UDP_POLL:
422  if(data != NULL) {
423  uip_udp_periodic_conn(data);
424  tcpip_ipv6_output();
425  }
426  break;
427 #endif /* UIP_UDP */
428 
429  case PACKET_INPUT:
430  packet_input();
431  break;
432  };
433 }

444 static void
445 output_fallback(void)
446 {
447 #ifdef UIP_FALLBACK_INTERFACE
448  LOG_INFO("fallback: removing ext hdrs & setting proto %d %d\n",
449  uip_ext_len, *((uint8_t *)UIP_IP_BUF + 40));
450  if(uip_ext_len > 0) {
451  uint8_t proto = *((uint8_t *)UIP_IP_BUF + 40);
452  remove_ext_hdr();
453  /* This should be copied from the ext header... */
454  UIP_IP_BUF->proto = proto;
455  }
456  /* Inform the other end that the destination is not reachable. If it's
457  * not informed routes might get lost unexpectedly until there's a need
458  * to send a new packet to the peer */
459  if(UIP_FALLBACK_INTERFACE.output() < 0) {
460  LOG_ERR("fallback: output error. Reporting DST UNREACH\n");
461  uip_icmp6_error_output(ICMP6_DST_UNREACH, ICMP6_DST_UNREACH_ADDR, 0);
462  uip_flags = 0;
463  tcpip_ipv6_output();
464  return;
465  }
466 #else
467  LOG_ERR("output: destination off-link and no default route\n");
468 #endif /* !UIP_FALLBACK_INTERFACE */
469 }

504 static uip_ipaddr_t*
505 get_nexthop(uip_ipaddr_t *addr)
506 {
507  uip_ipaddr_t *nexthop;
508  uip_ds6_route_t *route;
509 
510  LOG_INFO("output: processing packet from ");
511  LOG_INFO_6ADDR(&UIP_IP_BUF->srcipaddr);
512  LOG_INFO_(" to ");
513  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
514  LOG_INFO_("\n");
515 
516  if(NEXTHOP_NON_STORING(addr)) {
517  LOG_INFO("output: selected next hop from SRH: ");
518  LOG_INFO_6ADDR(addr);
519  LOG_INFO_("\n");
520  return addr;
521  }
522 
523  /* We first check if the destination address is on our immediate
524  link. If so, we simply use the destination address as our
525  nexthop address. */
526  if(uip_ds6_is_addr_onlink(&UIP_IP_BUF->destipaddr)) {
527  LOG_INFO("output: destination is on link\n");
528  return &UIP_IP_BUF->destipaddr;
529  }
530 
531  /* Check if we have a route to the destination address. */
532  route = uip_ds6_route_lookup(&UIP_IP_BUF->destipaddr);
533 
534  /* No route was found - we send to the default route instead. */
535  if(route == NULL) {
536  nexthop = uip_ds6_defrt_choose();
537  if(nexthop == NULL) {
538  output_fallback();
539  } else {
540  LOG_INFO("output: no route found, using default route: ");
541  LOG_INFO_6ADDR(nexthop);
542  LOG_INFO_("\n");
543  }
544 
545  } else {
546  /* A route was found, so we look up the nexthop neighbor for
547  the route. */
548  nexthop = uip_ds6_route_nexthop(route);
549 
550  /* If the nexthop is dead, for example because the neighbor
551  never responded to link-layer acks, we drop its route. */
552  if(nexthop == NULL) {
553  LOG_ERR("output: found dead route\n");
554  drop_route(route);
555  /* We don't have a nexthop to send the packet to, so we drop
556  it. */
557  } else {
558  LOG_INFO("output: found next hop from routing table: ");
559  LOG_INFO_6ADDR(nexthop);
560  LOG_INFO_("\n");
561  }
562  }
563 
564  return nexthop;
565 }

587 #if UIP_CONF_IPV6_QUEUE_PKT

603 static int
604 send_nd6_ns(uip_ipaddr_t *nexthop)
605 {
606  int err = 1;
607 
608 #if UIP_ND6_SEND_NS
609  uip_ds6_nbr_t *nbr = NULL;
610  if((nbr = uip_ds6_nbr_add(nexthop, NULL, 0, NBR_INCOMPLETE, NBR_TABLE_REASON_IPV6_ND, NULL)) != NULL) {
611  err = 0;
612 
613  queue_packet(nbr);
614  /* RFC4861, 7.2.2:
615  * "If the source address of the packet prompting the solicitation is the
616  * same as one of the addresses assigned to the outgoing interface, that
617  * address SHOULD be placed in the IP Source Address of the outgoing
618  * solicitation. Otherwise, any one of the addresses assigned to the
619  * interface should be used."*/
620  if(uip_ds6_is_my_addr(&UIP_IP_BUF->srcipaddr)){
621  uip_nd6_ns_output(&UIP_IP_BUF->srcipaddr, NULL, &nbr->ipaddr);
622  } else {
623  uip_nd6_ns_output(NULL, NULL, &nbr->ipaddr);
624  }
625 
626  stimer_set(&nbr->sendns, uip_ds6_if.retrans_timer / 1000);
627  nbr->nscount = 1;
628  /* Send the first NS try from here (multicast destination IP address). */
629  }
630 #else
631  LOG_ERR("output: neighbor not in cache: ");
632  LOG_ERR_6ADDR(nexthop);
633  LOG_ERR_("\n");
634 #endif
635 
636  return err;
637 }
638 /*---------------------------------------------------------------------------*/
639 void
640 tcpip_ipv6_output(void)
641 {
642  uip_ipaddr_t ipaddr;
643  uip_ds6_nbr_t *nbr = NULL;
644  const uip_lladdr_t *linkaddr;
645  uip_ipaddr_t *nexthop;
646 
647  if(uip_len == 0) {
648  return;
649  }
650 
651  if(uip_len > UIP_LINK_MTU) {
652  LOG_ERR("output: Packet too big");
653  goto exit;
654  }
655 
656  if(uip_is_addr_unspecified(&UIP_IP_BUF->destipaddr)){
657  LOG_ERR("output: Destination address unspecified");
658  goto exit;
659  }
660 
661 #if UIP_CONF_IPV6_RPL
662  if(!rpl_ext_header_update()) {
663  /* Packet can not be forwarded */
664  LOG_ERR("output: RPL header update error\n");
665  uip_clear_buf();
666  return;
667  }
668 #endif /* UIP_CONF_IPV6_RPL */
669 
670  if(uip_is_addr_mcast(&UIP_IP_BUF->destipaddr)) {
671  linkaddr = NULL;
672  goto send_packet;
673  }
674 
675  /* We first check if the destination address is one of ours. There is no
676  * loopback interface -- instead, process this directly as incoming. */
677  if(uip_ds6_is_my_addr(&UIP_IP_BUF->destipaddr)) {
678  LOG_INFO("output: sending to ourself\n");
679  packet_input();
680  return;
681  }
682 
683  /* Look for a next hop */
684  if((nexthop = get_nexthop(&ipaddr)) == NULL) {
685  goto exit;
686  }
687  annotate_transmission(nexthop);
688 
689  nbr = uip_ds6_nbr_lookup(nexthop);
690 
691 #if UIP_ND6_AUTOFILL_NBR_CACHE
692  if(nbr == NULL) {
693  /* Neighbor not found in cache? Derive its link-layer address from it's
694  link-local IPv6, assuming it used autoconfiguration. This is not
695  standard-compliant but this is a convenient way to keep the
696  neighbor cache out of the way in cases ND is not used */
697  uip_lladdr_t lladdr;
698  uip_ds6_set_lladdr_from_iid(&lladdr, nexthop);
699  if((nbr = uip_ds6_nbr_add(nexthop, &lladdr,
700  0, NBR_REACHABLE, NBR_TABLE_REASON_IPV6_ND_AUTOFILL, NULL)) == NULL) {
701  LOG_ERR("output: failed to autofill neighbor cache for host ");
702  LOG_ERR_6ADDR(nexthop);
703  LOG_ERR_(", link-layer addr ");
704  LOG_ERR_LLADDR((linkaddr_t*)&lladdr);
705  LOG_ERR_("\n");
706  goto exit;
707  }
708  }
709 #endif /* UIP_ND6_AUTOFILL_NBR_CACHE */
710 
711  if(nbr == NULL) {
712  if(send_nd6_ns(nexthop)) {
713  LOG_ERR("output: failed to add neighbor to cache\n");
714  goto exit;
715  } else {
716  /* We're sending NS here instead of original packet */
717  goto send_packet;
718  }
719  }
720 
721 #if UIP_ND6_SEND_NS
722  if(nbr->state == NBR_INCOMPLETE) {
723  LOG_ERR("output: nbr cache entry incomplete\n");
724  queue_packet(nbr);
725  goto exit;
726  }
727  /* Send in parallel if we are running NUD (nbc state is either STALE,
728  DELAY, or PROBE). See RFC 4861, section 7.3.3 on node behavior. */
729  if(nbr->state == NBR_STALE) {
730  nbr->state = NBR_DELAY;
731  stimer_set(&nbr->reachable, UIP_ND6_DELAY_FIRST_PROBE_TIME);
732  nbr->nscount = 0;
733  LOG_INFO("output: nbr cache entry stale moving to delay\n");
734  }
735 #endif /* UIP_ND6_SEND_NS */
736 
737 send_packet:
738  if(nbr) {
739  linkaddr = uip_ds6_nbr_get_ll(nbr);
740  } else {
741  linkaddr = NULL;
742  }
743 
744  LOG_INFO("output: sending to ");
745  LOG_INFO_LLADDR((linkaddr_t *)linkaddr);
746  LOG_INFO_("\n");
747  tcpip_output(linkaddr);
748 
749  if(nbr) {
750  send_queued(nbr);
751  }
752 
753 exit:
754  uip_clear_buf();
755  return;
756 }
