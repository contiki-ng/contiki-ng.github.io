45 void
46 ieee_addr_cpy_to(uint8_t *dst, uint8_t len)
47 {
48  if(IEEE_ADDR_CONF_HARDCODED) {
49  uint8_t ieee_addr_hc[8] = IEEE_ADDR_CONF_ADDRESS;
50 
51  memcpy(dst, &ieee_addr_hc[8 - len], len);
52  } else {
53  /*
54  * By default, we assume that the IEEE address is stored on flash using
55  * little-endian byte order.
56  *
57  * However, some SoCs ship with a different byte order, whereby the first
58  * four bytes are flipped with the four last ones.
59  *
60  * Using this address as an example: 00 12 4B 00 01 02 03 04
61  * We expect it stored as: 04 03 02 01 00 4B 12 00
62  * But it is also possible to encounter: 00 4B 12 00 04 03 02 01
63  *
64  * Thus: read locations [3, 2, 1] and if we encounter the TI OUI, flip the
65  * order of the two 4-byte sequences. Each of the 4-byte sequences is still
66  * little-endian.
67  */
68  int i;
69  uint8_t oui_ti[3] = IEEE_ADDR_OUI_TI;
70  if(((uint8_t *)IEEE_ADDR_LOCATION)[3] == oui_ti[0]
71  && ((uint8_t *)IEEE_ADDR_LOCATION)[2] == oui_ti[1]
72  && ((uint8_t *)IEEE_ADDR_LOCATION)[1] == oui_ti[2]) {
73  for(i = 0; i < len / 2; i++) {
74  dst[i] = ((uint8_t *)IEEE_ADDR_LOCATION)[len / 2 - 1 - i];
75  }
76  for(i = 0; i < len / 2; i++) {
77  dst[i + len / 2] = ((uint8_t *)IEEE_ADDR_LOCATION)[len - 1 - i];
78  }
79  } else {
80  for(i = 0; i < len; i++) {
81  dst[i] = ((uint8_t *)IEEE_ADDR_LOCATION)[len - 1 - i];
82  }
83  }
84  }
85 
86 #if IEEE_ADDR_NODE_ID
87  dst[len - 1] = IEEE_ADDR_NODE_ID & 0xFF;
88  dst[len - 2] = IEEE_ADDR_NODE_ID >> 8;
89 #endif
90 }
