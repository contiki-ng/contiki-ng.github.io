82 /*---------------------------------------------------------------------------*/
83 void
84 rpl_dag_leave(void)
85 {
86  LOG_INFO("leaving DAG ");
87  LOG_INFO_6ADDR(&curr_instance.dag.dag_id);
88  LOG_INFO_(", instance %u\n", curr_instance.instance_id);
89 
90  /* Issue a no-path DAO */
91  if(!rpl_dag_root_is_root()) {
92  RPL_LOLLIPOP_INCREMENT(curr_instance.dag.dao_curr_seqno);
93  rpl_icmp6_dao_output(0);
94  }
95 
96  /* Forget past link statistics */
97  link_stats_reset();
98 
99  /* Remove all neighbors and lnks */
100  rpl_neighbor_remove_all();
101  rpl_ns_free_all();
102 
103  /* Stop all timers */
104  rpl_timers_stop_dag_timers();
105 
106  /* Remove autoconfigured address */
107  if((curr_instance.dag.prefix_info.flags & UIP_ND6_RA_FLAG_AUTONOMOUS)) {
108  rpl_reset_prefix(&curr_instance.dag.prefix_info);
109  }
110 
111  /* Mark instance as unused */
112  curr_instance.used = 0;
113 }

121 /*---------------------------------------------------------------------------*/
122 void
123 rpl_dag_periodic(unsigned seconds)
124 {
125  if(curr_instance.used) {
126  if(curr_instance.dag.lifetime != RPL_LIFETIME(RPL_INFINITE_LIFETIME)) {
127  curr_instance.dag.lifetime =
128  curr_instance.dag.lifetime > seconds ? curr_instance.dag.lifetime - seconds : 0;
129  if(curr_instance.dag.lifetime == 0) {
130  LOG_WARN("DAG expired, poison and leave\n");
131  curr_instance.dag.state = DAG_POISONING;
132  rpl_timers_schedule_state_update();
133  } else if(curr_instance.dag.lifetime < 300 && curr_instance.dag.preferred_parent != NULL) {
134  /* Five minutes before expiring, start sending unicast DIS to get an update */
135  LOG_WARN("DAG expiring in %u seconds, send DIS to preferred parent\n", (unsigned)curr_instance.dag.lifetime);
136  rpl_icmp6_dis_output(rpl_neighbor_get_ipaddr(curr_instance.dag.preferred_parent));
137  }
138  }
139  }

173 void
174 rpl_global_repair(void)
175 {
176  if(rpl_dag_root_is_root()) {
177  LOG_WARN("initiating global repair, version %u, rank %u)\n",
178  curr_instance.dag.version, curr_instance.dag.rank);
179 #if LOG_INFO_ENABLED
180  rpl_neighbor_print_list("Global repair (before)");
181 #endif /* LOG_INFO_ENABLED */
182 
183  /* Initiate global repair */
184  RPL_LOLLIPOP_INCREMENT(curr_instance.dag.version); /* New DAG version */
185  RPL_LOLLIPOP_INCREMENT(curr_instance.dtsn_out); /* Request new DAOs */
186  rpl_local_repair("Global repair");
187  }
188 }

190 static void
191 global_repair_non_root(rpl_dio_t *dio)
192 {
193  if(!rpl_dag_root_is_root()) {
194  LOG_WARN("participating in global repair, version %u, rank %u)\n",
195  curr_instance.dag.version, curr_instance.dag.rank);
196 #if LOG_INFO_ENABLED

198 #endif /* LOG_INFO_ENABLED */

200  init_dag_from_dio(dio);
201  rpl_local_repair("Global repair");

204 /*---------------------------------------------------------------------------*/
205 void
206 rpl_local_repair(const char *str)
207 {
208  if(curr_instance.used) { /* Check needed because this is a public function */
209  LOG_WARN("local repair (%s)\n", str);
210  if(!rpl_dag_root_is_root()) {
211  curr_instance.dag.state = DAG_INITIALIZED; /* Reset DAG state */
212  }
213  curr_instance.of->reset(); /* Reset OF */
214  link_stats_reset(); /* Forget past link statistics */
215  rpl_neighbor_remove_all(); /* Remove all neighbors */
216  rpl_timers_dio_reset("Local repair"); /* Reset Trickle timer */
217  rpl_timers_schedule_state_update();
218  }
219 }

230 /*---------------------------------------------------------------------------*/
231 /* Updates rank and parent */
232 void
233 rpl_dag_update_state(void)
234 {
235  rpl_rank_t old_rank;
236 
237  if(!curr_instance.used) {
238  return;
239  }
240 
241  old_rank = curr_instance.dag.rank;
242  /* Any scheduled state update is no longer needed */
243  rpl_timers_unschedule_state_update();
244 
245  if(curr_instance.dag.state == DAG_POISONING) {
246  rpl_neighbor_set_preferred_parent(NULL);
247  curr_instance.dag.rank = RPL_INFINITE_RANK;
248  if(old_rank != RPL_INFINITE_RANK) {
249  /* Advertise that we are leaving, and leave after a delay */
250  LOG_WARN("poisoning and leaving after a delay\n");
251  rpl_timers_dio_reset("Poison routes");
252  rpl_timers_schedule_leaving();
253  }
254  } else if(!rpl_dag_root_is_root()) {
255  rpl_nbr_t *old_parent = curr_instance.dag.preferred_parent;
256  rpl_nbr_t *nbr;
257 
258  /* Select and set preferred parent */
259  rpl_neighbor_set_preferred_parent(rpl_neighbor_select_best());
260  /* Update rank */
261  curr_instance.dag.rank = rpl_neighbor_rank_via_nbr(curr_instance.dag.preferred_parent);
262 
263  /* Update better_parent_since flag for each neighbor */
264  nbr = nbr_table_head(rpl_neighbors);
265  while(nbr != NULL) {
266  if(rpl_neighbor_rank_via_nbr(nbr) < curr_instance.dag.rank) {
267  /* This neighbor would be a better parent than our current.
268  Set 'better_parent_since' if not already set. */
269  if(nbr->better_parent_since == 0) {
270  nbr->better_parent_since = clock_time(); /* Initialize */
271  }
272  } else {
273  nbr->better_parent_since = 0; /* Not a better parent */
274  }
275  nbr = nbr_table_next(rpl_neighbors, nbr);
276  }
277 
278  if(old_parent == NULL || curr_instance.dag.rank < curr_instance.dag.lowest_rank) {
279  /* This is a slight departure from RFC6550: if we had no preferred parent before,
280  * reset lowest_rank. This helps recovering from temporary bad link conditions. */
281  curr_instance.dag.lowest_rank = curr_instance.dag.rank;
282  }
283 
284  /* Reset DIO timer in case of significant rank update */
285  if(curr_instance.dag.last_advertised_rank != RPL_INFINITE_RANK
286  && curr_instance.dag.rank != RPL_INFINITE_RANK
287  && ABS((int32_t)curr_instance.dag.rank - curr_instance.dag.last_advertised_rank) > RPL_SIGNIFICANT_CHANGE_THRESHOLD) {
288  LOG_WARN("significant rank update %u->%u\n",
289  curr_instance.dag.last_advertised_rank, curr_instance.dag.rank);
290  /* Update already here to avoid multiple resets in a row */
291  curr_instance.dag.last_advertised_rank = curr_instance.dag.rank;
292  rpl_timers_dio_reset("Significant rank update");
293  }
294 
295  /* Parent switch */
296  if(curr_instance.dag.preferred_parent != old_parent) {
297  /* We just got a parent (was NULL), reset trickle timer to advertise this */
298  if(old_parent == NULL) {
299  curr_instance.dag.state = DAG_JOINED;
300  rpl_timers_dio_reset("Got parent");
301  LOG_WARN("found parent: ");
302  LOG_WARN_6ADDR(rpl_neighbor_get_ipaddr(curr_instance.dag.preferred_parent));
303  LOG_WARN_(", staying in DAG\n");
304  rpl_timers_unschedule_leaving();
305  }
306 
307  /* Schedule a DAO */
308  if(curr_instance.dag.preferred_parent != NULL) {
309  rpl_timers_schedule_dao();
310  } else {
311  /* We have no more parent, schedule DIS to get a chance to hear updated state */
312  curr_instance.dag.state = DAG_INITIALIZED;
313  LOG_WARN("no parent, scheduling periodic DIS, will leave if no parent is found\n");
314  rpl_timers_dio_reset("Poison routes");
315  rpl_timers_schedule_periodic_dis();
316  rpl_timers_schedule_leaving();
317  }
318 
319 #if LOG_INFO_ENABLED
320  rpl_neighbor_print_list("Parent switch");
321 #endif /* LOG_INFO_ENABLED */
322  }
323  }
324 
325  /* Finally, update metric container */
326  curr_instance.of->update_metric_container();
327 }

329 static rpl_nbr_t *
330 update_nbr_from_dio(uip_ipaddr_t *from, rpl_dio_t *dio)
331 {
332  rpl_nbr_t *nbr = NULL;
333  const uip_lladdr_t *lladdr;
334 
335  nbr = rpl_neighbor_get_from_ipaddr(from);
336  /* Neighbor not in RPL neighbor table, add it */
337  if(nbr == NULL) {
338  /* Is the neighbor known by ds6? Drop this request if not.
339  * Typically, the neighbor is added upon receiving a DIO. */
340  lladdr = uip_ds6_nbr_lladdr_from_ipaddr(from);
341  if(lladdr == NULL) {
342  return NULL;
343  }
344 
345  /* Add neighbor to RPL table */
346  nbr = nbr_table_add_lladdr(rpl_neighbors, (linkaddr_t *)lladdr,
347  NBR_TABLE_REASON_RPL_DIO, dio);
348  if(nbr == NULL) {
349  LOG_ERR("failed to add neighbor\n");
350  return NULL;
351  }
352  }
353 
354  /* Update neighbor info from DIO */
355  nbr->rank = dio->rank;
356  nbr->dtsn = dio->dtsn;
357 #if RPL_WITH_MC
358  memcpy(&nbr->mc, &dio->mc, sizeof(nbr->mc));
359 #endif /* RPL_WITH_MC */
360 
361  return nbr;
362 }

364 static void
365 process_dio_from_current_dag(uip_ipaddr_t *from, rpl_dio_t *dio)
366 {
367  rpl_nbr_t *nbr;
368  uint8_t last_dtsn;
369 
370  /* Does the rank make sense at all? */
371  if(dio->rank < ROOT_RANK) {
372  return;
373  }
374 
375  /* If the DIO sender is on an older version of the DAG, ignore it. The node
376  will eventually hear the global repair and catch up. */
377  if(rpl_lollipop_greater_than(curr_instance.dag.version, dio->version)) {
378  return;
379  }
380 
381  /* The DIO is valid, proceed further */
382 
383  /* Update DIO counter for redundancy mngt */
384  if(dio->rank != RPL_INFINITE_RANK) {
385  curr_instance.dag.dio_counter++;
386  }
387 
388  /* The DIO has a newer version: global repair.
389  * Must come first, as it might remove all neighbors, and we then need
390  * to re-add this source of the DIO to the neighbor table */
391  if(rpl_lollipop_greater_than(dio->version, curr_instance.dag.version)) {
392  if(curr_instance.dag.rank == ROOT_RANK) { /* The root should not hear newer versions */
393  LOG_ERR("inconsistent DIO version (current: %u, received: %u), initiate global repair\n",
394  curr_instance.dag.version, dio->version);
395  curr_instance.dag.version = dio->version; /* Update version and trigger global repair */
396  rpl_global_repair();
397  } else {
398  LOG_WARN("new DIO version (current: %u, received: %u), apply global repair\n",
399  curr_instance.dag.version, dio->version);
400  global_repair_non_root(dio);
401  }
402  }
403 
404  /* Update IPv6 neighbor cache */
405  if(!rpl_icmp6_update_nbr_table(from, NBR_TABLE_REASON_RPL_DIO, dio)) {
406  LOG_ERR("IPv6 cache full, dropping DIO\n");
407  return;
408  }
409 
410  /* Add neighbor to RPL neighbor table */
411  nbr = rpl_neighbor_get_from_ipaddr(from);
412  last_dtsn = nbr != NULL ? nbr->dtsn : RPL_LOLLIPOP_INIT;
413 
414  if(!update_nbr_from_dio(from, dio)) {
415  LOG_ERR("neighbor table full, dropping DIO\n");
416  return;
417  }
418 
419  /* Init lifetime if not set yet. Refresh it at every DIO from preferred parent. */
420  if(curr_instance.dag.lifetime == 0 ||
421  (nbr != NULL && nbr == curr_instance.dag.preferred_parent)) {
422  LOG_INFO("refreshing lifetime\n");
423  curr_instance.dag.lifetime = RPL_LIFETIME(RPL_DAG_LIFETIME);
424  }
425 
426  /* If the source is our preferred parent and it increased DTSN, we increment
427  * our DTSN in turn and schedule a DAO (see RFC6550 section 9.6.) */
428  if(curr_instance.mop != RPL_MOP_NO_DOWNWARD_ROUTES) {
429  if(nbr != NULL && nbr == curr_instance.dag.preferred_parent && rpl_lollipop_greater_than(dio->dtsn, last_dtsn)) {
430  RPL_LOLLIPOP_INCREMENT(curr_instance.dtsn_out);
431  LOG_INFO("DTSN increment %u->%u, schedule new DAO with DTSN %u",
432  last_dtsn, dio->dtsn, curr_instance.dtsn_out);
433  rpl_timers_schedule_dao();
434  }
435  }
436 }

438 static int
439 init_dag(uint8_t instance_id, uip_ipaddr_t *dag_id, rpl_ocp_t ocp,
440  uip_ipaddr_t *prefix, unsigned prefix_len, uint8_t prefix_flags)
441 {
442  rpl_of_t *of;
443 
444  memset(&curr_instance, 0, sizeof(curr_instance));
445 
446  /* OF */
447  of = find_objective_function(ocp);
448  if(of == NULL) {
449  LOG_ERR("ignoring DIO with an unsupported OF: %u\n", ocp);
450  return 0;
451  }
452 
453  /* Prefix */
454  if(!rpl_set_prefix_from_addr(prefix, prefix_len, prefix_flags)) {
455  LOG_ERR("failed to set prefix\n");
456  return 0;
457  }
458 
459  /* Instnace */
460  curr_instance.instance_id = instance_id;
461  curr_instance.of = of;
462  curr_instance.dtsn_out = RPL_LOLLIPOP_INIT;
463  curr_instance.used = 1;
464 
465  /* DAG */
466  curr_instance.dag.rank = RPL_INFINITE_RANK;
467  curr_instance.dag.last_advertised_rank = RPL_INFINITE_RANK;
468  curr_instance.dag.lowest_rank = RPL_INFINITE_RANK;
469  curr_instance.dag.dao_last_seqno = RPL_LOLLIPOP_INIT;
470  curr_instance.dag.dao_last_acked_seqno = RPL_LOLLIPOP_INIT;
471  curr_instance.dag.dao_curr_seqno = RPL_LOLLIPOP_INIT;
472  memcpy(&curr_instance.dag.dag_id, dag_id, sizeof(curr_instance.dag.dag_id));
473 
474  return 1;
475 }

477 static int
478 init_dag_from_dio(rpl_dio_t *dio)
479 {
480  if(!init_dag(dio->instance_id, &dio->dag_id, dio->ocp,
481  &dio->prefix_info.prefix, dio->prefix_info.length, dio->prefix_info.flags)) {
482  return 0;
483  }
484 
485  /* Instnace */
486  curr_instance.mop = dio->mop;
487  curr_instance.mc.type = dio->mc.type;
488  curr_instance.mc.flags = dio->mc.flags;
489  curr_instance.mc.aggr = dio->mc.aggr;
490  curr_instance.mc.prec = dio->mc.prec;
491  curr_instance.max_rankinc = dio->dag_max_rankinc;
492  curr_instance.min_hoprankinc = dio->dag_min_hoprankinc;
493  curr_instance.dio_intdoubl = dio->dag_intdoubl;
494  curr_instance.dio_intmin = dio->dag_intmin;
495  curr_instance.dio_redundancy = dio->dag_redund;
496  curr_instance.default_lifetime = dio->default_lifetime;
497  curr_instance.lifetime_unit = dio->lifetime_unit;
498 
499  /* DAG */
500  curr_instance.dag.state = DAG_INITIALIZED;
501  curr_instance.dag.preference = dio->preference;
502  curr_instance.dag.grounded = dio->grounded;
503  curr_instance.dag.version = dio->version;
504  curr_instance.dag.dio_intcurrent = dio->dag_intmin;
505 
506  return 1;
507 }
508 /*---------------------------------------------------------------------------*/
509 static int
510 process_dio_init_dag(uip_ipaddr_t *from, rpl_dio_t *dio)
511 {
512  /* Check MOP */
513  if(dio->mop != RPL_MOP_NO_DOWNWARD_ROUTES && dio->mop != RPL_MOP_NON_STORING) {
514  LOG_WARN("ignoring DIO with an unsupported MOP: %d\n", dio->mop);
515  return 0;
516  }
517 
518  /* Initialize instance and DAG data structures */
519  if(!init_dag_from_dio(dio)) {
520  LOG_WARN("failed to initialize DAG\n");
521  return 0;
522  }
523 
524  /* Init OF and timers */
525  curr_instance.of->reset();
526  rpl_timers_dio_reset("Join");
527 #if RPL_WITH_PROBING
528  rpl_schedule_probing();
529 #endif /* RPL_WITH_PROBING */
530  /* Leave the network after RPL_DELAY_BEFORE_LEAVING in case we do not
531  find a parent */
532  LOG_INFO("initialized DAG with instance ID %u, DAG ID ",
533  curr_instance.instance_id);
534  LOG_INFO_6ADDR(&curr_instance.dag.dag_id);
535  LOG_INFO_(", prexix ");
536  LOG_INFO_6ADDR(&dio->prefix_info.prefix);
537  LOG_INFO_("/%u, rank %u\n", dio->prefix_info.length, curr_instance.dag.rank);
538 
539  LOG_ANNOTATE("#A init=%u\n", curr_instance.dag.dag_id.u8[sizeof(curr_instance.dag.dag_id) - 1]);
540 
541  LOG_WARN_("just joined, no parent yet, setting timer for leaving\n");
542  rpl_timers_schedule_leaving();
543 
544  return 1;
545 }
546 /*---------------------------------------------------------------------------*/
547 void
548 rpl_process_dio(uip_ipaddr_t *from, rpl_dio_t *dio)
549 {
550  if(!curr_instance.used && !rpl_dag_root_is_root()) {
551  /* Attempt to init our DAG from this DIO */
552  if(!process_dio_init_dag(from, dio)) {
553  LOG_WARN("failed to init DAG\n");
554  return;
555  }
556  }
557 
558  if(curr_instance.used
559  && curr_instance.instance_id == dio->instance_id
560  && uip_ipaddr_cmp(&curr_instance.dag.dag_id, &dio->dag_id)) {
561  process_dio_from_current_dag(from, dio);
562  rpl_dag_update_state();
563  }
564 }
565 /*---------------------------------------------------------------------------*/
566 void
567 rpl_process_dis(uip_ipaddr_t *from, int is_multicast)
568 {
569  if(is_multicast) {
570  rpl_timers_dio_reset("Multicast DIS");
571  } else {
572  /* Add neighbor to cache and reply to the unicast DIS with a unicast DIO*/
573  if(rpl_icmp6_update_nbr_table(from, NBR_TABLE_REASON_RPL_DIS, NULL) != NULL) {
574  LOG_INFO("unicast DIS, reply to sender\n");
575  rpl_icmp6_dio_output(from);
576  }
577  }
578 }

580 void
581 rpl_process_dao(uip_ipaddr_t *from, rpl_dao_t *dao)
582 {
583  if(dao->lifetime == 0) {
584  rpl_ns_expire_parent(from, &dao->parent_addr);
585  } else {
586  if(!rpl_ns_update_node(from, &dao->parent_addr, RPL_LIFETIME(dao->lifetime))) {
587  LOG_ERR("failed to add link on incoming DAO\n");
588  return;
589  }
590  }
591 
592 #if RPL_WITH_DAO_ACK
593  if(dao->flags & RPL_DAO_K_FLAG) {
594  rpl_timers_schedule_dao_ack(from, dao->sequence);
595  }
596 #endif /* RPL_WITH_DAO_ACK */
597 }
598 /*---------------------------------------------------------------------------*/
599 #if RPL_WITH_DAO_ACK
600 void
601 rpl_process_dao_ack(uint8_t sequence, uint8_t status)
602 {
603  /* Update dao_last_acked_seqno */
604  if(rpl_lollipop_greater_than(sequence, curr_instance.dag.dao_last_acked_seqno)) {
605  curr_instance.dag.dao_last_acked_seqno = sequence;
606  }
607  /* Is this an ACK for our last DAO? */
608  if(sequence == curr_instance.dag.dao_last_seqno) {
609  int status_ok = status < RPL_DAO_ACK_UNABLE_TO_ACCEPT;
610  if(curr_instance.dag.state == DAG_JOINED && status_ok) {
611  curr_instance.dag.state = DAG_REACHABLE;
612  rpl_timers_dio_reset("Reachable");
613  }
614 
615  if(!status_ok) {
616  /* We got a NACK, start poisoning and leave */
617  LOG_WARN("DAO-NACK received with seqno %u, status %u, poison and leave\n",
618  sequence, status);
619  curr_instance.dag.state = DAG_POISONING;
620  }
621  }
622 }
623 #endif /* RPL_WITH_DAO_ACK */

625 int
626 rpl_process_hbh(rpl_nbr_t *sender, uint16_t sender_rank, int loop_detected, int rank_error_signaled)
627 {
628  int drop = 0;
629 
630  if(loop_detected) {
631  if(rank_error_signaled) {
632 #if RPL_LOOP_ERROR_DROP
633  /* Drop packet and reset trickle timer, as per RFC6550 - 11.2.2.2 */
634  rpl_timers_dio_reset("HBH error");
635  LOG_WARN("rank error and loop detected, dropping\n");
636  drop = 1;
637 #endif /* RPL_LOOP_ERROR_DROP */
638  }
639  /* Attempt to repair the loop by sending a unicast DIO back to the sender
640  * so that it gets a fresh update of our rank. */
641  rpl_timers_schedule_unicast_dio(sender);
642  }
643 
644  if(rank_error_signaled) {
645  /* A rank error was signalled, attempt to repair it by updating
646  * the sender's rank from ext header */
647  if(sender != NULL) {
648  sender->rank = sender_rank;
649  /* Select DAG and preferred parent. In case of a parent switch,
650  the new parent will be used to forward the current packet. */
651  rpl_dag_update_state();
652  }
653  }
654 
655  return !drop;
656 }
