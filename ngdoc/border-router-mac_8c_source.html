69 void
70 packet_sent(uint8_t sessionid, uint8_t status, uint8_t tx)
71 {
72  if(sessionid < MAX_CALLBACKS) {
73  struct tx_callback *callback;
74  callback = &callbacks[sessionid];
75  packetbuf_clear();
76  packetbuf_attr_copyfrom(callback->attrs, callback->addrs);
77  mac_call_sent_callback(callback->cback, callback->ptr, status, tx);
78  } else {
79  PRINTF("*** ERROR: too high session id %d\n", sessionid);
80  }
81 }

83 static int
84 setup_callback(mac_callback_t sent, void *ptr)
85 {
86  struct tx_callback *callback;
87  int tmp = callback_pos;
88  callback = &callbacks[callback_pos];
89  callback->cback = sent;
90  callback->ptr = ptr;
91  packetbuf_attr_copyto(callback->attrs, callback->addrs);
92 
93  callback_pos++;
94  if(callback_pos >= MAX_CALLBACKS) {
95  callback_pos = 0;
96  }
97 
98  return tmp;
99 }

101 static void
102 send_packet(mac_callback_t sent, void *ptr)
103 {
104  int size;
105  /* 3 bytes per packet attribute is required for serialization */
106  uint8_t buf[PACKETBUF_NUM_ATTRS * 3 + PACKETBUF_SIZE + 3];
107  uint8_t sid;
108 
109  packetbuf_set_addr(PACKETBUF_ADDR_SENDER, &linkaddr_node_addr);
110 
111  /* ack or not ? */
112  packetbuf_set_attr(PACKETBUF_ATTR_MAC_ACK, 1);
113 
114  /* Will make it send only DATA packets... for now */
115  packetbuf_set_attr(PACKETBUF_ATTR_FRAME_TYPE, FRAME802154_DATAFRAME);
116  /* printf("Sending packet of type: %s \n", get_frame_type(packetbuf_attr(PACKETBUF_ATTR_FRAME_TYPE))); */
117 
118  if(NETSTACK_FRAMER.create() < 0) {
119  /* Failed to allocate space for headers */
120  PRINTF("br-rdc: send failed, too large header\n");
121  mac_call_sent_callback(sent, ptr, MAC_TX_ERR_FATAL, 1);
122  } else {
123  /* here we send the data over SLIP to the radio-chip */
124  size = 0;
125 #if SERIALIZE_ATTRIBUTES
126  size = packetutils_serialize_atts(&buf[3], sizeof(buf) - 3);
127 #endif
128  if(size < 0 || size + packetbuf_totlen() + 3 > sizeof(buf)) {
129  PRINTF("br-rdc: send failed, too large header\n");
130  mac_call_sent_callback(sent, ptr, MAC_TX_ERR_FATAL, 1);
131  } else {
132  sid = setup_callback(sent, ptr);
133 
134  buf[0] = '!';
135  buf[1] = 'S';
136  buf[2] = sid; /* sequence or session number for this packet */
137 
138  /* Copy packet data */
139  memcpy(&buf[3 + size], packetbuf_hdrptr(), packetbuf_totlen());
140 
141  write_to_slip(buf, packetbuf_totlen() + size + 3);
142  }
143  }
144 }
