554 PROCESS(cc1200_process, "CC1200 driver");
555 /*---------------------------------------------------------------------------*/
556 PROCESS_THREAD(cc1200_process, ev, data)
557 {
558 
559  PROCESS_POLLHANDLER(pollhandler());
560 
561  PROCESS_BEGIN();
562 
563 #if CC1200_USE_RX_WATCHDOG
564  while(1) {
565 
566  if((rf_flags & (RF_ON | RF_TX_ACTIVE)) == RF_ON) {
567 
568  PROCESS_WAIT_EVENT_UNTIL(etimer_expired(&et));
569  etimer_reset(&et);
570 
571  /*
572  * We are on and not in TX. As every function of this driver
573  * assures that we are in RX mode
574  * (using BUSYWAIT_UNTIL_STATE(STATE_RX, ...) construct) in
575  * either rx_rx(), idle_calibrate_rx() or transmit(),
576  * something probably went wrong in the rx interrupt handler
577  * if we are not in RX at this point.
578  */
579 
580  if(cc1200_arch_gpio0_read_pin() == 0) {
581 
582  /*
583  * GPIO de-asserts as soon as we leave RX for what reason ever. No
584  * reason to check RX as long as it is asserted (we are receiving a
585  * packet). We should never interfere with the rx interrupt if we
586  * check GPIO0 in advance...
587  */
588 
589  LOCK_SPI();
590  if(state() != STATE_RX) {
591  WARNING("RF: RX watchdog triggered!\n");
592  rx_rx();
593  }
594  RELEASE_SPI();
595 
596  }
597 
598  } else {
599  PROCESS_YIELD();
600  }
601 
602  }
603 #endif /* #if CC1200_USE_RX_WATCHDOG */
604 
605  PROCESS_YIELD_UNTIL(ev == PROCESS_EVENT_EXIT);
606 
607  PROCESS_END();
608 
609 }

611 /* Handle tasks left over from rx interrupt or because SPI was locked */
612 static void
613 pollhandler(void)
614 {
615 
616  if((rf_flags & (RF_ON + RF_POLL_RX_INTERRUPT)) ==
617  (RF_ON + RF_POLL_RX_INTERRUPT)) {
618  cc1200_rx_interrupt();
619  }
620 
621  if(rf_flags & RF_UPDATE_CHANNEL) {
622  /* We couldn't set the channel because we were busy. Try again now. */
623  set_channel(new_rf_channel);
624  }
625 
626  if(rx_pkt_len > 0) {
627 
628  int len;
629 
630  /*

710 /* Prepare the radio with a packet to be sent. */
711 static int
712 prepare(const void *payload, unsigned short payload_len)
713 {
714 
715  INFO("RF: Prepare (%d)\n", payload_len);
716 
717  if((payload_len < ACK_LEN) ||
718  (payload_len > CC1200_MAX_PAYLOAD_LEN)) {
719  ERROR("RF: Invalid payload length!\n");
720  return RADIO_TX_ERR;
721  }
722

724  memcpy(tx_pkt, payload, tx_pkt_len);
725 
726  return RADIO_TX_OK;
727 
728 }
729 /*---------------------------------------------------------------------------*/
730 /* Send the packet that has previously been prepared. */
731 static int
732 transmit(unsigned short transmit_len)
733 {
734 
735  uint8_t was_off = 0;
736  int ret = RADIO_TX_OK;
737 
738  INFO("RF: Transmit (%d)\n", transmit_len);
739 
740  if(transmit_len != tx_pkt_len) {
741  ERROR("RF: TX length mismatch!\n");
742  return RADIO_TX_ERR;
743  }
744 
745  /* TX ongoing. Inhibit channel update & ACK as soon as possible */
746  rf_flags |= RF_TX_ACTIVE;
747 
748  if(!(rf_flags & RF_ON)) {
749  /* Radio is off - turn it on */
750  was_off = 1;
751  on();
752  /* Radio is in RX now (and calibrated...) */
753  }
754 
755  if(tx_mode_value & RADIO_TX_MODE_SEND_ON_CCA) {
756  /* Perform clear channel assessment */
757  if(!channel_clear()) {
758  /* Channel occupied */
759  if(was_off) {
760  off();
761  }
762  rf_flags &= ~RF_TX_ACTIVE;
763  return RADIO_TX_COLLISION;
764  }
765  }
766 
767  /*
768  * Lock SPI here because "on()" and "channel_clear()"
769  * won't work if SPI is locked!
770  */
771  LOCK_SPI();
772 
773  /*
774  * Make sure we start from a sane state. idle() also disables
775  * the GPIO interrupt(s).
776  */
777  idle();
778 
779  /* Update output power */
780  if(new_txpower != txpower) {
781  update_txpower(new_txpower);
782  }
783 
784 #if !CC1200_AUTOCAL
785  /* Perform manual calibration unless just turned on */
786  if(!was_off) {
787  calibrate();
788  }
789 #endif
790 
791  /* Send data using TX FIFO */
792  if(idle_tx_rx((const uint8_t *)tx_pkt, tx_pkt_len) == RADIO_TX_OK) {
793 
794  /*
795  * TXOFF_MODE is set to RX,
796  * let's wait until we are in RX and turn on the GPIO IRQs
797  * again as they were turned off in idle()
798  */
799 
800  BUSYWAIT_UNTIL_STATE(STATE_RX,
801  CC1200_RF_CFG.tx_rx_turnaround);
802 
803  ENABLE_GPIO_INTERRUPTS();
804 
805  } else {
806 
807  /*
808  * Something went wrong during TX, idle_tx_rx() returns in IDLE
809  * state in this case.
810  * Turn on RX again unless we turn off anyway
811  */
812 
813  ret = RADIO_TX_ERR;
814  if(!was_off) {
815 #ifdef RF_FORCE_CALIBRATION
816  rf_flags |= RF_FORCE_CALIBRATION;
817 #endif
818  idle_calibrate_rx();
819  }
820  }
821 
822  /* Release SPI here because "off()" won't work if SPI is locked! */
823  RELEASE_SPI();
824 
825  if(was_off) {
826  off();
827  }
828 
829  /* TX completed */
830  rf_flags &= ~RF_TX_ACTIVE;
831 
832  return ret;
833 
834 }

836 /* Prepare & transmit a packet. */
837 static int
838 send(const void *payload, unsigned short payload_len)
839 {
840 
841  int ret;
842 
843  INFO("RF: Send (%d)\n", payload_len);
844 
845  /* payload_len checked within prepare() */
846  if((ret = prepare(payload, payload_len)) == RADIO_TX_OK) {
847  ret = transmit(payload_len);
848  }
849 
850  return ret;
851 
852 }
853 /*---------------------------------------------------------------------------*/
854 /* Read a received packet into a buffer. */
855 static int
856 read(void *buf, unsigned short buf_len)
857 {
858 
859  int len = 0;
860 
861  if(rx_pkt_len > 0) {
862 
863  int8_t rssi = rx_pkt[rx_pkt_len - 2];
864  /* CRC is already checked */
865  uint8_t crc_lqi = rx_pkt[rx_pkt_len - 1];
866 
867  len = rx_pkt_len - APPENDIX_LEN;
868 
869  if(len > buf_len) {
870 
871  ERROR("RF: Failed to read packet (too big)!\n");
872 
873  } else {
874 
875  INFO("RF: Read (%d bytes, %d dBm)\n", len, rssi);
876 
877  memcpy((void *)buf, (const void *)rx_pkt, len);
878 
879  /* Release rx_pkt */
880  rx_pkt_len = 0;
881 
882  packetbuf_set_attr(PACKETBUF_ATTR_RSSI, rssi);
883  /* Mask out CRC bit */
884  packetbuf_set_attr(PACKETBUF_ATTR_LINK_QUALITY,
885  crc_lqi & ~(1 << 7));
886  }
887 
888  }
889 
890  return len;
891

893 /*---------------------------------------------------------------------------*/
894 /*
895  * Perform a Clear-Channel Assessment (CCA) to find out if there is a
896  * packet in the air or not.
897  */
898 static int
899 channel_clear(void)
900 {
901 
902  uint8_t cca, was_off = 0;
903 
904  if(SPI_IS_LOCKED()) {
905  /* Probably locked in rx interrupt. Return "channel occupied" */
906  return 0;
907  }
908 
909  if(!(rf_flags & RF_ON)) {
910  /* We are off */
911  was_off = 1;
912  on();
913  }
914 
915  LOCK_SPI();
916 
917  RF_ASSERT(state() == STATE_RX);
918 
919  /*
920  * At this point we should be in RX. If GPIO0 is set, we are currently
921  * receiving a packet, no need to check the RSSI. Or is there any situation
922  * when we want access the channel even if we are currently receiving a
923  * packet???
924  */
925 
926  if(cc1200_arch_gpio0_read_pin() == 1) {
927  /* Channel occupied */
928  INFO("RF: CCA (0)\n");
929  cca = 0;
930  } else {
931 
932  uint8_t rssi0;
933 
934  /* Update CCA threshold */
935  if(new_cca_threshold != cca_threshold) {
936  update_cca_threshold(new_cca_threshold);
937  }
938 
939  /* Wait for CARRIER_SENSE_VALID signal */
940  BUSYWAIT_UNTIL(((rssi0 = single_read(CC1200_RSSI0))
941  & CC1200_CARRIER_SENSE_VALID),
942  RTIMER_SECOND / 100);
943  RF_ASSERT(rssi0 & CC1200_CARRIER_SENSE_VALID);
944 
945  if(rssi0 & CC1200_CARRIER_SENSE) {
946  /* Channel occupied */
947  INFO("RF: CCA (0)\n");
948  cca = 0;
949  } else {
950  /* Channel clear */
951  INFO("RF: CCA (1)\n");
952  cca = 1;
953  }
954 
955  }
956 
957  RELEASE_SPI();
958 
959  if(was_off) {
960  off();
961  }
962 
963  return cca;
964 
965 }

967 /*
968  * Check if the radio driver is currently receiving a packet.
969  *
970  * CSMA uses this function
971  * - to detect a collision before transmit()
972  * - to detect an incoming ACK
973  */
974 static int
975 receiving_packet(void)
976 {
977 
978  int ret = 0;
979 
980  if((rf_flags & (RF_ON | RF_TX_ACTIVE)) == RF_ON) {
981  /* We are on and not in TX */
982  if((cc1200_arch_gpio0_read_pin() == 1) || (rx_pkt_len != 0)) {
983 
984  /*
985  * SYNC word found or packet just received. Changing the criteria
986  * for this event might make it necessary to review the MAC timing
987  * parameters! Instead of (or in addition to) using GPIO0 we could also
988  * read out MODEM_STATUS1 (e.g. PQT reached), but this would not change
989  * the situation at least for CSMA as it uses two "blocking" timers
990  * (does not perform polling...). Therefore the overall timing
991  * of the ACK handling wouldn't change. It would just allow to detect an
992  * incoming packet a little bit earlier and help us with respect to
993  * collision avoidance (why not use channel_clear()
994  * at this point?).
995  */
996 
997  ret = 1;
998 
999  }
1000  }
1001 
1002  INFO("RF: Receiving (%d)\n", ret);
1003  return ret;
1004

1765 /*---------------------------------------------------------------------------*/
1766 /* Fill TX FIFO, start TX and wait for TX to complete (blocking!). */
1767 static int
1768 idle_tx_rx(const uint8_t *payload, uint16_t payload_len)
1769 {
1770 
1771 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1772  uint16_t bytes_left_to_write;
1773  uint8_t to_write;
1774  const uint8_t *p;
1775 #endif
1776 
1777 #if CC1200_802154G
1778  /* Prepare PHR for 802.15.4g frames */
1779  struct {
1780  uint8_t phra;
1781  uint8_t phrb;
1782  } phr;
1783 #if CC1200_802154G_CRC16
1784  payload_len += 2;
1785 #else
1786  payload_len += 4;
1787 #endif
1788  /* Frame length */
1789  phr.phrb = (uint8_t)(payload_len & 0x00FF);
1790  phr.phra = (uint8_t)((payload_len >> 8) & 0x0007);
1791 #if CC1200_802154G_WHITENING
1792  /* Enable Whitening */
1793  phr.phra |= (1 << 3);
1794 #endif /* #if CC1200_802154G_WHITENING */
1795 #if CC1200_802154G_CRC16
1796  /* FCS type 1, 2 Byte CRC */
1797  phr.phra |= (1 << 4);
1798 #endif /* #if CC1200_802154G_CRC16 */
1799 #endif /* #if CC1200_802154G */
1800 
1801  /* Prepare for RX */
1802  rf_flags &= ~RF_RX_PROCESSING_PKT;
1803  strobe(CC1200_SFRX);
1804 
1805  /* Flush TX FIFO */
1806  strobe(CC1200_SFTX);
1807 
1808 #if USE_SFSTXON
1809  /*
1810  * Enable synthesizer. Saves us a few s especially if it takes
1811  * long enough to fill the FIFO. This strobe must not be
1812  * send before SFTX!
1813  */
1814  strobe(CC1200_SFSTXON);
1815 #endif
1816 
1817  /* Configure GPIO0 to detect TX state */
1818  single_write(CC1200_IOCFG0, CC1200_IOCFG_MARC_2PIN_STATUS_0);
1819 
1820 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1821  /*
1822  * We already checked that GPIO2 is used if
1823  * CC1200_MAX_PAYLOAD_LEN > 127 / 126 in the header of this file
1824  */
1825  single_write(CC1200_IOCFG2, CC1200_IOCFG_TXFIFO_THR);
1826 #endif
1827 
1828 #if CC1200_802154G
1829  /* Write PHR */
1830  burst_write(CC1200_TXFIFO, (uint8_t *)&phr, PHR_LEN);
1831 #else
1832  /* Write length byte */
1833  burst_write(CC1200_TXFIFO, (uint8_t *)&payload_len, PHR_LEN);
1834 #endif /* #if CC1200_802154G */
1835 
1836  /*
1837  * Fill FIFO with data. If SPI is slow it might make sense
1838  * to divide this process into several chunks.
1839  * The best solution would be to perform TX FIFO refill
1840  * using an interrupt, but we are blocking here (= in TX) anyway...
1841  */
1842 
1843 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1844  to_write = MIN(payload_len, (CC1200_FIFO_SIZE - PHR_LEN));
1845  burst_write(CC1200_TXFIFO, payload, to_write);
1846  bytes_left_to_write = payload_len - to_write;
1847  p = payload + to_write;
1848 #else
1849  burst_write(CC1200_TXFIFO, payload, payload_len);
1850 #endif
1851 
1852 #if USE_SFSTXON
1853  /* Wait for synthesizer to be ready */
1854  BUSYWAIT_UNTIL_STATE(STATE_FSTXON, RTIMER_SECOND / 100);
1855 #endif
1856 
1857  /* Start TX */
1858  strobe(CC1200_STX);
1859 
1860  /* Wait for TX to start. */
1861  BUSYWAIT_UNTIL((cc1200_arch_gpio0_read_pin() == 1), RTIMER_SECOND / 100);
1862 
1863  /* Turned off at the latest in idle() */
1864  TX_LEDS_ON();
1865 
1866  /* Turned off at the latest in idle() */
1867  ENERGEST_ON(ENERGEST_TYPE_TRANSMIT);
1868 
1869  if((cc1200_arch_gpio0_read_pin() == 0) &&
1870  (single_read(CC1200_NUM_TXBYTES) != 0)) {
1871 
1872  /*
1873  * TX didn't start in time. We also check NUM_TXBYES
1874  * in case we missed the rising edge of the GPIO signal
1875  */
1876 
1877  ERROR("RF: TX doesn't start!\n");
1878 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1879  single_write(CC1200_IOCFG2, GPIO2_IOCFG);
1880 #endif
1881  idle();
1882 
1883  /* Re-configure GPIO0 */
1884  single_write(CC1200_IOCFG0, GPIO0_IOCFG);
1885 
1886  return RADIO_TX_ERR;
1887 
1888  }
1889 
1890 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1891  if(bytes_left_to_write != 0) {
1892  rtimer_clock_t t0;
1893  uint8_t s;
1894  t0 = RTIMER_NOW();
1895  do {
1896  if((bytes_left_to_write != 0) &&
1897  (cc1200_arch_gpio2_read_pin() == 0)) {
1898  /* TX TIFO is drained below FIFO_THRESHOLD. Re-fill... */
1899  to_write = MIN(bytes_left_to_write, FIFO_THRESHOLD);
1900  burst_write(CC1200_TXFIFO, p, to_write);
1901  bytes_left_to_write -= to_write;
1902  p += to_write;
1903  t0 += CC1200_RF_CFG.tx_pkt_lifetime;
1904  }
1905  } while((cc1200_arch_gpio0_read_pin() == 1) &&
1906  RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + CC1200_RF_CFG.tx_pkt_lifetime));
1907 
1908  /*
1909  * At this point we either left TX or a timeout occurred. If all went
1910  * well, we are in RX (or at least settling) now.
1911  * If we didn't manage to refill the TX FIFO, an underflow might
1912  * have occur-ed - the radio might be still in TX here!
1913  */
1914 
1915  s = state();
1916  if((s != STATE_RX) && (s != STATE_SETTLING)) {
1917 
1918  /*
1919  * Something bad happened. Wait for radio to enter a
1920  * stable state (in case of an error we are in TX here)
1921  */
1922 
1923  INFO("RF: TX failure!\n");
1924  BUSYWAIT_UNTIL((state() != STATE_TX), RTIMER_SECOND / 100);
1925  /* Re-configure GPIO2 */
1926  single_write(CC1200_IOCFG2, GPIO2_IOCFG);
1927  idle();
1928 
1929  /* Re-configure GPIO0 */
1930  single_write(CC1200_IOCFG0, GPIO0_IOCFG);
1931 
1932  return RADIO_TX_ERR;
1933 
1934  }
1935 
1936  } else {
1937  /* Wait for TX to complete */
1938  BUSYWAIT_UNTIL((cc1200_arch_gpio0_read_pin() == 0),
1939  CC1200_RF_CFG.tx_pkt_lifetime);
1940  }
1941 #else
1942  /* Wait for TX to complete */
1943  BUSYWAIT_UNTIL((cc1200_arch_gpio0_read_pin() == 0),
1944  CC1200_RF_CFG.tx_pkt_lifetime);
1945 #endif
1946 
1947  if(cc1200_arch_gpio0_read_pin() == 1) {
1948  /* TX takes to long - abort */
1949  ERROR("RF: TX takes to long!\n");
1950 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1951  /* Re-configure GPIO2 */
1952  single_write(CC1200_IOCFG2, GPIO2_IOCFG);
1953 #endif
1954  idle();
1955 
1956  /* Re-configure GPIO0 */
1957  single_write(CC1200_IOCFG0, GPIO0_IOCFG);
1958 
1959  return RADIO_TX_ERR;
1960 
1961  }
1962 
1963 #if (CC1200_MAX_PAYLOAD_LEN > (CC1200_FIFO_SIZE - PHR_LEN))
1964  /* Re-configure GPIO2 */
1965  single_write(CC1200_IOCFG2, GPIO2_IOCFG);
1966 #endif
1967 
1968  /* Re-configure GPIO0 */
1969  single_write(CC1200_IOCFG0, GPIO0_IOCFG);
1970 
1971  TX_LEDS_OFF();
1972 
1973  ENERGEST_OFF(ENERGEST_TYPE_TRANSMIT);
1974  ENERGEST_ON(ENERGEST_TYPE_LISTEN);
1975 
1976  return RADIO_TX_OK;
1977 
1978 }
