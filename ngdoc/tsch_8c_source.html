225 tsch_reset(void)
226 {
227  int i;
228  frame802154_set_pan_id(0xffff);
229  /* First make sure pending packet callbacks are sent etc */
230  process_post_synch(&tsch_pending_events_process, PROCESS_EVENT_POLL, NULL);
231  /* Reset neighbor queues */
232  tsch_queue_reset();
233  /* Remove unused neighbors */
234  tsch_queue_free_unused_neighbors();
235  tsch_queue_update_time_source(NULL);
236  /* Initialize global variables */
237  tsch_join_priority = 0xff;
238  TSCH_ASN_INIT(tsch_current_asn, 0, 0);
239  current_link = NULL;
240  /* Reset timeslot timing to defaults */
241  for(i = 0; i < tsch_ts_elements_count; i++) {
242  tsch_timing[i] = US_TO_RTIMERTICKS(tsch_default_timing_us[i]);
243  }
244 #ifdef TSCH_CALLBACK_LEAVING_NETWORK
245  TSCH_CALLBACK_LEAVING_NETWORK();
246 #endif
247  linkaddr_copy(&last_eb_nbr_addr, &linkaddr_null);
248 #if TSCH_AUTOSELECT_TIME_SOURCE
249  struct nbr_sync_stat *stat;
250  best_neighbor_eb_count = 0;
251  /* Remove all nbr stats */
252  stat = nbr_table_head(sync_stats);
253  while(stat != NULL) {
254  nbr_table_remove(sync_stats, stat);
255  stat = nbr_table_next(sync_stats, stat);
256  }
257 #endif /* TSCH_AUTOSELECT_TIME_SOURCE */
258  tsch_set_eb_period(TSCH_EB_PERIOD);
259 }

263 /* Tx callback for keepalive messages */
264 static void
265 keepalive_packet_sent(void *ptr, int status, int transmissions)
266 {
267  /* Update neighbor link statistics */
268  link_stats_packet_sent(packetbuf_addr(PACKETBUF_ADDR_RECEIVER), status, transmissions);
269  /* Call RPL callback if RPL is enabled */
270 #ifdef TSCH_CALLBACK_KA_SENT
271  TSCH_CALLBACK_KA_SENT(status, transmissions);
272 #endif /* TSCH_CALLBACK_KA_SENT */
273  LOG_INFO("KA sent to ");
274  LOG_INFO_LLADDR(packetbuf_addr(PACKETBUF_ADDR_RECEIVER));
275  LOG_INFO_(", st %d-%d\n", status, transmissions);
276 
277  /* We got no ack, try to recover by switching to the last neighbor we received an EB from */
278  if(status != MAC_TX_OK) {
279  if(linkaddr_cmp(&last_eb_nbr_addr, &linkaddr_null)) {
280  LOG_WARN("not able to re-synchronize, received no EB from other neighbors\n");
281  if(sync_count == 0) {
282  /* We got no synchronization at all in this session, leave the network */
283  tsch_disassociate();
284  }
285  } else {
286  LOG_WARN("re-synchronizing on ");
287  LOG_WARN_LLADDR(&last_eb_nbr_addr);
288  LOG_WARN_("\n");
289  /* We simply pick the last neighbor we receiver sync information from */
290  tsch_queue_update_time_source(&last_eb_nbr_addr);
291  tsch_join_priority = last_eb_nbr_jp + 1;
292  /* Try to get in sync ASAP */
293  tsch_schedule_keepalive_immediately();
294  return;
295  }
296  }
297 
298  tsch_schedule_keepalive();
299 }

302 static void
303 keepalive_send(void *ptr)
304 {
305  if(tsch_is_associated) {
306  struct tsch_neighbor *n = tsch_queue_get_time_source();

316 /*---------------------------------------------------------------------------*/
317 /* Set ctimer to send a keepalive message after expiration of TSCH_KEEPALIVE_TIMEOUT */
318 void
319 tsch_schedule_keepalive(void)
320 {
321  /* Pick a delay in the range [tsch_current_ka_timeout*0.9, tsch_current_ka_timeout[ */
322  if(!tsch_is_coordinator && tsch_is_associated && tsch_current_ka_timeout > 0) {
323  unsigned long delay = (tsch_current_ka_timeout - tsch_current_ka_timeout / 10)
324  + random_rand() % (tsch_current_ka_timeout / 10);
325  ctimer_set(&keepalive_timer, delay, keepalive_send, NULL);
326  }
327 }
328 /*---------------------------------------------------------------------------*/
329 /* Set ctimer to send a keepalive message immediately */
330 void
331 tsch_schedule_keepalive_immediately(void)
332 {
333  /* Pick a delay in the range [tsch_current_ka_timeout*0.9, tsch_current_ka_timeout[ */
334  if(!tsch_is_coordinator && tsch_is_associated) {
335  ctimer_set(&keepalive_timer, 0, keepalive_send, NULL);

339 static void
340 eb_input(struct input_packet *current_input)
341 {
342  /* LOG_INFO("EB received\n"); */
343  frame802154_t frame;
344  /* Verify incoming EB (does its ASN match our Rx time?),
345  * and update our join priority. */
346  struct ieee802154_ies eb_ies;
347 
348  if(tsch_packet_parse_eb(current_input->payload, current_input->len,
349  &frame, &eb_ies, NULL, 1)) {
350  /* PAN ID check and authentication done at rx time */
351 
352  /* Got an EB from a different neighbor than our time source, keep enough data
353  * to switch to it in case we lose the link to our time source */
354  struct tsch_neighbor *ts = tsch_queue_get_time_source();
355  if(ts == NULL || !linkaddr_cmp(&last_eb_nbr_addr, &ts->addr)) {
356  linkaddr_copy(&last_eb_nbr_addr, (linkaddr_t *)&frame.src_addr);
357  last_eb_nbr_jp = eb_ies.ie_join_priority;
358  }
359 
360 #if TSCH_AUTOSELECT_TIME_SOURCE
361  if(!tsch_is_coordinator) {
362  /* Maintain EB received counter for every neighbor */
363  struct eb_stat *stat = (struct eb_stat *)nbr_table_get_from_lladdr(eb_stats, (linkaddr_t *)&frame.src_addr);
364  if(stat == NULL) {
365  stat = (struct eb_stat *)nbr_table_add_lladdr(eb_stats, (linkaddr_t *)&frame.src_addr, NBR_TABLE_REASON_MAC, NULL);
366  }
367  if(stat != NULL) {
368  stat->rx_count++;
369  stat->jp = eb_ies.ie_join_priority;
370  best_neighbor_eb_count = MAX(best_neighbor_eb_count, stat->rx_count);
371  }
372  /* Select best time source */
373  struct eb_stat *best_stat = NULL;
374  stat = nbr_table_head(eb_stats);
375  while(stat != NULL) {
376  /* Is neighbor eligible as a time source? */
377  if(stat->rx_count > best_neighbor_eb_count / 2) {
378  if(best_stat == NULL ||
379  stat->jp < best_stat->jp) {
380  best_stat = stat;
381  }
382  }
383  stat = nbr_table_next(eb_stats, stat);
384  }
385  /* Update time source */
386  if(best_stat != NULL) {
387  tsch_queue_update_time_source(nbr_table_get_lladdr(eb_stats, best_stat));
388  tsch_join_priority = best_stat->jp + 1;
389  }
390  }
391 #endif /* TSCH_AUTOSELECT_TIME_SOURCE */
392 
393  /* Did the EB come from our time source? */
394  if(ts != NULL && linkaddr_cmp((linkaddr_t *)&frame.src_addr, &ts->addr)) {
395  /* Check for ASN drift */
396  int32_t asn_diff = TSCH_ASN_DIFF(current_input->rx_asn, eb_ies.ie_asn);
397  if(asn_diff != 0) {
398  /* We disagree with our time source's ASN -- leave the network */
399  LOG_WARN("! ASN drifted by %ld, leaving the network\n", asn_diff);
400  tsch_disassociate();
401  }
402 
403  if(eb_ies.ie_join_priority >= TSCH_MAX_JOIN_PRIORITY) {
404  /* Join priority unacceptable. Leave network. */
405  LOG_WARN("! EB JP too high %u, leaving the network\n",
406  eb_ies.ie_join_priority);
407  tsch_disassociate();
408  } else {
409 #if TSCH_AUTOSELECT_TIME_SOURCE
410  /* Update join priority */
411  if(tsch_join_priority != eb_ies.ie_join_priority + 1) {
412  LOG_INFO("update JP from EB %u -> %u\n",
413  tsch_join_priority, eb_ies.ie_join_priority + 1);
414  tsch_join_priority = eb_ies.ie_join_priority + 1;
415  }
416 #endif /* TSCH_AUTOSELECT_TIME_SOURCE */
417  }
418  }
419  }

422 /* Process pending input packet(s) */
423 static void
424 tsch_rx_process_pending()
425 {
426  int16_t input_index;
427  /* Loop on accessing (without removing) a pending input packet */
428  while((input_index = ringbufindex_peek_get(&input_ringbuf)) != -1) {
429  struct input_packet *current_input = &input_array[input_index];
430  frame802154_t frame;
431  uint8_t ret = frame802154_parse(current_input->payload, current_input->len, &frame);
432  int is_data = ret && frame.fcf.frame_type == FRAME802154_DATAFRAME;
433  int is_eb = ret
434  && frame.fcf.frame_version == FRAME802154_IEEE802154_2015
435  && frame.fcf.frame_type == FRAME802154_BEACONFRAME;
436 
437  if(is_data) {
438  /* Skip EBs and other control messages */
439  /* Copy to packetbuf for processing */
440  packetbuf_copyfrom(current_input->payload, current_input->len);
441  packetbuf_set_attr(PACKETBUF_ATTR_RSSI, current_input->rssi);
442  packetbuf_set_attr(PACKETBUF_ATTR_CHANNEL, current_input->channel);
443  }
444 
445  if(is_data) {
446  /* Pass to upper layers */
447  packet_input();
448  } else if(is_eb) {
449  eb_input(current_input);
450  }
451 
452  /* Remove input from ringbuf */
453  ringbufindex_get(&input_ringbuf);
454  }
455 }

457 /* Pass sent packets to upper layer */
458 static void
459 tsch_tx_process_pending(void)
460 {
461  int16_t dequeued_index;
462  /* Loop on accessing (without removing) a pending input packet */
463  while((dequeued_index = ringbufindex_peek_get(&dequeued_ringbuf)) != -1) {
464  struct tsch_packet *p = dequeued_array[dequeued_index];
465  /* Put packet into packetbuf for packet_sent callback */
466  queuebuf_to_packetbuf(p->qb);
467  LOG_INFO("packet sent to ");
468  LOG_INFO_LLADDR(packetbuf_addr(PACKETBUF_ADDR_RECEIVER));
469  LOG_INFO_(", seqno %u, status %d, tx %d\n",
470  packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO), p->ret, p->transmissions);
471  /* Call packet_sent callback */
472  mac_call_sent_callback(p->sent, p->ptr, p->ret, p->transmissions);
473  /* Free packet queuebuf */
474  tsch_queue_free_packet(p);
475  /* Free all unused neighbors */
476  tsch_queue_free_unused_neighbors();
477  /* Remove dequeued packet from ringbuf */
478  ringbufindex_get(&dequeued_ringbuf);
479  }

513 /*---------------------------------------------------------------------------*/
514 /* Attempt to associate to a network form an incoming EB */
515 static int
516 tsch_associate(const struct input_packet *input_eb, rtimer_clock_t timestamp)
517 {
518  frame802154_t frame;
519  struct ieee802154_ies ies;
520  uint8_t hdrlen;
521  int i;
522 
523  if(input_eb == NULL || tsch_packet_parse_eb(input_eb->payload, input_eb->len,
524  &frame, &ies, &hdrlen, 0) == 0) {
525  LOG_ERR("! failed to parse EB (len %u)\n", input_eb->len);
526  return 0;
527  }
528 
529  tsch_current_asn = ies.ie_asn;
530  tsch_join_priority = ies.ie_join_priority + 1;
531 
532 #if TSCH_JOIN_SECURED_ONLY
533  if(frame.fcf.security_enabled == 0) {
534  LOG_ERR("! parse_eb: EB is not secured\n");
535  return 0;
536  }
537 #endif /* TSCH_JOIN_SECURED_ONLY */
538 #if LLSEC802154_ENABLED
539  if(!tsch_security_parse_frame(input_eb->payload, hdrlen,
540  input_eb->len - hdrlen - tsch_security_mic_len(&frame),
541  &frame, (linkaddr_t*)&frame.src_addr, &tsch_current_asn)) {
542  LOG_ERR("! parse_eb: failed to authenticate\n");
543  return 0;
544  }
545 #endif /* LLSEC802154_ENABLED */
546 
547 #if !LLSEC802154_ENABLED
548  if(frame.fcf.security_enabled == 1) {
549  LOG_ERR("! parse_eb: we do not support security, but EB is secured\n");
550  return 0;
551  }
552 #endif /* !LLSEC802154_ENABLED */
553 
554 #if TSCH_JOIN_MY_PANID_ONLY
555  /* Check if the EB comes from the PAN ID we expect */
556  if(frame.src_pid != IEEE802154_PANID) {
557  LOG_ERR("! parse_eb: PAN ID %x != %x\n", frame.src_pid, IEEE802154_PANID);
558  return 0;
559  }
560 #endif /* TSCH_JOIN_MY_PANID_ONLY */
561 
562  /* There was no join priority (or 0xff) in the EB, do not join */
563  if(ies.ie_join_priority == 0xff) {
564  LOG_ERR("! parse_eb: no join priority\n");
565  return 0;
566  }
567 
568  /* TSCH timeslot timing */
569  for(i = 0; i < tsch_ts_elements_count; i++) {
570  if(ies.ie_tsch_timeslot_id == 0) {
571  tsch_timing[i] = US_TO_RTIMERTICKS(tsch_default_timing_us[i]);
572  } else {
573  tsch_timing[i] = US_TO_RTIMERTICKS(ies.ie_tsch_timeslot[i]);
574  }
575  }
576 
577  /* TSCH hopping sequence */
578  if(ies.ie_channel_hopping_sequence_id == 0) {
579  memcpy(tsch_hopping_sequence, TSCH_DEFAULT_HOPPING_SEQUENCE, sizeof(TSCH_DEFAULT_HOPPING_SEQUENCE));
580  TSCH_ASN_DIVISOR_INIT(tsch_hopping_sequence_length, sizeof(TSCH_DEFAULT_HOPPING_SEQUENCE));
581  } else {
582  if(ies.ie_hopping_sequence_len <= sizeof(tsch_hopping_sequence)) {
583  memcpy(tsch_hopping_sequence, ies.ie_hopping_sequence_list, ies.ie_hopping_sequence_len);
584  TSCH_ASN_DIVISOR_INIT(tsch_hopping_sequence_length, ies.ie_hopping_sequence_len);
585  } else {
586  LOG_ERR("! parse_eb: hopping sequence too long (%u)\n", ies.ie_hopping_sequence_len);
587  return 0;
588  }
589  }
590 
591 #if TSCH_CHECK_TIME_AT_ASSOCIATION > 0
592  /* Divide by 4k and multiply again to avoid integer overflow */
593  uint32_t expected_asn = 4096 * TSCH_CLOCK_TO_SLOTS(clock_time() / 4096, tsch_timing_timeslot_length); /* Expected ASN based on our current time*/
594  int32_t asn_threshold = TSCH_CHECK_TIME_AT_ASSOCIATION * 60ul * TSCH_CLOCK_TO_SLOTS(CLOCK_SECOND, tsch_timing_timeslot_length);
595  int32_t asn_diff = (int32_t)tsch_current_asn.ls4b - expected_asn;
596  if(asn_diff > asn_threshold) {
597  LOG_ERR("! EB ASN rejected %lx %lx %ld\n",
598  tsch_current_asn.ls4b, expected_asn, asn_diff);
599  return 0;
600  }
601 #endif
602 
603 #if TSCH_INIT_SCHEDULE_FROM_EB
604  /* Create schedule */
605  if(ies.ie_tsch_slotframe_and_link.num_slotframes == 0) {
606 #if TSCH_SCHEDULE_WITH_6TISCH_MINIMAL
607  LOG_INFO("parse_eb: no schedule, setting up minimal schedule\n");
608  tsch_schedule_create_minimal();
609 #else
610  LOG_INFO("parse_eb: no schedule\n");
611 #endif
612  } else {
613  /* First, empty current schedule */
614  tsch_schedule_remove_all_slotframes();
615  /* We support only 0 or 1 slotframe in this IE */
616  int num_links = ies.ie_tsch_slotframe_and_link.num_links;
617  if(num_links <= FRAME802154E_IE_MAX_LINKS) {
618  int i;
619  struct tsch_slotframe *sf = tsch_schedule_add_slotframe(
620  ies.ie_tsch_slotframe_and_link.slotframe_handle,
621  ies.ie_tsch_slotframe_and_link.slotframe_size);
622  for(i = 0; i < num_links; i++) {
623  tsch_schedule_add_link(sf,
624  ies.ie_tsch_slotframe_and_link.links[i].link_options,
625  LINK_TYPE_ADVERTISING, &tsch_broadcast_address,
626  ies.ie_tsch_slotframe_and_link.links[i].timeslot, ies.ie_tsch_slotframe_and_link.links[i].channel_offset);
627  }
628  } else {
629  LOG_ERR("! parse_eb: too many links in schedule (%u)\n", num_links);
630  return 0;
631  }
632  }
633 #endif /* TSCH_INIT_SCHEDULE_FROM_EB */
634 
635  if(tsch_join_priority < TSCH_MAX_JOIN_PRIORITY) {
636  struct tsch_neighbor *n;
637 
638  /* Add coordinator to list of neighbors, lock the entry */
639  n = tsch_queue_add_nbr((linkaddr_t *)&frame.src_addr);
640 
641  if(n != NULL) {
642  tsch_queue_update_time_source((linkaddr_t *)&frame.src_addr);
643 
644  /* Set PANID */
645  frame802154_set_pan_id(frame.src_pid);
646 
647  /* Synchronize on EB */
648  tsch_slot_operation_sync(timestamp - tsch_timing[tsch_ts_tx_offset], &tsch_current_asn);
649 
650  /* Update global flags */
651  tsch_is_associated = 1;
652  tsch_is_pan_secured = frame.fcf.security_enabled;
653  tx_count = 0;
654  rx_count = 0;
655  sync_count = 0;
656 
657  /* Start sending keep-alives now that tsch_is_associated is set */
658  tsch_schedule_keepalive();
659 
660 #ifdef TSCH_CALLBACK_JOINING_NETWORK
661  TSCH_CALLBACK_JOINING_NETWORK();
662 #endif
663 
664  tsch_association_count++;
665  LOG_INFO("association done (%u), sec %u, PAN ID %x, asn-%x.%lx, jp %u, timeslot id %u, hopping id %u, slotframe len %u with %u links, from ",
666  tsch_association_count,
667  tsch_is_pan_secured,
668  frame.src_pid,
669  tsch_current_asn.ms1b, tsch_current_asn.ls4b, tsch_join_priority,
670  ies.ie_tsch_timeslot_id,
671  ies.ie_channel_hopping_sequence_id,
672  ies.ie_tsch_slotframe_and_link.slotframe_size,
673  ies.ie_tsch_slotframe_and_link.num_links);
674  LOG_INFO_LLADDR((const linkaddr_t *)&frame.src_addr);
675  LOG_INFO_("\n");
676 
677  return 1;
678  }
679  }
680  LOG_ERR("! did not associate.\n");
681  return 0;
682 }

685 /*---------------------------------------------------------------------------*/
686 /* Scanning protothread, called by tsch_process:
687  * Listen to different channels, and when receiving an EB,
688  * attempt to associate.
689  */
690 PT_THREAD(tsch_scan(struct pt *pt))
691 {
692  PT_BEGIN(pt);
693 
694  static struct input_packet input_eb;
695  static struct etimer scan_timer;
696  /* Time when we started scanning on current_channel */
697  static clock_time_t current_channel_since;
698 
699  TSCH_ASN_INIT(tsch_current_asn, 0, 0);
700 
701  etimer_set(&scan_timer, CLOCK_SECOND / TSCH_ASSOCIATION_POLL_FREQUENCY);
702  current_channel_since = clock_time();
703 
704  while(!tsch_is_associated && !tsch_is_coordinator) {
705  /* Hop to any channel offset */
706  static uint8_t current_channel = 0;
707 
708  /* We are not coordinator, try to associate */
709  rtimer_clock_t t0;
710  int is_packet_pending = 0;
711  clock_time_t now_time = clock_time();
712 
713  /* Switch to a (new) channel for scanning */
714  if(current_channel == 0 || now_time - current_channel_since > TSCH_CHANNEL_SCAN_DURATION) {
715  /* Pick a channel at random in TSCH_JOIN_HOPPING_SEQUENCE */
716  uint8_t scan_channel = TSCH_JOIN_HOPPING_SEQUENCE[
717  random_rand() % sizeof(TSCH_JOIN_HOPPING_SEQUENCE)];
718  if(current_channel != scan_channel) {
719  NETSTACK_RADIO.set_value(RADIO_PARAM_CHANNEL, scan_channel);
720  current_channel = scan_channel;
721  LOG_INFO("scanning on channel %u\n", scan_channel);
722  }
723  current_channel_since = now_time;
724  }
725 
726  /* Turn radio on and wait for EB */
727  NETSTACK_RADIO.on();
728 
729  is_packet_pending = NETSTACK_RADIO.pending_packet();
730  if(!is_packet_pending && NETSTACK_RADIO.receiving_packet()) {
731  /* If we are currently receiving a packet, wait until end of reception */
732  t0 = RTIMER_NOW();
733  BUSYWAIT_UNTIL_ABS((is_packet_pending = NETSTACK_RADIO.pending_packet()), t0, RTIMER_SECOND / 100);
734  }
735 
736  if(is_packet_pending) {
737  /* Read packet */
738  input_eb.len = NETSTACK_RADIO.read(input_eb.payload, TSCH_PACKET_MAX_LEN);
739 
740  /* Save packet timestamp */
741  NETSTACK_RADIO.get_object(RADIO_PARAM_LAST_PACKET_TIMESTAMP, &t0, sizeof(rtimer_clock_t));
742 
743  /* Parse EB and attempt to associate */
744  LOG_INFO("scan: received packet (%u bytes) on channel %u\n", input_eb.len, current_channel);
745 
746  tsch_associate(&input_eb, t0);
747  }
748 
749  if(tsch_is_associated) {
750  /* End of association, turn the radio off */
751  NETSTACK_RADIO.off();
752  } else if(!tsch_is_coordinator) {
753  /* Go back to scanning */
754  etimer_reset(&scan_timer);
755  PT_WAIT_UNTIL(pt, etimer_expired(&scan_timer));
756  }
757  }
758 
759  PT_END(pt);
760 }

762 /*---------------------------------------------------------------------------*/
763 /* The main TSCH process */
764 PROCESS_THREAD(tsch_process, ev, data)
765 {
766  static struct pt scan_pt;
767 
768  PROCESS_BEGIN();
769 
770  while(1) {
771 
772  while(!tsch_is_associated) {
773  if(tsch_is_coordinator) {
774  /* We are coordinator, start operating now */
775  tsch_start_coordinator();
776  } else {
777  /* Start scanning, will attempt to join when receiving an EB */
778  PROCESS_PT_SPAWN(&scan_pt, tsch_scan(&scan_pt));
779  }
780  }
781 
782  /* We are part of a TSCH network, start slot operation */
783  tsch_slot_operation_start();
784 
785  /* Yield our main process. Slot operation will re-schedule itself
786  * as long as we are associated */
787  PROCESS_YIELD_UNTIL(!tsch_is_associated);
788 
789  LOG_WARN("leaving the network, stats: tx %lu, rx %lu, sync %lu\n",
790  tx_count, rx_count, sync_count);
791 
792  /* Will need to re-synchronize */
793  tsch_reset();
794  }
795 
796  PROCESS_END();
797 }

799 /*---------------------------------------------------------------------------*/
800 /* A periodic process to send TSCH Enhanced Beacons (EB) */
801 PROCESS_THREAD(tsch_send_eb_process, ev, data)
802 {
803  static struct etimer eb_timer;
804 
805  PROCESS_BEGIN();
806 
807  /* Wait until association */
808  etimer_set(&eb_timer, CLOCK_SECOND / 10);
809  while(!tsch_is_associated) {
810  PROCESS_WAIT_UNTIL(etimer_expired(&eb_timer));
811  etimer_reset(&eb_timer);
812  }
813 
814  /* Set an initial delay except for coordinator, which should send an EB asap */
815  if(!tsch_is_coordinator) {
816  etimer_set(&eb_timer, random_rand() % TSCH_EB_PERIOD);
817  PROCESS_WAIT_UNTIL(etimer_expired(&eb_timer));
818  }
819 
820  while(1) {
821  unsigned long delay;
822 
823  if(tsch_is_associated && tsch_current_eb_period > 0) {
824  /* Enqueue EB only if there isn't already one in queue */
825  if(tsch_queue_packet_count(&tsch_eb_address) == 0) {
826  uint8_t hdr_len = 0;
827  uint8_t tsch_sync_ie_offset;
828  /* Prepare the EB packet and schedule it to be sent */
829  if(tsch_packet_create_eb(&hdr_len, &tsch_sync_ie_offset) > 0) {
830  struct tsch_packet *p;
831  /* Enqueue EB packet */
832  if(!(p = tsch_queue_add_packet(&tsch_eb_address, NULL, NULL))) {
833  LOG_ERR("! could not enqueue EB packet\n");
834  } else {
835  LOG_INFO("TSCH: enqueue EB packet %u %u\n",
836  packetbuf_totlen(), packetbuf_hdrlen());
837  p->tsch_sync_ie_offset = tsch_sync_ie_offset;
838  p->header_len = hdr_len;
839  }
840  }
841  }
842  }
843  if(tsch_current_eb_period > 0) {
844  /* Next EB transmission with a random delay
845  * within [tsch_current_eb_period*0.75, tsch_current_eb_period[ */
846  delay = (tsch_current_eb_period - tsch_current_eb_period / 4)
847  + random_rand() % (tsch_current_eb_period / 4);
848  } else {
849  delay = TSCH_EB_PERIOD;
850  }
851  etimer_set(&eb_timer, delay);
852  PROCESS_WAIT_UNTIL(etimer_expired(&eb_timer));
853  }
854  PROCESS_END();
855 }

875 static void
876 tsch_init(void)
877 {
878  radio_value_t radio_rx_mode;
879  radio_value_t radio_tx_mode;
880  rtimer_clock_t t;
881 
882  /* Radio Rx mode */
883  if(NETSTACK_RADIO.get_value(RADIO_PARAM_RX_MODE, &radio_rx_mode) != RADIO_RESULT_OK) {
884  LOG_ERR("! radio does not support getting RADIO_PARAM_RX_MODE. Abort init.\n");
885  return;
886  }
887  /* Disable radio in frame filtering */
888  radio_rx_mode &= ~RADIO_RX_MODE_ADDRESS_FILTER;
889  /* Unset autoack */
890  radio_rx_mode &= ~RADIO_RX_MODE_AUTOACK;
891  /* Set radio in poll mode */
892  radio_rx_mode |= RADIO_RX_MODE_POLL_MODE;
893  if(NETSTACK_RADIO.set_value(RADIO_PARAM_RX_MODE, radio_rx_mode) != RADIO_RESULT_OK) {
894  LOG_ERR("! radio does not support setting required RADIO_PARAM_RX_MODE. Abort init.\n");
895  return;
896  }
897 
898  /* Radio Tx mode */
899  if(NETSTACK_RADIO.get_value(RADIO_PARAM_TX_MODE, &radio_tx_mode) != RADIO_RESULT_OK) {
900  LOG_ERR("! radio does not support getting RADIO_PARAM_TX_MODE. Abort init.\n");
901  return;
902  }
903  /* Unset CCA */
904  radio_tx_mode &= ~RADIO_TX_MODE_SEND_ON_CCA;
905  if(NETSTACK_RADIO.set_value(RADIO_PARAM_TX_MODE, radio_tx_mode) != RADIO_RESULT_OK) {
906  LOG_ERR("! radio does not support setting required RADIO_PARAM_TX_MODE. Abort init.\n");
907  return;
908  }
909  /* Test setting channel */
910  if(NETSTACK_RADIO.set_value(RADIO_PARAM_CHANNEL, TSCH_DEFAULT_HOPPING_SEQUENCE[0]) != RADIO_RESULT_OK) {
911  LOG_ERR("! radio does not support setting channel. Abort init.\n");
912  return;
913  }
914  /* Test getting timestamp */
915  if(NETSTACK_RADIO.get_object(RADIO_PARAM_LAST_PACKET_TIMESTAMP, &t, sizeof(rtimer_clock_t)) != RADIO_RESULT_OK) {
916  LOG_ERR("! radio does not support getting last packet timestamp. Abort init.\n");
917  return;
918  }
919  /* Check max hopping sequence length vs default sequence length */
920  if(TSCH_HOPPING_SEQUENCE_MAX_LEN < sizeof(TSCH_DEFAULT_HOPPING_SEQUENCE)) {
921  LOG_ERR("! TSCH_HOPPING_SEQUENCE_MAX_LEN < sizeof(TSCH_DEFAULT_HOPPING_SEQUENCE). Abort init.\n");
922  }
923 
924  /* Init the queuebuf and TSCH sub-modules */
925  queuebuf_init();
926  tsch_reset();
927  tsch_queue_init();
928  tsch_schedule_init();
929  tsch_log_init();
930  ringbufindex_init(&input_ringbuf, TSCH_MAX_INCOMING_PACKETS);
931  ringbufindex_init(&dequeued_ringbuf, TSCH_DEQUEUED_ARRAY_SIZE);
932 #if TSCH_AUTOSELECT_TIME_SOURCE
933  nbr_table_register(sync_stats, NULL);
934 #endif /* TSCH_AUTOSELECT_TIME_SOURCE */
935 
936  tsch_packet_seqno = random_rand();
937  tsch_is_initialized = 1;
938 
939 #if TSCH_AUTOSTART
940  /* Start TSCH operation.
941  * If TSCH_AUTOSTART is not set, one needs to call NETSTACK_MAC.on() to start TSCH. */
942  NETSTACK_MAC.on();
943 #endif /* TSCH_AUTOSTART */
944 
945 #if TSCH_WITH_SIXTOP
946  sixtop_init();
947 #endif
948 }

950 /* Function send for TSCH-MAC, puts the packet in packetbuf in the MAC queue */
951 static void
952 send_packet(mac_callback_t sent, void *ptr)
953 {
954  int ret = MAC_TX_DEFERRED;
955  int hdr_len = 0;
956  const linkaddr_t *addr = packetbuf_addr(PACKETBUF_ADDR_RECEIVER);
957 
958  if(!tsch_is_associated) {
959  if(!tsch_is_initialized) {
960  LOG_WARN("! not initialized (see earlier logs), drop outgoing packet\n");
961  } else {
962  LOG_WARN("! not associated, drop outgoing packet\n");
963  }
964  ret = MAC_TX_ERR;
965  mac_call_sent_callback(sent, ptr, ret, 1);
966  return;
967  }
968 
969  /* Ask for ACK if we are sending anything other than broadcast */
970  if(!linkaddr_cmp(addr, &linkaddr_null)) {
971  /* PACKETBUF_ATTR_MAC_SEQNO cannot be zero, due to a pecuilarity
972  in framer-802154.c. */
973  if(++tsch_packet_seqno == 0) {
974  tsch_packet_seqno++;
975  }
976  packetbuf_set_attr(PACKETBUF_ATTR_MAC_SEQNO, tsch_packet_seqno);
977  packetbuf_set_attr(PACKETBUF_ATTR_MAC_ACK, 1);
978  } else {
979  /* Broadcast packets shall be added to broadcast queue
980  * The broadcast address in Contiki is linkaddr_null which is equal
981  * to tsch_eb_address */
982  addr = &tsch_broadcast_address;
983  }
984 
985  packetbuf_set_attr(PACKETBUF_ATTR_FRAME_TYPE, FRAME802154_DATAFRAME);
986 
987 #if LLSEC802154_ENABLED
988  if(tsch_is_pan_secured) {
989  /* Set security level, key id and index */
990  packetbuf_set_attr(PACKETBUF_ATTR_SECURITY_LEVEL, TSCH_SECURITY_KEY_SEC_LEVEL_OTHER);
991  packetbuf_set_attr(PACKETBUF_ATTR_KEY_ID_MODE, FRAME802154_1_BYTE_KEY_ID_MODE); /* Use 1-byte key index */
992  packetbuf_set_attr(PACKETBUF_ATTR_KEY_INDEX, TSCH_SECURITY_KEY_INDEX_OTHER);
993  }
994 #endif /* LLSEC802154_ENABLED */
995 
996 #if !NETSTACK_CONF_BRIDGE_MODE
997  /*
998  * In the Contiki stack, the source address of a frame is set at the RDC
999  * layer. Since TSCH doesn't use any RDC protocol and bypasses the layer to
1000  * transmit a frame, it should set the source address by itself.
1001  */
1002  packetbuf_set_addr(PACKETBUF_ADDR_SENDER, &linkaddr_node_addr);
1003 #endif
1004 
1005  if((hdr_len = NETSTACK_FRAMER.create()) < 0) {
1006  LOG_ERR("! can't send packet due to framer error\n");
1007  ret = MAC_TX_ERR;
1008  } else {
1009  struct tsch_packet *p;
1010  /* Enqueue packet */
1011  p = tsch_queue_add_packet(addr, sent, ptr);
1012  if(p == NULL) {
1013  LOG_ERR("! can't send packet to ");
1014  LOG_ERR_LLADDR(addr);
1015  LOG_ERR_(" with seqno %u, queue %u %u\n",
1016  tsch_packet_seqno, tsch_queue_packet_count(addr), tsch_queue_global_packet_count());
1017  ret = MAC_TX_ERR;
1018  } else {
1019  p->header_len = hdr_len;
1020  LOG_INFO("send packet to ");
1021  LOG_INFO_LLADDR(addr);
1022  LOG_INFO_(" with seqno %u, queue %u %u, len %u %u\n",
1023  tsch_packet_seqno,
1024  tsch_queue_packet_count(addr), tsch_queue_global_packet_count(),
1025  p->header_len, queuebuf_datalen(p->qb));
1026  }
1027  }
1028  if(ret != MAC_TX_DEFERRED) {
1029  mac_call_sent_callback(sent, ptr, ret, 1);
1030  }
1031 }

1033 static void
1034 packet_input(void)
1035 {
1036  int frame_parsed = 1;
1037 
1038  frame_parsed = NETSTACK_FRAMER.parse();
1039 
1040  if(frame_parsed < 0) {
1041  LOG_ERR("! failed to parse %u\n", packetbuf_datalen());
1042  } else {
1043  int duplicate = 0;
1044 
1045  /* Seqno of 0xffff means no seqno */
1046  if(packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO) != 0xffff) {
1047  /* Check for duplicates */
1048  duplicate = mac_sequence_is_duplicate();
1049  if(duplicate) {
1050  /* Drop the packet. */
1051  LOG_WARN("! drop dup ll from ");
1052  LOG_WARN_LLADDR(packetbuf_addr(PACKETBUF_ADDR_SENDER));
1053  LOG_WARN_(" seqno %u\n", packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO));
1054  } else {
1055  mac_sequence_register_seqno();
1056  }
1057  }
1058 
1059  if(!duplicate) {
1060  LOG_INFO("received from ");
1061  LOG_INFO_LLADDR(packetbuf_addr(PACKETBUF_ADDR_SENDER));
1062  LOG_INFO_(" with seqno %u\n", packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO));
1063 #if TSCH_WITH_SIXTOP
1064  sixtop_input();
1065 #endif /* TSCH_WITH_SIXTOP */
1066  NETSTACK_NETWORK.input();
1067  }
1068  }
