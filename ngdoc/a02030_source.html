59 static void
60 mac_callback(void *ptr, int status, int transmissions)
61 {
62  sixp_trans_t *trans = (sixp_trans_t *)ptr;
63  sixp_trans_state_t new_state, current_state;
64 
65  assert(trans != NULL);
66  if(trans == NULL) {
67  LOG_ERR("6P: mac_callback() fails because trans is NULL\n");
68  return;
69  }
70 
71  current_state = sixp_trans_get_state(trans);
72  if(status == MAC_TX_OK) {
73  switch(current_state) {
74  case SIXP_TRANS_STATE_INIT:
75  new_state = SIXP_TRANS_STATE_REQUEST_SENT;
76  break;
77  case SIXP_TRANS_STATE_REQUEST_RECEIVED:
78  new_state = SIXP_TRANS_STATE_RESPONSE_SENT;
79  break;
80  case SIXP_TRANS_STATE_RESPONSE_RECEIVED:
81  new_state = SIXP_TRANS_STATE_CONFIRMATION_SENT;
82  break;
83  default:
84  LOG_ERR("6P: mac_callback() fails because of an unexpected state (%u)\n",
85  current_state);
86  return;
87  }
88  } else {
89  /*
90  * In a case of transmission failure of a request, a corresponding SF would
91  * retransmit the request with a new transaction. For a response or a
92  * confirmation, the same transaction will be used for retransmission as
93  * long as it doesn't have timeout.
94  */
95  if(current_state == SIXP_TRANS_STATE_INIT) {
96  /* request case */
97  new_state = SIXP_TRANS_STATE_TERMINATING;
98  } else {
99  /* response or confirmation case: stay the same state */
100  new_state = current_state;
101  }
102  }
103 
104  if(new_state != current_state &&
105  sixp_trans_transit_state(trans, new_state) != 0) {
106  LOG_ERR("6P: mac_callback() fails because of state transition failure\n");
107  LOG_ERR("6P: something wrong; we're terminating the trans %p\n", trans);
108  (void)sixp_trans_transit_state(trans, SIXP_TRANS_STATE_TERMINATING);
109  return;
110  }
111 
112  sixp_trans_invoke_callback(trans,
113  status == MAC_TX_OK ?
114  SIXP_OUTPUT_STATUS_SUCCESS :
115  SIXP_OUTPUT_STATUS_FAILURE);
116  sixp_trans_set_callback(trans, NULL, NULL, 0);
117 }

119 static int
120 send_back_error(sixp_pkt_type_t type, sixp_pkt_code_t code,
121  uint8_t sfid, uint8_t seqno,
122  const linkaddr_t *dest_addr)
123 {
124  /* create a 6P packet within packetbuf */
125  if(sixp_pkt_create(type, code, sfid, seqno, 0, NULL, 0, NULL) < 0) {
126  LOG_ERR("6P: failed to create a 6P packet to return an error [rc:%u]\n",
127  code.value);
128  return -1;
129  }
130  /* we don't care about how the transmission goes; no need to set callback */
131  sixtop_output(dest_addr, NULL, NULL);
132  return 0;
133 }

135 void
136 sixp_input(const uint8_t *buf, uint16_t len, const linkaddr_t *src_addr)
137 {
138  sixp_pkt_t pkt;
139  sixp_nbr_t *nbr;
140  uint8_t invalid_schedule_generation;
141  sixp_trans_t *trans;
142  const sixtop_sf_t *sf;
143  int16_t seqno;
144  int16_t gen;
145  int ret;
146 
147  assert(buf != NULL && src_addr != NULL);
148  if(buf == NULL || src_addr == NULL) {
149  return;
150  }
151 
152  if(sixp_pkt_parse(buf, len, &pkt) < 0) {
153  LOG_ERR("6P: sixp_input() fails because off a malformed 6P packet\n");
154  return;
155  }
156 
157  if(pkt.type != SIXP_PKT_TYPE_REQUEST &&
158  pkt.type != SIXP_PKT_TYPE_RESPONSE &&
159  pkt.type != SIXP_PKT_TYPE_CONFIRMATION) {
160  LOG_ERR("6P: sixp_input() fails because of unsupported type [type:%u]\n",
161  pkt.type);
162  return;
163  }
164 
165  if((sf = sixtop_find_sf(pkt.sfid)) == NULL) {
166  LOG_ERR("6P: sixp_input() fails because SF [sfid:%u] is unavailable\n",
167  pkt.sfid);
168  /*
169  * XXX: what if the incoming packet is a response? confirmation should be
170  * sent back?
171  */
172  if(send_back_error(SIXP_PKT_TYPE_RESPONSE,
173  (sixp_pkt_code_t)(uint8_t)SIXP_PKT_RC_SFID,
174  pkt.sfid, pkt.seqno, src_addr) < 0) {
175  LOG_ERR("6P: sixp_input() fails to return an error response\n");
176  };
177  return;
178  }
179 
180  nbr = sixp_nbr_find(src_addr);
181  /* Generation Management */
182  if(pkt.code.value == SIXP_PKT_CMD_CLEAR) {
183  /* Not need to validate generation counters in a case of CMD_CLEAR */
184  invalid_schedule_generation = 0;
185  } else if(nbr == NULL) {
186  if(pkt.gen == 0) {
187  invalid_schedule_generation = 0; /* valid combination */
188  } else {
189  LOG_ERR("6P: GEN should be 0 because of no corresponding nbr\n");
190  invalid_schedule_generation = 1;
191  }
192  } else {
193  if((gen = sixp_nbr_get_gen(nbr)) < 0) {
194  LOG_ERR("6P: unexpected error; cannot get our GEN\n");
195  return;
196  }
197  LOG_INFO("6P: received GEN %u, our GEN: %u\n",
198  pkt.gen, sixp_nbr_get_gen(nbr));
199  if(pkt.gen == gen) {
200  invalid_schedule_generation = 0; /* valid combination */
201  } else {
202  invalid_schedule_generation = 1;
203  }
204  }
205  if(invalid_schedule_generation) {
206  LOG_ERR("6P: sixp_input() fails because of schedule generation mismatch\n");
207  return;
208  }
209 
210  /* Transaction Management */
211  trans = sixp_trans_find(src_addr);
212 
213  if(pkt.type == SIXP_PKT_TYPE_REQUEST) {
214  if(trans != NULL) {
215  /* Error: not supposed to have another transaction with the peer. */
216  LOG_ERR("6P: sixp_input() fails because another request [peer_addr:");
217  LOG_ERR_LLADDR((const linkaddr_t *)src_addr);
218  LOG_ERR_(" seqno:%u] is in process\n", sixp_trans_get_seqno(trans));
219  if(send_back_error(SIXP_PKT_TYPE_RESPONSE,
220  (sixp_pkt_code_t)(uint8_t)SIXP_PKT_RC_BUSY,
221  pkt.sfid, pkt.seqno, src_addr) < 0) {
222  LOG_ERR("6P: sixp_input() fails to return an error response");
223  }
224  return;
225  } else if((trans = sixp_trans_alloc(&pkt, src_addr)) == NULL) {
226  LOG_ERR("6P: sixp_input() fails because of lack of memory\n");
227  if(send_back_error(SIXP_PKT_TYPE_RESPONSE,
228  (sixp_pkt_code_t)(uint8_t)SIXP_PKT_RC_NORES,
229  pkt.sfid, pkt.seqno, src_addr) < 0) {
230  LOG_ERR("6P: sixp_input() fails to return an error response\n");
231  }
232  return;
233  }
234  } else if(pkt.type == SIXP_PKT_TYPE_RESPONSE ||
235  pkt.type == SIXP_PKT_TYPE_CONFIRMATION) {
236  if(trans == NULL) {
237  /* Error: should have a transaction for incoming packet */
238  LOG_ERR("6P: sixp_input() fails because of no trans [peer_addr:");
239  LOG_ERR_LLADDR((const linkaddr_t *)src_addr);
240  LOG_ERR_("]\n");
241  return;
242  } else if((seqno = sixp_trans_get_seqno(trans)) < 0 ||
243  seqno != pkt.seqno) {
244  LOG_ERR("6P: sixp_input() fails because of invalid seqno [seqno:%u, %u]\n",
245  seqno, pkt.seqno);
246  return;
247  }
248  }
249 
250  /* state transition */
251  assert(trans != NULL);
252  switch(pkt.type) {
253  case SIXP_PKT_TYPE_REQUEST:
254  ret = sixp_trans_transit_state(trans,
255  SIXP_TRANS_STATE_REQUEST_RECEIVED);
256  break;
257  case SIXP_PKT_TYPE_RESPONSE:
258  ret = sixp_trans_transit_state(trans,
259  SIXP_TRANS_STATE_RESPONSE_RECEIVED);
260  break;
261  case SIXP_PKT_TYPE_CONFIRMATION:
262  ret = sixp_trans_transit_state(trans,
263  SIXP_TRANS_STATE_CONFIRMATION_RECEIVED);
264  break;
265  default:
266  LOG_ERR("6P: sixp_input() fails because of unsupported type [type:%u]\n",
267  pkt.type);
268  return;
269  }
270  if(ret < 0) {
271  LOG_ERR("6P: sixp_input() fails because of state transition failure\n");
272  LOG_ERR("6P: something wrong; we're terminating the trans %p\n", trans);
273  (void)sixp_trans_transit_state(trans, SIXP_TRANS_STATE_TERMINATING);
274  return;
275  }
276 
277  if(sf->input != NULL) {
278  sf->input(pkt.type, pkt.code, pkt.body, pkt.body_len, src_addr);
279  }
280 
281  return;
282 }

284 int
285 sixp_output(sixp_pkt_type_t type, sixp_pkt_code_t code, uint8_t sfid,
286  const uint8_t *body, uint16_t body_len,
287  const linkaddr_t *dest_addr,
288  sixp_sent_callback_t func, void *arg, uint16_t arg_len)
289 {
290  sixp_trans_t *trans;
291  sixp_nbr_t *nbr;
292  sixp_pkt_cmd_t cmd;
293  int16_t seqno, gen;
294  sixp_pkt_t pkt;
295 
296  assert(dest_addr != NULL);
297 
298  /* validate the state of a transaction with a specified peer */
299  trans = sixp_trans_find(dest_addr);
300  if(type == SIXP_PKT_TYPE_REQUEST) {
301  if(trans != NULL) {
302  LOG_ERR("6P: sixp_output() fails because another trans for [peer_addr:");
303  LOG_ERR_LLADDR((const linkaddr_t *)dest_addr);
304  LOG_ERR_("] is in process\n");
305  return -1;
306  } else {
307  /* ready to send a request */
308  /* we're going to allocate a new transaction later */
309  }
310  } else if(type == SIXP_PKT_TYPE_RESPONSE) {
311  if(trans == NULL) {
312  LOG_ERR("6P: sixp_output() fails because of no transaction [peer_addr:");
313  LOG_ERR_LLADDR((const linkaddr_t *)dest_addr);
314  LOG_ERR_("]\n");
315  return -1;
316  } else if(sixp_trans_get_state(trans) !=
317  SIXP_TRANS_STATE_REQUEST_RECEIVED) {
318  LOG_ERR("6P: sixp_output() fails because of invalid transaction state\n");
319  return -1;
320  } else {
321  /* ready to send a response */
322  }
323  } else if(type == SIXP_PKT_TYPE_CONFIRMATION) {
324  if(trans == NULL) {
325  LOG_ERR("6P: sixp_output() fails because of no transaction [peer_addr:");
326  LOG_ERR_LLADDR((const linkaddr_t *)dest_addr);
327  LOG_ERR_("\n");
328  return -1;
329  } else if(sixp_trans_get_state(trans) !=
330  SIXP_TRANS_STATE_RESPONSE_RECEIVED) {
331  LOG_ERR("6P: sixp_output() fails because of invalid transaction state\n");
332  return -1;
333  } else {
334  /* ready to send a confirmation */
335  }
336  } else {
337  LOG_ERR("6P: sixp_output() fails because of unsupported type [type:%u]\n",
338  type);
339  return -1;
340  }
341 
342  nbr = sixp_nbr_find(dest_addr);
343 
344  /*
345  * Make sure we have a nbr for the peer if the packet is a response with
346  * success so that we can manage the schedule generation.
347  */
348  if(nbr == NULL &&
349  type == SIXP_PKT_TYPE_RESPONSE && code.value == SIXP_PKT_RC_SUCCESS &&
350  ((cmd = sixp_trans_get_cmd(trans)) == SIXP_PKT_CMD_ADD ||
351  cmd == SIXP_PKT_CMD_DELETE) &&
352  (nbr = sixp_nbr_alloc(dest_addr)) == NULL) {
353  LOG_ERR("6P: sixp_output() fails because of no memory for another nbr\n");
354  return -1;
355  }
356 
357  /* set SeqNum */
358  if(type == SIXP_PKT_TYPE_REQUEST) {
359  if(nbr == NULL &&
360  (nbr = sixp_nbr_alloc(dest_addr)) == NULL) {
361  LOG_ERR("6P: sixp_output() fails because it fails to allocate a nbr\n");
362  return -1;
363  }
364  if((seqno = sixp_nbr_get_next_seqno(nbr)) < 0){
365  LOG_ERR("6P: sixp_output() fails to get the next sequence number\n");
366  return -1;
367  }
368  if(sixp_nbr_increment_next_seqno(nbr) < 0) {
369  LOG_ERR("6P: sixp_output() fails to increment the next sequence number\n");
370  return -1;
371  }
372  } else {
373  assert(trans != NULL);
374  if((seqno = sixp_trans_get_seqno(trans)) < 0) {
375  LOG_ERR("6P: sixp_output() fails because it fails to get seqno\n");
376  return -1;
377  }
378  }
379 
380  /* set GEN */
381  if(nbr == NULL) {
382  gen = 0;
383  } else if((gen = sixp_nbr_get_gen(nbr)) < 0) {
384  LOG_ERR("6P: sixp_output() fails to get GEN\n");
385  return -1;
386  }
387 
388  /* create a 6P packet within packetbuf */
389  if(sixp_pkt_create(type, code, sfid,
390  (uint8_t)seqno, (uint8_t)gen,
391  body, body_len,
392  type == SIXP_PKT_TYPE_REQUEST ? &pkt : NULL) < 0) {
393  LOG_ERR("6P: sixp_output() fails to create a 6P packet\n");
394  return -1;
395  }
396 
397  /* allocate a transaction for a sending request */
398  if(type == SIXP_PKT_TYPE_REQUEST) {
399  assert(trans == NULL);
400  if((trans = sixp_trans_alloc(&pkt, dest_addr)) == NULL) {
401  LOG_ERR("6P: sixp_output() is aborted because of no memory\n");
402  return -1;
403  } else {
404  /* ready for proceed */
405  }
406  }
407 
408  assert(trans != NULL);
409  sixp_trans_set_callback(trans, func, arg, arg_len);
410  sixtop_output(dest_addr, mac_callback, trans);
411 
412  return 0;
413 }
