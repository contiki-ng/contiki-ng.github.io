120 /*
121  * Macro to define a periodic resource.
122  * The corresponding [name]_periodic_handler() function will be called every period.
123  * For instance polling a sensor and publishing a changed value to subscribed clients would be done there.
124  * The subscriber list will be maintained by the final_handler rest_subscription_handler() (see rest-mapping header file).
125  */
126 #define PERIODIC_RESOURCE(name, attributes, get_handler, post_handler, put_handler, delete_handler, period, periodic_handler) \
127  periodic_resource_t periodic_##name; \
128  resource_t name = { NULL, NULL, IS_OBSERVABLE | IS_PERIODIC, attributes, get_handler, post_handler, put_handler, delete_handler, { .periodic = &periodic_##name } }; \
129  periodic_resource_t periodic_##name = { NULL, &name, period, { { 0 } }, periodic_handler };
