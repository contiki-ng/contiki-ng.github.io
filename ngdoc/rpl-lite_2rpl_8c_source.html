55 int
56 rpl_lollipop_greater_than(int a, int b)
57 {
58  /* Check if we are comparing an initial value with an old value */
59  if(a > RPL_LOLLIPOP_CIRCULAR_REGION && b <= RPL_LOLLIPOP_CIRCULAR_REGION) {
60  return (RPL_LOLLIPOP_MAX_VALUE + 1 + b - a) > RPL_LOLLIPOP_SEQUENCE_WINDOWS;
61  }
62  /* Otherwise check if a > b and comparable => ok, or
63  if they have wrapped and are still comparable */
64  return (a > b && (a - b) < RPL_LOLLIPOP_SEQUENCE_WINDOWS) ||
65  (a < b && (b - a) > (RPL_LOLLIPOP_CIRCULAR_REGION + 1-
66  RPL_LOLLIPOP_SEQUENCE_WINDOWS));
67 }

69 const uip_ipaddr_t *
70 rpl_get_global_address(void)
71 {
72  int i;
73  uint8_t state;
74  uip_ipaddr_t *ipaddr = NULL;
75  uip_ipaddr_t *prefix = NULL;
76  uint8_t prefix_length = 0;
77 
78  if(curr_instance.used && curr_instance.dag.prefix_info.length != 0) {
79  prefix = &curr_instance.dag.prefix_info.prefix;
80  prefix_length = curr_instance.dag.prefix_info.length;
81  }
82 
83  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
84  state = uip_ds6_if.addr_list[i].state;
85  if(uip_ds6_if.addr_list[i].isused &&
86  state == ADDR_PREFERRED &&
87  !uip_is_addr_linklocal(&uip_ds6_if.addr_list[i].ipaddr) &&
88  (prefix == NULL || uip_ipaddr_prefixcmp(prefix, &uip_ds6_if.addr_list[i].ipaddr, prefix_length))) {
89  ipaddr = &uip_ds6_if.addr_list[i].ipaddr;
90  }
91  }
92  return ipaddr;
93 }

95 void
96 rpl_link_callback(const linkaddr_t *addr, int status, int numtx)
97 {
98  if(curr_instance.used == 1 ) {
99  rpl_nbr_t *nbr = rpl_neighbor_get_from_lladdr((uip_lladdr_t *)addr);
100  if(nbr != NULL) {
101  /* Link stats were updated, and we need to update our internal state.
102  Updating from here is unsafe; postpone */
103  LOG_INFO("packet sent to ");
104  LOG_INFO_LLADDR(addr);
105  LOG_INFO_(", status %u, tx %u, new link metric %u\n", status, numtx, rpl_neighbor_get_link_metric(nbr));
106  rpl_timers_schedule_state_update();
107  }
108  }
109 }

131 void
132 rpl_reset_prefix(rpl_prefix_t *last_prefix)
133 {
134  uip_ipaddr_t ipaddr;
135  uip_ds6_addr_t *rep;
136  set_ip_from_prefix(&ipaddr, last_prefix);
137  rep = uip_ds6_addr_lookup(&ipaddr);
138  if(rep != NULL) {
139  LOG_INFO("removing global IP address ");
140  LOG_INFO_6ADDR(&ipaddr);
141  LOG_INFO_("\n");
142  uip_ds6_addr_rm(rep);
143  }
144  curr_instance.dag.prefix_info.length = 0;
145 }
146 /*---------------------------------------------------------------------------*/
147 int
148 rpl_set_prefix_from_addr(uip_ipaddr_t *addr, unsigned len, uint8_t flags)
149 {
150  uip_ipaddr_t ipaddr;
151 
152  if(addr == NULL || len == 0 || len > 128 || !(flags & UIP_ND6_RA_FLAG_AUTONOMOUS)) {
153  LOG_WARN("prefix not included, not-supported or invalid\n");
154  return 0;
155  }
156 
157  /* Try and initialize prefix */
158  memset(&curr_instance.dag.prefix_info.prefix, 0, sizeof(rpl_prefix_t));
159  memcpy(&curr_instance.dag.prefix_info.prefix, addr, (len + 7) / 8);
160  curr_instance.dag.prefix_info.length = len;
161  curr_instance.dag.prefix_info.lifetime = RPL_ROUTE_INFINITE_LIFETIME;
162  curr_instance.dag.prefix_info.flags = flags;
163 
164  /* Add global address if not already there */
165  set_ip_from_prefix(&ipaddr, &curr_instance.dag.prefix_info);
166  if(uip_ds6_addr_lookup(&ipaddr) == NULL) {
167  LOG_INFO("adding global IP address ");
168  LOG_INFO_6ADDR(&ipaddr);
169  LOG_INFO_("\n");
170  uip_ds6_addr_add(&ipaddr, 0, ADDR_AUTOCONF);
171  }
172  return 1;
173 }
