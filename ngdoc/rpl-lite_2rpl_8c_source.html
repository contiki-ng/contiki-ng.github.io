54 int
55 rpl_lollipop_greater_than(int a, int b)
56 {
57  /* Check if we are comparing an initial value with an old value */
58  if(a > RPL_LOLLIPOP_CIRCULAR_REGION && b <= RPL_LOLLIPOP_CIRCULAR_REGION) {
59  return (RPL_LOLLIPOP_MAX_VALUE + 1 + b - a) > RPL_LOLLIPOP_SEQUENCE_WINDOWS;
60  }
61  /* Otherwise check if a > b and comparable => ok, or
62  if they have wrapped and are still comparable */
63  return (a > b && (a - b) < RPL_LOLLIPOP_SEQUENCE_WINDOWS) ||
64  (a < b && (b - a) > (RPL_LOLLIPOP_CIRCULAR_REGION + 1-
65  RPL_LOLLIPOP_SEQUENCE_WINDOWS));
66 }

68 const uip_ipaddr_t *
69 rpl_get_global_address(void)
70 {
71  int i;
72  uint8_t state;
73  uip_ipaddr_t *ipaddr = NULL;
74 
75  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
76  state = uip_ds6_if.addr_list[i].state;
77  if(uip_ds6_if.addr_list[i].isused &&
78  state == ADDR_PREFERRED &&
79  !uip_is_addr_linklocal(&uip_ds6_if.addr_list[i].ipaddr)) {
80  ipaddr = &uip_ds6_if.addr_list[i].ipaddr;
81  }
82  }
83  return ipaddr;
84 }

86 void
87 rpl_link_callback(const linkaddr_t *addr, int status, int numtx)
88 {
89  if(curr_instance.used == 1 ) {
90  rpl_nbr_t *nbr = rpl_neighbor_get_from_lladdr((uip_lladdr_t *)addr);
91  if(nbr != NULL) {
92  /* Link stats were updated, and we need to update our internal state.
93  Updating from here is unsafe; postpone */
94  LOG_INFO("packet sent to ");
95  LOG_INFO_LLADDR(addr);
96  LOG_INFO_(", status %u, tx %u, new link metric %u\n", status, numtx, rpl_neighbor_get_link_metric(nbr));
97  rpl_timers_schedule_state_update();
98  }
99  }
100 }

116 void
117 rpl_reset_prefix(rpl_prefix_t *last_prefix)
118 {
119  uip_ipaddr_t ipaddr;
120  uip_ds6_addr_t *rep;
121  set_ip_from_prefix(&ipaddr, last_prefix);
122  rep = uip_ds6_addr_lookup(&ipaddr);
123  if(rep != NULL) {
124  LOG_INFO("removing global IP address ");
125  LOG_INFO_6ADDR(&ipaddr);
126  LOG_INFO_("\n");
127  uip_ds6_addr_rm(rep);
128  }
129  curr_instance.dag.prefix_info.length = 0;
130 }

132 int
133 rpl_set_prefix_from_addr(uip_ipaddr_t *addr, unsigned len, uint8_t flags)
134 {
135  uip_ipaddr_t ipaddr;
136 
137  if(addr == NULL || len == 0 || len > 128 || !(flags & UIP_ND6_RA_FLAG_AUTONOMOUS)) {
138  LOG_WARN("prefix not included, not-supported or invalid\n");
139  return 0;
140  }
141 
142  /* Try and initialize prefix */
143  memset(&curr_instance.dag.prefix_info.prefix, 0, sizeof(rpl_prefix_t));
144  memcpy(&curr_instance.dag.prefix_info.prefix, addr, (len + 7) / 8);
145  curr_instance.dag.prefix_info.length = len;
146  curr_instance.dag.prefix_info.lifetime = RPL_ROUTE_INFINITE_LIFETIME;
147  curr_instance.dag.prefix_info.flags = flags;
148 
149  /* Add global address if not already there */
150  set_ip_from_prefix(&ipaddr, &curr_instance.dag.prefix_info);
151  if(uip_ds6_addr_lookup(&ipaddr) == NULL) {
152  LOG_INFO("adding global IP address ");
153  LOG_INFO_6ADDR(&ipaddr);
154  LOG_INFO_("\n");
155  uip_ds6_addr_add(&ipaddr, 0, ADDR_AUTOCONF);
156  }
157  return 1;
158 }
