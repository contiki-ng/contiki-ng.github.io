101 static void
102 set_lladdr(void)
103 {
104  linkaddr_t addr;
105  int i;
106 
107  memset(&addr, 0, sizeof(linkaddr_t));
108 #if NETSTACK_CONF_WITH_IPV6
109  memcpy(addr.u8, ds2411_id, sizeof(addr.u8));
110 #else
111  if(node_id == 0) {
112  for(i = 0; i < sizeof(linkaddr_t); ++i) {
113  addr.u8[i] = ds2411_id[7 - i];
114  }
115  } else {
116  addr.u8[0] = node_id & 0xff;
117  addr.u8[1] = node_id >> 8;
118  }
119 #endif
120  linkaddr_set_node_addr(&addr);
121  PRINTF("Contiki started with address ");
122  for(i = 0; i < sizeof(addr.u8) - 1; i++) {
123  PRINTF("%d.", addr.u8[i]);
124  }
125  PRINTF("%d\n", addr.u8[i]);
126 }

172  rtimer_init();
173  /*
174  * Hardware initialization done!
175  */
176 
177  /* Initialize energest first (but after rtimer)
178  */
179  energest_init();
180  ENERGEST_ON(ENERGEST_TYPE_CPU);
181 
182 #if WITH_TINYOS_AUTO_IDS
183  node_id = TOS_NODE_ID;
184 #else /* WITH_TINYOS_AUTO_IDS */
185  /* Restore node id if such has been stored in external mem */
186  node_id_restore();
187 #endif /* WITH_TINYOS_AUTO_IDS */
188 
189  /* for setting "hardcoded" IEEE 802.15.4 MAC addresses */
190 #ifdef IEEE_802154_MAC_ADDRESS
191  {
192  uint8_t ieee[] = IEEE_802154_MAC_ADDRESS;
193  memcpy(ds2411_id, ieee, sizeof(uip_lladdr.addr));
194  ds2411_id[7] = node_id & 0xff;
195  }
196 #endif
197 
198  random_init(ds2411_id[0] + node_id);
199 
200  leds_off(LEDS_BLUE);
201  /*

203  */

205  process_start(&etimer_process, NULL);

212 
213  cc2420_init();
214  {
215  uint8_t longaddr[8];
216  uint16_t shortaddr;
217 
218  shortaddr = (linkaddr_node_addr.u8[0] << 8) +
219  linkaddr_node_addr.u8[1];
220  memset(longaddr, 0, sizeof(longaddr));
221  linkaddr_copy((linkaddr_t *)&longaddr, &linkaddr_node_addr);
222  PRINTF("MAC %02x:%02x:%02x:%02x:%02x:%02x:%02x:%02x ",
223  longaddr[0], longaddr[1], longaddr[2], longaddr[3],
224  longaddr[4], longaddr[5], longaddr[6], longaddr[7]);
225 
226  cc2420_set_pan_addr(IEEE802154_PANID, shortaddr, longaddr);
227  }
228 
229  PRINTF(CONTIKI_VERSION_STRING " started. ");

239 
240 #if NETSTACK_CONF_WITH_IPV6
241  memcpy(&uip_lladdr.addr, ds2411_id, sizeof(uip_lladdr.addr));
242  /* Setup nullmac-like MAC for 802.15.4 */
243 /* sicslowpan_init(sicslowmac_init(&cc2420_driver)); */
244 /* PRINTF(" %s channel %u\n", sicslowmac_driver.name, CC2420_CONF_CCA_THRESH); */
245 
246  /* Setup X-MAC for 802.15.4 */
247  queuebuf_init();

249  NETSTACK_NETWORK.init();

257

289  NETSTACK_NETWORK.init();

295 #if !NETSTACK_CONF_WITH_IPV6
296  uart1_set_input(serial_line_input_byte);
297  serial_line_init();
298 #endif
299 
300  leds_off(LEDS_GREEN);
301 
302 #if TIMESYNCH_CONF_ENABLED
303  timesynch_init();

305 #endif /* TIMESYNCH_CONF_ENABLED */
306 
307  watchdog_start();
308 
309 #if !PROCESS_CONF_NO_PROCESS_NAMES
310  print_processes(autostart_processes);
311 #endif /* !PROCESS_CONF_NO_PROCESS_NAMES */
312  autostart_start(autostart_processes);
313 
314  /*
315  * This is the scheduler loop.
316  */
317 #if DCOSYNCH_CONF_ENABLED
318  timer_set(&mgt_timer, DCOSYNCH_PERIOD * CLOCK_SECOND);
319 #endif
320 
321  /* watchdog_stop();*/
322  while(1) {
323  int r;
324  do {
325  /* Reset watchdog. */
326  watchdog_periodic();
327  r = process_run();
328  } while(r > 0);
329 
330  /*
331  * Idle processing.
332  */
333  int s = splhigh(); /* Disable interrupts. */
334  /* uart1_active is for avoiding LPM3 when still sending or receiving */
335  if(process_nevents() != 0 || uart1_active()) {
336  splx(s); /* Re-enable interrupts. */
337  } else {
338  static unsigned long irq_energest = 0;
339 
340 #if DCOSYNCH_CONF_ENABLED
341  /* before going down to sleep possibly do some management */
342  if(timer_expired(&mgt_timer)) {
343  watchdog_periodic();
344  timer_reset(&mgt_timer);
345  msp430_sync_dco();
346 #if CC2420_CONF_SFD_TIMESTAMPS
347  cc2420_arch_sfd_init();
348 #endif /* CC2420_CONF_SFD_TIMESTAMPS */
349  }
350 #endif
351 
352  /* Re-enable interrupts and go to sleep atomically. */
353  ENERGEST_SWITCH(ENERGEST_TYPE_CPU, ENERGEST_TYPE_LPM);
354  /* We only want to measure the processing done in IRQs when we
355  are asleep, so we discard the processing time done when we
356  were awake. */
357  energest_type_set(ENERGEST_TYPE_IRQ, irq_energest);
358  watchdog_stop();
359  /* check if the DCO needs to be on - if so - only LPM 1 */
360  if (msp430_dco_required) {
361  _BIS_SR(GIE | CPUOFF); /* LPM1 sleep for DMA to work!. */
362  } else {
363  _BIS_SR(GIE | SCG0 | SCG1 | CPUOFF); /* LPM3 sleep. This
364  statement will block
365  until the CPU is
366  woken up by an
367  interrupt that sets
368  the wake up flag. */
369  }
370  /* We get the current processing time for interrupts that was
371  done during the LPM and store it for next time around. */
372  dint();
373  irq_energest = energest_type_time(ENERGEST_TYPE_IRQ);
374  eint();
375  watchdog_start();
376  ENERGEST_SWITCH(ENERGEST_TYPE_LPM, ENERGEST_TYPE_CPU);
377  }
378  }
379 
380  return 0;
381 }
