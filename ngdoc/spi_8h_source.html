42 /* Define macros to use for checking SPI transmission status depending
43  on if it is possible to wait for TX buffer ready. This is possible
44  on for example MSP430 but not on AVR. */
45 #ifdef SPI_WAITFORTxREADY
46 #define SPI_WAITFORTx_BEFORE() SPI_WAITFORTxREADY()
47 #define SPI_WAITFORTx_AFTER()
48 #define SPI_WAITFORTx_ENDED() SPI_WAITFOREOTx()
49 #else /* SPI_WAITFORTxREADY */
50 #define SPI_WAITFORTx_BEFORE()
51 #define SPI_WAITFORTx_AFTER() SPI_WAITFOREOTx()
52 #define SPI_WAITFORTx_ENDED()
53 #endif /* SPI_WAITFORTxREADY */

59 /* Write one character to SPI */
60 #define SPI_WRITE(data) \
61  do { \
62  SPI_WAITFORTx_BEFORE(); \
63  SPI_TXBUF = data; \
64  SPI_WAITFOREOTx(); \
65  } while(0)
66 
67 /* Write one character to SPI - will not wait for end
68  useful for multiple writes with wait after final */
69 #define SPI_WRITE_FAST(data) \
70  do { \
71  SPI_WAITFORTx_BEFORE(); \
72  SPI_TXBUF = data; \
73  SPI_WAITFORTx_AFTER(); \
74  } while(0)
75 
76 /* Read one character from SPI */
77 #define SPI_READ(data) \
78  do { \
79  SPI_TXBUF = 0; \
80  SPI_WAITFOREORx(); \
81  data = SPI_RXBUF; \
82  } while(0)
83 
84 /* Flush the SPI read register */
85 #ifndef SPI_FLUSH
86 #define SPI_FLUSH() \
87  do { \
88  SPI_RXBUF; \
89  } while(0)
90 #endif
