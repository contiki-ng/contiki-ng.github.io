42 void
43 leds_arch_init(void)
44 {
45  LEDS_PxDIR |= (LEDS_CONF_RED | LEDS_CONF_GREEN | LEDS_CONF_YELLOW);
46  LEDS_PxOUT |= (LEDS_CONF_RED | LEDS_CONF_GREEN | LEDS_CONF_YELLOW);
47 }
48 /*---------------------------------------------------------------------------*/
49 unsigned char
50 leds_arch_get(void)
51 {
52  unsigned char leds;
53  leds = LEDS_PxOUT;
54  return ((leds & LEDS_CONF_RED) ? 0 : LEDS_RED)
55  | ((leds & LEDS_CONF_GREEN) ? 0 : LEDS_GREEN)
56  | ((leds & LEDS_CONF_YELLOW) ? 0 : LEDS_YELLOW);
57 }
58 /*---------------------------------------------------------------------------*/
59 void
60 leds_arch_set(unsigned char leds)
61 {
62  LEDS_PxOUT = (LEDS_PxOUT & ~(LEDS_CONF_RED|LEDS_CONF_GREEN|LEDS_CONF_YELLOW))
63  | ((leds & LEDS_RED) ? 0 : LEDS_CONF_RED)
64  | ((leds & LEDS_GREEN) ? 0 : LEDS_CONF_GREEN)
65  | ((leds & LEDS_YELLOW) ? 0 : LEDS_CONF_YELLOW);
66 }
