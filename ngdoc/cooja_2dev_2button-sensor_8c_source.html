51 configure(int type, int c)
52 {
53  if(type == SENSORS_ACTIVE) {
54  simButtonIsActive = c;
55  return 1;
56  } else if(type == SENSORS_HW_INIT) {

70 static void
71 doInterfaceActionsBeforeTick(void)
72 {
73  // Check if button value has changed
74  if(simButtonChanged && simButtonIsActive && simButtonIsDown) {
75  if(timer_expired(&debouncetimer)) {
76  timer_set(&debouncetimer, CLOCK_SECOND / 10);
77  sensors_changed(&button_sensor);
78  }
79  }
80  simButtonChanged = 0;
81 }
