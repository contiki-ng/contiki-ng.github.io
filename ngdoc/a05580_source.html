59 /* RFC6551 and RFC6719 do not mandate the use of a specific formula to
60  * compute the ETX value. This MRHOF implementation relies on the value
61  * computed by the link-stats module. It has an optional feature,
62  * RPL_MRHOF_CONF_SQUARED_ETX, that consists in squaring this value.
63  * This basically penalizes bad links while preserving the semantics of ETX
64  * (1 = perfect link, more = worse link). As a result, MRHOF will favor
65  * good links over short paths. Recommended when reliability is a priority.
66  * Without this feature, a hop with 50% PRR (ETX=2) is equivalent to two
67  * perfect hops with 100% PRR (ETX=1+1=2). With this feature, the former
68  * path obtains ETX=2*2=4 and the former ETX=1*1+1*1=2. */
69 #ifdef RPL_MRHOF_CONF_SQUARED_ETX
70 #define RPL_MRHOF_SQUARED_ETX RPL_MRHOF_CONF_SQUARED_ETX
71 #else /* RPL_MRHOF_CONF_SQUARED_ETX */
72 #define RPL_MRHOF_SQUARED_ETX 1
73 #endif /* RPL_MRHOF_CONF_SQUARED_ETX */
74 
75 /* Configuration parameters of RFC6719. Reject parents that have a higher
76  * link metric than the following. The default value is 512. */
77 #define MAX_LINK_METRIC 512 /* Eq ETX of 4 */
78 
79 /* Reject parents that have a higher path cost than the following. */
80 #define MAX_PATH_COST 32768 /* Eq path ETX of 256 */
81 
82 #if !RPL_MRHOF_SQUARED_ETX
83 /* Hysteresis of MRHOF: the rank must differ more than PARENT_SWITCH_THRESHOLD_DIV
84  * in order to switch preferred parent. Default in RFC6719: 192, eq ETX of 1.5. */
85 #define RANK_THRESHOLD 192 /* Eq ETX of 1.5 */
86 #else /* !RPL_MRHOF_SQUARED_ETX */
87 #define RANK_THRESHOLD 384 /* Eq ETX of sqrt(3) */
88 #endif /* !RPL_MRHOF_SQUARED_ETX */
89 
90 /* Additional, custom hysteresis based on time. If a neighbor was consistently
91  * better than our preferred parent for at least TIME_THRESHOLD, switch to
92  * this neighbor regardless of RANK_THRESHOLD. */
93 #define TIME_THRESHOLD (10 * 60 * CLOCK_SECOND)
94

102 static uint16_t
103 nbr_link_metric(rpl_nbr_t *nbr)
104 {
105  const struct link_stats *stats = rpl_neighbor_get_link_stats(nbr);
106  return stats != NULL ? stats->etx : 0xffff;
107 }
108 /*---------------------------------------------------------------------------*/
109 static uint16_t
110 link_metric_to_rank(uint16_t etx)
111 {
112 #if RPL_MRHOF_SQUARED_ETX
113  uint32_t squared_etx = ((uint32_t)etx * etx) / LINK_STATS_ETX_DIVISOR;
114  return (uint16_t)MIN(squared_etx, 0xffff);
115 #else /* RPL_MRHOF_SQUARED_ETX */
116  return etx;
117 #endif /* RPL_MRHOF_SQUARED_ETX */
118 }
119 /*---------------------------------------------------------------------------*/
120 static uint16_t
121 nbr_path_cost(rpl_nbr_t *nbr)
122 {
123  uint16_t base;
124 
125  if(nbr == NULL) {
126  return 0xffff;
127  }
128 
129 #if RPL_WITH_MC
130  /* Handle the different MC types */
131  switch(curr_instance.mc.type) {
132  case RPL_DAG_MC_ETX:
133  base = nbr->mc.obj.etx;
134  break;
135  case RPL_DAG_MC_ENERGY:
136  base = nbr->mc.obj.energy.energy_est << 8;
137  break;
138  default:
139  base = nbr->rank;
140  break;
141  }
142 #else /* RPL_WITH_MC */
143  base = nbr->rank;
144 #endif /* RPL_WITH_MC */
145 
146  /* path cost upper bound: 0xffff */
147  return MIN((uint32_t)base + link_metric_to_rank(nbr_link_metric(nbr)), 0xffff);
148 }
149 /*---------------------------------------------------------------------------*/
150 static rpl_rank_t
151 rank_via_nbr(rpl_nbr_t *nbr)
152 {
153  uint16_t min_hoprankinc;
154  uint16_t path_cost;
155 
156  if(nbr == NULL) {
157  return RPL_INFINITE_RANK;
158  }
159 
160  min_hoprankinc = curr_instance.min_hoprankinc;
161  path_cost = nbr_path_cost(nbr);
162 
163  /* Rank lower-bound: nbr rank + min_hoprankinc */
164  return MAX(MIN((uint32_t)nbr->rank + min_hoprankinc, RPL_INFINITE_RANK), path_cost);
165 }
166 /*---------------------------------------------------------------------------*/
167 static int
168 nbr_has_usable_link(rpl_nbr_t *nbr)
169 {
170  uint16_t link_metric = nbr_link_metric(nbr);
171  /* Exclude links with too high link metrics */
172  return link_metric <= MAX_LINK_METRIC;
173 }
174 /*---------------------------------------------------------------------------*/
175 static int
176 nbr_is_acceptable_parent(rpl_nbr_t *nbr)
177 {
178  uint16_t path_cost = nbr_path_cost(nbr);
179  /* Exclude links with too high link metrics or path cost (RFC6719, 3.2.2) */
180  return nbr_has_usable_link(nbr) && path_cost <= MAX_PATH_COST;
181 }
182 /*---------------------------------------------------------------------------*/
183 static int
184 within_hysteresis(rpl_nbr_t *nbr)
185 {
186  uint16_t path_cost = nbr_path_cost(nbr);
187  uint16_t parent_path_cost = nbr_path_cost(curr_instance.dag.preferred_parent);
188 
189  int within_rank_hysteresis = path_cost + RANK_THRESHOLD > parent_path_cost;
190  int within_time_hysteresis = nbr->better_parent_since == 0
191  || (clock_time() - nbr->better_parent_since) <= TIME_THRESHOLD;
192 
193  /* As we want to consider neighbors that are either beyond the rank or time
194  hystereses, return 1 here iff the neighbor is within both hystereses. */
195  return within_rank_hysteresis && within_time_hysteresis;
196 }
197 /*---------------------------------------------------------------------------*/
198 static rpl_nbr_t *
199 best_parent(rpl_nbr_t *nbr1, rpl_nbr_t *nbr2)
200 {
201  int nbr1_is_acceptable;
202  int nbr2_is_acceptable;
203 
204  nbr1_is_acceptable = nbr1 != NULL && nbr_is_acceptable_parent(nbr1);
205  nbr2_is_acceptable = nbr2 != NULL && nbr_is_acceptable_parent(nbr2);
206 
207  if(!nbr1_is_acceptable) {
208  return nbr2_is_acceptable ? nbr2 : NULL;
209  }
210  if(!nbr2_is_acceptable) {
211  return nbr1_is_acceptable ? nbr1 : NULL;
212  }
213 
214  /* Maintain stability of the preferred parent. Switch only if the gain
215  is greater than RANK_THRESHOLD, or if the neighbor has been better than the
216  current parent for at more than TIME_THRESHOLD. */
217  if(nbr1 == curr_instance.dag.preferred_parent && within_hysteresis(nbr2)) {
218  return nbr1;
219  }
220  if(nbr2 == curr_instance.dag.preferred_parent && within_hysteresis(nbr1)) {
221  return nbr2;
222  }
223 
224  return nbr_path_cost(nbr1) < nbr_path_cost(nbr2) ? nbr1 : nbr2;
225 }

228 static void
229 update_metric_container(void)
230 {
231  curr_instance.mc.type = RPL_DAG_MC_NONE;
232 }
233 #else /* RPL_WITH_MC */
234 static void
235 update_metric_container(void)
236 {
237  uint16_t path_cost;
238  uint8_t type;
239 
240  if(!curr_instance.used) {
241  LOG_WARN("cannot update the metric container when not joined\n");
242  return;
243  }
244 
245  if(curr_instance.dag.rank == ROOT_RANK) {
246  /* Configure MC at root only, other nodes are auto-configured when joining */
247  curr_instance.mc.type = RPL_DAG_MC;
248  curr_instance.mc.flags = 0;
249  curr_instance.mc.aggr = RPL_DAG_MC_AGGR_ADDITIVE;
250  curr_instance.mc.prec = 0;
251  path_cost = curr_instance.dag.rank;
252  } else {
253  path_cost = nbr_path_cost(curr_instance.dag.preferred_parent);
254  }
255 
256  /* Handle the different MC types */
257  switch(curr_instance.mc.type) {
258  case RPL_DAG_MC_NONE:
259  break;
260  case RPL_DAG_MC_ETX:
261  curr_instance.mc.length = sizeof(curr_instance.mc.obj.etx);
262  curr_instance.mc.obj.etx = path_cost;
263  break;
264  case RPL_DAG_MC_ENERGY:
265  curr_instance.mc.length = sizeof(curr_instance.mc.obj.energy);
266  if(curr_instance.dag.rank == ROOT_RANK) {
267  type = RPL_DAG_MC_ENERGY_TYPE_MAINS;
268  } else {
269  type = RPL_DAG_MC_ENERGY_TYPE_BATTERY;
270  }
271  curr_instance.mc.obj.energy.flags = type << RPL_DAG_MC_ENERGY_TYPE;
272  /* Energy_est is only one byte, use the least significant byte of the path metric. */
273  curr_instance.mc.obj.energy.energy_est = path_cost >> 8;
274  break;
275  default:
276  LOG_WARN("MRHOF, non-supported MC %u\n", curr_instance.mc.type);
277  break;
278  }
279 }
280 #endif /* RPL_WITH_MC */
