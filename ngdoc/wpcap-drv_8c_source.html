42 /* It is not particularly easy to install tun interfaces in Windows/cygwin, so wpcap
43  * is used instead. The ip4 or ip6 address of the interface to connect to is passed
44  * on the command line that invokes the minimal-net and native executables.
45  *
46  * The minimal-net border router uses wpcap to connect to both primary
47  * and fallback interfaces. It is passed two addresses, and the uip stack is compiled
48  * with space for the ethernet headers on both interfaces.
49  *
50  * However the native border router uses wpcap to connect to a fallback interface only.
51  * The primary interface is the serial connection to the slip radio, and the
52  * uip stack is compiled without space for ethernet headers.
53  * The following define adds or strips ethernet headers from the fallback interface.
54  * Since it is at present used only with the native border router, it is also used
55  * as a hack to bypass polling of the primary interface.
56  *
57  * SELECT_CALLBACK is defined in /examples/ipv6/native-border-router/project-conf.h
58  */
59 #ifdef SELECT_CALLBACK
60 #define FALLBACK_HAS_ETHERNET_HEADERS 1
61 #endif

77 static void
78 pollhandler(void)
79 {
80 #if !FALLBACK_HAS_ETHERNET_HEADERS //native br is fallback only
81  process_poll(&wpcap_process);
82  uip_len = wpcap_poll();
83 
84  if(uip_len > 0) {
85 #if NETSTACK_CONF_WITH_IPV6
86  if(BUF->type == uip_htons(UIP_ETHTYPE_IPV6)) {
87 // printf("wpcap poll calls tcpip");
88  tcpip_input();
89  } else
90 #endif /* NETSTACK_CONF_WITH_IPV6 */
91  if(BUF->type == uip_htons(UIP_ETHTYPE_IP)) {
92  uip_len -= sizeof(struct uip_eth_hdr);
93  tcpip_input();
94 #if !NETSTACK_CONF_WITH_IPV6
95  } else if(BUF->type == uip_htons(UIP_ETHTYPE_ARP)) {
96  uip_arp_arpin(); //math
97  /* If the above function invocation resulted in data that
98  should be sent out on the network, the global variable
99  uip_len is set to a value > 0. */
100  if(uip_len > 0) {
101  wpcap_send();
102  }
103 #endif /* !NETSTACK_CONF_WITH_IPV6 */
104  } else {
105  uip_clear_buf();
106  }
107  }
108 #endif
109 #ifdef UIP_FALLBACK_INTERFACE
110 
111  process_poll(&wpcap_process);
112  uip_len = wfall_poll();
113 
114  if(uip_len > 0) {
115 #if FALLBACK_HAS_ETHERNET_HEADERS
116  if(BUF->type == uip_htons(UIP_ETHTYPE_IPV6)) {
117  //remove ethernet header and pass ipv6 packet to stack
118  uip_len-=14;
119 //{int i;printf("\n0000 ");for (i=0;i<uip_len;i++) printf("%02x ",*(unsigned char*)(uip_buf+i));printf("\n");}
120  // memcpy(uip_buf, uip_buf+14, uip_len);
121  memcpy(&uip_buf[UIP_LLH_LEN], uip_buf+14, uip_len); //LLH_LEN is zero for native border router to slip radio
122 // CopyMemory(uip_buf, uip_buf+14, uip_len);
123 //{int i;printf("\n0000 ");for (i=0;i<uip_len;i++) printf("%02x ",*(char*)(uip_buf+i));printf("\n");}
124  tcpip_input();
125  } else
126  goto bail;
127 #elif NETSTACK_CONF_WITH_IPV6
128  if(BUF->type == uip_htons(UIP_ETHTYPE_IPV6)) {
129  tcpip_input();
130  } else
131  goto bail;
132 #endif /* NETSTACK_CONF_WITH_IPV6 */
133  if(BUF->type == uip_htons(UIP_ETHTYPE_IP)) {
134  uip_len -= sizeof(struct uip_eth_hdr);
135  tcpip_input();
136 #if !NETSTACK_CONF_WITH_IPV6
137  } else if(BUF->type == uip_htons(UIP_ETHTYPE_ARP)) {
138  uip_arp_arpin(); //math
139  /* If the above function invocation resulted in data that
140  should be sent out on the network, the global variable
141  uip_len is set to a value > 0. */
142  if(uip_len > 0) {
143  wfall_send();
144  }
145 #endif /* !NETSTACK_CONF_WITH_IPV6 */
146  } else {
147 bail:
148  uip_clear_buf();
149  }
150  }
151 #endif
152 
153 }
