260 static radio_value_t
261 get_rssi(void)
262 {
263  uint32_t cmd_status;
264  int8_t rssi;
265  uint8_t attempts = 0;
266  uint8_t was_off = 0;
267  rfc_CMD_GET_RSSI_t cmd;
268 
269  /* If we are off, turn on first */
270  if(!rf_is_on()) {
271  was_off = 1;
272  if(on() != RF_CORE_CMD_OK) {
273  PRINTF("get_rssi: on() failed\n");
274  return RF_CMD_CCA_REQ_RSSI_UNKNOWN;
275  }
276  }
277 
278  rssi = RF_CMD_CCA_REQ_RSSI_UNKNOWN;
279 
280  while((rssi == RF_CMD_CCA_REQ_RSSI_UNKNOWN || rssi == 0) && ++attempts < 10) {
281  memset(&cmd, 0x00, sizeof(cmd));
282  cmd.commandNo = CMD_GET_RSSI;
283 
284  if(rf_core_send_cmd((uint32_t)&cmd, &cmd_status) == RF_CORE_CMD_ERROR) {
285  PRINTF("get_rssi: CMDSTA=0x%08lx\n", cmd_status);
286  break;
287  } else {
288  /* Current RSSI in bits 23:16 of cmd_status */
289  rssi = (cmd_status >> 16) & 0xFF;
290  }
291  }
292 
293  /* If we were off, turn back off */
294  if(was_off) {
295  off();
296  }
297 
298  return rssi;
299 }

301 static uint8_t
302 get_channel(void)
303 {
304  uint32_t freq_khz;
305 
306  freq_khz = smartrf_settings_cmd_fs.frequency * 1000;
307 
308  /*
309  * For some channels, fractFreq * 1000 / 65536 will return 324.99xx.
310  * Casting the result to uint32_t will truncate decimals resulting in the
311  * function returning channel - 1 instead of channel. Thus, we do a quick
312  * positive integer round up.
313  */
314  freq_khz += (((smartrf_settings_cmd_fs.fractFreq * 1000) + 65535) / 65536);
315 
316  return (freq_khz - DOT_15_4G_CHAN0_FREQUENCY) / DOT_15_4G_CHANNEL_SPACING;
317 }

319 static void
320 set_channel(uint8_t channel)
321 {
322  uint32_t new_freq;
323  uint16_t freq, frac;
324 
325  new_freq = DOT_15_4G_CHAN0_FREQUENCY + (channel * DOT_15_4G_CHANNEL_SPACING);
326 
327  freq = (uint16_t)(new_freq / 1000);
328  frac = (new_freq - (freq * 1000)) * 65536 / 1000;
329 
330  PRINTF("set_channel: %u = 0x%04x.0x%04x (%lu)\n", channel, freq, frac,
331  new_freq);
332 
333  smartrf_settings_cmd_prop_radio_div_setup.centerFreq = freq;
334  smartrf_settings_cmd_fs.frequency = freq;
335  smartrf_settings_cmd_fs.fractFreq = frac;
336 }

418 static uint8_t
419 rf_cmd_prop_rx()
420 {
421  uint32_t cmd_status;
422  rtimer_clock_t t0;
423  volatile rfc_CMD_PROP_RX_ADV_t *cmd_rx_adv;
424  int ret;
425 
426  cmd_rx_adv = (rfc_CMD_PROP_RX_ADV_t *)&smartrf_settings_cmd_prop_rx_adv;
427  cmd_rx_adv->status = RF_CORE_RADIO_OP_STATUS_IDLE;
428 
429  /*
430  * Set the max Packet length. This is for the payload only, therefore
431  * 2047 - length offset
432  */
433  cmd_rx_adv->maxPktLen = DOT_4G_MAX_FRAME_LEN - cmd_rx_adv->lenOffset;
434 
435  ret = rf_core_send_cmd((uint32_t)cmd_rx_adv, &cmd_status);
436 
437  if(ret != RF_CORE_CMD_OK) {
438  PRINTF("rf_cmd_prop_rx: send_cmd ret=%d, CMDSTA=0x%08lx, status=0x%04x\n",
439  ret, cmd_status, cmd_rx_adv->status);
440  return RF_CORE_CMD_ERROR;
441  }
442 
443  t0 = RTIMER_NOW();
444 
445  while(cmd_rx_adv->status != RF_CORE_RADIO_OP_STATUS_ACTIVE &&
446  (RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + ENTER_RX_WAIT_TIMEOUT)));
447 
448  /* Wait to enter RX */
449  if(cmd_rx_adv->status != RF_CORE_RADIO_OP_STATUS_ACTIVE) {
450  PRINTF("rf_cmd_prop_rx: CMDSTA=0x%08lx, status=0x%04x\n",
451  cmd_status, cmd_rx_adv->status);
452  return RF_CORE_CMD_ERROR;
453  }
454 
455  return ret;
456 }

497 static int
498 rx_off_prop(void)
499 {
500  uint32_t cmd_status;
501  int ret;
502 
503  /* If we are off, do nothing */
504  if(!rf_is_on()) {
505  return RF_CORE_CMD_OK;
506  }
507 
508  /* Send a CMD_ABORT command to RF Core */
509  if(rf_core_send_cmd(CMDR_DIR_CMD(CMD_ABORT), &cmd_status) != RF_CORE_CMD_OK) {
510  PRINTF("rx_off_prop: CMD_ABORT status=0x%08lx\n", cmd_status);
511  /* Continue nonetheless */
512  }
513 
514  while(rf_is_on());
515 
516  if(smartrf_settings_cmd_prop_rx_adv.status == PROP_DONE_STOPPED ||
517  smartrf_settings_cmd_prop_rx_adv.status == PROP_DONE_ABORT) {
518  /* Stopped gracefully */
519  ENERGEST_OFF(ENERGEST_TYPE_LISTEN);
520  ret = RF_CORE_CMD_OK;
521  } else {
522  PRINTF("rx_off_prop: status=0x%04x\n",
523  smartrf_settings_cmd_prop_rx_adv.status);
524  ret = RF_CORE_CMD_ERROR;
525  }
526 
527  return ret;
528 }

620  memset(rx_buf, 0, sizeof(rx_buf));
621

623  rx_data_queue.pCurrEntry = rx_buf[0];
624  rx_data_queue.pLastEntry = NULL;
625

656 /*---------------------------------------------------------------------------*/
657 static int
658 transmit(unsigned short transmit_len)
659 {
660  int ret;
661  uint8_t was_off = 0;
662  uint32_t cmd_status;
663  volatile rfc_CMD_PROP_TX_ADV_t *cmd_tx_adv;
664 
665  /* Length in .15.4g PHY HDR. Includes the CRC but not the HDR itself */
666  uint16_t total_length;
667 
668  if(!rf_is_on()) {
669  was_off = 1;
670  if(on() != RF_CORE_CMD_OK) {
671  PRINTF("transmit: on() failed\n");
672  return RADIO_TX_ERR;
673  }
674  }
675 
676  /*
677  * Prepare the .15.4g PHY header
678  * MS=0, Length MSBits=0, DW and CRC configurable
679  * Total length = transmit_len (payload) + CRC length
680  *
681  * The Radio will flip the bits around, so tx_buf[0] must have the length
682  * LSBs (PHR[15:8] and tx_buf[1] will have PHR[7:0]
683  */
684  total_length = transmit_len + CRC_LEN;
685 
686  tx_buf[0] = total_length & 0xFF;
687  tx_buf[1] = (total_length >> 8) + DOT_4G_PHR_DW_BIT + DOT_4G_PHR_CRC_BIT;
688 
689  /* Prepare the CMD_PROP_TX_ADV command */
690  cmd_tx_adv = (rfc_CMD_PROP_TX_ADV_t *)&smartrf_settings_cmd_prop_tx_adv;
691 
692  /*
693  * pktLen: Total number of bytes in the TX buffer, including the header if
694  * one exists, but not including the CRC (which is not present in the buffer)
695  */
696  cmd_tx_adv->pktLen = transmit_len + DOT_4G_PHR_LEN;
697  cmd_tx_adv->pPkt = tx_buf;
698 
699  /* Abort RX */
700  rx_off_prop();
701 
702  /* Enable the LAST_COMMAND_DONE interrupt to wake us up */
703  rf_core_cmd_done_en(false, false);
704 
705  ret = rf_core_send_cmd((uint32_t)cmd_tx_adv, &cmd_status);
706 
707  if(ret) {
708  /* If we enter here, TX actually started */
709  ENERGEST_OFF(ENERGEST_TYPE_LISTEN);
710  ENERGEST_ON(ENERGEST_TYPE_TRANSMIT);
711 
712  watchdog_periodic();
713 
714  /* Idle away while the command is running */
715  while((cmd_tx_adv->status & RF_CORE_RADIO_OP_MASKED_STATUS)
716  == RF_CORE_RADIO_OP_MASKED_STATUS_RUNNING) {
717  lpm_sleep();
718  }
719 
720  if(cmd_tx_adv->status == RF_CORE_RADIO_OP_STATUS_PROP_DONE_OK) {
721  /* Sent OK */
722  ret = RADIO_TX_OK;
723  } else {
724  /* Operation completed, but frame was not sent */
725  PRINTF("transmit: Not Sent OK status=0x%04x\n",
726  cmd_tx_adv->status);
727  ret = RADIO_TX_ERR;
728  }
729  } else {
730  /* Failure sending the CMD_PROP_TX command */
731  PRINTF("transmit: PROP_TX_ERR ret=%d, CMDSTA=0x%08lx, status=0x%04x\n",
732  ret, cmd_status, cmd_tx_adv->status);
733  ret = RADIO_TX_ERR;
734  }
735 
736  /*
737  * Update ENERGEST state here, before a potential call to off(), which
738  * will correctly update it if required.
739  */
740  ENERGEST_OFF(ENERGEST_TYPE_TRANSMIT);
741  ENERGEST_ON(ENERGEST_TYPE_LISTEN);
742 
743  /*
744  * Disable LAST_FG_COMMAND_DONE interrupt. We don't really care about it
745  * except when we are transmitting
746  */
747  rf_core_cmd_done_dis(false);
748 
749  /* Workaround. Set status to IDLE */
750  cmd_tx_adv->status = RF_CORE_RADIO_OP_STATUS_IDLE;
751 
752  rx_on_prop();
753 
754  if(was_off) {
755  off();
756  }
757 
758  return ret;
759 }

768 static int
769 read_frame(void *buf, unsigned short buf_len)
770 {
771  rfc_dataEntryGeneral_t *entry = (rfc_dataEntryGeneral_t *)rx_read_entry;
772  uint8_t *data_ptr = &entry->data;
773  int len = 0;
774 
775  if(entry->status == DATA_ENTRY_STATUS_FINISHED) {
776 
777  /*
778  * First 2 bytes in the data entry are the length.
779  * Our data entry consists of: Payload + RSSI (1 byte) + Status (1 byte)
780  * This length includes all of those.
781  */
782  len = (*(uint16_t *)data_ptr);
783  data_ptr += 2;
784  len -= 2;
785 
786  if(len > 0) {
787  if(len <= buf_len) {
788  memcpy(buf, data_ptr, len);
789  }
790 
791  packetbuf_set_attr(PACKETBUF_ATTR_RSSI, (int8_t)data_ptr[len]);
792  packetbuf_set_attr(PACKETBUF_ATTR_LINK_QUALITY, 0x7F);
793  }
794 
795  /* Move read entry pointer to next entry */
796  rx_read_entry = entry->pNextEntry;
797  entry->status = DATA_ENTRY_STATUS_PENDING;
798  }
799 
800  return len;
801 }

803 static int
804 channel_clear(void)
805 {
806  uint8_t was_off = 0;
807  uint32_t cmd_status;
808  int8_t rssi = RF_CMD_CCA_REQ_RSSI_UNKNOWN;
809 
810  /*
811  * If we are in the middle of a BLE operation, we got called by ContikiMAC
812  * from within an interrupt context. Indicate a clear channel
813  */
814  if(rf_ble_is_active() == RF_BLE_ACTIVE) {
815  return RF_CCA_CLEAR;
816  }
817 
818  if(!rf_core_is_accessible()) {
819  was_off = 1;
820  if(on() != RF_CORE_CMD_OK) {
821  PRINTF("channel_clear: on() failed\n");
822  if(was_off) {
823  off();
824  }
825  return RF_CCA_CLEAR;
826  }
827  } else {
828  if(transmitting()) {
829  PRINTF("channel_clear: called while in TX\n");
830  return RF_CCA_CLEAR;
831  }
832  }
833 
834  while(rssi == RF_CMD_CCA_REQ_RSSI_UNKNOWN || rssi == 0) {
835  if(rf_core_send_cmd(CMDR_DIR_CMD(CMD_GET_RSSI), &cmd_status)
836  != RF_CORE_CMD_OK) {
837  break;
838  }
839  /* Current RSSI in bits 23:16 of cmd_status */
840  rssi = (cmd_status >> 16) & 0xFF;
841  }
842 
843  if(was_off) {
844  off();
845  }
846 
847  if(rssi >= rssi_threshold) {
848  return RF_CCA_BUSY;
849  }
850 
851  return RF_CCA_CLEAR;
852 }

867 /*---------------------------------------------------------------------------*/
868 static int
869 pending_packet(void)
870 {
871  int rv = 0;
872  volatile rfc_dataEntry_t *entry = (rfc_dataEntry_t *)rx_data_queue.pCurrEntry;
873 
874  /* Go through all RX buffers and check their status */
875  do {
876  if(entry->status == DATA_ENTRY_STATUS_FINISHED) {
877  rv += 1;
878  process_poll(&rf_core_process);
879  }
880 
881  entry = (rfc_dataEntry_t *)entry->pNextEntry;
882  } while(entry != (rfc_dataEntry_t *)rx_data_queue.pCurrEntry);
883 
884  /* If we didn't find an entry at status finished, no frames are pending */
885  return rv;
886 }

888 static int
889 on(void)
890 {
891  /*
892  * If we are in the middle of a BLE operation, we got called by ContikiMAC
893  * from within an interrupt context. Abort, but pretend everything is OK.
894  */
895  if(rf_ble_is_active() == RF_BLE_ACTIVE) {
896  return RF_CORE_CMD_OK;
897  }
898 
899  /*
900  * Request the HF XOSC as the source for the HF clock. Needed before we can
901  * use the FS. This will only request, it will _not_ perform the switch.
902  */
903  oscillators_request_hf_xosc();
904 
905  if(rf_is_on()) {
906  PRINTF("on: We were on. PD=%u, RX=0x%04x \n", rf_core_is_accessible(),
907  smartrf_settings_cmd_prop_rx_adv.status);
908  return RF_CORE_CMD_OK;
909  }
910 
911  if(!rf_core_is_accessible()) {
912  if(rf_core_power_up() != RF_CORE_CMD_OK) {
913  PRINTF("on: rf_core_power_up() failed\n");
914 
915  rf_core_power_down();
916 
917  return RF_CORE_CMD_ERROR;
918  }
919 
920  /* Keep track of RF Core mode */
921  rf_core_set_modesel();
922 
923  /* Apply patches to radio core */
924  rf_patch_cpe_genfsk();
925  while(!HWREG(RFC_DBELL_BASE + RFC_DBELL_O_RFACKIFG));
926  HWREG(RFC_DBELL_BASE + RFC_DBELL_O_RFACKIFG) = 0;
927  rf_patch_rfe_genfsk();
928 
929  /* Initialize bus request */
930  HWREG(RFC_DBELL_BASE + RFC_DBELL_O_RFACKIFG) = 0;
931  HWREG(RFC_DBELL_BASE + RFC_DBELL_O_CMDR) =
932  CMDR_DIR_CMD_1BYTE(CMD_BUS_REQUEST, 1);
933 
934  /* set VCOLDO reference */
935  ti_lib_rfc_adi3vco_ldo_voltage_mode(true);
936 
937  /* Let CC13xxware automatically set a correct value for RTRIM for us */
938  ti_lib_rfc_rtrim((rfc_radioOp_t *)&smartrf_settings_cmd_prop_radio_div_setup);
939 
940  /* Make sure BUS_REQUEST is done */
941  while(!HWREG(RFC_DBELL_BASE + RFC_DBELL_O_RFACKIFG));
942  HWREG(RFC_DBELL_BASE + RFC_DBELL_O_RFACKIFG) = 0;
943 
944  if(rf_core_start_rat() != RF_CORE_CMD_OK) {
945  PRINTF("on: rf_core_start_rat() failed\n");
946 
947  rf_core_power_down();
948 
949  return RF_CORE_CMD_ERROR;
950  }
951  }
952 
953  rf_core_setup_interrupts(false);
954 
955  init_rx_buffers();
956 
957  /*
958  * Trigger a switch to the XOSC, so that we can subsequently use the RF FS
959  * This will block until the XOSC is actually ready, but give how we
960  * requested it early on, this won't be too long a wait/
961  */
962  oscillators_switch_to_hf_xosc();
963 
964  if(prop_div_radio_setup() != RF_CORE_CMD_OK) {
965  PRINTF("on: prop_div_radio_setup() failed\n");
966  return RF_CORE_CMD_ERROR;
967  }
968 
969  if(prop_fs() != RF_CORE_CMD_OK) {
970  PRINTF("on: prop_fs() failed\n");
971  return RF_CORE_CMD_ERROR;
972  }
973 
974  return rx_on_prop();
975 }

978 off(void)
979 {
980  /*
981  * If we are in the middle of a BLE operation, we got called by ContikiMAC
982  * from within an interrupt context. Abort, but pretend everything is OK.
983  */
984  if(rf_ble_is_active() == RF_BLE_ACTIVE) {
985  return RF_CORE_CMD_OK;
986  }
987 
988  rx_off_prop();
989  rf_core_power_down();

999  return RF_CORE_CMD_OK;

1006  return RADIO_RESULT_INVALID_VALUE;
1007  }
1008 
1009  switch(param) {
1010  case RADIO_PARAM_POWER_MODE:
1011  /* On / off */

1013  return RADIO_RESULT_OK;

1019  return RADIO_RESULT_OK;

1027  return RADIO_RESULT_ERROR;

1029  return RADIO_RESULT_OK;
1030  }

1032  *value = 0;

1039  return RADIO_RESULT_OK;
1040  case RADIO_CONST_TXPOWER_MAX:

1048 static radio_result_t
1049 set_value(radio_param_t param, radio_value_t value)
1050 {
1051  uint8_t was_off = 0;
1052  radio_result_t rv = RADIO_RESULT_OK;
1053 
1054  switch(param) {
1055  case RADIO_PARAM_POWER_MODE:
1056  if(value == RADIO_POWER_MODE_ON) {
1057  if(on() != RF_CORE_CMD_OK) {
1058  PRINTF("set_value: on() failed (1)\n");
1059  return RADIO_RESULT_ERROR;
1060  }
1061  return RADIO_RESULT_OK;
1062  }
1063  if(value == RADIO_POWER_MODE_OFF) {
1064  off();
1065  return RADIO_RESULT_OK;
1066  }
1067  return RADIO_RESULT_INVALID_VALUE;
1068  case RADIO_PARAM_CHANNEL:
1069  if(value < 0 ||
1070  value > DOT_15_4G_CHANNEL_MAX) {
1071  return RADIO_RESULT_INVALID_VALUE;
1072  }
1073 
1074  if(get_channel() == (uint8_t)value) {
1075  /* We already have that very same channel configured.
1076  * Nothing to do here. */
1077  return RADIO_RESULT_OK;
1078  }
1079 
1080  set_channel((uint8_t)value);
1081  break;
1082  case RADIO_PARAM_TXPOWER:
1083  if(value < TX_POWER_DRIVER[get_tx_power_array_last_element()].dbm ||
1084  value > OUTPUT_POWER_MAX) {
1085  return RADIO_RESULT_INVALID_VALUE;
1086  }
1087 
1088  soft_off_prop();
1089 
1090  set_tx_power(value);
1091 
1092  if(soft_on_prop() != RF_CORE_CMD_OK) {
1093  PRINTF("set_value: soft_on_prop() failed\n");
1094  rv = RADIO_RESULT_ERROR;
1095  }
1096 
1097  return RADIO_RESULT_OK;
1098  case RADIO_PARAM_RX_MODE:
1099  return RADIO_RESULT_OK;
1100  case RADIO_PARAM_CCA_THRESHOLD:
1101  rssi_threshold = (int8_t)value;
1102  break;
1103  default:
1104  return RADIO_RESULT_NOT_SUPPORTED;
1105  }
1106 
1107  /* If we reach here we had no errors. Apply new settings */
1108  if(!rf_is_on()) {
1109  was_off = 1;
1110  if(on() != RF_CORE_CMD_OK) {
1111  PRINTF("set_value: on() failed (2)\n");
1112  return RADIO_RESULT_ERROR;
1113  }
1114  }
1115 
1116  if(rx_off_prop() != RF_CORE_CMD_OK) {
1117  PRINTF("set_value: rx_off_prop() failed\n");
1118  rv = RADIO_RESULT_ERROR;
1119  }
1120 
1121  if(soft_on_prop() != RF_CORE_CMD_OK) {
1122  PRINTF("set_value: rx_on_prop() failed\n");
1123  rv = RADIO_RESULT_ERROR;
1124  }
1125 
1126  /* If we were off, turn back off */
1127  if(was_off) {
1128  off();
1129  }
1130 
1131  return rv;
1132 }
