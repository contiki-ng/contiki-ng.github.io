75 /* Adds and returns a slotframe (NULL if failure) */
76 struct tsch_slotframe *
77 tsch_schedule_add_slotframe(uint16_t handle, uint16_t size)
78 {
79  if(size == 0) {
80  return NULL;
81  }
82 
83  if(tsch_schedule_get_slotframe_by_handle(handle)) {
84  /* A slotframe with this handle already exists */
85  return NULL;
86  }
87 
88  if(tsch_get_lock()) {
89  struct tsch_slotframe *sf = memb_alloc(&slotframe_memb);
90  if(sf != NULL) {
91  /* Initialize the slotframe */
92  sf->handle = handle;
93  TSCH_ASN_DIVISOR_INIT(sf->size, size);
94  LIST_STRUCT_INIT(sf, links_list);
95  /* Add the slotframe to the global list */
96  list_add(slotframe_list, sf);
97  }
98  LOG_INFO("add_slotframe %u %u\n",
99  handle, size);
100  tsch_release_lock();
101  return sf;
102  }
103  return NULL;
104 }

119 /* Removes a slotframe Return 1 if success, 0 if failure */
120 int
121 tsch_schedule_remove_slotframe(struct tsch_slotframe *slotframe)
122 {
123  if(slotframe != NULL) {
124  /* Remove all links belonging to this slotframe */
125  struct tsch_link *l;
126  while((l = list_head(slotframe->links_list))) {
127  tsch_schedule_remove_link(slotframe, l);
128  }
129 
130  /* Now that the slotframe has no links, remove it. */
131  if(tsch_get_lock()) {
132  LOG_INFO("remove slotframe %u %u\n", slotframe->handle, slotframe->size.val);
133  memb_free(&slotframe_memb, slotframe);
134  list_remove(slotframe_list, slotframe);
135  tsch_release_lock();
136  return 1;
137  }
138  }
139  return 0;
140 }

163  struct tsch_slotframe *sf = list_head(slotframe_list);
164  while(sf != NULL) {
165  struct tsch_link *l = list_head(sf->links_list);
166  /* Loop over all items. Assume there is max one link per timeslot */
167  while(l != NULL) {
168  if(l->handle == handle) {
169  return l;

171  l = list_item_next(l);
172  }
173  sf = list_item_next(sf);
174  }
175  }
176  return NULL;

179 /* Adds a link to a slotframe, return a pointer to it (NULL if failure) */
180 struct tsch_link *
181 tsch_schedule_add_link(struct tsch_slotframe *slotframe,
182  uint8_t link_options, enum link_type link_type, const linkaddr_t *address,
183  uint16_t timeslot, uint16_t channel_offset)
184 {
185  struct tsch_link *l = NULL;
186  if(slotframe != NULL) {
187  /* We currently support only one link per timeslot in a given slotframe. */
188 
189  /* Validation of specified timeslot and channel_offset */
190  if(timeslot > (slotframe->size.val - 1)) {
191  LOG_ERR("! add_link invalid timeslot: %u\n", timeslot);
192  return NULL;
193  } else if(channel_offset > 15) {
194  LOG_ERR("! add_link invalid channel_offset: %u\n", channel_offset);
195  return NULL;
196  }
197 
198  /* Start with removing the link currently installed at this timeslot (needed
199  * to keep neighbor state in sync with link options etc.) */
200  tsch_schedule_remove_link_by_timeslot(slotframe, timeslot);
201  if(!tsch_get_lock()) {
202  LOG_ERR("! add_link memb_alloc couldn't take lock\n");
203  } else {
204  l = memb_alloc(&link_memb);
205  if(l == NULL) {
206  LOG_ERR("! add_link memb_alloc failed\n");
207  tsch_release_lock();
208  } else {
209  static int current_link_handle = 0;
210  struct tsch_neighbor *n;
211  /* Add the link to the slotframe */
212  list_add(slotframe->links_list, l);
213  /* Initialize link */
214  l->handle = current_link_handle++;
215  l->link_options = link_options;
216  l->link_type = link_type;
217  l->slotframe_handle = slotframe->handle;
218  l->timeslot = timeslot;
219  l->channel_offset = channel_offset;
220  l->data = NULL;
221  if(address == NULL) {
222  address = &linkaddr_null;
223  }
224  linkaddr_copy(&l->addr, address);
225 
226  LOG_INFO("add_link %u %u %u %u %u ",
227  slotframe->handle, link_options, link_type, timeslot, channel_offset);
228  LOG_INFO_LLADDR(address);
229  LOG_INFO_("\n");
230  /* Release the lock before we update the neighbor (will take the lock) */
231  tsch_release_lock();
232 
233  if(l->link_options & LINK_OPTION_TX) {
234  n = tsch_queue_add_nbr(&l->addr);
235  /* We have a tx link to this neighbor, update counters */
236  if(n != NULL) {
237  n->tx_links_count++;
238  if(!(l->link_options & LINK_OPTION_SHARED)) {
239  n->dedicated_tx_links_count++;
240  }
241  }
242  }
243  }
244  }
245  }
246  return l;
247 }

249 /* Removes a link from slotframe. Return 1 if success, 0 if failure */
250 int
251 tsch_schedule_remove_link(struct tsch_slotframe *slotframe, struct tsch_link *l)
252 {
253  if(slotframe != NULL && l != NULL && l->slotframe_handle == slotframe->handle) {
254  if(tsch_get_lock()) {
255  uint8_t link_options;
256  linkaddr_t addr;
257 
258  /* Save link option and addr in local variables as we need them
259  * after freeing the link */
260  link_options = l->link_options;
261  linkaddr_copy(&addr, &l->addr);
262 
263  /* The link to be removed is scheduled as next, set it to NULL
264  * to abort the next link operation */
265  if(l == current_link) {
266  current_link = NULL;
267  }
268  LOG_INFO("remove_link %u %u %u %u ",
269  slotframe->handle, l->link_options, l->timeslot, l->channel_offset);
270  LOG_INFO_LLADDR(&l->addr);
271  LOG_INFO_("\n");
272 
273  list_remove(slotframe->links_list, l);
274  memb_free(&link_memb, l);
275 
276  /* Release the lock before we update the neighbor (will take the lock) */
277  tsch_release_lock();
278 
279  /* This was a tx link to this neighbor, update counters */
280  if(link_options & LINK_OPTION_TX) {
281  struct tsch_neighbor *n = tsch_queue_add_nbr(&addr);
282  if(n != NULL) {
283  n->tx_links_count--;
284  if(!(link_options & LINK_OPTION_SHARED)) {
285  n->dedicated_tx_links_count--;
286  }
287  }
288  }
289 
290  return 1;
291  } else {
292  LOG_ERR("! remove_link memb_alloc couldn't take lock\n");
293  }
294  }
295  return 0;
296 }

309 {

311  if(slotframe != NULL) {
312  struct tsch_link *l = list_head(slotframe->links_list);

326 /* Returns the next active link after a given ASN, and a backup link (for the same ASN, with Rx flag) */
327 struct tsch_link *
328 tsch_schedule_get_next_active_link(struct tsch_asn_t *asn, uint16_t *time_offset,
329  struct tsch_link **backup_link)
330 {
331  uint16_t time_to_curr_best = 0;
332  struct tsch_link *curr_best = NULL;
333  struct tsch_link *curr_backup = NULL; /* Keep a back link in case the current link
334  turns out useless when the time comes. For instance, for a Tx-only link, if there is
335  no outgoing packet in queue. In that case, run the backup link instead. The backup link
336  must have Rx flag set. */
337  if(!tsch_is_locked()) {
338  struct tsch_slotframe *sf = list_head(slotframe_list);
339  /* For each slotframe, look for the earliest occurring link */
340  while(sf != NULL) {
341  /* Get timeslot from ASN, given the slotframe length */
342  uint16_t timeslot = TSCH_ASN_MOD(*asn, sf->size);
343  struct tsch_link *l = list_head(sf->links_list);
344  while(l != NULL) {
345  uint16_t time_to_timeslot =
346  l->timeslot > timeslot ?
347  l->timeslot - timeslot :
348  sf->size.val + l->timeslot - timeslot;
349  if(curr_best == NULL || time_to_timeslot < time_to_curr_best) {
350  time_to_curr_best = time_to_timeslot;
351  curr_best = l;
352  curr_backup = NULL;
353  } else if(time_to_timeslot == time_to_curr_best) {
354  struct tsch_link *new_best = NULL;
355  /* Two links are overlapping, we need to select one of them.
356  * By standard: prioritize Tx links first, second by lowest handle */
357  if((curr_best->link_options & LINK_OPTION_TX) == (l->link_options & LINK_OPTION_TX)) {
358  /* Both or neither links have Tx, select the one with lowest handle */
359  if(l->slotframe_handle < curr_best->slotframe_handle) {
360  new_best = l;
361  }
362  } else {
363  /* Select the link that has the Tx option */
364  if(l->link_options & LINK_OPTION_TX) {
365  new_best = l;
366  }
367  }
368 
369  /* Maintain backup_link */
370  if(curr_backup == NULL) {
371  /* Check if 'l' best can be used as backup */
372  if(new_best != l && (l->link_options & LINK_OPTION_RX)) { /* Does 'l' have Rx flag? */
373  curr_backup = l;
374  }
375  /* Check if curr_best can be used as backup */
376  if(new_best != curr_best && (curr_best->link_options & LINK_OPTION_RX)) { /* Does curr_best have Rx flag? */
377  curr_backup = curr_best;
378  }
379  }
380 
381  /* Maintain curr_best */
382  if(new_best != NULL) {
383  curr_best = new_best;
384  }
385  }
386 
387  l = list_item_next(l);
388  }
389  sf = list_item_next(sf);
390  }
391  if(time_offset != NULL) {
392  *time_offset = time_to_curr_best;
393  }
394  }
395  if(backup_link != NULL) {
396  *backup_link = curr_backup;
397  }
398  return curr_best;
399 }

416 /* Create a 6TiSCH minimal schedule */
417 void
418 tsch_schedule_create_minimal(void)
419 {
420  struct tsch_slotframe *sf_min;
421  /* First, empty current schedule */
422  tsch_schedule_remove_all_slotframes();
423  /* Build 6TiSCH minimal schedule.
424  * We pick a slotframe length of TSCH_SCHEDULE_DEFAULT_LENGTH */
425  sf_min = tsch_schedule_add_slotframe(0, TSCH_SCHEDULE_DEFAULT_LENGTH);
426  /* Add a single Tx|Rx|Shared slot using broadcast address (i.e. usable for unicast and broadcast).
427  * We set the link type to advertising, which is not compliant with 6TiSCH minimal schedule
428  * but is required according to 802.15.4e if also used for EB transmission.
429  * Timeslot: 0, channel offset: 0. */
430  tsch_schedule_add_link(sf_min,
431  LINK_OPTION_RX | LINK_OPTION_TX | LINK_OPTION_SHARED | LINK_OPTION_TIME_KEEPING,
432  LINK_TYPE_ADVERTISING, &tsch_broadcast_address,
433  0, 0);
434 }
