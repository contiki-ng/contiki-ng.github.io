96 static void
97 fade(unsigned char l)
98 {
99  volatile int i;
100  int k, j;
101  for(k = 0; k < 800; ++k) {
102  j = k > 400 ? 800 - k : k;
103 
104  leds_on(l);
105  for(i = 0; i < j; ++i) {
106  asm("nop");
107  }
108  leds_off(l);
109  for(i = 0; i < 400 - j; ++i) {
110  asm("nop");
111  }
112  }
113 }

115 static void
116 set_rf_params(void)
117 {
118  uint16_t short_addr;
119  uint8_t ext_addr[8];
120 
121  ieee_addr_cpy_to(ext_addr, 8);
122 
123  short_addr = ext_addr[7];
124  short_addr |= ext_addr[6] << 8;
125 
126  /* Populate linkaddr_node_addr. Maintain endianness */
127  memcpy(&linkaddr_node_addr, &ext_addr[8 - LINKADDR_SIZE], LINKADDR_SIZE);
128 
129 #if STARTUP_CONF_VERBOSE
130  {
131  int i;
132  printf("Contiki configured with address ");
133  for(i = 0; i < LINKADDR_SIZE - 1; i++) {
134  printf("%02x:", linkaddr_node_addr.u8[i]);
135  }
136  printf("%02x\n", linkaddr_node_addr.u8[i]);
137  }
138 #endif
139 
140  NETSTACK_RADIO.set_value(RADIO_PARAM_PAN_ID, IEEE802154_PANID);
141  NETSTACK_RADIO.set_value(RADIO_PARAM_16BIT_ADDR, short_addr);
142  NETSTACK_RADIO.set_value(RADIO_PARAM_CHANNEL, CC2538_RF_CHANNEL);
143  NETSTACK_RADIO.set_object(RADIO_PARAM_64BIT_ADDR, ext_addr, 8);
144 }

185  soc_print_info();
186 #endif
187 
188  random_init(0);
189 
190  udma_init();
191 
192  process_start(&etimer_process, NULL);
193  ctimer_init();
194 
195  board_init();
196

198  crypto_init();

201 
202  netstack_init();

204 
205  PRINTF("Net: ");

209

212  queuebuf_init();

214 #endif /* NETSTACK_CONF_WITH_IPV6 */
215 
216  process_start(&sensors_process, NULL);
217 
218  SENSORS_ACTIVATE(button_sensor);
219 
220  energest_init();
221  ENERGEST_ON(ENERGEST_TYPE_CPU);
222 
223  autostart_start(autostart_processes);
224 
225  watchdog_start();
226  fade(LEDS_GREEN);
227 
228  while(1) {
229  uint8_t r;
230  do {
231  watchdog_periodic();
232 
233  r = process_run();
234  } while(r > 0);
235 
236  lpm_enter();

238 }
