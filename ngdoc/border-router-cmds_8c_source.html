60 static const uint8_t *
61 hextoi(const uint8_t *buf, int len, int *v)
62 {
63  *v = 0;
64  for(; len > 0; len--, buf++) {
65  if(*buf >= '0' && *buf <= '9') {
66  *v = (*v << 4) + ((*buf - '0') & 0xf);
67  } else if(*buf >= 'a' && *buf <= 'f') {
68  *v = (*v << 4) + ((*buf - 'a' + 10) & 0xf);
69  } else if(*buf >= 'A' && *buf <= 'F') {
70  *v = (*v << 4) + ((*buf - 'A' + 10) & 0xf);
71  } else {
72  break;
73  }
74  }
75  return buf;
76 }
77 /*---------------------------------------------------------------------------*/
78 static const uint8_t *
79 dectoi(const uint8_t *buf, int len, int *v)
80 {
81  int negative = 0;
82  *v = 0;
83  if(len <= 0) {
84  return buf;
85  }
86  if(*buf == '$') {
87  return hextoi(buf + 1, len - 1, v);
88  }
89  if(*buf == '0' && *(buf + 1) == 'x' && len > 2) {
90  return hextoi(buf + 2, len - 2, v);
91  }
92  if(*buf == '-') {
93  negative = 1;
94  buf++;
95  }
96  for(; len > 0; len--, buf++) {
97  if(*buf < '0' || *buf > '9') {
98  break;
99  }
100  *v = (*v * 10) + ((*buf - '0') & 0xf);
101  }
102  if(negative) {
103  *v = - *v;
104  }
105  return buf;
106 }

113 int
114 border_router_cmd_handler(const uint8_t *data, int len)
115 {
116  /* handle global repair, etc here */
117  if(data[0] == '!') {
118  PRINTF("Got configuration message of type %c\n", data[1]);
119  if(command_context == CMD_CONTEXT_STDIO) {
120  switch(data[1]) {
121  case 'G':
122  /* This is supposed to be from stdin */
123  printf("Performing Global Repair...\n");
124  NETSTACK_ROUTING.global_repair("Command");
125  return 1;
126  case 'C': {
127  /* send on a set-param thing! */
128  uint8_t set_param[] = {'!', 'V', 0, RADIO_PARAM_CHANNEL, 0, 0 };
129  int channel = -1;
130  dectoi(&data[2], len - 2, &channel);
131  if(channel >= 0) {
132  set_param[5] = channel & 0xff;
133  write_to_slip(set_param, sizeof(set_param));
134  }
135  return 1;
136  }
137  case 'P': {
138  /* send on a set-param thing! */
139  uint8_t set_param[] = {'!', 'V', 0, RADIO_PARAM_PAN_ID, 0, 0 };
140  int pan_id;
141  dectoi(&data[2], len - 2, &pan_id);
142  set_param[4] = (pan_id >> 8) & 0xff;
143  set_param[5] = pan_id & 0xff;
144  write_to_slip(set_param, sizeof(set_param));
145  return 1;
146  }
147  default:
148  return 0;
149  }
150  } else if(command_context == CMD_CONTEXT_RADIO) {
151  /* We need to know that this is from the slip-radio here. */
152  switch(data[1]) {
153  case 'M':
154  PRINTF("Setting MAC address\n");
155  border_router_set_mac(&data[2]);
156  return 1;
157  case 'V':
158  if(data[3] == RADIO_PARAM_CHANNEL) {
159  printf("Channel is %d\n", data[5]);
160  }
161  if(data[3] == RADIO_PARAM_PAN_ID) {
162  printf("PAN_ID is 0x%04x\n", (data[4] << 8) + data[5]);
163  }
164  return 1;
165  case 'R':
166  PRINTF("Packet data report for sid:%d st:%d tx:%d\n",
167  data[2], data[3], data[4]);
168  packet_sent(data[2], data[3], data[4]);
169  return 1;
170  default:
171  return 0;
172  }
173  }
174  } else if(data[0] == '?') {
175  PRINTF("Got request message of type %c\n", data[1]);
176  if(data[1] == 'M' && command_context == CMD_CONTEXT_STDIO) {
177  uint8_t buf[20];
178  char *hexchar = "0123456789abcdef";
179  int j;
180  /* this is just a test so far... just to see if it works */
181  buf[0] = '!';
182  buf[1] = 'M';
183  for(j = 0; j < 8; j++) {
184  buf[2 + j * 2] = hexchar[uip_lladdr.addr[j] >> 4];
185  buf[3 + j * 2] = hexchar[uip_lladdr.addr[j] & 15];
186  }
187  cmd_send(buf, 18);
188  return 1;
189  } else if(data[1] == 'C' && command_context == CMD_CONTEXT_STDIO) {
190  /* send on a set-param thing! */
191  uint8_t set_param[] = {'?', 'V', 0, RADIO_PARAM_CHANNEL};
192  write_to_slip(set_param, sizeof(set_param));
193  return 1;
194  } else if(data[1] == 'P' && command_context == CMD_CONTEXT_STDIO) {
195  /* send on a set-param thing! */
196  uint8_t set_param[] = {'?', 'V', 0, RADIO_PARAM_PAN_ID};
197  write_to_slip(set_param, sizeof(set_param));
198  return 1;
199  } else if(data[1] == 'S') {
200  border_router_print_stat();
201  return 1;
202  }
203  }
204  return 0;
205 }

225 PROCESS_THREAD(border_router_cmd_process, ev, data)
226 {
227  static struct pt shell_input_pt;
228  PROCESS_BEGIN();
229  PRINTF("Started br-cmd process\n");
230 
231  shell_init();
232 
233  while(1) {
234  PROCESS_YIELD();
235  if(ev == serial_line_event_message && data != NULL) {
236  PRINTF("Got serial data!!! %s of len: %lu\n",
237  (char *)data, strlen((char *)data));
238  command_context = CMD_CONTEXT_STDIO;
239  if(cmd_input(data, strlen((char *)data))) {
240  /* Commnand executed - all is fine */
241  } else {
242  /* did not find command - run shell and see if ... */
243  PROCESS_PT_SPAWN(&shell_input_pt, shell_input(&shell_input_pt, serial_shell_output, data));
244  }
245  }
246  }
247  PROCESS_END();
248 }
