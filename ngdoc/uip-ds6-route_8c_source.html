95 static void
96 assert_nbr_routes_list_sane(void)
97 {
98  uip_ds6_route_t *r;
99  int count;
100 
101  /* Check if the route list has an infinite loop. */
102  for(r = uip_ds6_route_head(),
103  count = 0;
104  r != NULL &&
105  count < UIP_DS6_ROUTE_NB * 2;
106  r = uip_ds6_route_next(r),
107  count++);
108 
109  if(count > UIP_DS6_ROUTE_NB) {
110  printf("uip-ds6-route.c: assert_nbr_routes_list_sane route list is in infinite loop\n");
111  }
112 
113 #if (UIP_MAX_ROUTES != 0)
114  /* Make sure that the route list has as many entries as the
115  num_routes vairable. */
116  if(count < num_routes) {
117  printf("uip-ds6-route.c: assert_nbr_routes_list_sane too few entries on route list: should be %d, is %d, max %d\n",
118  num_routes, count, UIP_MAX_ROUTES);
119  }
120 #endif /* (UIP_MAX_ROUTES != 0) */
121 }

255 uip_ds6_route_t *
256 uip_ds6_route_lookup(uip_ipaddr_t *addr)
257 {
258 #if (UIP_MAX_ROUTES != 0)
259  uip_ds6_route_t *r;
260  uip_ds6_route_t *found_route;
261  uint8_t longestmatch;
262 
263  LOG_INFO("Looking up route for ");
264  LOG_INFO_6ADDR(addr);
265  LOG_INFO_("\n");
266 
267  if(addr == NULL) {
268  return NULL;
269  }
270 
271  found_route = NULL;
272  longestmatch = 0;
273  for(r = uip_ds6_route_head();
274  r != NULL;
275  r = uip_ds6_route_next(r)) {
276  if(r->length >= longestmatch &&
277  uip_ipaddr_prefixcmp(addr, &r->ipaddr, r->length)) {
278  longestmatch = r->length;
279  found_route = r;
280  /* check if total match - e.g. all 128 bits do match */
281  if(longestmatch == 128) {
282  break;
283  }
284  }
285  }
286 
287  if(found_route != NULL) {
288  LOG_INFO("Found route: ");
289  LOG_INFO_6ADDR(addr);
290  LOG_INFO_(" via ");
291  LOG_INFO_6ADDR(uip_ds6_route_nexthop(found_route));
292  LOG_INFO_("\n");
293  } else {
294  LOG_WARN("No route found\n");
295  }
296 
297  if(found_route != NULL && found_route != list_head(routelist)) {
298  /* If we found a route, we put it at the start of the routeslist
299  list. The list is ordered by how recently we looked them up:
300  the least recently used route will be at the end of the
301  list - for fast lookups (assuming multiple packets to the same node). */
302 
303  list_remove(routelist, found_route);
304  list_push(routelist, found_route);
305  }
306 
307  return found_route;
308 #else /* (UIP_MAX_ROUTES != 0) */
309  return NULL;
310 #endif /* (UIP_MAX_ROUTES != 0) */
311 }

313 uip_ds6_route_t *
314 uip_ds6_route_add(uip_ipaddr_t *ipaddr, uint8_t length,
315  uip_ipaddr_t *nexthop)
316 {
317 #if (UIP_MAX_ROUTES != 0)
318  uip_ds6_route_t *r;
319  struct uip_ds6_route_neighbor_route *nbrr;
320 
321 #if LOG_DBG_ENABLED
322  assert_nbr_routes_list_sane();
323 #endif /* LOG_DBG_ENABLED */
324 
325  if(ipaddr == NULL || nexthop == NULL) {
326  return NULL;
327  }
328 
329  /* Get link-layer address of next hop, make sure it is in neighbor table */
330  const uip_lladdr_t *nexthop_lladdr = uip_ds6_nbr_lladdr_from_ipaddr(nexthop);
331  if(nexthop_lladdr == NULL) {
332  LOG_WARN("Add: neighbor link-local address unknown for ");
333  LOG_WARN_6ADDR(nexthop);
334  LOG_WARN_("\n");
335  return NULL;
336  }
337 
338  /* First make sure that we don't add a route twice. If we find an
339  existing route for our destination, we'll delete the old
340  one first. */
341  r = uip_ds6_route_lookup(ipaddr);
342  if(r != NULL) {
343  uip_ipaddr_t *current_nexthop;
344  current_nexthop = uip_ds6_route_nexthop(r);
345  if(current_nexthop != NULL && uip_ipaddr_cmp(nexthop, current_nexthop)) {
346  /* no need to update route - already correct! */
347  return r;
348  }
349  LOG_INFO("Add: old route for ");
350  LOG_INFO_6ADDR(ipaddr);
351  LOG_INFO_(" found, deleting it\n");
352 
353  uip_ds6_route_rm(r);
354  }
355  {
356  struct uip_ds6_route_neighbor_routes *routes;
357  /* If there is no routing entry, create one. We first need to
358  check if we have room for this route. If not, we remove the
359  least recently used one we have. */
360 
361  if(uip_ds6_route_num_routes() == UIP_DS6_ROUTE_NB) {
362  uip_ds6_route_t *oldest;
363  oldest = NULL;
364 #if UIP_DS6_ROUTE_REMOVE_LEAST_RECENTLY_USED
365  /* Removing the oldest route entry from the route table. The
366  least recently used route is the first route on the list. */
367  oldest = list_tail(routelist);
368 #endif
369  if(oldest == NULL) {
370  return NULL;
371  }
372  LOG_INFO("Add: dropping route to ");
373  LOG_INFO_6ADDR(&oldest->ipaddr);
374  LOG_INFO_("\n");
375  uip_ds6_route_rm(oldest);
376  }
377 
378 
379  /* Every neighbor on our neighbor table holds a struct
380  uip_ds6_route_neighbor_routes which holds a list of routes that
381  go through the neighbor. We add our route entry to this list.
382 
383  We first check to see if we already have this neighbor in our
384  nbr_route table. If so, the neighbor already has a route entry
385  list.
386  */
387  routes = nbr_table_get_from_lladdr(nbr_routes,
388  (linkaddr_t *)nexthop_lladdr);
389 
390  if(routes == NULL) {
391  /* If the neighbor did not have an entry in our neighbor table,
392  we create one. The nbr_table_add_lladdr() function returns a
393  pointer to a pointer that we may use for our own purposes. We
394  initialize this pointer with the list of routing entries that
395  are attached to this neighbor. */
396  routes = nbr_table_add_lladdr(nbr_routes,
397  (linkaddr_t *)nexthop_lladdr,
398  NBR_TABLE_REASON_ROUTE, NULL);
399  if(routes == NULL) {
400  /* This should not happen, as we explicitly deallocated one
401  route table entry above. */
402  LOG_ERR("Add: could not allocate neighbor table entry\n");
403  return NULL;
404  }
405  LIST_STRUCT_INIT(routes, route_list);
406 #ifdef NETSTACK_CONF_ROUTING_NEIGHBOR_ADDED_CALLBACK
407  NETSTACK_CONF_ROUTING_NEIGHBOR_ADDED_CALLBACK((const linkaddr_t *)nexthop_lladdr);
408 #endif
409  }
410 
411  /* Allocate a routing entry and populate it. */
412  r = memb_alloc(&routememb);
413 
414  if(r == NULL) {
415  /* This should not happen, as we explicitly deallocated one
416  route table entry above. */
417  LOG_ERR("Add: could not allocate route\n");
418  return NULL;
419  }
420 
421  /* add new routes first - assuming that there is a reason to add this
422  and that there is a packet coming soon. */
423  list_push(routelist, r);
424 
425  nbrr = memb_alloc(&neighborroutememb);
426  if(nbrr == NULL) {
427  /* This should not happen, as we explicitly deallocated one
428  route table entry above. */
429  LOG_ERR("Add: could not allocate neighbor route list entry\n");
430  memb_free(&routememb, r);
431  return NULL;
432  }
433 
434  nbrr->route = r;
435  /* Add the route to this neighbor */
436  list_add(routes->route_list, nbrr);
437  r->neighbor_routes = routes;
438  num_routes++;
439 
440  LOG_INFO("Add: num %d\n", num_routes);
441 
442  /* lock this entry so that nexthop is not removed */
443  nbr_table_lock(nbr_routes, routes);
444  }
445 
446  uip_ipaddr_copy(&(r->ipaddr), ipaddr);
447  r->length = length;
448 
449 #ifdef UIP_DS6_ROUTE_STATE_TYPE
450  memset(&r->state, 0, sizeof(UIP_DS6_ROUTE_STATE_TYPE));
451 #endif
452 
453  LOG_INFO("Add: adding route: ");
454  LOG_INFO_6ADDR(ipaddr);
455  LOG_INFO_(" via ");
456  LOG_INFO_6ADDR(nexthop);
457  LOG_INFO_("\n");
458  LOG_ANNOTATE("#L %u 1;blue\n", nexthop->u8[sizeof(uip_ipaddr_t) - 1]);
459 
460 #if UIP_DS6_NOTIFICATIONS
461  call_route_callback(UIP_DS6_NOTIFICATION_ROUTE_ADD, ipaddr, nexthop);
462 #endif
463 
464 #if LOG_DBG_ENABLED
465  assert_nbr_routes_list_sane();
466 #endif /* LOG_DBG_ENABLED */
467  return r;
468 
469 #else /* (UIP_MAX_ROUTES != 0) */
470  return NULL;
471 #endif /* (UIP_MAX_ROUTES != 0) */
472 }

475 void
476 uip_ds6_route_rm(uip_ds6_route_t *route)
477 {
478 #if (UIP_MAX_ROUTES != 0)
479  struct uip_ds6_route_neighbor_route *neighbor_route;
480 #if LOG_DBG_ENABLED
481  assert_nbr_routes_list_sane();
482 #endif /* LOG_DBG_ENABLED */
483  if(route != NULL && route->neighbor_routes != NULL) {
484 
485  LOG_INFO("Rm: removing route: ");
486  LOG_INFO_6ADDR(&route->ipaddr);
487  LOG_INFO_("\n");
488 
489  /* Remove the route from the route list */
490  list_remove(routelist, route);
491 
492  /* Find the corresponding neighbor_route and remove it. */
493  for(neighbor_route = list_head(route->neighbor_routes->route_list);
494  neighbor_route != NULL && neighbor_route->route != route;
495  neighbor_route = list_item_next(neighbor_route));
496 
497  if(neighbor_route == NULL) {
498  LOG_INFO("Rm: neighbor_route was NULL for ");
499  LOG_INFO_6ADDR(&route->ipaddr);
500  LOG_INFO_("\n");
501  }
502  list_remove(route->neighbor_routes->route_list, neighbor_route);
503  if(list_head(route->neighbor_routes->route_list) == NULL) {
504  /* If this was the only route using this neighbor, remove the
505  neighbor from the table - this implicitly unlocks nexthop */
506 #if LOG_WITH_ANNOTATE
507  uip_ipaddr_t *nexthop = uip_ds6_route_nexthop(route);
508  if(nexthop != NULL) {
509  LOG_ANNOTATE("#L %u 0\n", nexthop->u8[sizeof(uip_ipaddr_t) - 1]);
510  }
511 #endif /* LOG_WITH_ANNOTATE */
512  LOG_INFO("Rm: removing neighbor too\n");
513  nbr_table_remove(nbr_routes, route->neighbor_routes->route_list);
514 #ifdef NETSTACK_CONF_ROUTING_NEIGHBOR_REMOVED_CALLBACK
515  NETSTACK_CONF_ROUTING_NEIGHBOR_REMOVED_CALLBACK(
516  (const linkaddr_t *)nbr_table_get_lladdr(nbr_routes, route->neighbor_routes->route_list));
517 #endif
518  }
519  memb_free(&routememb, route);
520  memb_free(&neighborroutememb, neighbor_route);
521 
522  num_routes--;
523 
524  LOG_INFO("Rm: num %d\n", num_routes);
525 
526 #if UIP_DS6_NOTIFICATIONS
527  call_route_callback(UIP_DS6_NOTIFICATION_ROUTE_RM,
528  &route->ipaddr, uip_ds6_route_nexthop(route));
529 #endif
530  }
531 
532 #if LOG_DBG_ENABLED
533  assert_nbr_routes_list_sane();
534 #endif /* LOG_DBG_ENABLED */
535 
536 #endif /* (UIP_MAX_ROUTES != 0) */
537  return;
538 }

541 static void
542 rm_routelist(struct uip_ds6_route_neighbor_routes *routes)
543 {
544 #if LOG_DBG_ENABLED
545  assert_nbr_routes_list_sane();
546 #endif /* LOG_DBG_ENABLED */
547  if(routes != NULL && routes->route_list != NULL) {
548  struct uip_ds6_route_neighbor_route *r;
549  r = list_head(routes->route_list);
550  while(r != NULL) {
551  uip_ds6_route_rm(r->route);
552  r = list_head(routes->route_list);
553  }
554  nbr_table_remove(nbr_routes, routes);
555  }
556 #if LOG_DBG_ENABLED
557  assert_nbr_routes_list_sane();
558 #endif /* LOG_DBG_ENABLED */
559 }

589 uip_ds6_defrt_t *
590 uip_ds6_defrt_add(uip_ipaddr_t *ipaddr, unsigned long interval)
591 {
592  uip_ds6_defrt_t *d;
593 
594 #if LOG_DBG_ENABLED
595  assert_nbr_routes_list_sane();
596 #endif /* LOG_DBG_ENABLED */
597 
598  if(ipaddr == NULL) {
599  return NULL;
600  }
601 
602  LOG_INFO("Add default\n");
603  d = uip_ds6_defrt_lookup(ipaddr);
604  if(d == NULL) {
605  d = memb_alloc(&defaultroutermemb);
606  if(d == NULL) {
607  LOG_ERR("Add default: could not add default route to ");
608  LOG_ERR_6ADDR(ipaddr);
609  LOG_ERR_(", out of memory\n");
