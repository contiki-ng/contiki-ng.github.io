118 /* Reads MAC from SoC
119  * Must be called before node_id_restore()
120  * and network addresses initialization */
121 static void
122 init_node_mac(void)
123 {
124  tuAddr psExtAddress;
125  vMMAC_GetMacAddress(&psExtAddress.sExt);
126  node_mac[7] = psExtAddress.sExt.u32L;
127  node_mac[6] = psExtAddress.sExt.u32L >> (uint32_t)8;
128  node_mac[5] = psExtAddress.sExt.u32L >> (uint32_t)16;
129  node_mac[4] = psExtAddress.sExt.u32L >> (uint32_t)24;
130  node_mac[3] = psExtAddress.sExt.u32H;
131  node_mac[2] = psExtAddress.sExt.u32H >> (uint32_t)8;
132  node_mac[1] = psExtAddress.sExt.u32H >> (uint32_t)16;
133  node_mac[0] = psExtAddress.sExt.u32H >> (uint32_t)24;
134 }

207 static void
208 set_linkaddr(void)
209 {
210  int i;
211  linkaddr_t addr;
212  memset(&addr, 0, LINKADDR_SIZE);
213 #if NETSTACK_CONF_WITH_IPV6
214  memcpy(addr.u8, node_mac, sizeof(addr.u8));
215 #else
216  if(node_id == 0) {
217  for(i = 0; i < LINKADDR_SIZE; ++i) {
218  addr.u8[i] = node_mac[LINKADDR_SIZE - 1 - i];
219  }
220  } else {
221  addr.u8[0] = node_id & 0xff;
222  addr.u8[1] = node_id >> 8;
223  }
224 #endif
225  linkaddr_set_node_addr(&addr);
226 #if DEBUG
227  PRINTF("Link-layer address: ");
228  for(i = 0; i < sizeof(addr.u8) - 1; i++) {
229  PRINTF("%d.", addr.u8[i]);
230  }
231  PRINTF("%d\n", addr.u8[i]);

358 static void
359 main_loop(void)
360 {
361  int r;
362  clock_time_t time_to_etimer;
363  rtimer_clock_t ticks_to_rtimer;
364 
365  while(1) {
366  do {
367  /* Reset watchdog. */
368  watchdog_periodic();
369  r = process_run();
370  } while(r > 0);
371  /*
372  * Idle processing.
373  */
374  watchdog_stop();
375 
376 #if DCOSYNCH_CONF_ENABLED
377  /* Calibrate the DCO every DCOSYNCH_PERIOD
378  * if we have more than 500uSec until next rtimer
379  * PS: Calibration disables interrupts and blocks for 200uSec.
380  * */
381  if(clock_seconds() - last_dco_calibration_time > DCOSYNCH_PERIOD) {
382  if(rtimer_arch_time_to_rtimer() > RTIMER_SECOND / 2000) {
383  /* PRINTF("ContikiMain: Calibrating the DCO\n"); */
384  eAHI_AttemptCalibration();
385  /* Patch to allow CpuDoze after calibration */
386  vREG_PhyWrite(REG_PHY_IS, REG_PHY_INT_VCO_CAL_MASK);
387  last_dco_calibration_time = clock_seconds();
388  }
389  }
390 #endif /* DCOSYNCH_CONF_ENABLED */
391 
392  /* flush standard output before sleeping */
393  uart_driver_flush(E_AHI_UART_0, TRUE, FALSE);
394 
395  /* calculate the time to the next etimer and rtimer */
396  time_to_etimer = clock_arch_time_to_etimer();
397  ticks_to_rtimer = rtimer_arch_time_to_rtimer();
398 
399 #if JN516X_SLEEP_ENABLED
400  /* we can sleep only up to the next rtimer/etimer */
401  rtimer_clock_t max_sleep_time = ticks_to_rtimer;
402  if(max_sleep_time >= JN516X_MIN_SLEEP_TIME) {
403  /* also take into account etimers */
404  uint64_t ticks_to_etimer = ((uint64_t)time_to_etimer * RTIMER_SECOND) / CLOCK_SECOND;
405  max_sleep_time = MIN(ticks_to_etimer, ticks_to_rtimer);
406  }
407 
408  if(max_sleep_time >= JN516X_MIN_SLEEP_TIME) {
409  max_sleep_time -= JN516X_SLEEP_GUARD_TIME;
410  /* bound the sleep time to 1 second */
411  max_sleep_time = MIN(max_sleep_time, JN516X_MAX_SLEEP_TIME);
412 
413 #if !RTIMER_USE_32KHZ
414  /* convert to 32.768 kHz oscillator ticks */
415  max_sleep_time = (uint64_t)max_sleep_time * JN516X_XOSC_SECOND / RTIMER_SECOND;
416 #endif
417  vAHI_WakeTimerEnable(WAKEUP_TIMER, TRUE);
418  /* sync with the tick timer */
419  WAIT_FOR_EDGE(sleep_start);
420  sleep_start_ticks = u32AHI_TickTimerRead();
421 
422  vAHI_WakeTimerStartLarge(WAKEUP_TIMER, max_sleep_time);
423  ENERGEST_SWITCH(ENERGEST_TYPE_CPU, ENERGEST_TYPE_EXTRA_LPM);
424  vAHI_Sleep(E_AHI_SLEEP_OSCON_RAMON);
425  } else {
426 #else
427  {
428 #endif /* JN516X_SLEEP_ENABLED */
429  clock_arch_schedule_interrupt(time_to_etimer, ticks_to_rtimer);
430  ENERGEST_SWITCH(ENERGEST_TYPE_CPU, ENERGEST_TYPE_LPM);
431  vAHI_CpuDoze();
432  watchdog_start();
433  ENERGEST_SWITCH(ENERGEST_TYPE_LPM, ENERGEST_TYPE_CPU);
434  }
435  }
436 }

445 void
446 AppWarmStart(void)
447 {
448  /* Wakeup after sleep with memory on.
449  * Need to initialize devices but not the application state.
450  * Note: the actual time this function is called is
451  * ~8 ticks (32kHz timer) later than the scheduled sleep end time.
452  */
453  uint32_t sleep_ticks;
454  uint64_t sleep_end;
455  rtimer_clock_t sleep_ticks_rtimer;
456 
457  clock_arch_calibrate();
458  leds_init();
459  uart0_init(UART_BAUD_RATE); /* Must come before first PRINTF */
460  NETSTACK_RADIO.init();
461  watchdog_init();
462  watchdog_stop();
463 
464  WAIT_FOR_EDGE(sleep_end);
465  sleep_ticks = (uint32_t)(sleep_start - sleep_end) + 1;
466 
467 #if RTIMER_USE_32KHZ
468  sleep_ticks_rtimer = sleep_ticks;
469 #else
470  {
471  static uint32_t remainder;
472  uint64_t t = (uint64_t)sleep_ticks * RTIMER_SECOND + remainder;
473  sleep_ticks_rtimer = (uint32_t)(t / JN516X_XOSC_SECOND);
474  remainder = t - sleep_ticks_rtimer * JN516X_XOSC_SECOND;
475  }
476 #endif
477 
478  /* reinitialize rtimers */
479  rtimer_arch_reinit(sleep_start_ticks, sleep_ticks_rtimer);
480 
481  ENERGEST_SWITCH(ENERGEST_TYPE_EXTRA_LPM, ENERGEST_TYPE_CPU);
482 
483  watchdog_start();
484 
485  /* reinitialize clock */
486  clock_arch_init(1);
487  /* schedule etimer interrupt */
488  clock_arch_schedule_interrupt(clock_arch_time_to_etimer(), rtimer_arch_time_to_rtimer());
489 
490 #if DCOSYNCH_CONF_ENABLED
491  /* The radio is recalibrated on wakeup */
492  last_dco_calibration_time = clock_seconds();
493 #endif
494 
495  main_loop();
496 }
