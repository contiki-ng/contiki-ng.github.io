55 void
56 shell_output_6addr(shell_output_func output, const uip_ipaddr_t *ipaddr)
57 {
58  uint16_t a;
59  unsigned int i;
60  int f;
61 
62  if(ipaddr == NULL) {
63  SHELL_OUTPUT(output, "(NULL IP addr)");
64  return;
65  }
66 
67  if(ip64_addr_is_ipv4_mapped_addr(ipaddr)) {
68  /* Printing IPv4-mapped addresses is done according to RFC 4291 */
69  SHELL_OUTPUT(output, "::FFFF:%u.%u.%u.%u", ipaddr->u8[12], ipaddr->u8[13], ipaddr->u8[14], ipaddr->u8[15]);
70  } else {
71  for(i = 0, f = 0; i < sizeof(uip_ipaddr_t); i += 2) {
72  a = (ipaddr->u8[i] << 8) + ipaddr->u8[i + 1];
73  if(a == 0 && f >= 0) {
74  if(f++ == 0) {
75  SHELL_OUTPUT(output, "::");
76  }
77  } else {
78  if(f > 0) {
79  f = -1;
80  } else if(i > 0) {
81  SHELL_OUTPUT(output, ":");
82  }
83  SHELL_OUTPUT(output, "%x", a);
84  }
85  }
86  }
87 }

96  unsigned int i;

99  SHELL_OUTPUT(output, ".");

101  SHELL_OUTPUT(output, "%02x", lladdr->u8[i]);
102  }
103  }

113 PT_THREAD(shell_input(struct pt *pt, shell_output_func output, const char *cmd))
114 {
115  static char *args;
116  static struct shell_command_t *cmd_ptr;
117 
118  PT_BEGIN(pt);
119 
120  /* Shave off any leading spaces. */
121  while(*cmd == ' ') {
122  cmd++;
123  }
124 
125  /* Look for arguments */
126  args = strchr(cmd, ' ');
127  if(args != NULL) {
128  *args = '\0';
129  args++;
130  }
131 
132  /* Lookup for command */
133  cmd_ptr = shell_commands;
134  while(cmd_ptr->name != NULL) {
135  if(strcmp(cmd, cmd_ptr->name) == 0) {
136  static struct pt cmd_pt;
137  PT_SPAWN(pt, &cmd_pt, cmd_ptr->func(&cmd_pt, output, args));
138  goto done;
139  }
140  cmd_ptr++;
141  }
142 
143  SHELL_OUTPUT(output, "Command not found. Type 'help' for a list of commands\n");
144 
145 done:
146  output_prompt(output);
147  PT_END(pt);
148 }
