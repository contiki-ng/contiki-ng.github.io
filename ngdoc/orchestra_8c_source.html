71 static void
72 orchestra_packet_sent(int mac_status)
73 {
74  /* Check if our parent just ACKed a DAO */
75  if(orchestra_parent_knows_us == 0
76  && mac_status == MAC_TX_OK
77  && packetbuf_attr(PACKETBUF_ATTR_NETWORK_ID) == UIP_PROTO_ICMP6
78  && packetbuf_attr(PACKETBUF_ATTR_CHANNEL) == (ICMP6_RPL << 8 | RPL_CODE_DAO)) {
79  if(!linkaddr_cmp(&orchestra_parent_linkaddr, &linkaddr_null)
80  && linkaddr_cmp(&orchestra_parent_linkaddr, packetbuf_addr(PACKETBUF_ADDR_RECEIVER))) {
81  orchestra_parent_knows_us = 1;
82  }
83  }
84 }

110 void
111 orchestra_callback_packet_ready(void)
112 {
113  int i;
114  /* By default, use any slotframe, any timeslot */
115  uint16_t slotframe = 9;
116  uint16_t timeslot = 0xffff;
117 
118  /* Loop over all rules until finding one able to handle the packet */
119  for(i = 0; i < NUM_RULES; i++) {
120  if(all_rules[i]->select_packet != NULL) {
121  if(all_rules[i]->select_packet(&slotframe, &timeslot)) {
122  break;
123  }
124  }
125  }
126 
127 #if TSCH_WITH_LINK_SELECTOR
128  packetbuf_set_attr(PACKETBUF_ATTR_TSCH_SLOTFRAME, slotframe);
129  packetbuf_set_attr(PACKETBUF_ATTR_TSCH_TIMESLOT, timeslot);
130 #endif
131 }

134 orchestra_callback_new_time_source(const struct tsch_neighbor *old, const struct tsch_neighbor *new)
135 {
136  /* Orchestra assumes that the time source is also the RPL parent.
137  * This is the case if the following is set:
138  * #define RPL_CALLBACK_PARENT_SWITCH tsch_rpl_callback_parent_switch
139  * */
140 
141  int i;
142  if(new != old) {
143  orchestra_parent_knows_us = 0;
144  }
145  for(i = 0; i < NUM_RULES; i++) {
146  if(all_rules[i]->new_time_source != NULL) {
147  all_rules[i]->new_time_source(old, new);
148  }
149  }
150 }

152 void
153 orchestra_init(void)
154 {
155  int i;
156  /* Snoop on packet transmission to know if our parent knows about us
157  * (i.e. has ACKed at one of our DAOs since we decided to use it as a parent) */
158  netstack_sniffer_add(&orchestra_sniffer);
159  linkaddr_copy(&orchestra_parent_linkaddr, &linkaddr_null);
160  /* Initialize all Orchestra rules */
161  for(i = 0; i < NUM_RULES; i++) {
162  if(all_rules[i]->init != NULL) {
163  PRINTF("Orchestra: initializing rule %u\n", i);
164  all_rules[i]->init(i);
165  }
166  }
167  PRINTF("Orchestra: initialization done\n");
168 }
