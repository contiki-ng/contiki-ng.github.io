55 /** \brief The PKA engine ISR
56  *
57  * This is the interrupt service routine for the PKA engine.
58  *
59  * This ISR is called at worst from PM0, so lpm_exit() does not need
60  * to be called.
61  */
62 void
63 pka_isr(void)
64 {
65  ENERGEST_ON(ENERGEST_TYPE_IRQ);
66 
67  NVIC_ClearPendingIRQ(PKA_IRQn);
68  NVIC_DisableIRQ(PKA_IRQn);
69 
70  if(notification_process != NULL) {
71  process_poll((struct process *)notification_process);
72  notification_process = NULL;
73  }
74 
75  ENERGEST_OFF(ENERGEST_TYPE_IRQ);
76 }
77 /*---------------------------------------------------------------------------*/
78 static bool
79 permit_pm1(void)
80 {
81  return (REG(PKA_FUNCTION) & PKA_FUNCTION_RUN) == 0;
82 }
83 /*---------------------------------------------------------------------------*/
84 void
85 pka_init(void)
86 {
87  volatile int i;
88 
89  lpm_register_peripheral(permit_pm1);
90 
91  pka_enable();
92 
93  /* Reset the PKA engine */
94  REG(SYS_CTRL_SRSEC) |= SYS_CTRL_SRSEC_PKA;
95  for(i = 0; i < 16; i++) {
96  REG(SYS_CTRL_SRSEC) &= ~SYS_CTRL_SRSEC_PKA;
97  }
98 }
99 /*---------------------------------------------------------------------------*/
100 void
101 pka_enable(void)
102 {
103  /* Enable the clock for the PKA engine */
104  REG(SYS_CTRL_RCGCSEC) |= SYS_CTRL_RCGCSEC_PKA;
105  REG(SYS_CTRL_SCGCSEC) |= SYS_CTRL_SCGCSEC_PKA;
106  REG(SYS_CTRL_DCGCSEC) |= SYS_CTRL_DCGCSEC_PKA;
107 }
108 /*---------------------------------------------------------------------------*/
109 void
110 pka_disable(void)
111 {
112  /* Gate the clock for the PKA engine */
113  REG(SYS_CTRL_RCGCSEC) &= ~SYS_CTRL_RCGCSEC_PKA;
114  REG(SYS_CTRL_SCGCSEC) &= ~SYS_CTRL_SCGCSEC_PKA;
115  REG(SYS_CTRL_DCGCSEC) &= ~SYS_CTRL_DCGCSEC_PKA;
116 }
117 /*---------------------------------------------------------------------------*/
118 uint8_t
119 pka_check_status(void)
120 {
121  return (REG(PKA_FUNCTION) & PKA_FUNCTION_RUN) == 0;
122 }
123 void
124 pka_register_process_notification(struct process *p)
125 {
126  notification_process = p;
127 }
