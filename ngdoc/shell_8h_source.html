54 #define SHELL_ARGS_NEXT(args, next_args) do { \
55  (args) = (next_args); \
56  if((args) != NULL) { \
57  if(*(args) == '\0') { \
58  (args) = NULL; \
59  } else { \
60  (next_args) = strchr((args), ' '); \

64  } \
65  } \
