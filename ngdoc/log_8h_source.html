104 #define LOG(newline, level, levelstr, ...) do { \
105  if(level <= (LOG_LEVEL)) { \
106  if(newline) { \
107  if(LOG_WITH_MODULE_PREFIX) { \
108  LOG_OUTPUT_PREFIX(level, levelstr, LOG_MODULE); \
109  } \
110  if(LOG_WITH_LOC) { \
111  LOG_OUTPUT("[%s: %d] ", __FILE__, __LINE__); \
112  } \
113  } \
114  LOG_OUTPUT(__VA_ARGS__); \
115  } \
116  } while (0)
117 
118 /* For Cooja annotations */
119 #define LOG_ANNOTATE(...) do { \
120  if(LOG_WITH_ANNOTATE) { \
121  LOG_OUTPUT(__VA_ARGS__); \
122  } \
123  } while (0)

172 /* For checking log level.
173  As this builds on curr_log_level variables, this should not be used
174  in pre-processor macros. Use in a C 'if' statement instead, e.g.:
175  if(LOG_INFO_ENABLED) { ... }
176  Note that most compilers will still be able to strip the code out
177  for low enough log levels configurations. */
178 #define LOG_ERR_ENABLED ((LOG_LEVEL) >= LOG_LEVEL_ERR)
179 #define LOG_WARN_ENABLED ((LOG_LEVEL) >= LOG_LEVEL_WARN)
180 #define LOG_INFO_ENABLED ((LOG_LEVEL) >= LOG_LEVEL_INFO)
181 #define LOG_DBG_ENABLED ((LOG_LEVEL) >= LOG_LEVEL_DBG)
