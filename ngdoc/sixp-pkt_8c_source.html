81 static int32_t
82 get_cell_options_offset(sixp_pkt_type_t type, sixp_pkt_code_t code)
83 {
84  if(type == SIXP_PKT_TYPE_REQUEST &&
85  (code.cmd == SIXP_PKT_CMD_ADD ||
86  code.cmd == SIXP_PKT_CMD_DELETE ||
87  code.cmd == SIXP_PKT_CMD_RELOCATE ||
88  code.cmd == SIXP_PKT_CMD_COUNT ||
89  code.cmd == SIXP_PKT_CMD_LIST)) {
90  return sizeof(sixp_pkt_metadata_t);
91  }
92  return -1;
93 }

180 sixp_pkt_set_metadata(sixp_pkt_type_t type, sixp_pkt_code_t code,
181  sixp_pkt_metadata_t metadata,
182  uint8_t *body, uint16_t body_len)
183 {
184  int32_t offset;
185 
186  if(body == NULL) {
187  LOG_ERR("6P-pkt: cannot set metadata; body is null\n");
188  return -1;
189  }
190 
191  if((offset = get_metadata_offset(type, code)) < 0) {
192  LOG_ERR("6P-pkt: cannot set metadata [type=%u, code=%u], invalid type\n",
193  type, code.value);
194  return -1;
195  }
196 
197  if(body_len < (offset + sizeof(metadata))) {
198  LOG_ERR("6P-pkt: cannot set metadata, body is too short [body_len=%u]\n",
199  body_len);
200  return -1;
201  }
202 
203  /*
204  * Copy the content into the Metadata field as it is since 6P has no idea
205  * about the internal structure of the field.
206  */
207  memcpy(body + offset, &metadata, sizeof(metadata));
208 
209  return 0;
210 }

418 sixp_pkt_set_offset(sixp_pkt_type_t type, sixp_pkt_code_t code,
419  sixp_pkt_offset_t cell_offset,
420  uint8_t *body, uint16_t body_len)
421 {
422  int32_t offset;
423 
424  if(body == NULL) {
425  LOG_ERR("6P-pkt: cannot set offset; invalid argument\n");
426  return -1;
427  }
428 
429  if((offset = get_offset_offset(type, code)) < 0) {
430  LOG_ERR("6P-pkt: cannot set offset; ");
431  LOG_ERR_("packet [type=%u, code=%u] won't have Offset\n",
432  type, code.value);
433  return -1;
434  }
435 
436  if(body_len < (offset + sizeof(cell_offset))) {
437  LOG_ERR("6P-pkt: cannot set offset; body is too short\n");
438  return -1;
439  }
440 
441  /*
442  * The (Cell)Offset field is 16-bit long; treat it as a little-endian value of
443  * unsigned integer following IEEE 802.15.4-2015.
444  */
445  (body + offset)[0] = *((uint16_t *)&cell_offset) & 0xff;
446  (body + offset)[1] = (*((uint16_t *)&cell_offset) >> 8) & 0xff;
447 
448  return 0;
449 }

863 int
864 sixp_pkt_parse(const uint8_t *buf, uint16_t len,
865  sixp_pkt_t *pkt)
866 {
867  assert(buf != NULL && pkt != NULL);
868  if(buf == NULL || pkt == NULL) {
869  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of invalid argument\n");
870  return -1;
871  }
872 
873  /* read the first 4 octets */
874  if(len < 4) {
875  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because it's a too short packet\n");
876  return -1;
877  }
878 
879  if((buf[0] & 0x0f) != SIXP_PKT_VERSION) {
880  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of invalid version [%u]\n",
881  buf[0] & 0x0f);
882  return -1;
883  }
884 
885  memset(pkt, 0, sizeof(sixp_pkt_t));
886  pkt->type = (buf[0] & 0x30) >> 4;
887  pkt->code.value = buf[1];
888  pkt->sfid = buf[2];
889  pkt->seqno = buf[3] & 0x0f;
890  pkt->gen = (buf[3] & 0xf0) >> 4;
891 
892  buf += 4;
893  len -= 4;
894 
895  LOG_INFO("6P-pkt: sixp_pkt_parse() is processing [type:%u, code:%u, len:%u]\n",
896  pkt->type, pkt->code.value, len);
897 
898  /* the rest is message body called "Other Fields" */
899  if(pkt->type == SIXP_PKT_TYPE_REQUEST) {
900  switch(pkt->code.cmd) {
901  case SIXP_PKT_CMD_ADD:
902  case SIXP_PKT_CMD_DELETE:
903  /* Add and Delete has the same request format */
904  if(len < (sizeof(sixp_pkt_metadata_t) +
905  sizeof(sixp_pkt_cell_options_t) +
906  sizeof(sixp_pkt_num_cells_t)) ||
907  (len % sizeof(uint32_t)) != 0) {
908  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of invalid length\n");
909  return -1;
910  }
911  break;
912  case SIXP_PKT_CMD_RELOCATE:
913  if(len < (sizeof(sixp_pkt_metadata_t) +
914  sizeof(sixp_pkt_cell_options_t) +
915  sizeof(sixp_pkt_num_cells_t)) ||
916  (len % sizeof(uint32_t)) != 0) {
917  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of invalid length\n");
918  return -1;
919  }
920  break;
921  case SIXP_PKT_CMD_COUNT:
922  if(len != (sizeof(sixp_pkt_metadata_t) +
923  sizeof(sixp_pkt_cell_options_t))) {
924  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of invalid length\n");
925  return -1;
926  }
927  break;
928  case SIXP_PKT_CMD_LIST:
929  if(len != (sizeof(sixp_pkt_metadata_t) +
930  sizeof(sixp_pkt_cell_options_t) +
931  sizeof(sixp_pkt_reserved_t) +
932  sizeof(sixp_pkt_offset_t) +
933  sizeof(sixp_pkt_max_num_cells_t))) {
934  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of invalid length\n");
935  return -1;
936  }
937  break;
938  case SIXP_PKT_CMD_CLEAR:
939  if(len != sizeof(sixp_pkt_metadata_t)) {
940  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of invalid length\n");
941  return -1;
942  }
943  break;
944  default:
945  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of unsupported cmd\n");
946  return -1;
947  }
948  } else if(pkt->type == SIXP_PKT_TYPE_RESPONSE ||
949  pkt->type == SIXP_PKT_TYPE_CONFIRMATION) {
950  switch(pkt->code.rc) {
951  case SIXP_PKT_RC_SUCCESS:
952  /*
953  * The "Other Field" contains
954  * - Res to CLEAR: Empty (length 0)
955  * - Res to STATUS: "Num. Cells" (total_num_cells)
956  * - Res to ADD, DELETE, LIST: 0, 1, or multiple 6P cells
957  */
958  if(len != 0 &&
959  len != sizeof(sixp_pkt_total_num_cells_t) &&
960  (len % sizeof(uint32_t)) != 0) {
961  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of invalid length\n");
962  return -1;
963  }
964  break;
965  case SIXP_PKT_RC_EOL:
966  if((len % sizeof(uint32_t)) != 0) {
967  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of invalid length\n");
968  return -1;
969  }
970  break;
971  case SIXP_PKT_RC_ERROR:
972  case SIXP_PKT_RC_RESET:
973  case SIXP_PKT_RC_VERSION:
974  case SIXP_PKT_RC_SFID:
975  case SIXP_PKT_RC_GEN:
976  case SIXP_PKT_RC_BUSY:
977  case SIXP_PKT_RC_NORES:
978  case SIXP_PKT_RC_CELLLIST:
979  if(len != 0) {
980  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of invalid length\n");
981  return -1;
982  }
983  break;
984  default:
985  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of unsupported code\n");
986  return -1;
987  }
988  } else {
989  LOG_ERR("6P-pkt: sixp_pkt_parse() fails because of unsupported type\n");
990  return -1;
991  }
992 
993  pkt->body = buf;
994  pkt->body_len = len;
995 
996  return 0;
997 }

999 int
1000 sixp_pkt_create(sixp_pkt_type_t type, sixp_pkt_code_t code,
1001  uint8_t sfid, uint8_t seqno, uint8_t gen,
1002  const uint8_t *body, uint16_t body_len, sixp_pkt_t *pkt)
1003 {
1004  uint8_t *hdr;
1005 
1006  assert((body == NULL && body_len == 0) || (body != NULL && body_len > 0));
1007  if((body == NULL && body_len > 0) || (body != NULL && body_len == 0)) {
1008  LOG_ERR("6P-pkt: sixp_pkt_create() fails because of invalid argument\n");
1009  return -1;
1010  }
1011 
1012  packetbuf_clear();
1013 
1014  /*
1015  * We're going to create a packet having 6top IE header (4 octets) and body
1016  * (body_len octets).
1017  */
1018  if(PACKETBUF_SIZE < (packetbuf_totlen() + body_len)) {
1019  LOG_ERR("6P-pkt: sixp_pkt_create() fails because body is too long\n");
1020  return -1;
1021  }
1022 
1023  if(packetbuf_hdralloc(4) != 1) {
1024  LOG_ERR("6P-pkt: sixp_pkt_create fails to allocate header space\n");
1025  return -1;
1026  }
1027  hdr = packetbuf_hdrptr();
1028  /* header: write the 6top IE header, 4 octets */
1029  hdr[0] = (type << 4) | SIXP_PKT_VERSION;
1030  hdr[1] = code.value;
1031  hdr[2] = sfid;
1032  hdr[3] = (gen << 4) | seqno;
1033 
1034  /* data: write body */
1035  if(body_len > 0 && body != NULL) {
1036  memcpy(packetbuf_dataptr(), body, body_len);
1037  packetbuf_set_datalen(body_len);
1038  }
1039 
1040  /* copy information of a sending packet into pkt if necessary */
1041  if(pkt != NULL) {
1042  pkt->type = type;
1043  pkt->code = code;
1044  pkt->sfid = sfid;
1045  pkt->seqno = seqno;
1046  pkt->gen = gen;
1047  pkt->body = body;
1048  pkt->body_len = body_len;
1049  }
1050 
1051  /* packetbuf is ready to be sent */
1052  return 0;
1053 }
