72 void
73 uart1_writeb(unsigned char c)
74 {
75  /* watchdog_periodic(); */
76 #if TX_WITH_INTERRUPT
77 
78  /* Put the outgoing byte on the transmission buffer. If the buffer
79  is full, we just keep on trying to put the byte into the buffer
80  until it is possible to put it there. */
81  while(ringbuf_put(&txbuf, c) == 0);
82 
83  /* If there is no transmission going, we need to start it by putting
84  the first byte into the UART. */
85  if(transmitting == 0) {
86  transmitting = 1;
87  UCA0TXBUF = ringbuf_get(&txbuf);
88  }
89 
90 #else /* TX_WITH_INTERRUPT */
91 
92  /* Loop until the transmission buffer is available. */
93  while(!(IFG2 & UCA0TXIFG));
94 
95  /* Transmit the data. */
96  UCA0TXBUF = c;
97 #endif /* TX_WITH_INTERRUPT */
98 }

104 void
105 uart1_init(unsigned long ubr)
106 {
107  /* RS232 */
108  P3SEL |= 0x30; /* P3.4,5 = USCI_A0 TXD/RXD */
109  UCA0CTL1 |= UCSSEL_2; /* CLK = SMCLK */
110  UCA0BR0 = 0x45; /* 8MHz/115200 = 69 = 0x45 */
111  UCA0BR1 = 0x00;
112  UCA0MCTL = UCBRS2; /* Modulation UCBRSx = 4 */
113  UCA0CTL1 &= ~UCSWRST; /* Initialize USCI state machine */
114 
115  transmitting = 0;
116 
117 }

119 ISR(USCIAB1RX, uart1_rx_interrupt)
120 {
121  uint8_t c;
122  ENERGEST_ON(ENERGEST_TYPE_IRQ);
123 
124  /* Check status register for receive errors. */
125  if(UCA0STAT & UCRXERR) {
126  c = UCA0RXBUF; /* Clear error flags by forcing a dummy read. */
127  } else {
128  c = UCA0RXBUF;
129  if(uart1_input_handler != NULL) {
130  if(uart1_input_handler(c)) {
131  LPM4_EXIT;
132  }
133  }
134  }
135  ENERGEST_OFF(ENERGEST_TYPE_IRQ);
136 }
