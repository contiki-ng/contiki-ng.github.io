74 static void
75 handle_rxdma_timer(void *ptr)
76 {
77  uint16_t size;
78  size = DMA0SZ; /* Note: loop requires that size is less or eq to RXBUFSIZE */
79  while(last_size != size) {
80  uart0_input_handler((unsigned char)rxbuf[RXBUFSIZE - last_size]);
81  last_size--;
82  if(last_size == 0) {
83  last_size = RXBUFSIZE;
84  }
85  }
86 
87  ctimer_reset(&rxdma_timer);
88 }

107 void
108 uart0_writeb(unsigned char c)
109 {
110  watchdog_periodic();
111 #if TX_WITH_INTERRUPT
112  /* Put the outgoing byte on the transmission buffer. If the buffer
113  is full, we just keep on trying to put the byte into the buffer
114  until it is possible to put it there. */
115  while(ringbuf_put(&txbuf, c) == 0);
116 
117  /* If there is no transmission going, we need to start it by putting
118  the first byte into the UART. */
119  if(transmitting == 0) {
120  transmitting = 1;
121  UCA0TXBUF = ringbuf_get(&txbuf);
122  }
123 
124 #else /* TX_WITH_INTERRUPT */
125 
126  /* Loop until the transmission buffer is available. */
127  /*Enric while((IFG2 & UCA0TXIFG) == 0); */
128  while((UCA0STAT & UCBUSY));
129 
130  /* Transmit the data. */
131  UCA0TXBUF = c;
132 #endif /* TX_WITH_INTERRUPT */
133 }

136  * Initalize the RS232 port.
137  *
138  */
139 void
140 uart0_init(unsigned long ubr)
141 {
142  /* RS232 */
143  UCA0CTL1 |= UCSWRST; /* Hold peripheral in reset state */
144  UCA0CTL1 |= UCSSEL_2; /* CLK = SMCLK */
145 
146  UCA0BR0 = ((uint8_t *)&ubr)[0]; /* 8MHz/115200 = 69 = 0x45 */
147  UCA0BR1 = ((uint8_t *)&ubr)[1];
148  UCA0MCTL = UCBRS_3; /* Modulation UCBRSx = 3 */
149 
150  P3DIR &= ~0x20; /* P3.5 = USCI_A0 RXD as input */
151  P3DIR |= 0x10; /* P3.4 = USCI_A0 TXD as output */
152  P3SEL |= 0x30; /* P3.4,5 = USCI_A0 TXD/RXD */
153  /*UCA0CTL1 &= ~UCSWRST;*/ /* Initialize USCI state machine */
154 
155  transmitting = 0;
156 
157  /* XXX Clear pending interrupts before enable */
158  IFG2 &= ~UCA0RXIFG;
159  IFG2 &= ~UCA0TXIFG;
160  UCA0CTL1 &= ~UCSWRST; /* Initialize USCI state machine **before** enabling interrupts */
161  IE2 |= UCA0RXIE; /* Enable UCA0 RX interrupt */
162  /* Enable USCI_A0 TX interrupts (if TX_WITH_INTERRUPT enabled) */
163 #if TX_WITH_INTERRUPT
164  ringbuf_init(&txbuf, txbuf_data, sizeof(txbuf_data));
165  IE2 |= UCA0TXIE; /* Enable UCA0 TX interrupt */
166 #endif /* TX_WITH_INTERRUPT */
167 
168 #if RX_WITH_DMA
169  IE2 &= ~UCA0RXIE; /* disable USART0 RX interrupt */
170  /* UART0_RX trigger */
171  DMACTL0 = DMA0TSEL_3;
172 
173  /* source address = UCA0RXBUF */
174  DMA0SA = (unsigned int)&UCA0RXBUF;
175  DMA0DA = (unsigned int)&rxbuf;
176  DMA0SZ = RXBUFSIZE;
177  last_size = RXBUFSIZE;
178  DMA0CTL = DMADT_4 + DMASBDB + DMADSTINCR_3 + DMAEN + DMAREQ;
179 
180  msp430_add_lpm_req(MSP430_REQUIRE_LPM1);
181 #endif /* RX_WITH_DMA */
182 }

218  IFG2 &= ~UCA0TXIFG;

221 }
