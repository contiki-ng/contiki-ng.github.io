207 /*---------------------------------------------------------------------------*/
208 /**
209  * \brief Reads the current signal strength (RSSI)
210  * \return The current RSSI in dBm
211  *
212  * This function reads the current RSSI on the currently configured
213  * channel.
214  */
215 static radio_value_t
216 get_rssi(void)
217 {
218  int8_t rssi;
219 
220  /* If we are off, turn on first */
221  if(RXENABLE == 0) {
222  rf_flags |= WAS_OFF;
223  on();
224  }
225 
226  /* Wait on RSSI_VALID */
227  while((RSSISTAT & RSSISTAT_RSSI_VALID) == 0);
228 
229  rssi = (radio_value_t)RSSI - RSSI_OFFSET;
230 
231  /* If we were off, turn back off */
232  if((rf_flags & WAS_OFF) == WAS_OFF) {
233  rf_flags &= ~WAS_OFF;
234  off();
235  }
236 
237  return rssi;
238 }

254 /* Returns the current TX power in dBm */
255 static radio_value_t
256 get_tx_power(void)
257 {
258  int i;
259  uint8_t reg_val = TXPOWER;
260 
261  if(TXCTRL == CC2530_RF_TX_POWER_TXCTRL_MIN_VAL) {
262  return OUTPUT_POWER_MIN;
263  }
264 
265  /*
266  * Find the TXPOWER value in the lookup table
267  * If the value has been written with set_tx_power, we should be able to
268  * find the exact value. However, in case the register has been written in
269  * a different fashion, we return the immediately lower value of the lookup
270  */
271  for(i = 0; i < OUTPUT_CONFIG_COUNT; i++) {
272  if(reg_val >= output_power[i].txpower_val) {
273  return output_power[i].power;
274  }
275  }
276  return OUTPUT_POWER_MIN;
277 }
278 /*---------------------------------------------------------------------------*/
279 /*
280  * Set TX power to 'at least' power dBm
281  * This works with a lookup table. If the value of 'power' does not exist in
282  * the lookup table, TXPOWER will be set to the immediately higher available
283  * value
284  */
285 static void
286 set_tx_power(radio_value_t power)
287 {
288  int i;
289 
290  if(power <= output_power[OUTPUT_CONFIG_COUNT - 1].power) {
291  TXCTRL = CC2530_RF_TX_POWER_TXCTRL_MIN_VAL;
292  TXPOWER = output_power[OUTPUT_CONFIG_COUNT - 1].txpower_val;
293  return;
294  }
295 
296  for(i = OUTPUT_CONFIG_COUNT - 2; i >= 0; --i) {
297  if(power <= output_power[i].power) {
298  /* Perhaps an earlier call set TXCTRL to 0x09. Restore */
299  TXCTRL = CC2530_RF_TX_POWER_TXCTRL_DEF_VAL;
300  TXPOWER = output_power[i].txpower_val;
301  return;
302  }
303  }
304 }

328 static int
329 init(void)
330 {
331  PUTSTRING("RF: Init\n");
332 
333  if(rf_flags & RF_ON) {
334  return 0;
335  }
336 
337 #if CC2530_RF_LOW_POWER_RX
338  /* Reduce RX power consumption current to 20mA at the cost of sensitivity */
339  RXCTRL = 0x00;
340  FSCTRL = 0x50;
341 #else
342  RXCTRL = 0x3F;
343  FSCTRL = 0x55;
344 #endif /* CC2530_RF_LOW_POWER_RX */
345 
346  CCACTRL0 = CC2530_RF_CCA_THRES;
347 
348  /*
349  * According to the user guide, these registers must be updated from their
350  * defaults for optimal performance
351  *
352  * Table 23-6, Sec. 23.15.1, p. 259
353  */
354  TXFILTCFG = 0x09; /* TX anti-aliasing filter */
355  AGCCTRL1 = 0x15; /* AGC target value */
356  FSCAL1 = 0x00; /* Reduce the VCO leakage */
357 
358  /* Auto ACKs and CRC calculation, default RX and TX modes with FIFOs */
359  FRMCTRL0 = FRMCTRL0_AUTOCRC;
360 #if CC2530_RF_AUTOACK
361  FRMCTRL0 |= FRMCTRL0_AUTOACK;
362 #endif
363 
364  /* Disable source address matching and autopend */
365  SRCMATCH = 0; /* investigate */
366 
367  /* MAX FIFOP threshold */
368  FIFOPCTRL = CC2530_RF_MAX_PACKET_LEN;
369 
370  TXPOWER = CC2530_RF_TX_POWER;
371 
372  RF_TX_LED_OFF();
373  RF_RX_LED_OFF();
374 
375  rf_flags |= RF_ON;
376 
377  return 1;
378 }
379 /*---------------------------------------------------------------------------*/
380 static int
381 prepare(const void *payload, unsigned short payload_len)
382 {
383  uint8_t i;
384 
385  PUTSTRING("RF: Prepare 0x");
386  PUTHEX(payload_len + CHECKSUM_LEN);
387  PUTSTRING(" bytes\n");
388 
389  /*
390  * When we transmit in very quick bursts, make sure previous transmission
391  * is not still in progress before re-writing to the TX FIFO
392  */
393  while(FSMSTAT1 & FSMSTAT1_TX_ACTIVE);
394 
395  if((rf_flags & RX_ACTIVE) == 0) {
396  on();
397  }
398 
399  CC2530_CSP_ISFLUSHTX();
400 
401  PUTSTRING("RF: data = ");
402  /* Send the phy length byte first */
403  RFD = payload_len + CHECKSUM_LEN; /* Payload plus FCS */
404  for(i = 0; i < payload_len; i++) {
405  RFD = ((unsigned char *)(payload))[i];
406  PUTHEX(((unsigned char *)(payload))[i]);
407  }
408  PUTSTRING("\n");
409 
410  /* Leave space for the FCS */
411  RFD = 0;
412  RFD = 0;
413 
414  return 0;
415 }
416 /*---------------------------------------------------------------------------*/
417 static int
418 transmit(unsigned short transmit_len)
419 {
420  uint8_t counter;
421  int ret = RADIO_TX_ERR;
422  rtimer_clock_t t0;
423  transmit_len; /* hush the warning */
424 
425  if(!(rf_flags & RX_ACTIVE)) {
426  t0 = RTIMER_NOW();
427  on();
428  rf_flags |= WAS_OFF;
429  while(RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + ONOFF_TIME));
430  }
431 
432  if(channel_clear() == CC2530_RF_CCA_BUSY) {
433  return RADIO_TX_COLLISION;
434  }
435 
436  /*
437  * prepare() double checked that TX_ACTIVE is low. If SFD is high we are
438  * receiving. Abort transmission and bail out with RADIO_TX_COLLISION
439  */
440  if(FSMSTAT1 & FSMSTAT1_SFD) {
441  return RADIO_TX_COLLISION;
442  }
443 
444  /* Start the transmission */
445  RF_TX_LED_ON();
446  ENERGEST_OFF(ENERGEST_TYPE_LISTEN);
447  ENERGEST_ON(ENERGEST_TYPE_TRANSMIT);
448 
449  CC2530_CSP_ISTXON();
450 
451  counter = 0;
452  while(!(FSMSTAT1 & FSMSTAT1_TX_ACTIVE) && (counter++ < 3)) {
453  clock_delay_usec(6);
454  }
455 
456  if(!(FSMSTAT1 & FSMSTAT1_TX_ACTIVE)) {
457  PUTSTRING("RF: TX never active.\n");
458  CC2530_CSP_ISFLUSHTX();
459  ret = RADIO_TX_ERR;
460  } else {
461  /* Wait for the transmission to finish */
462  while(FSMSTAT1 & FSMSTAT1_TX_ACTIVE);
463  ret = RADIO_TX_OK;
464  }
465  ENERGEST_OFF(ENERGEST_TYPE_TRANSMIT);
466  ENERGEST_ON(ENERGEST_TYPE_LISTEN);
467 
468  if(rf_flags & WAS_OFF) {
469  off();
470  }
471 
472  RF_TX_LED_OFF();
473 
474  /* OK, sent. We are now ready to send more */
475  return ret;
476 }

484 /*---------------------------------------------------------------------------*/
485 static int
486 read(void *buf, unsigned short bufsize)
487 {
488  uint8_t i;
489  uint8_t len;
490  uint8_t crc_corr;
491  int8_t rssi;
492 
493  PUTSTRING("RF: Read\n");
494 
495  /* Check the length */
496  len = RFD;
497 
498  /* Check for validity */
499  if(len > CC2530_RF_MAX_PACKET_LEN) {
500  /* Oops, we must be out of sync. */
501  PUTSTRING("RF: bad sync\n");
502 
503  CC2530_CSP_ISFLUSHRX();
504  return 0;
505  }
506 
507  if(len <= CC2530_RF_MIN_PACKET_LEN) {
508  PUTSTRING("RF: too short\n");
509 
510  CC2530_CSP_ISFLUSHRX();
511  return 0;
512  }
513 
514  if(len - CHECKSUM_LEN > bufsize) {
515  PUTSTRING("RF: too long\n");
516 
517  CC2530_CSP_ISFLUSHRX();
518  return 0;
519  }
520 
521  RF_RX_LED_ON();
522 
523  PUTSTRING("RF: read (0x");
524  PUTHEX(len);
525  PUTSTRING(" bytes) = ");
526  len -= CHECKSUM_LEN;
527  for(i = 0; i < len; ++i) {
528  ((unsigned char *)(buf))[i] = RFD;
529  PUTHEX(((unsigned char *)(buf))[i]);
530  }
531  PUTSTRING("\n");
532 
533  /* Read the RSSI and CRC/Corr bytes */
534  rssi = ((int8_t) RFD) - RSSI_OFFSET;
535  crc_corr = RFD;
536 
537  /* MS bit CRC OK/Not OK, 7 LS Bits, Correlation value */
538  if(crc_corr & CRC_BIT_MASK) {
539  packetbuf_set_attr(PACKETBUF_ATTR_RSSI, rssi);
540  packetbuf_set_attr(PACKETBUF_ATTR_LINK_QUALITY, crc_corr & LQI_BIT_MASK);
541  } else {
542  CC2530_CSP_ISFLUSHRX();
543  RF_RX_LED_OFF();
544  return 0;
545  }
546 
547  /* If FIFOP==1 and FIFO==0 then we had a FIFO overflow at some point. */
548  if((FSMSTAT1 & (FSMSTAT1_FIFO | FSMSTAT1_FIFOP)) == FSMSTAT1_FIFOP) {
549  /*
550  * If we reach here means that there might be more intact packets in the
551  * FIFO despite the overflow. This can happen with bursts of small packets.
552  *
553  * Only flush if the FIFO is actually empty. If not, then next pass we will
554  * pick up one more packet or flush due to an error.
555  */
556  if(!RXFIFOCNT) {
557  CC2530_CSP_ISFLUSHRX();
558  }
559  }
560 
561  RF_RX_LED_OFF();
562 
563  return (len);
564 }

595 static int
596 on(void)
597 {
598  if(!(rf_flags & RX_ACTIVE)) {
599  CC2530_CSP_ISFLUSHRX();
600  CC2530_CSP_ISRXON();
601 
602  rf_flags |= RX_ACTIVE;
603  }
604 
605  ENERGEST_ON(ENERGEST_TYPE_LISTEN);
606  return 1;

614

616 
617  ENERGEST_OFF(ENERGEST_TYPE_LISTEN);
618  return 1;

621 static radio_result_t
622 get_value(radio_param_t param, radio_value_t *value)
623 {
624  if(!value) {
625  return RADIO_RESULT_INVALID_VALUE;
626  }
627 
628  switch(param) {
629  case RADIO_PARAM_POWER_MODE:
630  *value = RXENABLE == 0 ? RADIO_POWER_MODE_OFF : RADIO_POWER_MODE_ON;
631  return RADIO_RESULT_OK;
632  case RADIO_PARAM_CHANNEL:
633  *value = (radio_value_t)get_channel();
634  return RADIO_RESULT_OK;
635  case RADIO_PARAM_PAN_ID:
636  *value = get_pan_id();
637  return RADIO_RESULT_OK;
638  case RADIO_PARAM_16BIT_ADDR:
639  *value = get_short_addr();
640  return RADIO_RESULT_OK;
641  case RADIO_PARAM_RX_MODE:
642  *value = 0;
643  if(FRMFILT0 & FRMFILT0_FRAME_FILTER_EN) {
644  *value |= RADIO_RX_MODE_ADDRESS_FILTER;
645  }
646  if(FRMCTRL0 & FRMCTRL0_AUTOACK) {
647  *value |= RADIO_RX_MODE_AUTOACK;
648  }
649  return RADIO_RESULT_OK;
650  case RADIO_PARAM_TXPOWER:
651  *value = get_tx_power();
652  return RADIO_RESULT_OK;
653  case RADIO_PARAM_CCA_THRESHOLD:
654  *value = get_cca_threshold();
655  return RADIO_RESULT_OK;
656  case RADIO_PARAM_RSSI:
657  *value = get_rssi();
658  return RADIO_RESULT_OK;
659  case RADIO_CONST_CHANNEL_MIN:
660  *value = CC2530_RF_CHANNEL_MIN;
661  return RADIO_RESULT_OK;
662  case RADIO_CONST_CHANNEL_MAX:
663  *value = CC2530_RF_CHANNEL_MAX;
664  return RADIO_RESULT_OK;
665  case RADIO_CONST_TXPOWER_MIN:
666  *value = OUTPUT_POWER_MIN;
667  return RADIO_RESULT_OK;
668  case RADIO_CONST_TXPOWER_MAX:
669  *value = OUTPUT_POWER_MAX;
670  return RADIO_RESULT_OK;
671  default:
672  return RADIO_RESULT_NOT_SUPPORTED;
673  }
674 }
675 /*---------------------------------------------------------------------------*/
676 static radio_result_t
677 set_value(radio_param_t param, radio_value_t value)
678 {
679  switch(param) {
680  case RADIO_PARAM_POWER_MODE:
681  if(value == RADIO_POWER_MODE_ON) {
682  on();
683  return RADIO_RESULT_OK;
684  }
685  if(value == RADIO_POWER_MODE_OFF) {
686  off();
687  return RADIO_RESULT_OK;
688  }
689  return RADIO_RESULT_INVALID_VALUE;
690  case RADIO_PARAM_CHANNEL:
691  if(value < CC2530_RF_CHANNEL_MIN || value > CC2530_RF_CHANNEL_MAX) {
692  return RADIO_RESULT_INVALID_VALUE;
693  }
694  if(set_channel(value) == CC2530_RF_CHANNEL_SET_ERROR) {
695  return RADIO_RESULT_ERROR;
696  }
