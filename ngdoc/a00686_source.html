56 bme280_arch_i2c_write_mem(uint8_t addr, uint8_t reg, uint8_t value)
57 {
58  uint8_t buf[2];
59 
60  buf[0] = reg;
61  buf[1] = value;
62 
63  i2c_master_enable();

65 }

67 void
68 bme280_arch_i2c_read_mem(uint8_t addr, uint8_t reg, uint8_t *buf, uint8_t bytes)
69 {
70  i2c_master_enable();
71  if(i2c_single_send(addr, reg) == I2C_MASTER_ERR_NONE) {
72  while(i2c_master_busy());
73  i2c_burst_receive(addr, buf, bytes);
74  }
75 }
