51 rgb_led_set(uint8_t colour)
52 {
53  leds_mask_t leds =
54  ((colour & RGB_LED_RED) ? LEDS_RED : LEDS_COLOUR_NONE) |
55  ((colour & RGB_LED_GREEN) ? LEDS_GREEN : LEDS_COLOUR_NONE) |
56  ((colour & RGB_LED_BLUE) ? LEDS_BLUE : LEDS_COLOUR_NONE);
57 
58  leds_off(LEDS_ALL);
59  leds_on(leds);
60 }
