67 node_matches_address(const rpl_ns_node_t *node, const uip_ipaddr_t *addr)
68 {
69  return addr != NULL
70  && node != NULL
71  && !memcmp(addr, &curr_instance.dag.dag_id, 8)
72  && !memcmp(((const unsigned char *)addr) + 8, node->link_identifier, 8);
73 }

78  rpl_ns_node_t *l;
79  for(l = list_head(nodelist); l != NULL; l = list_item_next(l)) {

88 int
89 rpl_ns_is_addr_reachable(const uip_ipaddr_t *addr)
90 {
91  int max_depth = RPL_NS_LINK_NUM;
92  rpl_ns_node_t *node = rpl_ns_get_node(addr);
93  rpl_ns_node_t *root_node = rpl_ns_get_node(&curr_instance.dag.dag_id);
94  while(node != NULL && node != root_node && max_depth > 0) {
95  node = node->parent;
96  max_depth--;
97  }
98  return node != NULL && node == root_node;
99 }

111 rpl_ns_node_t *
112 rpl_ns_update_node(const uip_ipaddr_t *child, const uip_ipaddr_t *parent, uint32_t lifetime)
113 {
114  rpl_ns_node_t *child_node = rpl_ns_get_node(child);
115  rpl_ns_node_t *parent_node = rpl_ns_get_node(parent);
116  rpl_ns_node_t *old_parent_node;
117 
118  if(parent != NULL) {
119  /* No node for the parent, add one with infinite lifetime */
120  if(parent_node == NULL) {
121  parent_node = rpl_ns_update_node(parent, NULL, RPL_ROUTE_INFINITE_LIFETIME);
122  if(parent_node == NULL) {
123  LOG_ERR("NS: no space left for root node!\n");
124  return NULL;
125  }
126  }
127  }
128 
129  /* No node for this child, add one */
130  if(child_node == NULL) {
131  child_node = memb_alloc(&nodememb);
132  /* No space left, abort */
133  if(child_node == NULL) {
134  LOG_ERR("NS: no space left for child ");
135  LOG_ERR_6ADDR(child);
136  LOG_ERR_("\n");
137  return NULL;
138  }
139  child_node->parent = NULL;
140  list_add(nodelist, child_node);
141  num_nodes++;
142  }
143 
144  /* Initialize node */
145  child_node->lifetime = lifetime;
146  memcpy(child_node->link_identifier, ((const unsigned char *)child) + 8, 8);
147 
148  /* Is the node reachable before the update? */
149  if(rpl_ns_is_addr_reachable(child)) {
150  old_parent_node = child_node->parent;
151  /* Update node */
152  child_node->parent = parent_node;
153  /* Has the node become unreachable? May happen if we create a loop. */
154  if(!rpl_ns_is_addr_reachable(child)) {
155  /* The new parent makes the node unreachable, restore old parent.
156  * We will take the update next time, with chances we know more of
157  * the topology and the loop is gone. */
158  child_node->parent = old_parent_node;
159  }
160  } else {
161  child_node->parent = parent_node;
162  }
163 
164  LOG_INFO("NS: updating link, child ");
165  LOG_INFO_6ADDR(child);
166  LOG_INFO_(", parent ");
167  LOG_INFO_6ADDR(parent);
168  LOG_INFO_(", lifetime %u, num_nodes %u\n", (unsigned)lifetime, num_nodes);
169 
170  return child_node;
171 }

205 void
206 rpl_ns_periodic(unsigned seconds)
207 {
208  rpl_ns_node_t *l;
209  rpl_ns_node_t *next;
210  /* First pass, decrement lifetime for all nodes with non-infinite lifetime */
211  for(l = list_head(nodelist); l != NULL; l = list_item_next(l)) {
212  /* Don't touch infinite lifetime nodes */
213  if(l->lifetime != RPL_ROUTE_INFINITE_LIFETIME) {
214  l->lifetime = l->lifetime > seconds ? l->lifetime - seconds : 0;
215  }
216  }
217  /* Second pass, for all expired nodes, deallocate them iff no child points to them */
218  for(l = list_head(nodelist); l != NULL; l = next) {
219  next = list_item_next(l);
220  if(l->lifetime == 0) {
221  rpl_ns_node_t *l2;
222  for(l2 = list_head(nodelist); l2 != NULL; l2 = list_item_next(l2)) {
223  if(l2->parent == l) {
224  break;
225  }
226  }
227 #if LOG_INFO_ENABLED
228  uip_ipaddr_t node_addr;
229  rpl_ns_get_node_global_addr(&node_addr, l);
230  LOG_INFO("NS: removing expired node ");
231  LOG_INFO_6ADDR(&node_addr);
232  LOG_INFO_("\n");
233 #endif /* LOG_INFO_ENABLED */
234  /* No child found, deallocate node */
235  list_remove(nodelist, l);
236  memb_free(&nodememb, l);
237  num_nodes--;
238  }
239  }
240 }
