55 /* Per-neighbor RPL information. According to RFC 6550, there exist three
56  * types of neighbors:
57  * - Candidate neighbor set: any neighbor, selected in an implementation
58  * and OF-specific way. The nodes in rpl_neighbors constitute the candidate neighbor set.
59  * - Parent set: the subset of the candidate neighbor set with rank below our rank
60  * - Preferred parent: one node of the parent set
61  */
62 NBR_TABLE_DECLARE(rpl_neighbors);
