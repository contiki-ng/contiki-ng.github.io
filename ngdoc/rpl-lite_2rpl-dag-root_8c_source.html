57 static void
58 create_dag_callback(void *ptr)
59 {
60  const uip_ipaddr_t *root, *ipaddr;
61 
62  root = curr_instance.used ? &curr_instance.dag.dag_id : NULL;
63  ipaddr = rpl_get_global_address();
64 
65  if(root == NULL || uip_ipaddr_cmp(root, ipaddr)) {
66  /* The RPL network we are joining is one that we created, so we
67  become root. */
68  if(to_become_root) {
69  rpl_dag_root_init_dag_immediately();
70  to_become_root = 0;
71  }
72  } else {
73  LOG_WARN("found a network we did not create: version %d grounded %d preference %d rank %d\n",
74  curr_instance.dag.version, curr_instance.dag.grounded,
75  curr_instance.dag.preference, curr_instance.dag.rank);
76 
77  /* We found a RPL network that we did not create so we just join
78  it without becoming root. But if the network has an infinite
79  rank, we assume the network has broken, and we become the new
80  root of the network. */
81 
82  if(curr_instance.dag.rank == RPL_INFINITE_RANK) {
83  if(to_become_root) {
84  rpl_dag_root_init_dag_immediately();
85  to_become_root = 0;
86  }
87  }
88 
89  /* Try again after the grace period */
90  ctimer_set(&c, RPL_DAG_GRACE_PERIOD, create_dag_callback, NULL);
91  }
92 }

94 static void
95 set_global_address(uip_ipaddr_t *prefix, uip_ipaddr_t *iid)
96 {
97  static uip_ipaddr_t root_ipaddr;
98  int i;
99  uint8_t state;
100 
101  /* Assign a unique local address (RFC4193,
102  http://tools.ietf.org/html/rfc4193). */
103  if(prefix == NULL) {
104  uip_ip6addr(&root_ipaddr, UIP_DS6_DEFAULT_PREFIX, 0, 0, 0, 0, 0, 0, 0);
105  } else {
106  memcpy(&root_ipaddr, prefix, 8);
107  }
108  if(iid == NULL) {
109  uip_ds6_set_addr_iid(&root_ipaddr, &uip_lladdr);
110  } else {
111  memcpy(((uint8_t*)&root_ipaddr) + 8, ((uint8_t*)iid) + 8, 8);
112  }
113 
114  uip_ds6_addr_add(&root_ipaddr, 0, ADDR_AUTOCONF);
115 
116  LOG_INFO("IPv6 addresses: ");
117  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
118  state = uip_ds6_if.addr_list[i].state;
119  if(uip_ds6_if.addr_list[i].isused &&
120  (state == ADDR_TENTATIVE || state == ADDR_PREFERRED)) {
121  LOG_INFO_6ADDR(&uip_ds6_if.addr_list[i].ipaddr);
122  LOG_INFO_("\n");
123  }
124  }
125 }

139 int
140 rpl_dag_root_init_dag_immediately(void)
141 {
142  struct uip_ds6_addr *root_if;
143  int i;
144  uint8_t state;
145  uip_ipaddr_t *ipaddr = NULL;
146 
147  rpl_dag_root_init(NULL, NULL);
148 
149  for(i = 0; i < UIP_DS6_ADDR_NB; i++) {
150  state = uip_ds6_if.addr_list[i].state;
151  if(uip_ds6_if.addr_list[i].isused &&
152  state == ADDR_PREFERRED &&
153  !uip_is_addr_linklocal(&uip_ds6_if.addr_list[i].ipaddr)) {
154  ipaddr = &uip_ds6_if.addr_list[i].ipaddr;
155  }
156  }
157 
158  root_if = uip_ds6_addr_lookup(ipaddr);
159  if(ipaddr != NULL || root_if != NULL) {
160 
161  rpl_dag_init_root(RPL_DEFAULT_INSTANCE, ipaddr,
162  (uip_ipaddr_t *)rpl_get_global_address(), 64, UIP_ND6_RA_FLAG_AUTONOMOUS);
163  rpl_dag_update_state();
164 
165  LOG_INFO("created a new RPL DAG\n");
166  return 0;
167  } else {
168  LOG_ERR("failed to create a new RPL DAG\n");
169  return -1;
170  }
171 }
