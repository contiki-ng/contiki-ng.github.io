81 static void
82 fade(unsigned char l)
83 {
84  volatile int i;
85  int k, j;
86  for(k = 0; k < 800; ++k) {
87  j = k > 400 ? 800 - k : k;
88 
89  leds_on(l);
90  for(i = 0; i < j; ++i) {
91  __asm("nop");
92  }
93  leds_off(l);
94  for(i = 0; i < 400 - j; ++i) {
95  __asm("nop");
96  }
97  }
98 }

100 static void
101 set_rf_params(void)
102 {
103  uint16_t short_addr;
104  uint8_t ext_addr[8];
105  radio_value_t val = 0;
106 
107  ieee_addr_cpy_to(ext_addr, 8);
108 
109  short_addr = ext_addr[7];
110  short_addr |= ext_addr[6] << 8;
111 
112  /* Populate linkaddr_node_addr. Maintain endianness */
113  memcpy(&linkaddr_node_addr, &ext_addr[8 - LINKADDR_SIZE], LINKADDR_SIZE);
114 
115  NETSTACK_RADIO.set_value(RADIO_PARAM_PAN_ID, IEEE802154_PANID);
116  NETSTACK_RADIO.set_value(RADIO_PARAM_16BIT_ADDR, short_addr);
117  NETSTACK_RADIO.set_value(RADIO_PARAM_CHANNEL, RF_CORE_CHANNEL);
118  NETSTACK_RADIO.set_object(RADIO_PARAM_64BIT_ADDR, ext_addr, 8);
119 
120  NETSTACK_RADIO.get_value(RADIO_PARAM_CHANNEL, &val);
121  printf(" RF: Channel %d\n", val);
122 
123 #if STARTUP_CONF_VERBOSE
124  {
125  int i;
126  printf(" Link layer addr: ");
127  for(i = 0; i < LINKADDR_SIZE - 1; i++) {
128  printf("%02x:", linkaddr_node_addr.u8[i]);
129  }
130  printf("%02x\n", linkaddr_node_addr.u8[i]);
131  }
132 #endif
133 
134  /* also set the global node id */
135  node_id = short_addr;
136  printf(" Node ID: %d\n", node_id);
137 }

139 /**
140  * \brief Main function for CC26xx-based platforms
141  *
142  * The same main() is used for all supported boards
143  */
144 int
145 main(void)
146 {
147  /* Enable flash cache and prefetch. */
148  ti_lib_vims_mode_set(VIMS_BASE, VIMS_MODE_ENABLED);
149  ti_lib_vims_configure(VIMS_BASE, true, true);
150 
151  ti_lib_int_master_disable();
152 
153  /* Set the LF XOSC as the LF system clock source */
154  oscillators_select_lf_xosc();
155 
156  lpm_init();
157 
158  board_init();
159 
160  gpio_interrupt_init();
161 
162  leds_init();
163 
164  /*
165  * Disable I/O pad sleep mode and open I/O latches in the AON IOC interface
166  * This is only relevant when returning from shutdown (which is what froze
167  * latches in the first place. Before doing these things though, we should
168  * allow software to first regain control of pins
169  */
170  ti_lib_pwr_ctrl_io_freeze_disable();
171 
172  fade(LEDS_RED);
173 
174  ti_lib_int_master_enable();
175 
176  soc_rtc_init();
177  clock_init();
178  rtimer_init();
179 
180  watchdog_init();
181  process_init();
182 
183  random_init(0x1234);
184 
185  /* Character I/O Initialisation */
186 #if CC26XX_UART_CONF_ENABLE
187  cc26xx_uart_init();
188 #endif
189 
190  serial_line_init();
191 
192  printf("Starting " CONTIKI_VERSION_STRING "\n");
193  printf("With DriverLib v%u.%u\n", DRIVERLIB_RELEASE_GROUP,
194  DRIVERLIB_RELEASE_BUILD);
195  printf(BOARD_STRING "\n");
196  printf("IEEE 802.15.4: %s, Sub-GHz: %s, BLE: %s, Prop: %s\n",
197  ti_lib_chipinfo_supports_ieee_802_15_4() == true ? "Yes" : "No",
198  ti_lib_chipinfo_chip_family_is_cc13xx() == true ? "Yes" : "No",
199  ti_lib_chipinfo_supports_ble() == true ? "Yes" : "No",
200  ti_lib_chipinfo_supports_proprietary() == true ? "Yes" : "No");
201 
202  process_start(&etimer_process, NULL);
203  ctimer_init();
204 
205  energest_init();
206  ENERGEST_ON(ENERGEST_TYPE_CPU);
207 
208  fade(LEDS_YELLOW);
209 
210  printf(" Net: ");
211  printf("%s\n", NETSTACK_NETWORK.name);
212  printf(" MAC: ");
213  printf("%s\n", NETSTACK_MAC.name);
214 
215  netstack_init();
216 
217  set_rf_params();
218 
219 #if NETSTACK_CONF_WITH_IPV6
220  memcpy(&uip_lladdr.addr, &linkaddr_node_addr, sizeof(uip_lladdr.addr));
221  queuebuf_init();
222  process_start(&tcpip_process, NULL);
223 #endif /* NETSTACK_CONF_WITH_IPV6 */
224 
225  fade(LEDS_GREEN);
226 
227  process_start(&sensors_process, NULL);
228 
229  autostart_start(autostart_processes);
230 
231  watchdog_start();
232 
233  fade(LEDS_ORANGE);
234 
235  while(1) {
236  uint8_t r;
237  do {
238  r = process_run();
239  watchdog_periodic();
240  } while(r > 0);
241 
242  /* Drop to some low power mode */
243  lpm_drop();
244  }
245 }
