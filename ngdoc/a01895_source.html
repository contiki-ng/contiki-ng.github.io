71 int
72 mac_sequence_is_duplicate(void)
73 {
74  int i;
75  clock_time_t now = clock_time();
76 
77  /*
78  * Check for duplicate packet by comparing the sequence number of the incoming
79  * packet with the last few ones we saw.
80  */
81  for(i = 0; i < MAX_SEQNOS; ++i) {
82  if(linkaddr_cmp(packetbuf_addr(PACKETBUF_ADDR_SENDER),
83  &received_seqnos[i].sender)) {
84  if(packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO) == received_seqnos[i].seqno) {
85 #if SEQNO_MAX_AGE > 0
86  if(now - received_seqnos[i].timestamp <= SEQNO_MAX_AGE) {
87  /* Duplicate packet. */
88  return 1;
89  }
90 #else /* SEQNO_MAX_AGE > 0 */
91  return 1;
92 #endif /* SEQNO_MAX_AGE > 0 */
93  }
94  break;
95  }
96  }
97  return 0;
98 }

100 void
101 mac_sequence_register_seqno(void)
102 {
103  int i, j;
104 
105  /* Locate possible previous sequence number for this address. */
106  for(i = 0; i < MAX_SEQNOS; ++i) {
107  if(linkaddr_cmp(packetbuf_addr(PACKETBUF_ADDR_SENDER),
108  &received_seqnos[i].sender)) {
109  i++;
110  break;
111  }
112  }
113 
114  /* Keep the last sequence number for each address as per 802.15.4e. */
115  for(j = i - 1; j > 0; --j) {
116  memcpy(&received_seqnos[j], &received_seqnos[j - 1], sizeof(struct seqno));
117  }
118  received_seqnos[0].seqno = packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO);
119  received_seqnos[0].timestamp = clock_time();
120  linkaddr_copy(&received_seqnos[0].sender,
121  packetbuf_addr(PACKETBUF_ADDR_SENDER));
122 }
