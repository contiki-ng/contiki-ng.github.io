152 /*---------------------------------------------------------------------------*/
153 void
154 uip_ds6_periodic(void)
155 {
156 
157  /* Periodic processing on unicast addresses */
158  for(locaddr = uip_ds6_if.addr_list;
159  locaddr < uip_ds6_if.addr_list + UIP_DS6_ADDR_NB; locaddr++) {
160  if(locaddr->isused) {
161  if((!locaddr->isinfinite) && (stimer_expired(&locaddr->vlifetime))) {
162  uip_ds6_addr_rm(locaddr);
163 #if UIP_ND6_DEF_MAXDADNS > 0
164  } else if((locaddr->state == ADDR_TENTATIVE)
165  && (locaddr->dadnscount <= uip_ds6_if.maxdadns)
166  && (timer_expired(&locaddr->dadtimer))
167  && (uip_len == 0)) {
168  uip_ds6_dad(locaddr);
169 #endif /* UIP_ND6_DEF_MAXDADNS > 0 */
170  }
171  }
172  }
173 
174  /* Periodic processing on default routers */
175  uip_ds6_defrt_periodic();
176  /* for(locdefrt = uip_ds6_defrt_list;
177  locdefrt < uip_ds6_defrt_list + UIP_DS6_DEFRT_NB; locdefrt++) {
178  if((locdefrt->isused) && (!locdefrt->isinfinite) &&
179  (stimer_expired(&(locdefrt->lifetime)))) {
180  uip_ds6_defrt_rm(locdefrt);
181  }
182  }*/
183 
184 #if !UIP_CONF_ROUTER
185  /* Periodic processing on prefixes */
186  for(locprefix = uip_ds6_prefix_list;
187  locprefix < uip_ds6_prefix_list + UIP_DS6_PREFIX_NB;
188  locprefix++) {
189  if(locprefix->isused && !locprefix->isinfinite
190  && stimer_expired(&(locprefix->vlifetime))) {
191  uip_ds6_prefix_rm(locprefix);
192  }
193  }
194 #endif /* !UIP_CONF_ROUTER */
195 
196 #if UIP_ND6_SEND_NS
197  uip_ds6_neighbor_periodic();
198 #endif /* UIP_ND6_SEND_NS */
199 
200 #if UIP_CONF_ROUTER && UIP_ND6_SEND_RA
201  /* Periodic RA sending */
202  if(stimer_expired(&uip_ds6_timer_ra) && (uip_len == 0)) {
203  uip_ds6_send_ra_periodic();
204  }
205 #endif /* UIP_CONF_ROUTER && UIP_ND6_SEND_RA */
206  etimer_reset(&uip_ds6_timer_periodic);
207  return;
208 }

211 uint8_t
212 uip_ds6_list_loop(uip_ds6_element_t *list, uint8_t size,
213  uint16_t elementsize, uip_ipaddr_t *ipaddr,
214  uint8_t ipaddrlen, uip_ds6_element_t **out_element)
215 {
216  uip_ds6_element_t *element;
217 
218  if(list == NULL || ipaddr == NULL || out_element == NULL) {
219  return NOSPACE;
220  }
221 
222  *out_element = NULL;
223 
224  for(element = list;
225  element <
226  (uip_ds6_element_t *)((uint8_t *)list + (size * elementsize));
227  element = (uip_ds6_element_t *)((uint8_t *)element + elementsize)) {
228  if(element->isused) {
229  if(uip_ipaddr_prefixcmp(&element->ipaddr, ipaddr, ipaddrlen)) {
230  *out_element = element;
231  return FOUND;
232  }
233  } else {
234  *out_element = element;
235  }
236  }
237 
238  return *out_element != NULL ? FREESPACE : NOSPACE;
239 }

513 /*---------------------------------------------------------------------------*/
514 void
515 uip_ds6_select_src(uip_ipaddr_t *src, uip_ipaddr_t *dst)
516 {
517  uint8_t best = 0; /* number of bit in common with best match */
518  uint8_t n = 0;
519  uip_ds6_addr_t *matchaddr = NULL;
520 
521  if(!uip_is_addr_linklocal(dst) && !uip_is_addr_mcast(dst)) {
522  /* find longest match */
523  for(locaddr = uip_ds6_if.addr_list;
524  locaddr < uip_ds6_if.addr_list + UIP_DS6_ADDR_NB; locaddr++) {
525  /* Only preferred global (not link-local) addresses */
526  if(locaddr->isused && locaddr->state == ADDR_PREFERRED &&
527  !uip_is_addr_linklocal(&locaddr->ipaddr)) {
528  n = get_match_length(dst, &locaddr->ipaddr);
529  if(n >= best) {
530  best = n;
531  matchaddr = locaddr;
532  }
533  }
534  }
535 #if UIP_IPV6_MULTICAST
536  } else if(uip_is_addr_mcast_routable(dst)) {
537  matchaddr = uip_ds6_get_global(ADDR_PREFERRED);
538 #endif
539  } else {
540  matchaddr = uip_ds6_get_link_local(ADDR_PREFERRED);
541  }
542 
543  /* use the :: (unspecified address) as source if no match found */
544  if(matchaddr == NULL) {
545  uip_create_unspecified(src);
546  } else {
547  uip_ipaddr_copy(src, &matchaddr->ipaddr);
548  }
549 }

552 void
553 uip_ds6_set_addr_iid(uip_ipaddr_t *ipaddr, uip_lladdr_t *lladdr)
554 {
555  /* We consider only links with IEEE EUI-64 identifier or
556  * IEEE 48-bit MAC addresses */
557 #if (UIP_LLADDR_LEN == 8)
558  memcpy(ipaddr->u8 + 8, lladdr, UIP_LLADDR_LEN);
559  ipaddr->u8[8] ^= 0x02;
560 #elif (UIP_LLADDR_LEN == 6)
561  memcpy(ipaddr->u8 + 8, lladdr, 3);
562  ipaddr->u8[11] = 0xff;
563  ipaddr->u8[12] = 0xfe;
564  memcpy(ipaddr->u8 + 13, (uint8_t *)lladdr + 3, 3);
565  ipaddr->u8[8] ^= 0x02;

584 uint8_t
585 get_match_length(uip_ipaddr_t *src, uip_ipaddr_t *dst)
586 {
587  uint8_t j, k, x_or;
588  uint8_t len = 0;
589 
590  for(j = 0; j < 16; j++) {
591  if(src->u8[j] == dst->u8[j]) {
592  len += 8;
593  } else {
594  x_or = src->u8[j] ^ dst->u8[j];
595  for(k = 0; k < 8; k++) {
596  if((x_or & 0x80) == 0) {
597  len++;
598  x_or <<= 1;
599  } else {
600  break;
601  }
602  }
603  break;
604  }
605  }
606  return len;
607 }
