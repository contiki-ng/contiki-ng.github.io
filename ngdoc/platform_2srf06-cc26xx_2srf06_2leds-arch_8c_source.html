67 /*---------------------------------------------------------------------------*/
68 void
69 leds_arch_set(unsigned char leds)
70 {
71  c = leds;
72 
73  /* Clear everything */
74  ti_lib_gpio_clear_multi_dio(BOARD_LED_ALL);
75 
76  if((leds & LEDS_RED) == LEDS_RED) {
77  ti_lib_gpio_set_dio(BOARD_IOID_LED_1);
78  }
79  if((leds & LEDS_YELLOW) == LEDS_YELLOW) {
80  ti_lib_gpio_set_dio(BOARD_IOID_LED_2);
81  }
82  if((leds & LEDS_GREEN) == LEDS_GREEN) {
83  ti_lib_gpio_set_dio(BOARD_IOID_LED_3);
84  }
85  if((leds & LEDS_ORANGE) == LEDS_ORANGE) {
86  ti_lib_gpio_set_dio(BOARD_IOID_LED_4);
87  }
88 }
