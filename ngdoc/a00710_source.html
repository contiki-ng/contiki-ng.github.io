125 uint_fast8_t
126 rf_core_send_cmd(uint32_t cmd, uint32_t *status)
127 {
128  uint32_t timeout_count = 0;
129  bool interrupts_disabled;
130  bool is_radio_op = false;
131 
132  /*
133  * If cmd is 4-byte aligned, then it's either a radio OP or an immediate
134  * command. Clear the status field if it's a radio OP
135  */
136  if((cmd & 0x03) == 0) {
137  uint32_t cmd_type;
138  cmd_type = ((rfc_command_t *)cmd)->commandNo & RF_CORE_COMMAND_TYPE_MASK;
139  if(cmd_type == RF_CORE_COMMAND_TYPE_IEEE_FG_RADIO_OP ||
140  cmd_type == RF_CORE_COMMAND_TYPE_RADIO_OP) {
141  is_radio_op = true;
142  ((rfc_radioOp_t *)cmd)->status = RF_CORE_RADIO_OP_STATUS_IDLE;
143  }
144  }
145 
146  /*
147  * Make sure ContikiMAC doesn't turn us off from within an interrupt while
148  * we are accessing RF Core registers
149  */
150  interrupts_disabled = ti_lib_int_master_disable();
151 
152  if(!rf_core_is_accessible()) {
153  PRINTF("rf_core_send_cmd: RF was off\n");
154  if(!interrupts_disabled) {
155  ti_lib_int_master_enable();
156  }
157  return RF_CORE_CMD_ERROR;
158  }
159 
160  if(is_radio_op) {
161  uint16_t command_no = ((rfc_radioOp_t *)cmd)->commandNo;
162  if((command_no & RF_CORE_COMMAND_PROTOCOL_MASK) != RF_CORE_COMMAND_PROTOCOL_COMMON &&
163  (command_no & RF_CORE_COMMAND_TYPE_MASK) == RF_CORE_COMMAND_TYPE_RADIO_OP) {
164  last_radio_op = (rfc_radioOp_t *)cmd;
165  }
166  }
167 
168  HWREG(RFC_DBELL_BASE + RFC_DBELL_O_CMDR) = cmd;
169  do {
170  *status = HWREG(RFC_DBELL_BASE + RFC_DBELL_O_CMDSTA);
171  if(++timeout_count > 50000) {
172  PRINTF("rf_core_send_cmd: 0x%08lx Timeout\n", cmd);
173  if(!interrupts_disabled) {
174  ti_lib_int_master_enable();
175  }
176  return RF_CORE_CMD_ERROR;
177  }
178  } while((*status & RF_CORE_CMDSTA_RESULT_MASK) == RF_CORE_CMDSTA_PENDING);
179 
180  if(!interrupts_disabled) {
181  ti_lib_int_master_enable();
182  }
183 
184  /*
185  * If we reach here the command is no longer pending. It is either completed
186  * successfully or with error
187  */
188  return (*status & RF_CORE_CMDSTA_RESULT_MASK) == RF_CORE_CMDSTA_DONE;
189 }
190 /*---------------------------------------------------------------------------*/
191 uint_fast8_t
192 rf_core_wait_cmd_done(void *cmd)
193 {
194  volatile rfc_radioOp_t *command = (rfc_radioOp_t *)cmd;
195  uint32_t timeout_cnt = 0;
196 
197  /*
198  * 0xn4nn=DONE, 0x0400=DONE_OK while all other "DONE" values means done
199  * but with some kind of error (ref. "Common radio operation status codes")
200  */
201  do {
202  if(++timeout_cnt > 500000) {
203  return RF_CORE_CMD_ERROR;
204  }
205  } while((command->status & RF_CORE_RADIO_OP_MASKED_STATUS)
206  != RF_CORE_RADIO_OP_MASKED_STATUS_DONE);
207 
208  return (command->status & RF_CORE_RADIO_OP_MASKED_STATUS)
209  == RF_CORE_RADIO_OP_STATUS_DONE_OK;
210 }

212 static int
213 fs_powerdown(void)
214 {
215  rfc_CMD_FS_POWERDOWN_t cmd;
216  uint32_t cmd_status;
217 
218  rf_core_init_radio_op((rfc_radioOp_t *)&cmd, sizeof(cmd), CMD_FS_POWERDOWN);
219 
220  if(rf_core_send_cmd((uint32_t)&cmd, &cmd_status) != RF_CORE_CMD_OK) {
221  PRINTF("fs_powerdown: CMDSTA=0x%08lx\n", cmd_status);
222  return RF_CORE_CMD_ERROR;
223  }
224 
225  if(rf_core_wait_cmd_done(&cmd) != RF_CORE_CMD_OK) {
226  PRINTF("fs_powerdown: CMDSTA=0x%08lx, status=0x%04x\n",
227  cmd_status, cmd.status);
228  return RF_CORE_CMD_ERROR;
229  }
230 
231  return RF_CORE_CMD_OK;
232 }

234 int
235 rf_core_power_up()
236 {
237  uint32_t cmd_status;
238  bool interrupts_disabled = ti_lib_int_master_disable();
239 
240  ti_lib_int_pend_clear(INT_RFC_CPE_0);
241  ti_lib_int_pend_clear(INT_RFC_CPE_1);
242  ti_lib_int_disable(INT_RFC_CPE_0);
243  ti_lib_int_disable(INT_RFC_CPE_1);
244 
245  /* Enable RF Core power domain */
246  ti_lib_prcm_power_domain_on(PRCM_DOMAIN_RFCORE);
247  while(ti_lib_prcm_power_domain_status(PRCM_DOMAIN_RFCORE)
248  != PRCM_DOMAIN_POWER_ON);
249 
250  ti_lib_prcm_domain_enable(PRCM_DOMAIN_RFCORE);
251  ti_lib_prcm_load_set();
252  while(!ti_lib_prcm_load_get());
253 
254  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIFG) = 0x0;
255  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIEN) = 0x0;
256  ti_lib_int_enable(INT_RFC_CPE_0);
257  ti_lib_int_enable(INT_RFC_CPE_1);
258 
259  if(!interrupts_disabled) {
260  ti_lib_int_master_enable();
261  }
262 
263  rf_switch_power_up();
264 
265  /* Let CPE boot */
266  HWREG(RFC_PWR_NONBUF_BASE + RFC_PWR_O_PWMCLKEN) = RF_CORE_CLOCKS_MASK;
267 
268  /* Turn on additional clocks on boot */
269  HWREG(RFC_DBELL_BASE + RFC_DBELL_O_RFACKIFG) = 0;
270  HWREG(RFC_DBELL_BASE+RFC_DBELL_O_CMDR) =
271  CMDR_DIR_CMD_2BYTE(RF_CMD0,
272  RFC_PWR_PWMCLKEN_MDMRAM | RFC_PWR_PWMCLKEN_RFERAM);
273 
274  /* Send ping (to verify RFCore is ready and alive) */
275  if(rf_core_send_cmd(CMDR_DIR_CMD(CMD_PING), &cmd_status) != RF_CORE_CMD_OK) {
276  PRINTF("rf_core_power_up: CMD_PING fail, CMDSTA=0x%08lx\n", cmd_status);
277  return RF_CORE_CMD_ERROR;
278  }
279 
280  return RF_CORE_CMD_OK;
281 }

283 uint8_t
284 rf_core_start_rat(void)
285 {
286  uint32_t cmd_status;
287  rfc_CMD_SYNC_START_RAT_t cmd_start;
288 
289  /* Start radio timer (RAT) */
290  rf_core_init_radio_op((rfc_radioOp_t *)&cmd_start, sizeof(cmd_start), CMD_SYNC_START_RAT);
291 
292  /* copy the value and send back */
293  cmd_start.rat0 = rat_offset;
294 
295  if(rf_core_send_cmd((uint32_t)&cmd_start, &cmd_status) != RF_CORE_CMD_OK) {
296  PRINTF("rf_core_get_rat_rtc_offset: SYNC_START_RAT fail, CMDSTA=0x%08lx\n",
297  cmd_status);
298  return RF_CORE_CMD_ERROR;
299  }
300 
301  /* Wait until done (?) */
302  if(rf_core_wait_cmd_done(&cmd_start) != RF_CORE_CMD_OK) {
303  PRINTF("rf_core_cmd_ok: SYNC_START_RAT wait, CMDSTA=0x%08lx, status=0x%04x\n",
304  cmd_status, cmd_start.status);
305  return RF_CORE_CMD_ERROR;
306  }
307 
308  return RF_CORE_CMD_OK;
309 }
310 /*---------------------------------------------------------------------------*/
311 uint8_t
312 rf_core_stop_rat(void)
313 {
314  rfc_CMD_SYNC_STOP_RAT_t cmd_stop;
315  uint32_t cmd_status;
316 
317  rf_core_init_radio_op((rfc_radioOp_t *)&cmd_stop, sizeof(cmd_stop), CMD_SYNC_STOP_RAT);
318 
319  int ret = rf_core_send_cmd((uint32_t)&cmd_stop, &cmd_status);
320  if(ret != RF_CORE_CMD_OK) {
321  PRINTF("rf_core_get_rat_rtc_offset: SYNC_STOP_RAT fail, ret %d CMDSTA=0x%08lx\n",
322  ret, cmd_status);
323  return ret;
324  }
325 
326  /* Wait until done */
327  ret = rf_core_wait_cmd_done(&cmd_stop);
328  if(ret != RF_CORE_CMD_OK) {
329  PRINTF("rf_core_cmd_ok: SYNC_STOP_RAT wait, CMDSTA=0x%08lx, status=0x%04x\n",
330  cmd_status, cmd_stop.status);
331  return ret;
332  }
333 
334  if(!rat_offset_known) {
335  /* save the offset, but only if this is the first time */
336  rat_offset_known = true;
337  rat_offset = cmd_stop.rat0;
338  }
339 
340  return RF_CORE_CMD_OK;
341 }

343 void
344 rf_core_power_down()
345 {
346  bool interrupts_disabled = ti_lib_int_master_disable();
347  ti_lib_int_disable(INT_RFC_CPE_0);
348  ti_lib_int_disable(INT_RFC_CPE_1);
349 
350  if(rf_core_is_accessible()) {
351  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIFG) = 0x0;
352  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIEN) = 0x0;
353 
354  /* need to send FS_POWERDOWN or analog components will use power */
355  fs_powerdown();
356  }
357 
358  rf_core_stop_rat();
359 
360  /* Shut down the RFCORE clock domain in the MCU VD */
361  ti_lib_prcm_domain_disable(PRCM_DOMAIN_RFCORE);
362  ti_lib_prcm_load_set();
363  while(!ti_lib_prcm_load_get());
364 
365  /* Turn off RFCORE PD */
366  ti_lib_prcm_power_domain_off(PRCM_DOMAIN_RFCORE);
367  while(ti_lib_prcm_power_domain_status(PRCM_DOMAIN_RFCORE)
368  != PRCM_DOMAIN_POWER_OFF);
369 
370  rf_switch_power_down();
371 
372  ti_lib_int_pend_clear(INT_RFC_CPE_0);
373  ti_lib_int_pend_clear(INT_RFC_CPE_1);
374  ti_lib_int_enable(INT_RFC_CPE_0);
375  ti_lib_int_enable(INT_RFC_CPE_1);
376  if(!interrupts_disabled) {
377  ti_lib_int_master_enable();
378  }
379 }

445 void
446 rf_core_setup_interrupts(bool poll_mode)
447 {
448  bool interrupts_disabled;
449  const uint32_t enabled_irqs = poll_mode ? ENABLED_IRQS_POLL_MODE : ENABLED_IRQS;
450 
451  /* We are already turned on by the caller, so this should not happen */
452  if(!rf_core_is_accessible()) {
453  PRINTF("setup_interrupts: No access\n");
454  return;
455  }
456 
457  /* Disable interrupts */
458  interrupts_disabled = ti_lib_int_master_disable();
459 
460  /* Set all interrupt channels to CPE0 channel, error to CPE1 */
461  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEISL) = ERROR_IRQ;
462 
463  /* Acknowledge configured interrupts */
464  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIEN) = enabled_irqs;
465 
466  /* Clear interrupt flags, active low clear(?) */
467  HWREG(RFC_DBELL_NONBUF_BASE + RFC_DBELL_O_RFCPEIFG) = 0x0;
468 
469  ti_lib_int_pend_clear(INT_RFC_CPE_0);
470  ti_lib_int_pend_clear(INT_RFC_CPE_1);
471  ti_lib_int_enable(INT_RFC_CPE_0);
472  ti_lib_int_enable(INT_RFC_CPE_1);
473 
474  if(!interrupts_disabled) {
475  ti_lib_int_master_enable();
476  }
477 }

586 void
587 cc26xx_rf_cpe0_isr(void)
588 {
589  ENERGEST_ON(ENERGEST_TYPE_IRQ);
590 
591  if(!rf_core_is_accessible()) {
592  printf("RF ISR called but RF not ready... PANIC!!\n");
593  if(rf_core_power_up() != RF_CORE_CMD_OK) {
