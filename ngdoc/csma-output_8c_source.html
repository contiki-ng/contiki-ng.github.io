154 static clock_time_t
155 backoff_period(void)
156 {
157  /* Use the default in IEEE 802.15.4: aUnitBackoffPeriod which is
158  * 20 symbols i.e. 320 usec. That is, 1/3125 second. */
159  return MAX(CLOCK_SECOND / 3125, 1);

162 static int
163 send_one_packet(void *ptr)
164 {
165  int ret;
166  int last_sent_ok = 0;
167 
168  packetbuf_set_addr(PACKETBUF_ADDR_SENDER, &linkaddr_node_addr);
169  packetbuf_set_attr(PACKETBUF_ATTR_MAC_ACK, 1);
170 
171  if(NETSTACK_FRAMER.create() < 0) {
172  /* Failed to allocate space for headers */
173  LOG_ERR("failed to create packet\n");
174  ret = MAC_TX_ERR_FATAL;
175  } else {
176  int is_broadcast;
177  uint8_t dsn;
178  dsn = ((uint8_t *)packetbuf_hdrptr())[2] & 0xff;
179 
180  NETSTACK_RADIO.prepare(packetbuf_hdrptr(), packetbuf_totlen());
181 
182  is_broadcast = packetbuf_holds_broadcast();
183 
184  if(NETSTACK_RADIO.receiving_packet() ||
185  (!is_broadcast && NETSTACK_RADIO.pending_packet())) {
186 
187  /* Currently receiving a packet over air or the radio has
188  already received a packet that needs to be read before
189  sending with auto ack. */
190  ret = MAC_TX_COLLISION;
191  } else {
192 
193  switch(NETSTACK_RADIO.transmit(packetbuf_totlen())) {
194  case RADIO_TX_OK:
195  if(is_broadcast) {
196  ret = MAC_TX_OK;
197  } else {
198  rtimer_clock_t wt;
199 
200  /* Check for ack */
201  wt = RTIMER_NOW();
202  watchdog_periodic();
203  while(RTIMER_CLOCK_LT(RTIMER_NOW(), wt + CSMA_ACK_WAIT_TIME)) {
204 #if CONTIKI_TARGET_COOJA || CONTIKI_TARGET_COOJA_IP64
205  simProcessRunValue = 1;
206  cooja_mt_yield();
207 #endif /* CONTIKI_TARGET_COOJA || CONTIKI_TARGET_COOJA_IP64 */
208  }
209 
210  ret = MAC_TX_NOACK;
211  if(NETSTACK_RADIO.receiving_packet() ||
212  NETSTACK_RADIO.pending_packet() ||
213  NETSTACK_RADIO.channel_clear() == 0) {
214  int len;
215  uint8_t ackbuf[CSMA_ACK_LEN];
216 
217  if(CSMA_AFTER_ACK_DETECTED_WAIT_TIME > 0) {
218  wt = RTIMER_NOW();
219  watchdog_periodic();
220  while(RTIMER_CLOCK_LT(RTIMER_NOW(),
221  wt + CSMA_AFTER_ACK_DETECTED_WAIT_TIME)) {
222 #if CONTIKI_TARGET_COOJA || CONTIKI_TARGET_COOJA_IP64
223  simProcessRunValue = 1;
224  cooja_mt_yield();
225 #endif /* CONTIKI_TARGET_COOJA || CONTIKI_TARGET_COOJA_IP64 */
226  }
227  }
228 
229  if(NETSTACK_RADIO.pending_packet()) {
230  len = NETSTACK_RADIO.read(ackbuf, CSMA_ACK_LEN);
231  if(len == CSMA_ACK_LEN && ackbuf[2] == dsn) {
232  /* Ack received */
233  ret = MAC_TX_OK;
234  } else {
235  /* Not an ack or ack not for us: collision */
236  ret = MAC_TX_COLLISION;
237  }
238  }
239  }
240  }
241  break;
242  case RADIO_TX_COLLISION:
243  ret = MAC_TX_COLLISION;
244  break;
245  default:
246  ret = MAC_TX_ERR;
247  break;
248  }
249  }
250  }
251  if(ret == MAC_TX_OK) {
252  last_sent_ok = 1;
253  }
254 
255  packet_sent(ptr, ret, 1);
256  return last_sent_ok;
257 }

259 static void
260 transmit_from_queue(void *ptr)
261 {
262  struct neighbor_queue *n = ptr;
263  if(n) {
264  struct packet_queue *q = list_head(n->packet_queue);
265  if(q != NULL) {
266  LOG_INFO("preparing packet for ");
267  LOG_INFO_LLADDR(&n->addr);
268  LOG_INFO_(", seqno %u, tx %u, queue %d\n",
269  queuebuf_attr(q->buf, PACKETBUF_ATTR_MAC_SEQNO),
270  n->transmissions, list_length(n->packet_queue));
271  /* Send first packet in the neighbor queue */
272  queuebuf_to_packetbuf(q->buf);
273  send_one_packet(n);
274  }
275  }
276 }

278 static void
279 schedule_transmission(struct neighbor_queue *n)
280 {
281  clock_time_t delay;
282  int backoff_exponent; /* BE in IEEE 802.15.4 */
283 
284  backoff_exponent = MIN(n->collisions, CSMA_MAX_BE);
285 
286  /* Compute max delay as per IEEE 802.15.4: 2^BE-1 backoff periods */
287  delay = ((1 << backoff_exponent) - 1) * backoff_period();
288  if(delay > 0) {
289  /* Pick a time for next transmission */
290  delay = random_rand() % delay;
291  }
292 
293  LOG_DBG("scheduling transmission in %u ticks, NB=%u, BE=%u\n",
294  (unsigned)delay, n->collisions, backoff_exponent);
295  ctimer_set(&n->transmit_timer, delay, transmit_from_queue, n);
296 }

298 static void
299 free_packet(struct neighbor_queue *n, struct packet_queue *p, int status)
300 {
301  if(p != NULL) {
302  /* Remove packet from queue and deallocate */
303  list_remove(n->packet_queue, p);
304 
305  queuebuf_free(p->buf);
306  memb_free(&metadata_memb, p->ptr);
307  memb_free(&packet_memb, p);
308  LOG_DBG("free_queued_packet, queue length %d, free packets %d\n",
309  list_length(n->packet_queue), memb_numfree(&packet_memb));
310  if(list_head(n->packet_queue) != NULL) {
311  /* There is a next packet. We reset current tx information */
312  n->transmissions = 0;
313  n->collisions = CSMA_MIN_BE;
314  /* Schedule next transmissions */
315  schedule_transmission(n);
316  } else {
317  /* This was the last packet in the queue, we free the neighbor */
318  ctimer_stop(&n->transmit_timer);
319  list_remove(neighbor_list, n);
320  memb_free(&neighbor_memb, n);
321  }
322  }
323 }

325 static void
326 tx_done(int status, struct packet_queue *q, struct neighbor_queue *n)
327 {
328  mac_callback_t sent;
329  struct qbuf_metadata *metadata;
330  void *cptr;
331  uint8_t ntx;
332 
333  metadata = (struct qbuf_metadata *)q->ptr;
334  sent = metadata->sent;
335  cptr = metadata->cptr;
336  ntx = n->transmissions;
337 
338  LOG_INFO("packet sent to ");
339  LOG_INFO_LLADDR(&n->addr);
340  LOG_INFO_(", seqno %u, status %u, tx %u, coll %u\n",
341  packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO),
342  status, n->transmissions, n->collisions);
343 
344  free_packet(n, q, status);
345  mac_call_sent_callback(sent, cptr, status, ntx);
346 }

357 static void
358 collision(struct packet_queue *q, struct neighbor_queue *n,
359  int num_transmissions)
360 {
361  struct qbuf_metadata *metadata;
362 
363  metadata = (struct qbuf_metadata *)q->ptr;
364 
365  n->collisions += num_transmissions;
366 
367  if(n->collisions > CSMA_MAX_BACKOFF) {
368  n->collisions = CSMA_MIN_BE;
369  /* Increment to indicate a next retry */
370  n->transmissions++;
371  }
372 
373  if(n->transmissions >= metadata->max_transmissions) {
374  tx_done(MAC_TX_COLLISION, q, n);
375  } else {
376  rexmit(q, n);
377  }
378 }
379 /*---------------------------------------------------------------------------*/
380 static void
381 noack(struct packet_queue *q, struct neighbor_queue *n, int num_transmissions)
382 {
383  struct qbuf_metadata *metadata;
384 
385  metadata = (struct qbuf_metadata *)q->ptr;
386 
387  n->collisions = CSMA_MIN_BE;
388  n->transmissions += num_transmissions;
389 
390  if(n->transmissions >= metadata->max_transmissions) {
391  tx_done(MAC_TX_NOACK, q, n);
392  } else {
393  rexmit(q, n);
394  }
395 }

400  n->collisions = CSMA_MIN_BE;

405 static void
406 packet_sent(void *ptr, int status, int num_transmissions)
407 {
408  struct neighbor_queue *n;
409  struct packet_queue *q;
410 
411  n = ptr;
412  if(n == NULL) {
413  return;
414  }
415 
416  /* Find out what packet this callback refers to */
417  for(q = list_head(n->packet_queue);
418  q != NULL; q = list_item_next(q)) {
419  if(queuebuf_attr(q->buf, PACKETBUF_ATTR_MAC_SEQNO) ==
420  packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO)) {
421  break;
422  }
423  }
424 
425  if(q == NULL) {
426  LOG_WARN("packet sent: seqno %u not found\n",
427  packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO));
428  return;
429  } else if(q->ptr == NULL) {
430  LOG_WARN("packet sent: no metadata\n");
431  return;
432  }
433 
434  LOG_INFO("tx to ");
435  LOG_INFO_LLADDR(&n->addr);
436  LOG_INFO_(", seqno %u, status %u, tx %u, coll %u\n",
437  packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO),
438  status, n->transmissions, n->collisions);
439 
440  switch(status) {
441  case MAC_TX_OK:
442  tx_ok(q, n, num_transmissions);
443  break;
444  case MAC_TX_NOACK:
445  noack(q, n, num_transmissions);
446  break;
447  case MAC_TX_COLLISION:
448  collision(q, n, num_transmissions);
449  break;
450  case MAC_TX_DEFERRED:
451  break;
452  default:
453  tx_done(status, q, n);
454  break;
455  }
456 }

458 void
459 csma_output_packet(mac_callback_t sent, void *ptr)
460 {
461  struct packet_queue *q;
462  struct neighbor_queue *n;
463  static uint8_t initialized = 0;
464  static uint8_t seqno;
465  const linkaddr_t *addr = packetbuf_addr(PACKETBUF_ADDR_RECEIVER);
466 
467  if(!initialized) {
468  initialized = 1;
469  /* Initialize the sequence number to a random value as per 802.15.4. */
470  seqno = random_rand();
471  }
472 
473  if(seqno == 0) {
474  /* PACKETBUF_ATTR_MAC_SEQNO cannot be zero, due to a pecuilarity
475  in framer-802154.c. */
476  seqno++;
477  }
478  packetbuf_set_attr(PACKETBUF_ATTR_MAC_SEQNO, seqno++);
479  packetbuf_set_attr(PACKETBUF_ATTR_FRAME_TYPE, FRAME802154_DATAFRAME);
480 
481  /* Look for the neighbor entry */
482  n = neighbor_queue_from_addr(addr);
483  if(n == NULL) {
484  /* Allocate a new neighbor entry */
485  n = memb_alloc(&neighbor_memb);
486  if(n != NULL) {
487  /* Init neighbor entry */
488  linkaddr_copy(&n->addr, addr);
489  n->transmissions = 0;
490  n->collisions = CSMA_MIN_BE;
491  /* Init packet queue for this neighbor */
492  LIST_STRUCT_INIT(n, packet_queue);
493  /* Add neighbor to the neighbor list */
494  list_add(neighbor_list, n);
495  }
496  }
497 
498  if(n != NULL) {
499  /* Add packet to the neighbor's queue */
500  if(list_length(n->packet_queue) < CSMA_MAX_PACKET_PER_NEIGHBOR) {
501  q = memb_alloc(&packet_memb);
502  if(q != NULL) {
503  q->ptr = memb_alloc(&metadata_memb);
504  if(q->ptr != NULL) {
505  q->buf = queuebuf_new_from_packetbuf();
506  if(q->buf != NULL) {
507  struct qbuf_metadata *metadata = (struct qbuf_metadata *)q->ptr;
508  /* Neighbor and packet successfully allocated */
509  metadata->max_transmissions = CSMA_MAX_MAX_FRAME_RETRIES + 1;
510  metadata->sent = sent;
511  metadata->cptr = ptr;
512  list_add(n->packet_queue, q);
513 
514  LOG_INFO("sending to ");
515  LOG_INFO_LLADDR(addr);
516  LOG_INFO_(", seqno %u, queue length %d, free packets %d\n",
517  packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO),
518  list_length(n->packet_queue), memb_numfree(&packet_memb));
519  /* If q is the first packet in the neighbor's queue, send asap */
520  if(list_head(n->packet_queue) == q) {
521  schedule_transmission(n);
522  }
523  return;
524  }
525  memb_free(&metadata_memb, q->ptr);
526  LOG_WARN("could not allocate queuebuf, dropping packet\n");
527  }
528  memb_free(&packet_memb, q);
529  LOG_WARN("could not allocate queuebuf, dropping packet\n");
530  }
531  /* The packet allocation failed. Remove and free neighbor entry if empty. */
532  if(list_length(n->packet_queue) == 0) {
533  list_remove(neighbor_list, n);
534  memb_free(&neighbor_memb, n);
535  }
536  } else {
537  LOG_WARN("Neighbor queue full\n");
538  }
539  LOG_WARN("could not allocate packet, dropping packet\n");
540  } else {
541  LOG_WARN("could not allocate neighbor, dropping packet\n");
542  }
543  mac_call_sent_callback(sent, ptr, MAC_TX_ERR, 1);
544 }
