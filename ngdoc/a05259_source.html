75 /**
76  * \brief A process that handles adding/removing
77  * BLE IPSP interfaces.
78  */
79 PROCESS_THREAD(ble_iface_observer, ev, data)
80 {
81  static struct etimer led_timer;
82 
83  PROCESS_BEGIN();
84 
85  etimer_set(&led_timer, CLOCK_SECOND/2);
86 
87  while(1) {
88  PROCESS_WAIT_EVENT();
89  if(ev == ble_event_interface_added) {
90  etimer_stop(&led_timer);
91  leds_off(LEDS_1);
92  leds_on(LEDS_2);
93  } else if(ev == ble_event_interface_deleted) {
94  etimer_set(&led_timer, CLOCK_SECOND/2);
95  leds_off(LEDS_2);
96  } else if(ev == PROCESS_EVENT_TIMER && etimer_expired(&led_timer)) {
97  etimer_reset(&led_timer);
98  leds_toggle(LEDS_1);
99  }
100  }
101  PROCESS_END();
102 }
103 #endif

154 
155  process_start(&etimer_process, NULL);
156  ctimer_init();
157 
158 #if ENERGEST_CONF_ON
159  energest_init();
160  ENERGEST_ON(ENERGEST_TYPE_CPU);
161 #endif
162 
163 #ifdef SOFTDEVICE_PRESENT
164  ble_stack_init();
165  ble_advertising_init(DEVICE_NAME);
166 
167 #if NETSTACK_CONF_WITH_IPV6
168  netstack_init();
169  linkaddr_t linkaddr;
170  ble_get_mac(linkaddr.u8);
171  /* Set link layer address */
172  linkaddr_set_node_addr(&linkaddr);
173  /* Set device link layer address in uip stack */
174  memcpy(&uip_lladdr.addr, &linkaddr, sizeof(uip_lladdr.addr));
175  process_start(&ble_iface_observer, NULL);
176  process_start(&tcpip_process, NULL);
177 #endif /* NETSTACK_CONF_WITH_IPV6 */
178 #endif /* SOFTDEVICE_PRESENT */
179 
180  process_start(&sensors_process, NULL);
181  autostart_start(autostart_processes);
182 
183  watchdog_start();
184 
185 #ifdef SOFTDEVICE_PRESENT
186  ble_advertising_start();
187  PRINTF("Advertising name [%s]\n", DEVICE_NAME);
188 #endif
189 
190  while(1) {
191  uint8_t r;
192  do {
193  r = process_run();
194  watchdog_periodic();
195  } while(r > 0);
196 
197  lpm_drop();
198  }
199 }
