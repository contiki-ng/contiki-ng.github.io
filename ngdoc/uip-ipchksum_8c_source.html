50 uint16_t
51 uip_ipchksum(void)
52 {
53  /* Assumes proper alignement of uip_buf. */
54  uint16_t *p = (uint16_t *)&uip_buf[UIP_LLH_LEN];
55  register uint16_t sum;
56 
57  sum = p[0];
58  asmv("add %[p], %[sum]": [sum] "+r" (sum): [p] "m" (p[1]));
59  asmv("addc %[p], %[sum]": [sum] "+r" (sum): [p] "m" (p[2]));
60  asmv("addc %[p], %[sum]": [sum] "+r" (sum): [p] "m" (p[3]));
61  asmv("addc %[p], %[sum]": [sum] "+r" (sum): [p] "m" (p[4]));
62  asmv("addc %[p], %[sum]": [sum] "+r" (sum): [p] "m" (p[5]));
63  asmv("addc %[p], %[sum]": [sum] "+r" (sum): [p] "m" (p[6]));
64  asmv("addc %[p], %[sum]": [sum] "+r" (sum): [p] "m" (p[7]));
65  asmv("addc %[p], %[sum]": [sum] "+r" (sum): [p] "m" (p[8]));
66  asmv("addc %[p], %[sum]": [sum] "+r" (sum): [p] "m" (p[9]));
67 
68  /* Finally, add the remaining carry bit. */
69  asmv("addc #0, %[sum]": [sum] "+r" (sum));
70 
71  /* Return sum in network byte order. */
72  return (sum == 0) ? 0xffff : sum;
73 }
