44 void
45 uip_debug_ipaddr_print(const uip_ipaddr_t *addr)
46 {
47  uint16_t a;
48  unsigned int i;
49  int f;
50 
51  if(addr == NULL) {
52  PRINTA("(NULL IP addr)");
53  return;
54  }
55 
56  if(ip64_addr_is_ipv4_mapped_addr(addr)) {
57  /*
58  * Printing IPv4-mapped addresses is done according to RFC 4291 [1]
59  *
60  * "An alternative form that is sometimes more
61  * convenient when dealing with a mixed environment
62  * of IPv4 and IPv6 nodes is x:x:x:x:x:x:d.d.d.d,
63  * where the 'x's are the hexadecimal values of the
64  * six high-order 16-bit pieces of the address, and
65  * the 'd's are the decimal values of the four
66  * low-order 8-bit pieces of the address (standard
67  * IPv4 representation)."
68  *
69  * [1] https://tools.ietf.org/html/rfc4291#page-4
70  */
71  PRINTA("::FFFF:%u.%u.%u.%u", addr->u8[12], addr->u8[13], addr->u8[14], addr->u8[15]);
72  } else {
73  for(i = 0, f = 0; i < sizeof(uip_ipaddr_t); i += 2) {
74  a = (addr->u8[i] << 8) + addr->u8[i + 1];
75  if(a == 0 && f >= 0) {
76  if(f++ == 0) {
77  PRINTA("::");
78  }
79  } else {
80  if(f > 0) {
81  f = -1;
82  } else if(i > 0) {
83  PRINTA(":");
84  }
85  PRINTA("%x", a);
86  }
87  }
88  }
89 }
