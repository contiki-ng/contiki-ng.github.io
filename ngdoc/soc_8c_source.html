69 uint8_t
70 soc_get_rev(void)
71 {
72  uint8_t rev = (REG(DIECFG2) & DIECFG2_DIE_REV_MSK) >> DIECFG2_DIE_REV_OFS;
73 
74  /* PG1.0 is encoded as 0x00. */
75  if(!(rev >> 4))
76  rev += 0x10;
77  return rev;
78 }
79 /*----------------------------------------------------------------------------*/
80 uint32_t
81 soc_get_sram_size(void)
82 {
83  uint32_t size_code = (REG(DIECFG0) & DIECFG0_SRAM_SIZE_MSK) >>
84  DIECFG0_SRAM_SIZE_OFS;
85 
86  return size_code <= 1 ? (2 - size_code) << 13 : 32 << 10;
87 }
88 /*----------------------------------------------------------------------------*/
89 uint32_t
90 soc_get_features(void)
91 {
92  return REG(DIECFG2) & (DIECFG2_AES_EN | DIECFG2_PKA_EN);
93 }

95 void
96 soc_print_info(void)
97 {
98  uint8_t rev = soc_get_rev();
99  uint32_t features = soc_get_features();
100 
101  LOG_DBG("CC2538: ID: 0x%04lx, rev.: PG%d.%d, Flash: %lu KiB, "
102  "SRAM: %lu KiB, AES/SHA: %u, ECC/RSA: %u\n"
103  "System clock: %lu Hz\n"
104  "I/O clock: %lu Hz\n"
105  "Reset cause: %s\n",
106  rom_util_get_chip_id(),
107  rev >> 4, rev & 0x0f,
108  rom_util_get_flash_size() >> 10,
109  soc_get_sram_size() >> 10,
110  !!(features & SOC_FEATURE_AES_SHA),
111  !!(features & SOC_FEATURE_ECC_RSA),
112  sys_ctrl_get_sys_clock(),
113  sys_ctrl_get_io_clock(),
114  sys_ctrl_get_reset_cause_str());
115 }
