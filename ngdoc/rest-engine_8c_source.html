64 /**
65  * \brief Initializes and starts the REST Engine process
66  *
67  * This function must be called by server processes before any resources are
68  * registered through rest_activate_resource().
69  */
70 void
71 rest_init_engine(void)
72 {
73  /* avoid initializing twice */
74  static uint8_t initialized = 0;
75 
76  if(initialized) {
77  PRINTF("REST engine process already running - double initialization?\n");
78  return;
79  }
80  initialized = 1;
81 
82  list_init(restful_services);
83 
84  REST.set_service_callback(rest_invoke_restful_service);
85 
86  /* Start the RESTful server implementation. */
87  REST.init();
88 
89  /*Start REST engine process */
90  process_start(&rest_engine_process, NULL);
91 }

93 /**
94  * \brief Makes a resource available under the given URI path
95  * \param resource A pointer to a resource implementation
96  * \param path The URI path string for this resource
97  *
98  * The resource implementation must be imported first using the
99  * extern keyword. The build system takes care of compiling every
100  * *.c file in the ./resources/ sub-directory (see example Makefile).
101  */
102 void
103 rest_activate_resource(resource_t *resource, char *path)
104 {
105  resource->url = path;
106  list_add(restful_services, resource);
107 
108  PRINTF("Activating: %s\n", resource->url);
109 
110  /* Only add periodic resources with a periodic_handler and a period > 0. */
111  if(resource->flags & IS_PERIODIC && resource->periodic->periodic_handler
112  && resource->periodic->period) {
113  PRINTF("Periodic resource: %p (%s)\n", resource->periodic,
114  resource->periodic->resource->url);
115  list_add(restful_periodic_services, resource->periodic);
116  if(process_is_running(&rest_engine_process)) {
117  PRINTF("Periodic: Set timer for /%s to %lu\n",
118  resource->url, resource->periodic->period);
119  PROCESS_CONTEXT_BEGIN(&rest_engine_process);
120  etimer_set(&resource->periodic->periodic_timer,
121  resource->periodic->period);
122  PROCESS_CONTEXT_END(&rest_engine_process);
123  }
124  }
125 }

135 int
136 rest_invoke_restful_service(void *request, void *response, uint8_t *buffer,
137  uint16_t buffer_size, int32_t *offset)
138 {
139  uint8_t found = 0;
140  uint8_t allowed = 1;
141 
142  resource_t *resource = NULL;
143  const char *url = NULL;
144  int url_len, res_url_len;
145 
146  url_len = REST.get_url(request, &url);
147  for(resource = (resource_t *)list_head(restful_services);
148  resource; resource = resource->next) {
149 
150  /* if the web service handles that kind of requests and urls matches */
151  res_url_len = strlen(resource->url);
152  if((url_len == res_url_len
153  || (url_len > res_url_len
154  && (resource->flags & HAS_SUB_RESOURCES)
155  && url[res_url_len] == '/'))
156  && strncmp(resource->url, url, res_url_len) == 0) {
157  found = 1;
158  rest_resource_flags_t method = REST.get_method_type(request);
159 
160  PRINTF("/%s, method %u, resource->flags %u\n", resource->url,
161  (uint16_t)method, resource->flags);
162 
163  if((method & METHOD_GET) && resource->get_handler != NULL) {
164  /* call handler function */
165  resource->get_handler(request, response, buffer, buffer_size, offset);
166  } else if((method & METHOD_POST) && resource->post_handler != NULL) {
167  /* call handler function */
168  resource->post_handler(request, response, buffer, buffer_size,
169  offset);
170  } else if((method & METHOD_PUT) && resource->put_handler != NULL) {
171  /* call handler function */
172  resource->put_handler(request, response, buffer, buffer_size, offset);
173  } else if((method & METHOD_DELETE) && resource->delete_handler != NULL) {
174  /* call handler function */
175  resource->delete_handler(request, response, buffer, buffer_size,
176  offset);
177  } else {
178  allowed = 0;
179  REST.set_response_status(response, REST.status.METHOD_NOT_ALLOWED);
180  }
181  break;
182  }
183  }
184  if(!found) {
185  REST.set_response_status(response, REST.status.NOT_FOUND);
186  } else if(allowed) {
187  /* final handler for special flags */
188  if(resource->flags & IS_OBSERVABLE) {
189  REST.subscription_handler(resource, request, response);
190  }
191  }
192  return found & allowed;
193 }

195 PROCESS_THREAD(rest_engine_process, ev, data)
196 {
197  PROCESS_BEGIN();
198 
199  /* pause to let REST server finish adding resources. */
200  PROCESS_PAUSE();
201 
202  /* initialize the PERIODIC_RESOURCE timers, which will be handled by this process. */
203  periodic_resource_t *periodic_resource = NULL;
204 
205  for(periodic_resource =
206  (periodic_resource_t *)list_head(restful_periodic_services);
207  periodic_resource; periodic_resource = periodic_resource->next) {
208  if(periodic_resource->periodic_handler && periodic_resource->period) {
209  PRINTF("Periodic: Set timer for /%s to %lu\n",
210  periodic_resource->resource->url, periodic_resource->period);
211  etimer_set(&periodic_resource->periodic_timer,
212  periodic_resource->period);
213  }
214  }
215  while(1) {
216  PROCESS_WAIT_EVENT();
217 
218  if(ev == PROCESS_EVENT_TIMER) {
219  for(periodic_resource =
220  (periodic_resource_t *)list_head(restful_periodic_services);
221  periodic_resource; periodic_resource = periodic_resource->next) {
222  if(periodic_resource->period
223  && etimer_expired(&periodic_resource->periodic_timer)) {
224 
225  PRINTF("Periodic: etimer expired for /%s (period: %lu)\n",
226  periodic_resource->resource->url, periodic_resource->period);
227 
228  /* Call the periodic_handler function, which was checked during adding to list. */
229  (periodic_resource->periodic_handler)();
230 
231  etimer_reset(&periodic_resource->periodic_timer);
232  }
233  }
234  }
235  }
236 
237  PROCESS_END();
238 }
