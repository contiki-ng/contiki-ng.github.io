68 static void
69 progress_request(coap_request_state_t *state) {
70  coap_message_t *request = state->request;
71  request->mid = coap_get_mid();
72  if((state->transaction =
73  coap_new_transaction(request->mid, state->remote_endpoint))) {
74  state->transaction->callback = coap_request_callback;
75  state->transaction->callback_data = state;
76 
77  if(state->block_num > 0) {
78  coap_set_header_block2(request, state->block_num, 0,
79  COAP_MAX_CHUNK_SIZE);
80  }
81  state->transaction->message_len =
82  coap_serialize_message(request, state->transaction->message);
83 
84  coap_send_transaction(state->transaction);
85  LOG_DBG("Requested #%"PRIu32" (MID %u)\n", state->block_num, request->mid);
86  }
87 }

91 static void
92 coap_request_callback(void *callback_data, coap_message_t *response)
93 {
94  coap_request_state_t *state = (coap_request_state_t *)callback_data;
95  uint32_t res_block1;
96 
97  state->response = response;
98 
99  LOG_DBG("request callback\n");
100 
101  if(!state->response) {
102  LOG_WARN("Server not responding giving up...\n");
103  state->callback(state);
104  return;
105  }
106 
107  /* Got a response */
108  coap_get_header_block2(state->response, &res_block, &more, NULL, NULL);
109  coap_get_header_block1(state->response, &res_block1, NULL, NULL, NULL);
110 
111  LOG_DBG("Received #%lu%s B1:%lu (%u bytes)\n",
112  (unsigned long)res_block, (unsigned)more ? "+" : "",
113  (unsigned long)res_block1,
114  state->response->payload_len);
115 
116  if(res_block == state->block_num) {
117  /* Call the callback function as we have more data */
118  state->callback(state);
119  /* this is only for counting BLOCK2 blocks.*/
120  ++(state->block_num);
121  } else {
122  LOG_WARN("WRONG BLOCK %"PRIu32"/%"PRIu32"\n", res_block, state->block_num);
123  ++block_error;
124  }
125 
126  if(more && block_error < COAP_MAX_ATTEMPTS) {
127  progress_request(state);
128  } else {
129  /* failure - now we give up and notify the callback */
130  state->response = NULL;
131  state->callback(state);
132  }
133 }

137 void
138 coap_send_request(coap_request_state_t *state, coap_endpoint_t *endpoint,
139  coap_message_t *request,
140  void (*callback)(coap_request_state_t *state))
141 {
142  /* can we have these variables shared between multiple requests? */
143  /* ripped from blocking request */
144  more = 0;
145  res_block = 0;
146  block_error = 0;
147 
148  state->block_num = 0;
149  state->response = NULL;
150  state->request = request;
151  state->remote_endpoint = endpoint;
152  state->callback = callback;
153 
154  progress_request(state);
155 }
