109 /**
110  * \brief A data strcuture representing the radio's primary mode of operation
111  *
112  * The CC13xx / CC26xx radio supports up to potentially 3 modes: IEEE, Prop and
113  * BLE. Within Contiki, we assume that the radio is by default in one of IEEE
114  * or Prop in order to support standard 6LoWPAN / .15.4 operation. The BLE
115  * mode interrupts this so called "primary" mode in order to send BLE adv
116  * messages. Once BLE is done advertising, we need to be able to restore the
117  * previous .15.4 mode. Unfortunately, the only way this can be done with
118  * NETSTACK_RADIO API is by fully power-cycling the radio, which is something
119  * we do not want to do.
120  *
121  * Thus, we declare a secondary data structure for primary mode drivers (IEEE
122  * or Prop). We use this data structure to issue "soft off" and "back on"
123  * commands. Soft off in this context means stopping RX (e.g. the respective
124  * IEEE RX operation), but without shutting down the RF core (which is what
125  * NETSTACK_RADIO.off() would have done). We then remember what mode we were
126  * using in order to be able to re-enter RX mode for this mode.
127  *
128  * A NETSTACK_RADIO driver will declare those two functions somewhere within
129  * its module of implementation. During its init() routine, it will notify
130  * the RF core module so that the latter can abort and restore operations.
131  */
132 typedef struct rf_core_primary_mode_s {
133  /**
134  * \brief A pointer to a function used to abort the current radio op
135  */
136  void (*abort)(void);
137 
138  /**
139  * \brief A pointer to a function that will restore the previous radio op
140  * \return RF_CORE_CMD_OK or RF_CORE_CMD_ERROR
141  */
142  uint8_t (*restore)(void);
143 } rf_core_primary_mode_t;

286 /**
287  * \brief Sends a command to the RF core.
288  *
289  * \param cmd The command value or a pointer to a command buffer
290  * \param status A pointer to a variable which will hold the status
291  * \return RF_CORE_CMD_OK or RF_CORE_CMD_ERROR
292  *
293  * This function supports all three types of command (Radio OP, immediate and
294  * direct)
295  *
296  * For immediate and Radio OPs, cmd is a pointer to the data structure
297  * containing the command and its parameters. This data structure must be
298  * 4-byte aligned.
299  *
300  * For direct commands, cmd contains the value of the command alongside its
301  * parameters. This value will be written to CMDSTA verbatim, so the command
302  * ID must be in the 16 high bits, and the 2 LS bits must be set to 01 by the
303  * caller.
304  *
305  * The caller is responsible of allocating and populating cmd for Radio OP and
306  * immediate commands
307  *
308  * The caller is responsible for allocating status
309  *
310  * For immediate commands and radio Ops, this function will set the command's
311  * status field to RF_CORE_RADIO_OP_STATUS_IDLE before sending it to the RF
312  */
313 uint_fast8_t rf_core_send_cmd(uint32_t cmd, uint32_t *status);
