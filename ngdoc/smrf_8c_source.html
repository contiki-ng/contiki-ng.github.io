83 mcast_fwd(void *p)
84 {
85  memcpy(uip_buf, &mcast_buf, mcast_len);
86  uip_len = mcast_len;
87  UIP_IP_BUF->ttl--;
88  tcpip_output(NULL);
89  uip_clear_buf();
90 }

92 static uint8_t
93 in()
94 {
95  rpl_dag_t *d; /* Our DODAG */
96  uip_ipaddr_t *parent_ipaddr; /* Our pref. parent's IPv6 address */
97  const uip_lladdr_t *parent_lladdr; /* Our pref. parent's LL address */
98 
99  /*
100  * Fetch a pointer to the LL address of our preferred parent
101  *
102  * ToDo: This rpl_get_any_dag() call is a dirty replacement of the previous
103  * rpl_get_dag(RPL_DEFAULT_INSTANCE);
104  * so that things can compile with the new RPL code. This needs updated to
105  * read instance ID from the RPL HBHO and use the correct parent accordingly
106  */
107  d = rpl_get_any_dag();
108  if(!d) {
109  UIP_MCAST6_STATS_ADD(mcast_dropped);
110  return UIP_MCAST6_DROP;
111  }
112 
113  /* Retrieve our preferred parent's LL address */
114  parent_ipaddr = rpl_parent_get_ipaddr(d->preferred_parent);
115  parent_lladdr = uip_ds6_nbr_lladdr_from_ipaddr(parent_ipaddr);
116 
117  if(parent_lladdr == NULL) {
118  UIP_MCAST6_STATS_ADD(mcast_dropped);
119  return UIP_MCAST6_DROP;
120  }
121 
122  /*
123  * We accept a datagram if it arrived from our preferred parent, discard
124  * otherwise.
125  */
126  if(memcmp(parent_lladdr, packetbuf_addr(PACKETBUF_ADDR_SENDER),
127  UIP_LLADDR_LEN)) {
128  PRINTF("SMRF: Routable in but SMRF ignored it\n");
129  UIP_MCAST6_STATS_ADD(mcast_dropped);
130  return UIP_MCAST6_DROP;
131  }
132 
133  if(UIP_IP_BUF->ttl <= 1) {
134  UIP_MCAST6_STATS_ADD(mcast_dropped);
135  return UIP_MCAST6_DROP;
136  }
137 
138  UIP_MCAST6_STATS_ADD(mcast_in_all);
139  UIP_MCAST6_STATS_ADD(mcast_in_unique);
140 
141  /* If we have an entry in the mcast routing table, something with
142  * a higher RPL rank (somewhere down the tree) is a group member */
143  if(uip_mcast6_route_lookup(&UIP_IP_BUF->destipaddr)) {
144  /* If we enter here, we will definitely forward */
145  UIP_MCAST6_STATS_ADD(mcast_fwd);
146 
147  /*
148  * Add a delay (D) of at least SMRF_FWD_DELAY() to compensate for how
149  * contikimac handles broadcasts. We can't start our TX before the sender
150  * has finished its own.
151  */
152  fwd_delay = SMRF_FWD_DELAY();
153 
154  /* Finalise D: D = min(SMRF_FWD_DELAY(), SMRF_MIN_FWD_DELAY) */
155 #if SMRF_MIN_FWD_DELAY
156  if(fwd_delay < SMRF_MIN_FWD_DELAY) {
157  fwd_delay = SMRF_MIN_FWD_DELAY;
158  }
159 #endif
160 
161  if(fwd_delay == 0) {
162  /* No delay required, send it, do it now, why wait? */
163  UIP_IP_BUF->ttl--;
164  tcpip_output(NULL);
165  UIP_IP_BUF->ttl++; /* Restore before potential upstack delivery */
166  } else {
167  /* Randomise final delay in [D , D*Spread], step D */
168  fwd_spread = SMRF_INTERVAL_COUNT;
169  if(fwd_spread > SMRF_MAX_SPREAD) {
170  fwd_spread = SMRF_MAX_SPREAD;
171  }
172  if(fwd_spread) {
173  fwd_delay = fwd_delay * (1 + ((random_rand() >> 11) % fwd_spread));
174  }
175 
176  memcpy(&mcast_buf, uip_buf, uip_len);
177  mcast_len = uip_len;
178  ctimer_set(&mcast_periodic, fwd_delay, mcast_fwd, NULL);
179  }
180  PRINTF("SMRF: %u bytes: fwd in %u [%u]\n",
181  uip_len, fwd_delay, fwd_spread);
182  }
183 
184  /* Done with this packet unless we are a member of the mcast group */
185  if(!uip_ds6_is_my_maddr(&UIP_IP_BUF->destipaddr)) {
186  PRINTF("SMRF: Not a group member. No further processing\n");
187  return UIP_MCAST6_DROP;
188  } else {
189  PRINTF("SMRF: Ours. Deliver to upper layers\n");
190  UIP_MCAST6_STATS_ADD(mcast_in_ours);
191  return UIP_MCAST6_ACCEPT;
192  }
193 }
