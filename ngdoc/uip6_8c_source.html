48 /*
49  * uIP is a small implementation of the IP, UDP and TCP protocols (as
50  * well as some basic ICMP stuff). The implementation couples the IP,
51  * UDP, TCP and the application layers very tightly. To keep the size
52  * of the compiled code down, this code frequently uses the goto
53  * statement. While it would be possible to break the uip_process()
54  * function into many smaller functions, this would increase the code
55  * size because of the overhead of parameter passing and the fact that
56  * the optimizer would not be as efficient.
57  *
58  * The principle is that we have a small buffer, called the uip_buf,
59  * in which the device driver puts an incoming packet. The TCP/IP
60  * stack parses the headers in the packet, and calls the
61  * application. If the remote host has sent data to the application,
62  * this data is present in the uip_buf and the application read the
63  * data from there. It is up to the application to put this data into
64  * a byte stream if needed. The application will not be fed with data
65  * that is out of sequence.
66  *
67  * If the application wishes to send data to the peer, it should put
68  * its data into the uip_buf. The uip_appdata pointer points to the
69  * first available byte. The TCP/IP stack will calculate the
70  * checksums, and fill in the necessary header fields and finally send
71  * the packet back to the peer.
72  */

308 static uint16_t
309 chksum(uint16_t sum, const uint8_t *data, uint16_t len)
310 {
311  uint16_t t;
312  const uint8_t *dataptr;
313  const uint8_t *last_byte;
314 
315  dataptr = data;
316  last_byte = data + len - 1;
317 
318  while(dataptr < last_byte) { /* At least two more bytes */
319  t = (dataptr[0] << 8) + dataptr[1];
320  sum += t;
321  if(sum < t) {
322  sum++; /* carry */
323  }
324  dataptr += 2;
325  }
326 
327  if(dataptr == last_byte) {
328  t = (dataptr[0] << 8) + 0;
329  sum += t;
330  if(sum < t) {
331  sum++; /* carry */
332  }
333  }
334 
335  /* Return sum in host byte order. */
336  return sum;
337 }

357 static uint16_t
358 upper_layer_chksum(uint8_t proto)
359 {
360 /* gcc 4.4.0 - 4.6.1 (maybe 4.3...) with -Os on 8 bit CPUS incorrectly compiles:
361  * int bar (int);
362  * int foo (unsigned char a, unsigned char b) {
363  * int len = (a << 8) + b; //len becomes 0xff00&<random>+b
364  * return len + bar (len);
365  * }
366  * upper_layer_len triggers this bug unless it is declared volatile.
367  * See https://sourceforge.net/apps/mantisbt/contiki/view.php?id=3
368  */
369  volatile uint16_t upper_layer_len;
370  uint16_t sum;
371 
372  upper_layer_len = (((uint16_t)(UIP_IP_BUF->len[0]) << 8) + UIP_IP_BUF->len[1] - uip_ext_len);
373 
374  LOG_DBG("Upper layer checksum len: %d from: %d\n", upper_layer_len,
375  UIP_IPH_LEN + UIP_LLH_LEN + uip_ext_len);
376 
377  /* First sum pseudoheader. */
378  /* IP protocol and length fields. This addition cannot carry. */
379  sum = upper_layer_len + proto;
380  /* Sum IP source and destination addresses. */
381  sum = chksum(sum, (uint8_t *)&UIP_IP_BUF->srcipaddr, 2 * sizeof(uip_ipaddr_t));
382 
383  /* Sum TCP header and data. */
384  sum = chksum(sum, &uip_buf[UIP_IPH_LEN + UIP_LLH_LEN + uip_ext_len],
385  upper_layer_len);
386 
387  return (sum == 0) ? 0xffff : uip_htons(sum);
388 }

448 struct uip_conn *
449 uip_connect(const uip_ipaddr_t *ripaddr, uint16_t rport)
450 {
451  register struct uip_conn *conn, *cconn;
452  int c;
453 
454  /* Find an unused local port. */
455  again:
456  ++lastport;
457 
458  if(lastport >= 32000) {
459  lastport = 4096;
460  }
461 
462  /* Check if this port is already in use, and if so try to find
463  another one. */
464  for(c = 0; c < UIP_TCP_CONNS; ++c) {
465  conn = &uip_conns[c];
466  if(conn->tcpstateflags != UIP_CLOSED &&
467  conn->lport == uip_htons(lastport)) {
468  goto again;
469  }
470  }
471 
472  conn = 0;
473  for(c = 0; c < UIP_TCP_CONNS; ++c) {
474  cconn = &uip_conns[c];
475  if(cconn->tcpstateflags == UIP_CLOSED) {
476  conn = cconn;
477  break;
478  }
479  if(cconn->tcpstateflags == UIP_TIME_WAIT) {
480  if(conn == 0 ||
481  cconn->timer > conn->timer) {
482  conn = cconn;
483  }
484  }
485  }
486 
487  if(conn == 0) {
488  return 0;
489  }
490 
491  conn->tcpstateflags = UIP_SYN_SENT;
492 
493  conn->snd_nxt[0] = iss[0];
494  conn->snd_nxt[1] = iss[1];
495  conn->snd_nxt[2] = iss[2];
496  conn->snd_nxt[3] = iss[3];
497 
498  conn->rcv_nxt[0] = 0;
499  conn->rcv_nxt[1] = 0;
500  conn->rcv_nxt[2] = 0;
501  conn->rcv_nxt[3] = 0;
502 
503  conn->initialmss = conn->mss = UIP_TCP_MSS;
504 
505  conn->len = 1; /* TCP length of the SYN is one. */
506  conn->nrtx = 0;
507  conn->timer = 1; /* Send the SYN next time around. */
508  conn->rto = UIP_RTO;
509  conn->sa = 0;
510  conn->sv = 16; /* Initial value of the RTT variance. */
511  conn->lport = uip_htons(lastport);
512  conn->rport = rport;
513  uip_ipaddr_copy(&conn->ripaddr, ripaddr);
514 
515  return conn;
516 }

520 remove_ext_hdr(void)
521 {
522  /* Remove ext header before TCP/UDP processing. */
523  if(uip_ext_len > 0) {
524  LOG_DBG("Cutting ext-header before processing (extlen: %d, uiplen: %d)\n",
525  uip_ext_len, uip_len);
526  if(uip_len < UIP_IPH_LEN + uip_ext_len) {
527  LOG_ERR("ERROR: uip_len too short compared to ext len\n");
528  uip_clear_buf();
529  return;
530  }
531  memmove(((uint8_t *)UIP_TCP_BUF), (uint8_t *)UIP_TCP_BUF + uip_ext_len,
532  uip_len - UIP_IPH_LEN - uip_ext_len);
533 
534  uip_len -= uip_ext_len;
535 
536  /* Update the IP length. */
537  UIP_IP_BUF->len[0] = (uip_len - UIP_IPH_LEN) >> 8;
538  UIP_IP_BUF->len[1] = (uip_len - UIP_IPH_LEN) & 0xff;
539  uip_ext_len = 0;
540  }

644 struct etimer uip_reass_timer; /**< Timer for reassembly */
645 uint8_t uip_reass_on; /* equal to 1 if we are currently reassembling a packet */
646 
647 static uint32_t uip_id; /* For every packet that is to be fragmented, the source
648  node generates an Identification value that is present
649  in all the fragments */
650 #define IP_MF 0x0001
651 
652 static uint16_t
653 uip_reass(void)
654 {
655  uint16_t offset=0;
656  uint16_t len;
657  uint16_t i;
658 
659  /* If ip_reasstmr is zero, no packet is present in the buffer */
660  /* We first write the unfragmentable part of IP header into the reassembly
661  buffer. The reset the other reassembly variables. */
662  if(uip_reass_on == 0) {
663  LOG_INFO("Starting reassembly\n");
664  memcpy(FBUF, UIP_IP_BUF, uip_ext_len + UIP_IPH_LEN);
665  /* temporary in case we do not receive the fragment with offset 0 first */
666  etimer_set(&uip_reass_timer, UIP_REASS_MAXAGE*CLOCK_SECOND);
667  uip_reass_on = 1;
668  uip_reassflags = 0;
669  uip_id = UIP_FRAG_BUF->id;
670  /* Clear the bitmap. */
671  memset(uip_reassbitmap, 0, sizeof(uip_reassbitmap));
672  }
673  /*
674  * Check if the incoming fragment matches the one currently present
675  * in the reasembly buffer. If so, we proceed with copying the fragment
676  * into the buffer.
677  */
678  if(uip_ipaddr_cmp(&FBUF->srcipaddr, &UIP_IP_BUF->srcipaddr) &&
679  uip_ipaddr_cmp(&FBUF->destipaddr, &UIP_IP_BUF->destipaddr) &&
680  UIP_FRAG_BUF->id == uip_id) {
681  len = uip_len - uip_ext_len - UIP_IPH_LEN - UIP_FRAGH_LEN;
682  offset = (uip_ntohs(UIP_FRAG_BUF->offsetresmore) & 0xfff8);
683  /* in byte, originaly in multiple of 8 bytes*/
684  LOG_INFO("len %d\n", len);
685  LOG_INFO("offset %d\n", offset);
686  if(offset == 0){
687  uip_reassflags |= UIP_REASS_FLAG_FIRSTFRAG;
688  /*
689  * The Next Header field of the last header of the Unfragmentable
690  * Part is obtained from the Next Header field of the first
691  * fragment's Fragment header.
692  */
693  *uip_next_hdr = UIP_FRAG_BUF->next;
694  memcpy(FBUF, UIP_IP_BUF, uip_ext_len + UIP_IPH_LEN);
695  LOG_INFO("src ");
696  LOG_INFO_6ADDR(&FBUF->srcipaddr);
697  LOG_INFO_("dest ");
698  LOG_INFO_6ADDR(&FBUF->destipaddr);
699  LOG_INFO_("next %d\n", UIP_IP_BUF->proto);
700 
701  }
702 
703  /* If the offset or the offset + fragment length overflows the
704  reassembly buffer, we discard the entire packet. */
705  if(offset > UIP_REASS_BUFSIZE ||
706  offset + len > UIP_REASS_BUFSIZE) {
707  uip_reass_on = 0;
708  etimer_stop(&uip_reass_timer);
709  return 0;
710  }
711 
712  /* If this fragment has the More Fragments flag set to zero, it is the
713  last fragment*/
714  if((uip_ntohs(UIP_FRAG_BUF->offsetresmore) & IP_MF) == 0) {
715  uip_reassflags |= UIP_REASS_FLAG_LASTFRAG;
716  /*calculate the size of the entire packet*/
717  uip_reasslen = offset + len;
718  LOG_INFO("last fragment reasslen %d\n", uip_reasslen);
719  } else {
720  /* If len is not a multiple of 8 octets and the M flag of that fragment
721  is 1, then that fragment must be discarded and an ICMP Parameter
722  Problem, Code 0, message should be sent to the source of the fragment,
723  pointing to the Payload Length field of the fragment packet. */
724  if(len % 8 != 0){
725  uip_icmp6_error_output(ICMP6_PARAM_PROB, ICMP6_PARAMPROB_HEADER, 4);
726  uip_reassflags |= UIP_REASS_FLAG_ERROR_MSG;
727  /* not clear if we should interrupt reassembly, but it seems so from
728  the conformance tests */
729  uip_reass_on = 0;
730  etimer_stop(&uip_reass_timer);
731  return uip_len;
732  }
733  }
734 
735  /* Copy the fragment into the reassembly buffer, at the right
736  offset. */
737  memcpy((uint8_t *)FBUF + UIP_IPH_LEN + uip_ext_len + offset,
738  (uint8_t *)UIP_FRAG_BUF + UIP_FRAGH_LEN, len);
739 
740  /* Update the bitmap. */
741  if(offset >> 6 == (offset + len) >> 6) {
742  uip_reassbitmap[offset >> 6] |=
743  bitmap_bits[(offset >> 3) & 7] &
744  ~bitmap_bits[((offset + len) >> 3) & 7];
745  } else {
746  /* If the two endpoints are in different bytes, we update the
747  bytes in the endpoints and fill the stuff inbetween with
748  0xff. */
749  uip_reassbitmap[offset >> 6] |= bitmap_bits[(offset >> 3) & 7];
750 
751  for(i = (1 + (offset >> 6)); i < ((offset + len) >> 6); ++i) {
752  uip_reassbitmap[i] = 0xff;
753  }
754  uip_reassbitmap[(offset + len) >> 6] |=
755  ~bitmap_bits[((offset + len) >> 3) & 7];
756  }
757 
758  /* Finally, we check if we have a full packet in the buffer. We do
759  this by checking if we have the last fragment and if all bits
760  in the bitmap are set. */
761 
762  if(uip_reassflags & UIP_REASS_FLAG_LASTFRAG) {
763  /* Check all bytes up to and including all but the last byte in
764  the bitmap. */
765  for(i = 0; i < (uip_reasslen >> 6); ++i) {
766  if(uip_reassbitmap[i] != 0xff) {
767  return 0;
768  }
769  }
770  /* Check the last byte in the bitmap. It should contain just the
771  right amount of bits. */
772  if(uip_reassbitmap[uip_reasslen >> 6] !=
773  (uint8_t)~bitmap_bits[(uip_reasslen >> 3) & 7]) {
774  return 0;
775  }
776 
777  /* If we have come this far, we have a full packet in the
778  buffer, so we copy it to uip_buf. We also reset the timer. */
779  uip_reass_on = 0;
780  etimer_stop(&uip_reass_timer);
781 
782  uip_reasslen += UIP_IPH_LEN + uip_ext_len;
783  memcpy(UIP_IP_BUF, FBUF, uip_reasslen);
784  UIP_IP_BUF->len[0] = ((uip_reasslen - UIP_IPH_LEN) >> 8);
785  UIP_IP_BUF->len[1] = ((uip_reasslen - UIP_IPH_LEN) & 0xff);
786  LOG_INFO("reassembled packet %d (%d)\n", uip_reasslen,
787  (UIP_IP_BUF->len[0] << 8) | UIP_IP_BUF->len[1]);
788 
789  return uip_reasslen;
790 
791  }
792  } else {
793  LOG_WARN("Already reassembling another paquet\n");
794  }
795  return 0;
796 }

843 /**
844  * \brief Process the options in Destination and Hop By Hop extension headers
845  */
846 static uint8_t
847 ext_hdr_options_process(void)
848 {
849  /*
850  * Length field in the extension header: length of the header in units of
851  * 8 bytes, excluding the first 8 bytes
852  * length field in an option : the length of data in the option
853  */
854  uip_ext_opt_offset = 2;
855  while(uip_ext_opt_offset < ((UIP_EXT_BUF->len << 3) + 8)) {
856  switch(UIP_EXT_HDR_OPT_BUF->type) {
857  /*
858  * for now we do not support any options except padding ones
859  * PAD1 does not make sense as the header must be 8bytes aligned,
860  * hence we can only have
861  */
862  case UIP_EXT_HDR_OPT_PAD1:
863  LOG_DBG("Processing PAD1 option\n");
864  uip_ext_opt_offset += 1;
865  break;
866  case UIP_EXT_HDR_OPT_PADN:
867  LOG_DBG("Processing PADN option\n");
868  uip_ext_opt_offset += UIP_EXT_HDR_OPT_PADN_BUF->opt_len + 2;
869  break;
870  case UIP_EXT_HDR_OPT_RPL:
871  /* Fixes situation when a node that is not using RPL
872  * joins a network which does. The received packages will include the
873  * RPL header and processed by the "default" case of the switch
874  * (0x63 & 0xC0 = 0x40). Hence, the packet is discarded as the header
875  * is considered invalid.
876  * Using this fix, the header is ignored, and the next header (if
877  * present) is processed.
878  */
879 #if UIP_CONF_IPV6_RPL
880  LOG_DBG("Processing RPL option\n");
881  if(!rpl_ext_header_hbh_update(uip_ext_opt_offset)) {
882  LOG_ERR("RPL Option Error: Dropping Packet\n");
883  return 1;
884  }
885 #endif /* UIP_CONF_IPV6_RPL */
886  uip_ext_opt_offset += (UIP_EXT_HDR_OPT_BUF->len) + 2;
887  return 0;
888  default:
889  /*
890  * check the two highest order bits of the option
891  * - 00 skip over this option and continue processing the header.
892  * - 01 discard the packet.
893  * - 10 discard the packet and, regardless of whether or not the
894  * packet's Destination Address was a multicast address, send an
895  * ICMP Parameter Problem, Code 2, message to the packet's
896  * Source Address, pointing to the unrecognized Option Type.
897  * - 11 discard the packet and, only if the packet's Destination
898  * Address was not a multicast address, send an ICMP Parameter
899  * Problem, Code 2, message to the packet's Source Address,
900  * pointing to the unrecognized Option Type.
901  */
902  LOG_DBG("MSB %x\n", UIP_EXT_HDR_OPT_BUF->type);
903  switch(UIP_EXT_HDR_OPT_BUF->type & 0xC0) {
904  case 0:
905  break;
906  case 0x40:
907  return 1;
908  case 0xC0:
909  if(uip_is_addr_mcast(&UIP_IP_BUF->destipaddr)) {
910  return 1;
911  }
912  case 0x80:
913  uip_icmp6_error_output(ICMP6_PARAM_PROB, ICMP6_PARAMPROB_OPTION,
914  (uint32_t)UIP_IPH_LEN + uip_ext_len + uip_ext_opt_offset);
915  return 2;
916  }
917  /* in the cases were we did not discard, update ext_opt* */
918  uip_ext_opt_offset += UIP_EXT_HDR_OPT_BUF->len + 2;
919  break;
920  }
921  }
922  return 0;
923 }

926 /*---------------------------------------------------------------------------*/
927 void
928 uip_process(uint8_t flag)
929 {
930 #if UIP_TCP
931  int c;
932  uint16_t tmp16;
933  uint8_t opt;
934  register struct uip_conn *uip_connr = uip_conn;
935 #endif /* UIP_TCP */
936 #if UIP_UDP
937  if(flag == UIP_UDP_SEND_CONN) {
938  goto udp_send;
939  }
940 #endif /* UIP_UDP */
941  uip_sappdata = uip_appdata = &uip_buf[UIP_IPTCPH_LEN + UIP_LLH_LEN];
942 
943  /* Check if we were invoked because of a poll request for a
944  particular connection. */
945  if(flag == UIP_POLL_REQUEST) {
946 #if UIP_TCP
947  if((uip_connr->tcpstateflags & UIP_TS_MASK) == UIP_ESTABLISHED &&
948  !uip_outstanding(uip_connr)) {
949  uip_flags = UIP_POLL;
950  UIP_APPCALL();
951  goto appsend;
952 #if UIP_ACTIVE_OPEN
953  } else if((uip_connr->tcpstateflags & UIP_TS_MASK) == UIP_SYN_SENT) {
954  /* In the SYN_SENT state, we retransmit out SYN. */
955  UIP_TCP_BUF->flags = 0;
956  goto tcp_send_syn;
957 #endif /* UIP_ACTIVE_OPEN */
958  }
959  goto drop;
960 #endif /* UIP_TCP */
961  /* Check if we were invoked because of the perodic timer fireing. */
962  } else if(flag == UIP_TIMER) {
963  /* Reset the length variables. */
964 #if UIP_TCP
965  uip_clear_buf();
966  uip_slen = 0;
967 
968  /* Increase the initial sequence number. */
969  if(++iss[3] == 0) {
970  if(++iss[2] == 0) {
971  if(++iss[1] == 0) {
972  ++iss[0];
973  }
974  }
975  }
976 
977  /*
978  * Check if the connection is in a state in which we simply wait
979  * for the connection to time out. If so, we increase the
980  * connection's timer and remove the connection if it times
981  * out.
982  */
983  if(uip_connr->tcpstateflags == UIP_TIME_WAIT ||
984  uip_connr->tcpstateflags == UIP_FIN_WAIT_2) {
985  ++(uip_connr->timer);
986  if(uip_connr->timer == UIP_TIME_WAIT_TIMEOUT) {
987  uip_connr->tcpstateflags = UIP_CLOSED;
988  }
989  } else if(uip_connr->tcpstateflags != UIP_CLOSED) {
990  /*
991  * If the connection has outstanding data, we increase the
992  * connection's timer and see if it has reached the RTO value
993  * in which case we retransmit.
994  */
995  if(uip_outstanding(uip_connr)) {
996  if(uip_connr->timer-- == 0) {
997  if(uip_connr->nrtx == UIP_MAXRTX ||
998  ((uip_connr->tcpstateflags == UIP_SYN_SENT ||
999  uip_connr->tcpstateflags == UIP_SYN_RCVD) &&
1000  uip_connr->nrtx == UIP_MAXSYNRTX)) {
1001  uip_connr->tcpstateflags = UIP_CLOSED;
1002 
1003  /*
1004  * We call UIP_APPCALL() with uip_flags set to
1005  * UIP_TIMEDOUT to inform the application that the
1006  * connection has timed out.
1007  */
1008  uip_flags = UIP_TIMEDOUT;
1009  UIP_APPCALL();
1010 
1011  /* We also send a reset packet to the remote host. */
1012  UIP_TCP_BUF->flags = TCP_RST | TCP_ACK;
1013  goto tcp_send_nodata;
1014  }
1015 
1016  /* Exponential backoff. */
1017  uip_connr->timer = UIP_RTO << (uip_connr->nrtx > 4?
1018  4:
1019  uip_connr->nrtx);
1020  ++(uip_connr->nrtx);
1021 
1022  /*
1023  * Ok, so we need to retransmit. We do this differently
1024  * depending on which state we are in. In ESTABLISHED, we
1025  * call upon the application so that it may prepare the
1026  * data for the retransmit. In SYN_RCVD, we resend the
1027  * SYNACK that we sent earlier and in LAST_ACK we have to
1028  * retransmit our FINACK.
1029  */
1030  UIP_STAT(++uip_stat.tcp.rexmit);
1031  switch(uip_connr->tcpstateflags & UIP_TS_MASK) {
1032  case UIP_SYN_RCVD:
1033  /* In the SYN_RCVD state, we should retransmit our SYNACK. */
1034  goto tcp_send_synack;
1035 
1036 #if UIP_ACTIVE_OPEN
1037  case UIP_SYN_SENT:
1038  /* In the SYN_SENT state, we retransmit out SYN. */
1039  UIP_TCP_BUF->flags = 0;
1040  goto tcp_send_syn;
1041 #endif /* UIP_ACTIVE_OPEN */
1042 
1043  case UIP_ESTABLISHED:
1044  /*
1045  * In the ESTABLISHED state, we call upon the application
1046  * to do the actual retransmit after which we jump into
1047  * the code for sending out the packet (the apprexmit
1048  * label).
1049  */
1050  uip_flags = UIP_REXMIT;
1051  UIP_APPCALL();
1052  goto apprexmit;
1053 
1054  case UIP_FIN_WAIT_1:
1055  case UIP_CLOSING:
1056  case UIP_LAST_ACK:
1057  /* In all these states we should retransmit a FINACK. */
1058  goto tcp_send_finack;
1059  }
1060  }
1061  } else if((uip_connr->tcpstateflags & UIP_TS_MASK) == UIP_ESTABLISHED) {
1062  /*
1063  * If there was no need for a retransmission, we poll the
1064  * application for new data.
1065  */
1066  uip_flags = UIP_POLL;
1067  UIP_APPCALL();
1068  goto appsend;
1069  }
1070  }
1071  goto drop;
1072 #endif /* UIP_TCP */
1073  }
1074 #if UIP_UDP
1075  if(flag == UIP_UDP_TIMER) {
1076  if(uip_udp_conn->lport != 0) {
1077  uip_conn = NULL;
1078  uip_sappdata = uip_appdata = &uip_buf[UIP_IPUDPH_LEN + UIP_LLH_LEN];
1079  uip_len = uip_slen = 0;
1080  uip_flags = UIP_POLL;
1081  UIP_UDP_APPCALL();
1082  goto udp_send;
1083  } else {
1084  goto drop;
1085  }
1086  }
1087 #endif /* UIP_UDP */
1088 
1089 
1090  /* This is where the input processing starts. */
1091  UIP_STAT(++uip_stat.ip.recv);
1092 
1093  /* Start of IP input header processing code. */
1094 
1095  /* Check validity of the IP header. */
1096  if((UIP_IP_BUF->vtc & 0xf0) != 0x60) { /* IP version and header length. */
1097  UIP_STAT(++uip_stat.ip.drop);
1098  UIP_STAT(++uip_stat.ip.vhlerr);
1099  LOG_ERR("invalid version.");
1100  goto drop;
1101  }
1102  /*
1103  * Check the size of the packet. If the size reported to us in
1104  * uip_len is smaller the size reported in the IP header, we assume
1105  * that the packet has been corrupted in transit. If the size of
1106  * uip_len is larger than the size reported in the IP packet header,
1107  * the packet has been padded and we set uip_len to the correct
1108  * value..
1109  */
1110 
1111  if((UIP_IP_BUF->len[0] << 8) + UIP_IP_BUF->len[1] <= uip_len) {
1112  uip_len = (UIP_IP_BUF->len[0] << 8) + UIP_IP_BUF->len[1] + UIP_IPH_LEN;
1113  /*
1114  * The length reported in the IPv6 header is the
1115  * length of the payload that follows the
1116  * header. However, uIP uses the uip_len variable
1117  * for holding the size of the entire packet,
1118  * including the IP header. For IPv4 this is not a
1119  * problem as the length field in the IPv4 header
1120  * contains the length of the entire packet. But
1121  * for IPv6 we need to add the size of the IPv6
1122  * header (40 bytes).
1123  */
1124  } else {
1125  LOG_ERR("packet shorter than reported in IP header.");
1126  goto drop;
1127  }
1128 
1129  LOG_INFO("packet received from ");
1130  LOG_INFO_6ADDR(&UIP_IP_BUF->srcipaddr);
1131  LOG_INFO_(" to ");
1132  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
1133  LOG_INFO_("\n");
1134 
1135  if(uip_is_addr_mcast(&UIP_IP_BUF->srcipaddr)){
1136  UIP_STAT(++uip_stat.ip.drop);
1137  LOG_ERR("Dropping packet, src is mcast\n");
1138  goto drop;
1139  }
1140 
1141  /* Refresh neighbor state after receiving a unicast message */
1142 #if UIP_ND6_SEND_NS
1143  if(!uip_is_addr_mcast(&UIP_IP_BUF->destipaddr)) {
1144  uip_ds6_nbr_refresh_reachable_state(&UIP_IP_BUF->srcipaddr);
1145  }
1146 #endif /* UIP_ND6_SEND_NS */
1147 
1148 #if UIP_CONF_ROUTER
1149  /*
1150  * Next header field processing. In IPv6, we can have extension headers,
1151  * if present, the Hop-by-Hop Option must be processed before forwarding
1152  * the packet.
1153  */
1154  uip_next_hdr = &UIP_IP_BUF->proto;
1155  uip_ext_len = 0;
1156  uip_ext_bitmap = 0;
1157  if(*uip_next_hdr == UIP_PROTO_HBHO) {
1158 #if UIP_CONF_IPV6_CHECKS
1159  uip_ext_bitmap |= UIP_EXT_HDR_BITMAP_HBHO;
1160 #endif /* UIP_CONF_IPV6_CHECKS */
1161  switch(ext_hdr_options_process()) {
1162  case 0:
1163  /* continue */
1164  uip_next_hdr = &UIP_EXT_BUF->next;
1165  uip_ext_len += (UIP_EXT_BUF->len << 3) + 8;
1166  break;
1167  case 1:
1168  LOG_ERR("Dropping packet after extension header processing\n");
1169  /* silently discard */
1170  goto drop;
1171  case 2:
1172  LOG_ERR("Sending error message after extension header processing\n");
1173  /* send icmp error message (created in ext_hdr_options_process)
1174  * and discard*/
1175  goto send;
1176  }
1177  }
1178 
1179  /*
1180  * Process Packets with a routable multicast destination:
1181  * - We invoke the multicast engine and let it do its thing
1182  * (cache, forward etc).
1183  * - We never execute the datagram forwarding logic in this file here. When
1184  * the engine returns, forwarding has been handled if and as required.
1185  * - Depending on the return value, we either discard or deliver up the stack
1186  *
1187  * All multicast engines must hook in here. After this function returns, we
1188  * expect UIP_BUF to be unmodified
1189  */
1190 #if UIP_IPV6_MULTICAST
1191  if(uip_is_addr_mcast_routable(&UIP_IP_BUF->destipaddr)) {
1192  if(UIP_MCAST6.in() == UIP_MCAST6_ACCEPT) {
1193  /* Deliver up the stack */
1194  goto process;
1195  } else {
1196  /* Don't deliver up the stack */
1197  goto drop;
1198  }
1199  }
1200 #endif /* UIP_IPV6_MULTICAST */
1201 
1202  /* TBD Some Parameter problem messages */
1203  if(!uip_ds6_is_my_addr(&UIP_IP_BUF->destipaddr) &&
1204  !uip_ds6_is_my_maddr(&UIP_IP_BUF->destipaddr)) {
1205  if(!uip_is_addr_mcast(&UIP_IP_BUF->destipaddr) &&
1206  !uip_is_addr_linklocal(&UIP_IP_BUF->destipaddr) &&
1207  !uip_is_addr_linklocal(&UIP_IP_BUF->srcipaddr) &&
1208  !uip_is_addr_unspecified(&UIP_IP_BUF->srcipaddr) &&
1209  !uip_is_addr_loopback(&UIP_IP_BUF->destipaddr)) {
1210 
1211 
1212  /* Check MTU */
1213  if(uip_len > UIP_LINK_MTU) {
1214  uip_icmp6_error_output(ICMP6_PACKET_TOO_BIG, 0, UIP_LINK_MTU);
1215  UIP_STAT(++uip_stat.ip.drop);
1216  goto send;
1217  }
1218  /* Check Hop Limit */
1219  if(UIP_IP_BUF->ttl <= 1) {
1220  uip_icmp6_error_output(ICMP6_TIME_EXCEEDED,
1221  ICMP6_TIME_EXCEED_TRANSIT, 0);
1222  UIP_STAT(++uip_stat.ip.drop);
1223  goto send;
1224  }
1225 
1226  UIP_IP_BUF->ttl = UIP_IP_BUF->ttl - 1;
1227  LOG_INFO("Forwarding packet to ");
1228  LOG_INFO_6ADDR(&UIP_IP_BUF->destipaddr);
1229  LOG_INFO_("\n");
1230  UIP_STAT(++uip_stat.ip.forwarded);
1231  goto send;
1232  } else {
1233  if((uip_is_addr_linklocal(&UIP_IP_BUF->srcipaddr)) &&
1234  (!uip_is_addr_unspecified(&UIP_IP_BUF->srcipaddr)) &&
1235  (!uip_is_addr_loopback(&UIP_IP_BUF->destipaddr)) &&
1236  (!uip_is_addr_mcast(&UIP_IP_BUF->destipaddr)) &&
1237  (!uip_ds6_is_addr_onlink((&UIP_IP_BUF->destipaddr)))) {
1238  LOG_ERR("LL source address with off link destination, dropping\n");
1239  uip_icmp6_error_output(ICMP6_DST_UNREACH,
1240  ICMP6_DST_UNREACH_NOTNEIGHBOR, 0);
1241  goto send;
1242  }
1243  LOG_ERR("Dropping packet, not for me and link local or multicast\n");
1244  UIP_STAT(++uip_stat.ip.drop);
1245  goto drop;
1246  }
1247  }
1248 #else /* UIP_CONF_ROUTER */
1249  if(!uip_ds6_is_my_addr(&UIP_IP_BUF->destipaddr) &&
1250  !uip_ds6_is_my_maddr(&UIP_IP_BUF->destipaddr) &&
1251  !uip_is_addr_mcast(&UIP_IP_BUF->destipaddr)) {
1252  LOG_ERR("Dropping packet, not for me\n");
1253  UIP_STAT(++uip_stat.ip.drop);
1254  goto drop;
1255  }
1256 
1257  /*
1258  * Next header field processing. In IPv6, we can have extension headers,
1259  * they are processed here
1260  */
1261  uip_next_hdr = &UIP_IP_BUF->proto;
1262  uip_ext_len = 0;
1263  uip_ext_bitmap = 0;
1264 #endif /* UIP_CONF_ROUTER */
1265 
1266 #if UIP_IPV6_MULTICAST
1267  process:
1268 #endif
1269 
1270  while(1) {
1271  switch(*uip_next_hdr){
1272 #if UIP_TCP
1273  case UIP_PROTO_TCP:
1274  /* TCP, for both IPv4 and IPv6 */
1275  goto tcp_input;
1276 #endif /* UIP_TCP */
1277 #if UIP_UDP
1278  case UIP_PROTO_UDP:
1279  /* UDP, for both IPv4 and IPv6 */
1280  goto udp_input;
1281 #endif /* UIP_UDP */
1282  case UIP_PROTO_ICMP6:
1283  /* ICMPv6 */
1284  goto icmp6_input;
1285  case UIP_PROTO_HBHO:
1286  LOG_DBG("Processing hbh header\n");
1287  /* Hop by hop option header */
1288 #if UIP_CONF_IPV6_CHECKS
1289  /* Hop by hop option header. If we saw one HBH already, drop */
1290  if(uip_ext_bitmap & UIP_EXT_HDR_BITMAP_HBHO) {
1291  goto bad_hdr;
1292  } else {
1293  uip_ext_bitmap |= UIP_EXT_HDR_BITMAP_HBHO;
1294  }
1295 #endif /*UIP_CONF_IPV6_CHECKS*/
1296  switch(ext_hdr_options_process()) {
1297  case 0:
1298  /*continue*/
1299  uip_next_hdr = &UIP_EXT_BUF->next;
1300  uip_ext_len += (UIP_EXT_BUF->len << 3) + 8;
1301  break;
1302  case 1:
1303  /*silently discard*/
1304  goto drop;
1305  case 2:
1306  /* send icmp error message (created in ext_hdr_options_process)
1307  * and discard*/
1308  goto send;
1309  }
1310  break;
1311  case UIP_PROTO_DESTO:
1312 #if UIP_CONF_IPV6_CHECKS
1313  /* Destination option header. if we saw two already, drop */
1314  LOG_DBG("Processing desto header\n");
1315  if(uip_ext_bitmap & UIP_EXT_HDR_BITMAP_DESTO1) {
1316  if(uip_ext_bitmap & UIP_EXT_HDR_BITMAP_DESTO2) {
1317  goto bad_hdr;
1318  } else{
1319  uip_ext_bitmap |= UIP_EXT_HDR_BITMAP_DESTO2;
1320  }
1321  } else {
1322  uip_ext_bitmap |= UIP_EXT_HDR_BITMAP_DESTO1;
1323  }
1324 #endif /*UIP_CONF_IPV6_CHECKS*/
1325  switch(ext_hdr_options_process()) {
1326  case 0:
1327  /*continue*/
1328  uip_next_hdr = &UIP_EXT_BUF->next;
1329  uip_ext_len += (UIP_EXT_BUF->len << 3) + 8;
1330  break;
1331  case 1:
1332  /*silently discard*/
1333  goto drop;
1334  case 2:
1335  /* send icmp error message (created in ext_hdr_options_process)
1336  * and discard*/
1337  goto send;
1338  }
1339  break;
1340  case UIP_PROTO_ROUTING:
1341 #if UIP_CONF_IPV6_CHECKS
1342  /* Routing header. If we saw one already, drop */
1343  if(uip_ext_bitmap & UIP_EXT_HDR_BITMAP_ROUTING) {
1344  goto bad_hdr;
1345  } else {
1346  uip_ext_bitmap |= UIP_EXT_HDR_BITMAP_ROUTING;
1347  }
1348 #endif /*UIP_CONF_IPV6_CHECKS*/
1349  /*
1350  * Routing Header length field is in units of 8 bytes, excluding
1351  * As per RFC2460 section 4.4, if routing type is unrecognized:
1352  * if segments left = 0, ignore the header
1353  * if segments left > 0, discard packet and send icmp error pointing
1354  * to the routing type
1355  */
1356 
1357  LOG_DBG("Processing Routing header\n");
1358  if(UIP_ROUTING_BUF->seg_left > 0) {
1359 #if UIP_CONF_IPV6_RPL && RPL_WITH_NON_STORING
1360  if(rpl_ext_header_srh_update()) {
1361  goto send; /* Proceed to forwarding */
1362  }
1363 #endif /* UIP_CONF_IPV6_RPL && RPL_WITH_NON_STORING */
1364  uip_icmp6_error_output(ICMP6_PARAM_PROB, ICMP6_PARAMPROB_HEADER, UIP_IPH_LEN + uip_ext_len + 2);
1365  UIP_STAT(++uip_stat.ip.drop);
1366  LOG_ERR("unrecognized routing type");
1367  goto send;
1368  }
1369  uip_next_hdr = &UIP_EXT_BUF->next;
1370  uip_ext_len += (UIP_EXT_BUF->len << 3) + 8;
1371  break;
1372  case UIP_PROTO_FRAG:
1373  /* Fragmentation header:call the reassembly function, then leave */
1374 #if UIP_CONF_IPV6_REASSEMBLY
1375  LOG_INFO("Processing fragmentation header\n");
1376  uip_len = uip_reass();
1377  if(uip_len == 0) {
1378  goto drop;
1379  }
1380  if(uip_reassflags & UIP_REASS_FLAG_ERROR_MSG){
1381  /* we are not done with reassembly, this is an error message */
1382  goto send;
1383  }
1384  /*packet is reassembled, reset the next hdr to the beginning
1385  of the IP header and restart the parsing of the reassembled pkt*/
1386  LOG_INFO("Processing reassembled packet\n");
1387  uip_ext_len = 0;
1388  uip_ext_bitmap = 0;
1389  uip_next_hdr = &UIP_IP_BUF->proto;
1390  break;
1391 #else /* UIP_CONF_IPV6_REASSEMBLY */
1392  UIP_STAT(++uip_stat.ip.drop);
1393  UIP_STAT(++uip_stat.ip.fragerr);
1394  LOG_ERR("fragment dropped.");
1395  goto drop;
1396 #endif /* UIP_CONF_IPV6_REASSEMBLY */
1397  case UIP_PROTO_NONE:
1398  goto drop;
1399  default:
1400  goto bad_hdr;
1401  }
1402  }
1403  bad_hdr:
1404  /*
1405  * RFC 2460 send error message parameterr problem, code unrecognized
1406  * next header, pointing to the next header field
1407  */
1408  uip_icmp6_error_output(ICMP6_PARAM_PROB, ICMP6_PARAMPROB_NEXTHEADER, (uint32_t)(uip_next_hdr - (uint8_t *)UIP_IP_BUF));
1409  UIP_STAT(++uip_stat.ip.drop);
1410  UIP_STAT(++uip_stat.ip.protoerr);
1411  LOG_ERR("unrecognized header");
1412  goto send;
1413  /* End of headers processing */
1414 
1415  icmp6_input:
1416  /* This is IPv6 ICMPv6 processing code. */
1417  LOG_INFO("icmp6: input length %d type: %d \n", uip_len, UIP_ICMP_BUF->type);
1418 
1419 #if UIP_CONF_IPV6_CHECKS
1420  /* Compute and check the ICMP header checksum */
1421  if(uip_icmp6chksum() != 0xffff) {
1422  UIP_STAT(++uip_stat.icmp.drop);
1423  UIP_STAT(++uip_stat.icmp.chkerr);
1424  LOG_ERR("icmpv6: bad checksum\n");
1425  goto drop;
1426  }
1427 #endif /*UIP_CONF_IPV6_CHECKS*/
1428 
1429  UIP_STAT(++uip_stat.icmp.recv);
1430  /*
1431  * Here we process incoming ICMPv6 packets
1432  * For echo request, we send echo reply
1433  * For ND pkts, we call the appropriate function in uip-nd6.c
1434  * We do not treat Error messages for now
1435  * If no pkt is to be sent as an answer to the incoming one, we
1436  * "goto drop". Else we just break; then at the after the "switch"
1437  * we "goto send"
1438  */
1439 #if UIP_CONF_ICMP6
1440  UIP_ICMP6_APPCALL(UIP_ICMP_BUF->type);
1441 #endif /*UIP_CONF_ICMP6*/
1442 
1443  /*
1444  * Search generic input handlers.
1445  * The handler is in charge of setting uip_len to 0
1446  */
1447  if(uip_icmp6_input(UIP_ICMP_BUF->type,
1448  UIP_ICMP_BUF->icode) == UIP_ICMP6_INPUT_ERROR) {
1449  LOG_ERR("Unknown ICMPv6 message type/code %d\n", UIP_ICMP_BUF->type);
1450  UIP_STAT(++uip_stat.icmp.drop);
1451  UIP_STAT(++uip_stat.icmp.typeerr);
1452  uip_clear_buf();
1453  }
1454 
1455  if(uip_len > 0) {
1456  goto send;
1457  } else {
1458  goto drop;
1459  }
1460  /* End of IPv6 ICMP processing. */
1461 
1462 
1463 #if UIP_UDP
1464  /* UDP input processing. */
1465  udp_input:
1466 
1467  remove_ext_hdr();
1468  UIP_IP_BUF->proto = UIP_PROTO_UDP;
1469 
1470  LOG_INFO("Receiving UDP packet\n");
1471 
1472  /* UDP processing is really just a hack. We don't do anything to the
1473  UDP/IP headers, but let the UDP application do all the hard
1474  work. If the application sets uip_slen, it has a packet to
1475  send. */
1476 #if UIP_UDP_CHECKSUMS
1477  /* XXX hack: UDP/IPv6 receivers should drop packets with UDP
1478  checksum 0. Here, we explicitly receive UDP packets with checksum
1479  0. This is to be able to debug code that for one reason or
1480  another miscomputes UDP checksums. The reception of zero UDP
1481  checksums should be turned into a configration option. */
1482  if(UIP_UDP_BUF->udpchksum != 0 && uip_udpchksum() != 0xffff) {
1483  UIP_STAT(++uip_stat.udp.drop);
1484  UIP_STAT(++uip_stat.udp.chkerr);
1485  LOG_ERR("udp: bad checksum 0x%04x 0x%04x\n", UIP_UDP_BUF->udpchksum,
1486  uip_udpchksum());
1487  goto drop;
1488  }
1489 #endif /* UIP_UDP_CHECKSUMS */
1490 
1491  /* Make sure that the UDP destination port number is not zero. */
1492  if(UIP_UDP_BUF->destport == 0) {
1493  LOG_ERR("udp: zero port.\n");
1494  goto drop;
1495  }
1496 
1497  /* Demultiplex this UDP packet between the UDP "connections". */
1498  for(uip_udp_conn = &uip_udp_conns[0];
1499  uip_udp_conn < &uip_udp_conns[UIP_UDP_CONNS];
1500  ++uip_udp_conn) {
1501  /* If the local UDP port is non-zero, the connection is considered
1502  to be used. If so, the local port number is checked against the
1503  destination port number in the received packet. If the two port
1504  numbers match, the remote port number is checked if the
1505  connection is bound to a remote port. Finally, if the
1506  connection is bound to a remote IP address, the source IP
1507  address of the packet is checked. */
1508  if(uip_udp_conn->lport != 0 &&
1509  UIP_UDP_BUF->destport == uip_udp_conn->lport &&
1510  (uip_udp_conn->rport == 0 ||
1511  UIP_UDP_BUF->srcport == uip_udp_conn->rport) &&
1512  (uip_is_addr_unspecified(&uip_udp_conn->ripaddr) ||
1513  uip_ipaddr_cmp(&UIP_IP_BUF->srcipaddr, &uip_udp_conn->ripaddr))) {
1514  goto udp_found;
1515  }
1516  }
1517  LOG_ERR("udp: no matching connection found\n");
1518  UIP_STAT(++uip_stat.udp.drop);
1519 
1520  uip_icmp6_error_output(ICMP6_DST_UNREACH, ICMP6_DST_UNREACH_NOPORT, 0);
1521  goto send;
1522 
1523  udp_found:
1524  LOG_DBG("In udp_found\n");
1525  UIP_STAT(++uip_stat.udp.recv);
1526 
1527  uip_len = uip_len - UIP_IPUDPH_LEN;
1528  uip_appdata = &uip_buf[UIP_IPUDPH_LEN + UIP_LLH_LEN];
1529  uip_conn = NULL;
1530  uip_flags = UIP_NEWDATA;
1531  uip_sappdata = uip_appdata = &uip_buf[UIP_IPUDPH_LEN + UIP_LLH_LEN];
1532  uip_slen = 0;
1533  UIP_UDP_APPCALL();
1534 
1535  udp_send:
1536  LOG_DBG("In udp_send\n");
1537 
1538  if(uip_slen == 0) {
1539  goto drop;
1540  }
1541  uip_len = uip_slen + UIP_IPUDPH_LEN;
1542 
1543  /* For IPv6, the IP length field does not include the IPv6 IP header
1544  length. */
1545  UIP_IP_BUF->len[0] = ((uip_len - UIP_IPH_LEN) >> 8);
1546  UIP_IP_BUF->len[1] = ((uip_len - UIP_IPH_LEN) & 0xff);
1547 
1548  UIP_IP_BUF->ttl = uip_udp_conn->ttl;
1549  UIP_IP_BUF->proto = UIP_PROTO_UDP;
1550 
1551  UIP_UDP_BUF->udplen = UIP_HTONS(uip_slen + UIP_UDPH_LEN);
1552  UIP_UDP_BUF->udpchksum = 0;
1553 
1554  UIP_UDP_BUF->srcport = uip_udp_conn->lport;
1555  UIP_UDP_BUF->destport = uip_udp_conn->rport;
1556 
1557  uip_ipaddr_copy(&UIP_IP_BUF->destipaddr, &uip_udp_conn->ripaddr);
1558  uip_ds6_select_src(&UIP_IP_BUF->srcipaddr, &UIP_IP_BUF->destipaddr);
1559 
1560  uip_appdata = &uip_buf[UIP_LLH_LEN + UIP_IPTCPH_LEN];
1561 
1562 #if UIP_UDP_CHECKSUMS
1563  /* Calculate UDP checksum. */
1564  UIP_UDP_BUF->udpchksum = ~(uip_udpchksum());
1565  if(UIP_UDP_BUF->udpchksum == 0) {
1566  UIP_UDP_BUF->udpchksum = 0xffff;
1567  }
1568 #endif /* UIP_UDP_CHECKSUMS */
1569 
1570  UIP_STAT(++uip_stat.udp.sent);
1571  goto ip_send_nolen;
1572 #endif /* UIP_UDP */
1573 
1574 #if UIP_TCP
1575  /* TCP input processing. */
1576  tcp_input:
1577 
1578  remove_ext_hdr();
1579  UIP_IP_BUF->proto = UIP_PROTO_TCP;
1580 
1581  UIP_STAT(++uip_stat.tcp.recv);
1582  LOG_INFO("Receiving TCP packet\n");
1583  /* Start of TCP input header processing code. */
1584 
1585  if(uip_tcpchksum() != 0xffff) { /* Compute and check the TCP
1586  checksum. */
1587  UIP_STAT(++uip_stat.tcp.drop);
1588  UIP_STAT(++uip_stat.tcp.chkerr);
1589  LOG_ERR("tcp: bad checksum 0x%04x 0x%04x\n", UIP_TCP_BUF->tcpchksum,
1590  uip_tcpchksum());
1591  goto drop;
1592  }
1593 
1594  /* Make sure that the TCP port number is not zero. */
1595  if(UIP_TCP_BUF->destport == 0 || UIP_TCP_BUF->srcport == 0) {
1596  LOG_ERR("tcp: zero port.");
1597  goto drop;
1598  }
1599 
1600  /* Demultiplex this segment. */
1601  /* First check any active connections. */
1602  for(uip_connr = &uip_conns[0]; uip_connr <= &uip_conns[UIP_TCP_CONNS - 1];
1603  ++uip_connr) {
1604  if(uip_connr->tcpstateflags != UIP_CLOSED &&
1605  UIP_TCP_BUF->destport == uip_connr->lport &&
1606  UIP_TCP_BUF->srcport == uip_connr->rport &&
1607  uip_ipaddr_cmp(&UIP_IP_BUF->srcipaddr, &uip_connr->ripaddr)) {
1608  goto found;
1609  }
1610  }
1611 
1612  /* If we didn't find and active connection that expected the packet,
1613  either this packet is an old duplicate, or this is a SYN packet
1614  destined for a connection in LISTEN. If the SYN flag isn't set,
1615  it is an old packet and we send a RST. */
1616  if((UIP_TCP_BUF->flags & TCP_CTL) != TCP_SYN) {
1617  goto reset;
1618  }
1619 
1620  tmp16 = UIP_TCP_BUF->destport;
1621  /* Next, check listening connections. */
1622  for(c = 0; c < UIP_LISTENPORTS; ++c) {
1623  if(tmp16 == uip_listenports[c]) {
1624  goto found_listen;
1625  }
1626  }
1627 
1628  /* No matching connection found, so we send a RST packet. */
1629  UIP_STAT(++uip_stat.tcp.synrst);
1630 
1631  reset:
1632  LOG_WARN("In reset\n");
1633  /* We do not send resets in response to resets. */
1634  if(UIP_TCP_BUF->flags & TCP_RST) {
1635  goto drop;
1636  }
1637 
1638  UIP_STAT(++uip_stat.tcp.rst);
1639 
1640  UIP_TCP_BUF->flags = TCP_RST | TCP_ACK;
1641  uip_len = UIP_IPTCPH_LEN;
1642  UIP_TCP_BUF->tcpoffset = 5 << 4;
1643 
1644  /* Flip the seqno and ackno fields in the TCP header. */
1645  c = UIP_TCP_BUF->seqno[3];
1646  UIP_TCP_BUF->seqno[3] = UIP_TCP_BUF->ackno[3];
1647  UIP_TCP_BUF->ackno[3] = c;
1648 
1649  c = UIP_TCP_BUF->seqno[2];
1650  UIP_TCP_BUF->seqno[2] = UIP_TCP_BUF->ackno[2];
1651  UIP_TCP_BUF->ackno[2] = c;
1652 
1653  c = UIP_TCP_BUF->seqno[1];
1654  UIP_TCP_BUF->seqno[1] = UIP_TCP_BUF->ackno[1];
1655  UIP_TCP_BUF->ackno[1] = c;
1656 
1657  c = UIP_TCP_BUF->seqno[0];
1658  UIP_TCP_BUF->seqno[0] = UIP_TCP_BUF->ackno[0];
1659  UIP_TCP_BUF->ackno[0] = c;
1660 
1661  /* We also have to increase the sequence number we are
1662  acknowledging. If the least significant byte overflowed, we need
1663  to propagate the carry to the other bytes as well. */
1664  if(++UIP_TCP_BUF->ackno[3] == 0) {
1665  if(++UIP_TCP_BUF->ackno[2] == 0) {
1666  if(++UIP_TCP_BUF->ackno[1] == 0) {
1667  ++UIP_TCP_BUF->ackno[0];
1668  }
1669  }
1670  }
1671 
1672  /* Swap port numbers. */
1673  tmp16 = UIP_TCP_BUF->srcport;
1674  UIP_TCP_BUF->srcport = UIP_TCP_BUF->destport;
1675  UIP_TCP_BUF->destport = tmp16;
1676 
1677  /* Swap IP addresses. */
1678  uip_ipaddr_copy(&UIP_IP_BUF->destipaddr, &UIP_IP_BUF->srcipaddr);
1679  uip_ds6_select_src(&UIP_IP_BUF->srcipaddr, &UIP_IP_BUF->destipaddr);
1680  /* And send out the RST packet! */
1681  goto tcp_send_noconn;
1682 
1683  /* This label will be jumped to if we matched the incoming packet
1684  with a connection in LISTEN. In that case, we should create a new
1685  connection and send a SYNACK in return. */
1686  found_listen:
1687  LOG_DBG("In found listen\n");
1688  /* First we check if there are any connections avaliable. Unused
1689  connections are kept in the same table as used connections, but
1690  unused ones have the tcpstate set to CLOSED. Also, connections in
1691  TIME_WAIT are kept track of and we'll use the oldest one if no
1692  CLOSED connections are found. Thanks to Eddie C. Dost for a very
1693  nice algorithm for the TIME_WAIT search. */
1694  uip_connr = 0;
1695  for(c = 0; c < UIP_TCP_CONNS; ++c) {
1696  if(uip_conns[c].tcpstateflags == UIP_CLOSED) {
1697  uip_connr = &uip_conns[c];
1698  break;
1699  }
1700  if(uip_conns[c].tcpstateflags == UIP_TIME_WAIT) {
1701  if(uip_connr == 0 ||
1702  uip_conns[c].timer > uip_connr->timer) {
1703  uip_connr = &uip_conns[c];
1704  }

1755  (uint16_t)uip_buf[UIP_IPTCPH_LEN + UIP_LLH_LEN + 3 + c];

1763  can skip past them. */
1764  if(uip_buf[UIP_TCPIP_HLEN + UIP_LLH_LEN + 1 + c] == 0) {
1765  /* If the length field is zero, the options are malformed
1766  and we don't process them further. */

1985  retransmitting an old segment, and the application should put that

1994  if(uip_outstanding(uip_connr)) {
1995  goto drop;

2052  "persistent timer" and uses the retransmission mechanim.
