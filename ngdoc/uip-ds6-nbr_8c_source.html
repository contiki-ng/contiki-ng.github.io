77 uip_ds6_nbr_t *
78 uip_ds6_nbr_add(const uip_ipaddr_t *ipaddr, const uip_lladdr_t *lladdr,
79  uint8_t isrouter, uint8_t state, nbr_table_reason_t reason,
80  void *data)
81 {
82  uip_ds6_nbr_t *nbr = nbr_table_add_lladdr(ds6_neighbors, (linkaddr_t*)lladdr
83  , reason, data);
84  if(nbr) {
85  uip_ipaddr_copy(&nbr->ipaddr, ipaddr);
86 #if UIP_ND6_SEND_RA || !UIP_CONF_ROUTER
87  nbr->isrouter = isrouter;
88 #endif /* UIP_ND6_SEND_RA || !UIP_CONF_ROUTER */
89  nbr->state = state;
90 #if UIP_CONF_IPV6_QUEUE_PKT
91  uip_packetqueue_new(&nbr->packethandle);
92 #endif /* UIP_CONF_IPV6_QUEUE_PKT */
93 #if UIP_ND6_SEND_NS
94  if(nbr->state == NBR_REACHABLE) {
95  stimer_set(&nbr->reachable, UIP_ND6_REACHABLE_TIME / 1000);
96  } else {
97  /* We set the timer in expired state */
98  stimer_set(&nbr->reachable, 0);
99  }
100  stimer_set(&nbr->sendns, 0);
101  nbr->nscount = 0;
102 #endif /* UIP_ND6_SEND_NS */
103  LOG_INFO("Adding neighbor with ip addr ");
104  LOG_INFO_6ADDR(ipaddr);
105  LOG_INFO_(" link addr ");
106  LOG_INFO_LLADDR((linkaddr_t*)lladdr);
107  LOG_INFO_(" state %u\n", state);
108  NEIGHBOR_STATE_CHANGED(nbr);
109  return nbr;
110  } else {
111  LOG_INFO("Add drop ip addr ");
112  LOG_INFO_6ADDR(ipaddr);
113  LOG_INFO_(" link addr (%p) ", lladdr);
114  LOG_INFO_LLADDR((linkaddr_t*)lladdr);
115  LOG_INFO_(" state %u\n", state);
116  return NULL;
117  }
118 }

212 void
213 uip_ds6_link_callback(int status, int numtx)
214 {
215 #if UIP_DS6_LL_NUD
216  const linkaddr_t *dest = packetbuf_addr(PACKETBUF_ADDR_RECEIVER);
217  if(linkaddr_cmp(dest, &linkaddr_null)) {
218  return;
219  }
220 
221  /* From RFC4861, page 72, last paragraph of section 7.3.3:
222  *
223  * "In some cases, link-specific information may indicate that a path to
224  * a neighbor has failed (e.g., the resetting of a virtual circuit). In
225  * such cases, link-specific information may be used to purge Neighbor
226  * Cache entries before the Neighbor Unreachability Detection would do
227  * so. However, link-specific information MUST NOT be used to confirm
228  * the reachability of a neighbor; such information does not provide
229  * end-to-end confirmation between neighboring IP layers."
230  *
231  * However, we assume that receiving a link layer ack ensures the delivery
232  * of the transmitted packed to the IP stack of the neighbour. This is a
233  * fair assumption and allows battery powered nodes save some battery by
234  * not re-testing the state of a neighbour periodically if it
235  * acknowledges link packets. */
236  if(status == MAC_TX_OK) {
237  uip_ds6_nbr_t *nbr;
238  nbr = uip_ds6_nbr_ll_lookup((uip_lladdr_t *)dest);
239  if(nbr != NULL && nbr->state != NBR_INCOMPLETE) {
240  nbr->state = NBR_REACHABLE;
241  stimer_set(&nbr->reachable, UIP_ND6_REACHABLE_TIME / 1000);
242  LOG_INFO("received a link layer ACK : ");
243  LOG_INFO_LLADDR((uip_lladdr_t *)dest);
244  LOG_INFO_(" is reachable.\n");
245  }
246  }
247 #endif /* UIP_DS6_LL_NUD */
248 }

251 /** Periodic processing on neighbors */
252 void
253 uip_ds6_neighbor_periodic(void)
254 {
255  uip_ds6_nbr_t *nbr = nbr_table_head(ds6_neighbors);
256  while(nbr != NULL) {
257  switch(nbr->state) {
258  case NBR_REACHABLE:
259  if(stimer_expired(&nbr->reachable)) {
260 #if UIP_CONF_IPV6_RPL
261  /* when a neighbor leave its REACHABLE state and is a default router,
262  instead of going to STALE state it enters DELAY state in order to
263  force a NUD on it. Otherwise, if there is no upward traffic, the
264  node never knows if the default router is still reachable. This
265  mimics the 6LoWPAN-ND behavior.
266  */
267  if(uip_ds6_defrt_lookup(&nbr->ipaddr) != NULL) {
268  LOG_INFO("REACHABLE: defrt moving to DELAY (");
269  LOG_INFO_6ADDR(&nbr->ipaddr);
270  LOG_INFO_(")\n");
271  nbr->state = NBR_DELAY;
272  stimer_set(&nbr->reachable, UIP_ND6_DELAY_FIRST_PROBE_TIME);
273  nbr->nscount = 0;
274  } else {
275  LOG_INFO("REACHABLE: moving to STALE (");
276  LOG_INFO_6ADDR(&nbr->ipaddr);
277  LOG_INFO_(")\n");
278  nbr->state = NBR_STALE;
279  }
280 #else /* UIP_CONF_IPV6_RPL */
281  LOG_INFO("REACHABLE: moving to STALE (");
282  LOG_INFO_6ADDR(&nbr->ipaddr);
283  LOG_INFO_(")\n");
284  nbr->state = NBR_STALE;
285 #endif /* UIP_CONF_IPV6_RPL */
286  }
287  break;
288  case NBR_INCOMPLETE:
289  if(nbr->nscount >= UIP_ND6_MAX_MULTICAST_SOLICIT) {
290  uip_ds6_nbr_rm(nbr);
291  } else if(stimer_expired(&nbr->sendns) && (uip_len == 0)) {
292  nbr->nscount++;
293  LOG_INFO("NBR_INCOMPLETE: NS %u\n", nbr->nscount);
294  uip_nd6_ns_output(NULL, NULL, &nbr->ipaddr);
295  stimer_set(&nbr->sendns, uip_ds6_if.retrans_timer / 1000);
296  }
297  break;
298  case NBR_DELAY:
299  if(stimer_expired(&nbr->reachable)) {
300  nbr->state = NBR_PROBE;
301  nbr->nscount = 0;
302  LOG_INFO("DELAY: moving to PROBE\n");
303  stimer_set(&nbr->sendns, 0);
304  }
305  break;
306  case NBR_PROBE:
307  if(nbr->nscount >= UIP_ND6_MAX_UNICAST_SOLICIT) {
308  uip_ds6_defrt_t *locdefrt;
309  LOG_INFO("PROBE END\n");
310  if((locdefrt = uip_ds6_defrt_lookup(&nbr->ipaddr)) != NULL) {
311  if (!locdefrt->isinfinite) {
312  uip_ds6_defrt_rm(locdefrt);
313  }
314  }
315  uip_ds6_nbr_rm(nbr);
316  } else if(stimer_expired(&nbr->sendns) && (uip_len == 0)) {
317  nbr->nscount++;
318  LOG_INFO("PROBE: NS %u\n", nbr->nscount);
319  uip_nd6_ns_output(NULL, &nbr->ipaddr, &nbr->ipaddr);
320  stimer_set(&nbr->sendns, uip_ds6_if.retrans_timer / 1000);
321  }
322  break;
323  default:
324  break;
325  }
326  nbr = nbr_table_next(ds6_neighbors, nbr);
327  }
328 }

342 uip_ds6_nbr_t *
343 uip_ds6_get_least_lifetime_neighbor(void)
344 {
345  uip_ds6_nbr_t *nbr = nbr_table_head(ds6_neighbors);
346  uip_ds6_nbr_t *nbr_expiring = NULL;
347  while(nbr != NULL) {
348  if(nbr_expiring != NULL) {
349  clock_time_t curr = stimer_remaining(&nbr->reachable);
350  if(curr < stimer_remaining(&nbr->reachable)) {
351  nbr_expiring = nbr;
352  }
353  } else {
354  nbr_expiring = nbr;
355  }
356  nbr = nbr_table_next(ds6_neighbors, nbr);
357  }
358  return nbr_expiring;
359 }
