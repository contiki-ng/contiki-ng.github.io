43 static struct uip_udp_conn *broadcast_conn;
44 /*---------------------------------------------------------------------*/
45 PROCESS_THREAD(test_uip_process, ev, data)
46 {
47  PROCESS_BEGIN();
48 
49  printf("uIP test process started\n");
50 
51  broadcast_conn = udp_broadcast_new(COOJA_PORT, NULL);
52  button_sensor.configure(SENSORS_ACTIVE, 1);
53 
54  while(1) {
55  PROCESS_WAIT_EVENT();
56  printf("An event occured: ");
57 
58  if(ev == PROCESS_EVENT_EXIT) {
59  printf("shutting down\n");
60  break;
61  }
62 
63  if(ev == sensors_event && data == &button_sensor) {
64  printf("button clicked, sending packet\n");
65 
66  tcpip_poll_udp(broadcast_conn);
67  PROCESS_WAIT_UNTIL(ev == tcpip_event && uip_poll());
68  uip_send("cooyah COOJA", 12);
69  } else if(ev == sensors_event) {
70  printf("unknown sensor event: %s\n", ((struct sensors_sensor *)data)->type);
71  } else if(ev == tcpip_event && uip_newdata()) {
72  printf("a packet was received, toggling leds\n");
73  printf("packet data: '%s'\n", (char*) uip_appdata);
74  leds_toggle(LEDS_ALL);
75  } else {
76  printf("unknown event: %d\n", ev);
77  }
78  }
79 
80  PROCESS_END();
81 }
