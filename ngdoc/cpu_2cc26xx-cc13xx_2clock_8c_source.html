76 void
77 clock_init(void)
78 {
79  count = 0;
80 
81  /*
82  * Here, we configure GPT0 Timer A, which we subsequently use in
83  * clock_delay_usec
84  *
85  * We need to access registers, so firstly power up the PD and then enable
86  * the clock to GPT0.
87  */
88  if(ti_lib_prcm_power_domain_status(PRCM_DOMAIN_PERIPH) !=
89  PRCM_DOMAIN_POWER_ON) {
90  power_domain_on();
91  }
92 
93  ti_lib_prcm_peripheral_run_enable(PRCM_PERIPH_TIMER0);
94  ti_lib_prcm_load_set();
95  while(!ti_lib_prcm_load_get());
96 
97  /* Disable both GPT0 timers */
98  HWREG(GPT0_BASE + GPT_O_CTL) &= ~(GPT_CTL_TAEN | GPT_CTL_TBEN);
99 
100  /*
101  * We assume that the clock is running at 48MHz, we use GPT0 Timer A,
102  * one-shot, countdown, prescaled by 48 gives us 1 tick per usec
103  */
104  ti_lib_timer_configure(GPT0_BASE,
105  TIMER_CFG_SPLIT_PAIR | TIMER_CFG_B_ONE_SHOT);
106 
107  /* Global config: split pair (2 x 16-bit wide) */
108  HWREG(GPT0_BASE + GPT_O_CFG) = TIMER_CFG_SPLIT_PAIR >> 24;
109 
110  /*
111  * Pre-scale value 47 pre-scales by 48
112  *
113  * ToDo: The theoretical value here should be 47 (to provide x48 prescale)
114  * However, 49 seems to give results much closer to the desired delay
115  */
116  ti_lib_timer_prescale_set(GPT0_BASE, TIMER_B, 49);
117 
118  /* GPT0 / Timer B: One shot, PWM interrupt enable */
119  HWREG(GPT0_BASE + GPT_O_TBMR) =
120  ((TIMER_CFG_B_ONE_SHOT >> 8) & 0xFF) | GPT_TBMR_TBPWMIE;
121 
122  /* enable sync with radio timer */
123  HWREGBITW(AON_RTC_BASE + AON_RTC_O_CTL, AON_RTC_CTL_RTC_UPD_EN_BITN) = 1;
124 }

126 static void
127 update_clock_variable(void)
128 {
129  uint32_t aon_rtc_secs_now;
130  uint32_t aon_rtc_secs_now2;
131  uint16_t aon_rtc_ticks_now;
132 
133  do {
134  aon_rtc_secs_now = HWREG(AON_RTC_BASE + AON_RTC_O_SEC);
135  aon_rtc_ticks_now = HWREG(AON_RTC_BASE + AON_RTC_O_SUBSEC) >> 16;
136  aon_rtc_secs_now2 = HWREG(AON_RTC_BASE + AON_RTC_O_SEC);
137  } while(aon_rtc_secs_now != aon_rtc_secs_now2);
138 
139  /* Convert AON RTC ticks to clock tick counter */
140  count = (aon_rtc_secs_now * CLOCK_SECOND) + (aon_rtc_ticks_now >> 9);
141 }

188 void
189 clock_delay_usec(uint16_t len)
190 {
191  uint32_t clock_status;
192 
193  if(ti_lib_prcm_power_domain_status(PRCM_DOMAIN_PERIPH) !=
194  PRCM_DOMAIN_POWER_ON) {
195  power_domain_on();
196  }
197 
198  clock_status = HWREG(PRCM_BASE + PRCM_O_GPTCLKGR) & PRCM_GPIOCLKGR_CLK_EN;
199 
200  ti_lib_prcm_peripheral_run_enable(PRCM_PERIPH_TIMER0);
201  ti_lib_prcm_load_set();
202  while(!ti_lib_prcm_load_get());
203 
204  ti_lib_timer_load_set(GPT0_BASE, TIMER_B, len);
205  ti_lib_timer_enable(GPT0_BASE, TIMER_B);
206 
207  /*
208  * Wait for TBEN to clear. CC26xxware does not provide us with a convenient
209  * function, hence the direct register access here
210  */
211  while(HWREG(GPT0_BASE + GPT_O_CTL) & GPT_CTL_TBEN);
212 
213  if(clock_status == 0) {
214  ti_lib_prcm_peripheral_run_disable(PRCM_PERIPH_TIMER0);
215  ti_lib_prcm_load_set();
216  while(!ti_lib_prcm_load_get());
217  }
218 }
