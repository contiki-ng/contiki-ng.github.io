91 /**
92  * \brief Function initialization and configuration of RTC driver instance.
93  */
94 static void
95 rtc_config(void)
96 {
97  uint32_t err_code;
98 
99  //Initialize RTC instance
100  err_code = nrf_drv_rtc_init(&rtc, NULL, rtc_handler);
101  APP_ERROR_CHECK(err_code);
102 
103  //Enable tick event & interrupt
104  nrf_drv_rtc_tick_enable(&rtc, true);
105 
106  //Power on RTC instance
107  nrf_drv_rtc_enable(&rtc);
108 }
