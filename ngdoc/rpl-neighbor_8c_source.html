67 static int
68 max_acceptable_rank(void)
69 {
70  if(curr_instance.max_rankinc == 0) {
71  /* There is no max rank increment */
72  return RPL_INFINITE_RANK;
73  } else {
74  /* Make sure not to exceed RPL_INFINITE_RANK */
75  return MIN((uint32_t)curr_instance.dag.lowest_rank + curr_instance.max_rankinc, RPL_INFINITE_RANK);
76  }
77 }

79 /* As per RFC 6550, section 8.2.2.4 */
80 static int
81 acceptable_rank(rpl_rank_t rank)
82 {
83  return rank != RPL_INFINITE_RANK
84  && rank >= ROOT_RANK
85  && rank <= max_acceptable_rank();
86 }

88 void
89 rpl_neighbor_print_list(const char *str)
90 {
91  if(curr_instance.used) {
92  int curr_dio_interval = curr_instance.dag.dio_intcurrent;
93  int curr_rank = curr_instance.dag.rank;
94  rpl_nbr_t *nbr = nbr_table_head(rpl_neighbors);
95  rpl_nbr_t *best = best_parent(0);
96  clock_time_t clock_now = clock_time();
97 
98  LOG_INFO("nbr: own state, addr ");
99  LOG_INFO_6ADDR(rpl_get_global_address());
100  LOG_INFO_(", DAG state: %s, MOP %u OCP %u rank %u max-rank %u, dioint %u, nbr count %u (%s)\n",
101  rpl_dag_state_to_str(curr_instance.dag.state),
102  curr_instance.mop, curr_instance.of->ocp, curr_rank,
103  max_acceptable_rank(),
104  curr_dio_interval, rpl_neighbor_count(), str);
105  while(nbr != NULL) {
106  const struct link_stats *stats = rpl_neighbor_get_link_stats(nbr);
107  LOG_INFO("nbr: ");
108  LOG_INFO_6ADDR(rpl_neighbor_get_ipaddr(nbr));
109  LOG_INFO_(" %5u, %5u => %5u -- %2u %c%c%c%c%c",
110  nbr->rank,
111  rpl_neighbor_get_link_metric(nbr),
112  rpl_neighbor_rank_via_nbr(nbr),
113  stats != NULL ? stats->freshness : 0,
114  (nbr->rank == ROOT_RANK) ? 'r' : ' ',
115  nbr == best ? 'b' : ' ',
116  (acceptable_rank(rpl_neighbor_rank_via_nbr(nbr)) && rpl_neighbor_is_acceptable_parent(nbr)) ? 'a' : ' ',
117  link_stats_is_fresh(stats) ? 'f' : ' ',
118  nbr == curr_instance.dag.preferred_parent ? 'p' : ' '
119  );
120  if(stats->last_tx_time > 0) {
121  LOG_INFO_(" (last tx %u min ago", (unsigned)((clock_now - stats->last_tx_time) / (60 * CLOCK_SECOND)));
122  } else {
123  LOG_INFO_(" (no tx");
124  }
125  if(nbr->better_parent_since > 0) {
126  LOG_INFO_(", better since %u min)\n", (unsigned)((clock_now - nbr->better_parent_since) / (60 * CLOCK_SECOND)));
127  } else {
128  LOG_INFO_(")\n");
129  }
130  nbr = nbr_table_next(rpl_neighbors, nbr);
131  }
132  LOG_INFO("nbr: end of list\n");
133  }
134 }

162 static void
163 remove_neighbor(rpl_nbr_t *nbr)
164 {
165  /* Make sure we don't point to a removed neighbor. Note that we do not need
166  to worry about preferred_parent here, as it is locked in the the table
167  and will never be removed by external modules. */
168  if(nbr == curr_instance.dag.urgent_probing_target) {
169  curr_instance.dag.urgent_probing_target = NULL;
170  }
171  if(nbr == curr_instance.dag.unicast_dio_target) {
172  curr_instance.dag.unicast_dio_target = NULL;
173  }
174  nbr_table_remove(rpl_neighbors, nbr);
175  rpl_timers_schedule_state_update(); /* Updating from here is unsafe; postpone */
176 }

238 int
239 rpl_neighbor_is_reachable(rpl_nbr_t *nbr) {
240  if(nbr == NULL) {
241  return 0;
242  } else {
243 #if UIP_ND6_SEND_NS
244  uip_ds6_nbr_t *ds6_nbr = rpl_get_ds6_nbr(nbr);
245  /* Exclude links to a neighbor that is not reachable at a NUD level */
246  if(ds6_nbr == NULL || ds6_nbr->state != NBR_REACHABLE) {
247  return 0;
248  }
249 #endif /* UIP_ND6_SEND_NS */
250  /* If we don't have fresh link information, assume the nbr is reachable. */
251  return !rpl_neighbor_is_fresh(nbr) || curr_instance.of->nbr_has_usable_link(nbr);
252  }
253 }

261 void
262 rpl_neighbor_set_preferred_parent(rpl_nbr_t *nbr)
263 {
264  if(curr_instance.dag.preferred_parent != nbr) {
265  LOG_INFO("parent switch: ");
266  LOG_INFO_6ADDR(rpl_neighbor_get_ipaddr(curr_instance.dag.preferred_parent));
267  LOG_INFO_(" -> ");
268  LOG_INFO_6ADDR(rpl_neighbor_get_ipaddr(nbr));
269  LOG_INFO_("\n");
270 
271 #ifdef RPL_CALLBACK_PARENT_SWITCH
272  RPL_CALLBACK_PARENT_SWITCH(curr_instance.dag.preferred_parent, nbr);
273 #endif /* RPL_CALLBACK_PARENT_SWITCH */
274 
275  /* Always keep the preferred parent locked, so it remains in the
276  * neighbor table. */
277  nbr_table_unlock(rpl_neighbors, curr_instance.dag.preferred_parent);
278  nbr_table_lock(rpl_neighbors, nbr);
279 
280  /* Update DS6 default route. Use an infinite lifetime */
281  uip_ds6_defrt_rm(uip_ds6_defrt_lookup(
282  rpl_neighbor_get_ipaddr(curr_instance.dag.preferred_parent)));
283  uip_ds6_defrt_add(rpl_neighbor_get_ipaddr(nbr), 0);
284 
285  curr_instance.dag.preferred_parent = nbr;
286  }
287 }

289 /* Remove DAG neighbors with a rank that is at least the same as minimum_rank. */
290 void
291 rpl_neighbor_remove_all(void)
292 {
293  rpl_nbr_t *nbr;
294 
295  LOG_INFO("removing all neighbors\n");
296 
297  nbr = nbr_table_head(rpl_neighbors);
298  while(nbr != NULL) {
299  remove_neighbor(nbr);
300  nbr = nbr_table_next(rpl_neighbors, nbr);
301  }
302 
303  /* Update needed immediately so as to ensure preferred_parent becomes NULL,
304  * and no longer points to a de-allocated neighbor. */
305  rpl_dag_update_state();
306 }

316 static rpl_nbr_t *
317 best_parent(int fresh_only)
318 {
319  rpl_nbr_t *nbr;
320  rpl_nbr_t *best = NULL;
321 
322  if(curr_instance.used == 0) {
323  return NULL;
324  }
325 
326  /* Search for the best parent according to the OF */
327  for(nbr = nbr_table_head(rpl_neighbors); nbr != NULL; nbr = nbr_table_next(rpl_neighbors, nbr)) {
328 
329  if(!acceptable_rank(nbr->rank) || !curr_instance.of->nbr_is_acceptable_parent(nbr)) {
330  /* Exclude neighbors with a rank that is not acceptable) */
331  continue;
332  }
333 
334  if(fresh_only && !rpl_neighbor_is_fresh(nbr)) {
335  /* Filter out non-fresh nerighbors if fresh_only is set */
336  continue;
337  }
338 
339 #if UIP_ND6_SEND_NS
340  {
341  uip_ds6_nbr_t *ds6_nbr = rpl_get_ds6_nbr(nbr);
342  /* Exclude links to a neighbor that is not reachable at a NUD level */
343  if(ds6_nbr == NULL || ds6_nbr->state != NBR_REACHABLE) {
344  continue;
345  }
346  }
347 #endif /* UIP_ND6_SEND_NS */
348 
349  /* Now we have an acceptable parent, check if it is the new best */
350  best = curr_instance.of->best_parent(best, nbr);
351  }
352 
353  return best;
354 }

356 rpl_nbr_t *
357 rpl_neighbor_select_best(void)
358 {
359  rpl_nbr_t *best;
360 
361  if(rpl_dag_root_is_root()) {
362  return NULL; /* The root has no parent */
363  }
364 
365  /* Look for best parent (regardless of freshness) */
366  best = best_parent(0);
367 
368 #if RPL_WITH_PROBING
369  if(best != NULL) {
370  if(rpl_neighbor_is_fresh(best)) {
371  /* Return best if it is fresh */
372  return best;
373  } else {
374  rpl_nbr_t *best_fresh;
375 
376  /* The best is not fresh. Probe it (unless there is already an urgent
377  probing target). We will be called back after the probing anyway. */
378  if(curr_instance.dag.urgent_probing_target == NULL) {
379  LOG_WARN("best parent is not fresh, schedule urgent probing to ");
380  LOG_WARN_6ADDR(rpl_neighbor_get_ipaddr(best));
381  LOG_WARN_("\n");
382  curr_instance.dag.urgent_probing_target = best;
383  rpl_schedule_probing();
384  }
385 
386  /* The best is our preferred parent. It is not fresh but used to be,
387  else we would not have selected it in the first place. Stick to it
388  for a little while and rely on urgent probing to make a call. */
389  if(best == curr_instance.dag.preferred_parent) {
390  return best;
391  }
392 
393  /* Look for the best fresh parent. */
394  best_fresh = best_parent(1);
395  if(best_fresh == NULL) {
396  if(curr_instance.dag.preferred_parent == NULL) {
397  /* We will wait to find a fresh node before selecting our first parent */
398  return NULL;
399  } else {
400  /* We already have a parent, now stick to the best and count on
401  urgent probing to get a fresh parent soon */
402  return best;
403  }
404  } else {
405  /* Select best fresh */
406  return best_fresh;
407  }
408  }
409  } else {
410  /* No acceptable parent */
411  return NULL;
412  }
413 #else /* RPL_WITH_PROBING */
414  return best;
415 #endif /* RPL_WITH_PROBING */
416 }
