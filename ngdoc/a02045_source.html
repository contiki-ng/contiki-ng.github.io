64 /*--------------------------------------------------------------------*/
65 static void
66 input(void)
67 {
68  if(current_callback != NULL) {
69  LOG_INFO("received %u bytes from ", packetbuf_datalen());
70  LOG_INFO_LLADDR(packetbuf_addr(PACKETBUF_ADDR_SENDER));
71  LOG_INFO_("\n");
72  current_callback(packetbuf_dataptr(), packetbuf_datalen(),
73  packetbuf_addr(PACKETBUF_ADDR_SENDER), packetbuf_addr(PACKETBUF_ADDR_RECEIVER));
74  }
75 }
76 /*--------------------------------------------------------------------*/
77 void
78 nullnet_set_input_callback(nullnet_input_callback callback)
79 {
80  current_callback = callback;
81 }
82 /*--------------------------------------------------------------------*/
83 void
84 nullnet_output(const void *data, uint16_t len, const linkaddr_t *dest)
85 {
86  packetbuf_clear();
87  packetbuf_copyfrom(data, len);
88  if(dest != NULL) {
89  packetbuf_set_addr(PACKETBUF_ADDR_RECEIVER, dest);
90  } else {
91  packetbuf_set_addr(PACKETBUF_ADDR_RECEIVER, &linkaddr_null);
92  }
93  packetbuf_set_addr(PACKETBUF_ADDR_SENDER, &linkaddr_node_addr);
94  LOG_INFO("sending %u bytes to ", packetbuf_datalen());
95  LOG_INFO_LLADDR(packetbuf_addr(PACKETBUF_ADDR_RECEIVER));
96  LOG_INFO_("\n");
97  NETSTACK_MAC.send(NULL, NULL);
98 }
