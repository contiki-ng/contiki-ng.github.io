82 static void
83 handle_trans_timeout(void *ptr)
84 {
85  sixp_trans_t *trans = (sixp_trans_t *)ptr;
86 
87  assert(trans != NULL);
88  if(trans == NULL) {
89  return;
90  }
91 
92  if(trans->sf->timeout != NULL) {
93  trans->sf->timeout(trans->cmd,
94  (const linkaddr_t *)&trans->peer_addr);
95  }
96 
97  (void)sixp_trans_transit_state(trans, SIXP_TRANS_STATE_TERMINATING);
98 }
99 /*---------------------------------------------------------------------------*/
100 static void
101 start_trans_timer(sixp_trans_t *trans)
102 {
103  ctimer_set(&trans->timer, trans->sf->timeout_interval,
104  handle_trans_timeout, trans);
105 }
106 /*---------------------------------------------------------------------------*/
107 static void
108 process_trans(void *ptr)
109 {
110  sixp_trans_t *trans = (sixp_trans_t *)ptr;
111 
112  assert(trans != NULL);
113  if(trans == NULL) {
114  return;
115  }
116 
117  /* make sure that the timer is stopped */
118  ctimer_stop(&trans->timer);
119 
120  /* state-specific operation */
121  if(trans->state == SIXP_TRANS_STATE_TERMINATING) {
122  /* handle the terminating state first */
123  LOG_INFO("6P-trans: trans [peer_addr:");
124  LOG_INFO_LLADDR((const linkaddr_t *)&trans->peer_addr);
125  LOG_INFO_(", seqno:%u] is going to be freed\n", trans->seqno);
126  free_trans(trans);
127  return;
128  }
129 
130  switch(trans->state) {
131  /* do for others */
132  case SIXP_TRANS_STATE_RESPONSE_SENT:
133  case SIXP_TRANS_STATE_RESPONSE_RECEIVED:
134  if(trans->mode == SIXP_TRANS_MODE_2_STEP) {
135  (void)sixp_trans_transit_state(trans, SIXP_TRANS_STATE_TERMINATING);
136  }
137  break;
138  case SIXP_TRANS_STATE_CONFIRMATION_SENT:
139  case SIXP_TRANS_STATE_CONFIRMATION_RECEIVED:
140  (void)sixp_trans_transit_state(trans, SIXP_TRANS_STATE_TERMINATING);
141  break;
142  case SIXP_TRANS_STATE_TERMINATING:
143  default:
144  break;
145  }
146 
147  if(trans->state != SIXP_TRANS_STATE_TERMINATING) {
148  /* set the timer with a timeout values defined by the SF */
149  start_trans_timer(trans);
150  }
151 }

153 static void
154 schedule_trans_process(sixp_trans_t *trans)
155 {
156  assert(trans != NULL);
157  if(trans == NULL) {
158  return;
159  }
160 
161  ctimer_stop(&trans->timer);
162  ctimer_set(&trans->timer, 0, process_trans, trans); /* expires immediately */
163 }

177 static sixp_trans_mode_t
178 determine_trans_mode(const sixp_pkt_t *req)
179 {
180  uint16_t cell_list_len;
181 
182  assert(req != NULL);
183  if(req == NULL) {
184  return SIXP_TRANS_MODE_UNAVAILABLE;
185  }
186 
187  /*
188  * We consider a transaction as 3-step if and only if its request command is
189  * either Add or Delete AND its cell_list is empty. Otherwise, 2-step.
190  */
191  if(req->type == SIXP_PKT_TYPE_REQUEST &&
192  (req->code.cmd == SIXP_PKT_CMD_ADD ||
193  req->code.cmd == SIXP_PKT_CMD_DELETE) &&
194  sixp_pkt_get_cell_list(req->type, (sixp_pkt_code_t)req->code.value,
195  NULL, &cell_list_len,
196  req->body, req->body_len) == 0 &&
197  cell_list_len == 0) {
198  return SIXP_TRANS_MODE_3_STEP;
199  }
200 
201  return SIXP_TRANS_MODE_2_STEP;
202 }
203 /*---------------------------------------------------------------------------*/
204 int
205 sixp_trans_transit_state(sixp_trans_t *trans, sixp_trans_state_t new_state)
206 {
207  assert(trans != NULL);
208  if(trans == NULL) {
209  LOG_ERR("6top: invalid argument, trans is NULL\n");
210  return -1;
211  }
212 
213  /* enforce state transition rules */
214  if(new_state == SIXP_TRANS_STATE_TERMINATING ||
215  (new_state == SIXP_TRANS_STATE_REQUEST_SENT &&
216  trans->state == SIXP_TRANS_STATE_INIT) ||
217  (new_state == SIXP_TRANS_STATE_REQUEST_RECEIVED &&
218  trans->state == SIXP_TRANS_STATE_INIT) ||
219  (new_state == SIXP_TRANS_STATE_RESPONSE_SENT &&
220  trans->state == SIXP_TRANS_STATE_REQUEST_RECEIVED) ||
221  (new_state == SIXP_TRANS_STATE_RESPONSE_RECEIVED &&
222  trans->state == SIXP_TRANS_STATE_REQUEST_SENT) ||
223  (new_state == SIXP_TRANS_STATE_CONFIRMATION_RECEIVED &&
224  trans->state == SIXP_TRANS_STATE_RESPONSE_SENT &&
225  trans->mode == SIXP_TRANS_MODE_3_STEP) ||
226  (new_state == SIXP_TRANS_STATE_CONFIRMATION_SENT &&
227  trans->state == SIXP_TRANS_STATE_RESPONSE_RECEIVED &&
228  trans->mode == SIXP_TRANS_MODE_3_STEP)) {
229  LOG_INFO("6P-trans: trans %p state changes from %u to %u\n",
230  trans, trans->state, new_state);
231  trans->state = new_state;
232  schedule_trans_process(trans);
233  return 0;
234  }
235 
236  /* invalid transition */
237  LOG_ERR("6P-trans: invalid transaction, from %u to %u, detected on trans %p\n",
238  trans->state, new_state, trans);
239  return -1;
240 }

309 sixp_trans_t *
310 sixp_trans_alloc(const sixp_pkt_t *pkt, const linkaddr_t *peer_addr)
311 {
312  const sixtop_sf_t *sf;
313  sixp_trans_t *trans;
314 
315  assert(pkt != NULL && peer_addr != NULL);
316  if(pkt == NULL || peer_addr == NULL) {
317  LOG_ERR("6P-trans: sixp_trans_alloc() fails because of invalid argument\n");
318  return NULL;
319  }
320 
321  if((sf = sixtop_find_sf(pkt->sfid)) == NULL) {
322  LOG_ERR("6P-trans: sixp_trans_alloc() fails; no suitable SF [sfid:%u]\n",
323  pkt->sfid);
324  return NULL;
325  }
326 
327  if(sixp_trans_find(peer_addr) != NULL) {
328  LOG_ERR("6P-trans: sixp_trans_alloc() fails because another trans with ");
329  LOG_ERR_LLADDR((const linkaddr_t *)peer_addr);
330  LOG_ERR_("is in process\n");
331  return NULL;
332  }
333 
334  if((trans = memb_alloc(&trans_memb)) == NULL) {
335  LOG_ERR("6P-trans: sixp_trans_alloc() fails because of lack of memory\n");
336  return NULL;
337  }
338 
339  memset(trans, 0, sizeof(sixp_trans_t));
340  trans->sf = sf;
341  trans->peer_addr = *peer_addr;
342  trans->seqno = pkt->seqno;
343  trans->cmd = pkt->code.value;
344  trans->state = SIXP_TRANS_STATE_INIT;
345  trans->mode = determine_trans_mode(pkt);
346  list_add(trans_list, trans);
347  start_trans_timer(trans);
348 
349  return trans;
350 }

352 sixp_trans_t *
353 sixp_trans_find(const linkaddr_t *peer_addr)
354 {
355  sixp_trans_t *trans;
356 
357  assert(peer_addr != NULL);
358  if(peer_addr == NULL) {
359  return NULL;
360  }
361 
362  /*
363  * XXX: we don't support concurrent 6P transactions which is mentioned in
364  * Section 4.3.3, draft-ietf-6tisch-6top-protocol-03.
365  *
366  * The assumption here is that there is one transactions for a single peer at
367  * most.
368  */
369  for(trans = list_head(trans_list);
370  trans != NULL; trans = trans->next) {
371  if(memcmp(peer_addr, &trans->peer_addr, sizeof(linkaddr_t)) == 0) {
372  return trans;
373  }
374  }
375 
376  return NULL;
377 }

379 int
380 sixp_trans_init(void)
381 {
382  sixp_trans_t *trans, *next_trans;
383 
384  /* make sure there's no timer task left before the initialization */
385  for(trans = list_head(trans_list);
386  trans != NULL; trans = next_trans) {
387  next_trans = trans->next;
388  ctimer_stop(&trans->timer);
389  free_trans(trans);
390  }
391 
392  list_init(trans_list);
393  memb_init(&trans_memb);
394  return 0;
395 }
